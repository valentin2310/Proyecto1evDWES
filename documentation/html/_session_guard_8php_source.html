141 public function user()
142 {
143 if ($this->loggedOut) {
144 return;
145 }
146
147 // If we've already retrieved the user for the current request we can just
148 // return it back immediately. We do not want to fetch the user data on
149 // every call to this method because that would be tremendously slow.
150 if (! is_null($this->user)) {
151 return $this->user;
152 }
153
154 $id = $this->session->get($this->getName());
155
156 // First we will try to load the user using the identifier in the session if
157 // one exists. Otherwise we will check for a "remember me" cookie in this
158 // request, and if one exists, attempt to retrieve the user using that.
159 if (! is_null($id) && $this->user = $this->provider->retrieveById($id)) {
160 $this->fireAuthenticatedEvent($this->user);
161 }
162
163 // If the user is null, but we decrypt a "recaller" cookie we can attempt to
164 // pull the user data on that cookie which serves as a remember cookie on
165 // the application. Once we have a user we can return it to the caller.
166 if (is_null($this->user) && ! is_null($recaller = $this->recaller())) {
167 $this->user = $this->userFromRecaller($recaller);
168
169 if ($this->user) {
170 $this->updateSession($this->user->getAuthIdentifier());
171
172 $this->fireLoginEvent($this->user, true);
173 }
174 }
175
176 return $this->user;

377 public function attempt(array $credentials = [], $remember = false)
378 {
379 $this->fireAttemptEvent($credentials, $remember);
380
381 $this->lastAttempted = $user = $this->provider->retrieveByCredentials($credentials);
382
383 // If an implementation of UserInterface was returned, we'll ask the provider
384 // to validate the user against the given credentials, and if they are in
385 // fact valid we'll log the users into the application and return true.
386 if ($this->hasValidCredentials($user, $credentials)) {
387 $this->login($user, $remember);
388
389 return true;
390 }
391
392 // If the authentication attempt fails we will fire an event so that the user
393 // may be notified of any suspicious attempts to access their account from
394 // an unrecognized user. A developer may listen to this event as needed.
395 $this->fireFailedEvent($user, $credentials);
396
397 return false;
