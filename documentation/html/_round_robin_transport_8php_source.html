49 public function send(RawMessage $message, Envelope $envelope = null): ?SentMessage
50 {
51 $exception = null;
52
53 while ($transport = $this->getNextTransport()) {
54 try {
55 return $transport->send($message, $envelope);
56 } catch (TransportExceptionInterface $e) {
57 $exception ??= new TransportException('All transports failed.');
58 $exception->appendDebug(sprintf("Transport \"%s\": %s\n", $transport, $e->getDebug()));

60 }
61 }
62

64 }

74 protected function getNextTransport(): ?TransportInterface
75 {
76 if (-1 === $this->cursor) {
77 $this->cursor = $this->getInitialCursor();
78 }
79
80 $cursor = $this->cursor;
81 while (true) {
82 $transport = $this->transports[$cursor];
83
84 if (!$this->isTransportDead($transport)) {
85 break;
86 }
87
88 if ((microtime(true) - $this->deadTransports[$transport]) > $this->retryPeriod) {
89 $this->deadTransports->detach($transport);
90
91 break;
92 }
93
94 if ($this->cursor === $cursor = $this->moveCursor($cursor)) {

97 }

102 }
