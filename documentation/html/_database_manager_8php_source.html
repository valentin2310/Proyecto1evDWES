134 protected function makeConnection($name)
135 {
136 $config = $this->configuration($name);
137
138 // First we will check by the connection name to see if an extension has been
139 // registered specifically for that connection. If it has we will call the
140 // Closure and pass it the config allowing it to resolve the connection.
141 if (isset($this->extensions[$name])) {
142 return call_user_func($this->extensions[$name], $config, $name);
143 }
144
145 // Next we will check to see if an extension has been registered for a driver
146 // and will call the Closure if so, which allows us to have a more generic
147 // resolver for the drivers themselves which applies to all connections.
148 if (isset($this->extensions[$driver = $config['driver']])) {
149 return call_user_func($this->extensions[$driver], $config, $name);
150 }
151
152 return $this->factory->make($config, $name);
153 }

187 protected function configure(Connection $connection, $type)
188 {
189 $connection = $this->setPdoForType($connection, $type)->setReadWriteType($type);
190
191 // First we'll set the fetch mode and a few other dependencies of the database
192 // connection. This method basically just configures and prepares it to get
193 // used by the application. Once we're finished we'll return it back out.
194 if ($this->app->bound('events')) {
195 $connection->setEventDispatcher($this->app['events']);
196 }
197
198 if ($this->app->bound('db.transactions')) {
199 $connection->setTransactionManager($this->app['db.transactions']);
200 }
201
202 // Here we'll set a reconnector callback. This reconnector can be any callable
203 // so we will set a Closure to reconnect from this manager with the name of
204 // the connection, which will allow us to reconnect from the connections.
205 $connection->setReconnector($this->reconnector);
206
207 $this->registerConfiguredDoctrineTypes($connection);
208
209 return $connection;
210 }
