64 protected function getMethods(bool $showAll, \ReflectionClass $reflector, bool $noInherit = false): array
65 {
66 $className = $reflector->getName();
67
68 $methods = [];
69 foreach ($reflector->getMethods() as $name => $method) {
70 // For some reason PHP reflection shows private methods from the parent class, even
71 // though they're effectively worthless. Let's suppress them here, like --no-inherit
72 if (($noInherit || $method->isPrivate()) && $method->getDeclaringClass()->getName() !== $className) {
73 continue;
74 }
75
76 if ($showAll || $method->isPublic()) {
77 $methods[$method->getName()] = $method;
78 }
79 }
80
81 \ksort($methods, \SORT_NATURAL | \SORT_FLAG_CASE);
82
83 return $methods;
84 }
