104 protected static function prefixLength(array $lines): int
105 {
106 // find only lines with interesting things
107 $lines = \array_filter($lines, function ($line) {
108 return \substr($line, \strspn($line, "* \t\n\r\0\x0B"));
109 });
110
111 // if we sort the lines, we only have to compare two items
112 \sort($lines);
113
114 $first = \reset($lines);
115 $last = \end($lines);
116
117 // Special case for single-line comments
118 if (\count($lines) === 1) {
119 return \strspn($first, "* \t\n\r\0\x0B");
120 }
121
122 // find the longest common substring
123 $count = \min(\strlen($first), \strlen($last));
124 for ($i = 0; $i < $count; $i++) {
125 if ($first[$i] !== $last[$i]) {
126 return $i;
127 }
128 }
129
130 return $count;
131 }

138 protected function parseComment(string $comment)
139 {
140 // Strip the opening and closing tags of the docblock
141 $comment = \substr($comment, 3, -2);
142
143 // Split into arrays of lines
144 $comment = \array_filter(\preg_split('/\r?\n\r?/', $comment));
145
146 // Trim asterisks and whitespace from the beginning and whitespace from the end of lines
147 $prefixLength = self::prefixLength($comment);
148 $comment = \array_map(function ($line) use ($prefixLength) {
149 return \rtrim(\substr($line, $prefixLength));
150 }, $comment);
151
152 // Group the lines together by @tags
153 $blocks = [];
154 $b = -1;
155 foreach ($comment as $line) {
156 if (self::isTagged($line)) {
157 $b++;
158 $blocks[] = [];
159 } elseif ($b === -1) {
160 $b = 0;
161 $blocks[] = [];
162 }
163 $blocks[$b][] = $line;
164 }
165
166 // Parse the blocks
167 foreach ($blocks as $block => $body) {
168 $body = \trim(\implode("\n", $body));
169
170 if ($block === 0 && !self::isTagged($body)) {
171 // This is the description block
172 $this->desc = $body;
173 } else {
174 // This block is tagged
175 $tag = \substr(self::strTag($body), 1);
176 $body = \ltrim(\substr($body, \strlen($tag) + 2));
177
178 if (isset(self::$vectors[$tag])) {
179 // The tagged block is a vector
180 $count = \count(self::$vectors[$tag]);
181 if ($body) {
182 $parts = \preg_split('/\s+/', $body, $count);
183 } else {
184 $parts = [];
185 }
186
187 // Default the trailing values
188 $parts = \array_pad($parts, $count, null);
189
190 // Store as a mapped array
191 $this->tags[$tag][] = \array_combine(self::$vectors[$tag], $parts);
192 } else {
193 // The tagged block is only text
194 $this->tags[$tag][] = $body;
195 }
196 }
197 }
