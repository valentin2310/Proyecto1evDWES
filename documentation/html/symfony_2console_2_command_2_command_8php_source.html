278 public function run(InputInterface $input, OutputInterface $output): int
279 {
280 // add the application arguments and options
281 $this->mergeApplicationDefinition();
282
283 // bind the input against the command specific arguments/options
284 try {
285 $input->bind($this->getDefinition());
286 } catch (ExceptionInterface $e) {
287 if (!$this->ignoreValidationErrors) {
288 throw $e;
289 }
290 }
291
292 $this->initialize($input, $output);
293
294 if (null !== $this->processTitle) {
295 if (\function_exists('cli_set_process_title')) {
296 if (!@cli_set_process_title($this->processTitle)) {
297 if ('Darwin' === \PHP_OS) {
298 $output->writeln('<comment>Running "cli_set_process_title" as an unprivileged user is not supported on MacOS.</comment>', OutputInterface::VERBOSITY_VERY_VERBOSE);
299 } else {
300 cli_set_process_title($this->processTitle);
301 }
302 }
303 } elseif (\function_exists('setproctitle')) {
304 setproctitle($this->processTitle);
305 } elseif (OutputInterface::VERBOSITY_VERY_VERBOSE === $output->getVerbosity()) {
306 $output->writeln('<comment>Install the proctitle PECL to be able to change the process title.</comment>');
307 }
308 }
309
310 if ($input->isInteractive()) {
311 $this->interact($input, $output);
312 }
313
314 // The command name argument is often omitted when a command is executed directly with its run() method.
315 // It would fail the validation if we didn't make sure the command argument is present,
316 // since it's required by the application.
317 if ($input->hasArgument('command') && null === $input->getArgument('command')) {
318 $input->setArgument('command', $this->getName());
319 }
320
321 $input->validate();
322
323 if ($this->code) {
324 $statusCode = ($this->code)($input, $output);
325 } else {
326 $statusCode = $this->execute($input, $output);
327
328 if (!\is_int($statusCode)) {
329 throw new \TypeError(sprintf('Return value of "%s::execute()" must be of the type int, "%s" returned.', static::class, get_debug_type($statusCode)));
330 }
331 }
332
333 return is_numeric($statusCode) ? (int) $statusCode : 0;
334 }
