953 {

955 // which simply lets the developers tweak the attribute as it is set on
956 // this model, such as "json_encoding" a listing of data for storage.
957 if ($this->hasSetMutator($key)) {
958 return $this->setMutatedAttributeValue($key, $value);
959 } elseif ($this->hasAttributeSetMutator($key)) {
960 return $this->setAttributeMarkedMutatedAttributeValue($key, $value);
961 }
962
963 // If an attribute is listed as a "date", we'll convert it from a DateTime
964 // instance into a form proper for storage on the database tables using
965 // the connection grammar's date format. We will auto set the values.
966 elseif (! is_null($value) && $this->isDateAttribute($key)) {
967 $value = $this->fromDateTime($value);
968 }
969
970 if ($this->isEnumCastable($key)) {
971 $this->setEnumCastableAttribute($key, $value);
972
973 return $this;
974 }
975
976 if ($this->isClassCastable($key)) {
977 $this->setClassCastableAttribute($key, $value);
978
979 return $this;

983 $value = $this->castAttributeAsJson($key, $value);
984 }

987 // attribute's underlying array. This takes care of properly nesting an
988 // attribute in the array's value in the case of deeply nested items.
989 if (str_contains($key, '->')) {
990 return $this->fillJsonAttribute($key, $value);

994 $value = $this->castAttributeAsEncryptedString($key, $value);
