35 public function codeUnitsToSourceLines(CodeUnitCollection $codeUnits): array
36 {
37 $result = [];
38
39 foreach ($codeUnits as $codeUnit) {
40 $sourceFileName = $codeUnit->sourceFileName();
41
42 if (!isset($result[$sourceFileName])) {
43 $result[$sourceFileName] = [];
44 }
45
46 $result[$sourceFileName] = array_merge($result[$sourceFileName], $codeUnit->sourceLines());
47 }
48
49 foreach (array_keys($result) as $sourceFileName) {
50 $result[$sourceFileName] = array_values(array_unique($result[$sourceFileName]));
51
52 sort($result[$sourceFileName]);
53 }
54
55 ksort($result);
56
57 return $result;
58 }
