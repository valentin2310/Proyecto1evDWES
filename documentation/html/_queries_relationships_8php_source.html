606 public function withAggregate($relations, $column, $function = null)
607 {
608 if (empty($relations)) {
609 return $this;
610 }
611
612 if (is_null($this->query->columns)) {
613 $this->query->select([$this->query->from.'.*']);
614 }

617
618 foreach ($this->parseWithRelations($relations) as $name => $constraints) {

620 // and if it has we will extract the actual relationship name and the desired name of
621 // the resulting column. This allows multiple aggregates on the same relationships.
622 $segments = explode(' ', $name);
623
624 unset($alias);
625
626 if (count($segments) === 3 && Str::lower($segments[1]) === 'as') {
627 [$name, $alias] = [$segments[0], $segments[2]];
628 }
629
630 $relation = $this->getRelationWithoutConstraints($name);
631
632 if ($function) {
633 $hashedColumn = $this->getRelationHashedColumn($column, $relation);
634
635 $wrappedColumn = $this->getQuery()->getGrammar()->wrap(
636 $column === '*' ? $column : $relation->getRelated()->qualifyColumn($hashedColumn)
637 );
638
639 $expression = $function === 'exists' ? $wrappedColumn : sprintf('%s(%s)', $function, $wrappedColumn);
640 } else {
641 $expression = $column;
642 }
643
644 // Here, we will grab the relationship sub-query and prepare to add it to the main query
645 // as a sub-select. First, we'll get the "has" query and use that to get the relation
646 // sub-query. We'll format this relationship name and append this column if needed.
647 $query = $relation->getRelationExistenceQuery(
648 $relation->getRelated()->newQuery(), $this, new Expression($expression)
649 )->setBindings([], 'select');

652

654
655 // If the query contains certain elements like orderings / more than one column selected
656 // then we will remove those elements from the query so that it will execute properly
657 // when given to the database. Otherwise, we may receive SQL errors or poor syntax.
658 $query->orders = null;

660
661 if (count($query->columns) > 1) {
662 $query->columns = [$query->columns[0]];
663 $query->bindings['select'] = [];
664 }
665
666 // Finally, we will make the proper column alias to the query and run this sub-select on
667 // the query builder. Then, we will return the builder instance back to the developer
668 // for further constraint chaining that needs to take place on the query as needed.
669 $alias ??= Str::snake(

671 );
672

674 $this->selectRaw(
675 sprintf('exists(%s) as %s', $query->toSql(), $this->getQuery()->grammar->wrap($alias)),
676 $query->getBindings()
677 )->withCasts([$alias => 'bool']);
678 } else {

680 $function ? $query : $query->limit(1),

682 );
683 }
684 }
