145 public function run(InputInterface $input = null, OutputInterface $output = null): int
146 {
147 if (\function_exists('putenv')) {
148 @putenv('LINES='.$this->terminal->getHeight());
149 @putenv('COLUMNS='.$this->terminal->getWidth());
150 }
151
152 $input ??= new ArgvInput();
153 $output ??= new ConsoleOutput();
154
155 $renderException = function (\Throwable $e) use ($output) {
156 if ($output instanceof ConsoleOutputInterface) {
157 $this->renderThrowable($e, $output->getErrorOutput());
158 } else {
159 $this->renderThrowable($e, $output);
160 }
161 };
162 if ($phpHandler = set_exception_handler($renderException)) {
163 restore_exception_handler();
164 if (!\is_array($phpHandler) || !$phpHandler[0] instanceof ErrorHandler) {
165 $errorHandler = true;
166 } elseif ($errorHandler = $phpHandler[0]->setExceptionHandler($renderException)) {
167 $phpHandler[0]->setExceptionHandler($errorHandler);
168 }
169 }
170
171 $this->configureIO($input, $output);
172
173 try {
174 $exitCode = $this->doRun($input, $output);
175 } catch (\Exception $e) {
176 if (!$this->catchExceptions) {
177 throw $e;
178 }
179
180 $renderException($e);
181
182 $exitCode = $e->getCode();
183 if (is_numeric($exitCode)) {
184 $exitCode = (int) $exitCode;
185 if ($exitCode <= 0) {
186 $exitCode = 1;
187 }
188 } else {
189 $exitCode = 1;
190 }
191 } finally {
192 // if the exception handler changed, keep it
193 // otherwise, unregister $renderException
194 if (!$phpHandler) {
195 if (set_exception_handler($renderException) === $renderException) {
196 restore_exception_handler();
197 }
198 restore_exception_handler();
199 } elseif (!$errorHandler) {
200 $finalHandler = $phpHandler[0]->setExceptionHandler(null);
201 if ($finalHandler !== $renderException) {
202 $phpHandler[0]->setExceptionHandler($finalHandler);
203 }
204 }
205 }
206
207 if ($this->autoExit) {
208 if ($exitCode > 255) {
209 $exitCode = 255;
210 }
211
212 exit($exitCode);
213 }
214
215 return $exitCode;

223 public function doRun(InputInterface $input, OutputInterface $output)
224 {
225 if (true === $input->hasParameterOption(['--version', '-V'], true)) {
226 $output->writeln($this->getLongVersion());
227
228 return 0;
229 }
230
231 try {
232 // Makes ArgvInput::getFirstArgument() able to distinguish an option from an argument.
233 $input->bind($this->getDefinition());
234 } catch (ExceptionInterface) {
235 // Errors must be ignored, full binding/validation happens later when the command is known.
236 }
237
238 $name = $this->getCommandName($input);
239 if (true === $input->hasParameterOption(['--help', '-h'], true)) {
240 if (!$name) {
241 $name = 'help';
242 $input = new ArrayInput(['command_name' => $this->defaultCommand]);
243 } else {
244 $this->wantHelps = true;
245 }
246 }
247
248 if (!$name) {
249 $name = $this->defaultCommand;
250 $definition = $this->getDefinition();
251 $definition->setArguments(array_merge(
252 $definition->getArguments(),
253 [
254 'command' => new InputArgument('command', InputArgument::OPTIONAL, $definition->getArgument('command')->getDescription(), $name),
255 ]
256 ));
257 }
258
259 try {
260 $this->runningCommand = null;
261 // the command name MUST be the first element of the input
262 $command = $this->find($name);
263 } catch (\Throwable $e) {
264 if (($e instanceof CommandNotFoundException && !$e instanceof NamespaceNotFoundException) && 1 === \count($alternatives = $e->getAlternatives()) && $input->isInteractive()) {
265 $alternative = $alternatives[0];
266
267 $style = new SymfonyStyle($input, $output);
268 $output->writeln('');
269 $formattedBlock = (new FormatterHelper())->formatBlock(sprintf('Command "%s" is not defined.', $name), 'error', true);
270 $output->writeln($formattedBlock);
271 if (!$style->confirm(sprintf('Do you want to run "%s" instead? ', $alternative), false)) {
272 if (null !== $this->dispatcher) {
273 $event = new ConsoleErrorEvent($input, $output, $e);
274 $this->dispatcher->dispatch($event, ConsoleEvents::ERROR);
275
276 return $event->getExitCode();
277 }
278
279 return 1;
280 }
281
282 $command = $this->find($alternative);
283 } else {
284 if (null !== $this->dispatcher) {
285 $event = new ConsoleErrorEvent($input, $output, $e);
286 $this->dispatcher->dispatch($event, ConsoleEvents::ERROR);
287
288 if (0 === $event->getExitCode()) {
289 return 0;
290 }
291
292 $e = $event->getError();
293 }
294
295 try {
296 if ($e instanceof CommandNotFoundException && $namespace = $this->findNamespace($name)) {
297 $helper = new DescriptorHelper();
298 $helper->describe($output instanceof ConsoleOutputInterface ? $output->getErrorOutput() : $output, $this, [
299 'format' => 'txt',
300 'raw_text' => false,
301 'namespace' => $namespace,
302 'short' => false,
303 ]);
304
305 return isset($event) ? $event->getExitCode() : 1;
306 }
307
308 throw $e;
309 } catch (NamespaceNotFoundException) {
310 throw $e;
311 }
312 }
313 }
314
315 if ($command instanceof LazyCommand) {
316 $command = $command->getCommand();
317 }
318
319 $this->runningCommand = $command;
320 $exitCode = $this->doRunCommand($command, $input, $output);
321 $this->runningCommand = null;
322
323 return $exitCode;
324 }

638 public function findNamespace(string $namespace): string
639 {
640 $allNamespaces = $this->getNamespaces();
641 $expr = implode('[^:]*:', array_map('preg_quote', explode(':', $namespace))).'[^:]*';
642 $namespaces = preg_grep('{^'.$expr.'}', $allNamespaces);
643
644 if (empty($namespaces)) {
645 $message = sprintf('There are no commands defined in the "%s" namespace.', $namespace);
646
647 if ($alternatives = $this->findAlternatives($namespace, $allNamespaces)) {
648 if (1 == \count($alternatives)) {
649 $message .= "\n\nDid you mean this?\n ";
650 } else {
651 $message .= "\n\nDid you mean one of these?\n ";
652 }
653
654 $message .= implode("\n ", $alternatives);
655 }
656
657 throw new NamespaceNotFoundException($message, $alternatives);
658 }
659
660 $exact = \in_array($namespace, $namespaces, true);
661 if (\count($namespaces) > 1 && !$exact) {
662 throw new NamespaceNotFoundException(sprintf("The namespace \"%s\" is ambiguous.\nDid you mean one of these?\n%s.", $namespace, $this->getAbbreviationSuggestions(array_values($namespaces))), array_values($namespaces));
663 }
664
665 return $exact ? $namespace : reset($namespaces);
666 }

678 public function find(string $name)
679 {
680 $this->init();
681
682 $aliases = [];
683
684 foreach ($this->commands as $command) {
685 foreach ($command->getAliases() as $alias) {
686 if (!$this->has($alias)) {
687 $this->commands[$alias] = $command;
688 }
689 }
690 }
691
692 if ($this->has($name)) {
693 return $this->get($name);
694 }
695
696 $allCommands = $this->commandLoader ? array_merge($this->commandLoader->getNames(), array_keys($this->commands)) : array_keys($this->commands);
697 $expr = implode('[^:]*:', array_map('preg_quote', explode(':', $name))).'[^:]*';
698 $commands = preg_grep('{^'.$expr.'}', $allCommands);
699
700 if (empty($commands)) {
701 $commands = preg_grep('{^'.$expr.'}i', $allCommands);
702 }
703
704 // if no commands matched or we just matched namespaces
705 if (empty($commands) || \count(preg_grep('{^'.$expr.'$}i', $commands)) < 1) {
706 if (false !== $pos = strrpos($name, ':')) {
707 // check if a namespace exists and contains commands
708 $this->findNamespace(substr($name, 0, $pos));
709 }
710
711 $message = sprintf('Command "%s" is not defined.', $name);
712
713 if ($alternatives = $this->findAlternatives($name, $allCommands)) {
714 // remove hidden commands
715 $alternatives = array_filter($alternatives, fn ($name) => !$this->get($name)->isHidden());
716
717 if (1 == \count($alternatives)) {
718 $message .= "\n\nDid you mean this?\n ";
719 } else {
720 $message .= "\n\nDid you mean one of these?\n ";
721 }
722 $message .= implode("\n ", $alternatives);
723 }
724
725 throw new CommandNotFoundException($message, array_values($alternatives));
726 }
727
728 // filter out aliases for commands which are already on the list
729 if (\count($commands) > 1) {
730 $commandList = $this->commandLoader ? array_merge(array_flip($this->commandLoader->getNames()), $this->commands) : $this->commands;
731 $commands = array_unique(array_filter($commands, function ($nameOrAlias) use (&$commandList, $commands, &$aliases) {
732 if (!$commandList[$nameOrAlias] instanceof Command) {
733 $commandList[$nameOrAlias] = $this->commandLoader->get($nameOrAlias);
734 }
735
736 $commandName = $commandList[$nameOrAlias]->getName();
737
738 $aliases[$nameOrAlias] = $commandName;
739
740 return $commandName === $nameOrAlias || !\in_array($commandName, $commands);
741 }));
742 }
743
744 if (\count($commands) > 1) {
745 $usableWidth = $this->terminal->getWidth() - 10;
746 $abbrevs = array_values($commands);
747 $maxLen = 0;
748 foreach ($abbrevs as $abbrev) {
749 $maxLen = max(Helper::width($abbrev), $maxLen);
750 }
751 $abbrevs = array_map(function ($cmd) use ($commandList, $usableWidth, $maxLen, &$commands) {
752 if ($commandList[$cmd]->isHidden()) {
753 unset($commands[array_search($cmd, $commands)]);
754
755 return false;
756 }
757
758 $abbrev = str_pad($cmd, $maxLen, ' ').' '.$commandList[$cmd]->getDescription();
759
760 return Helper::width($abbrev) > $usableWidth ? Helper::substr($abbrev, 0, $usableWidth - 3).'...' : $abbrev;
761 }, array_values($commands));
762
763 if (\count($commands) > 1) {
764 $suggestions = $this->getAbbreviationSuggestions(array_filter($abbrevs));
765
766 throw new CommandNotFoundException(sprintf("Command \"%s\" is ambiguous.\nDid you mean one of these?\n%s.", $name, $suggestions), array_values($commands));
767 }
768 }
769
770 $command = $this->get(reset($commands));
771
772 if ($command->isHidden()) {
773 throw new CommandNotFoundException(sprintf('The command "%s" does not exist.', $name));
774 }
775
776 return $command;
777 }

853 protected function doRenderThrowable(\Throwable $e, OutputInterface $output): void
854 {
855 do {
856 $message = trim($e->getMessage());
857 if ('' === $message || OutputInterface::VERBOSITY_VERBOSE <= $output->getVerbosity()) {
858 $class = get_debug_type($e);
859 $title = sprintf(' [%s%s] ', $class, 0 !== ($code = $e->getCode()) ? ' ('.$code.')' : '');
860 $len = Helper::width($title);
861 } else {
862 $len = 0;
863 }
864
865 if (str_contains($message, "@anonymous\0")) {
866 $message = preg_replace_callback('/[a-zA-Z_\x7f-\xff][\\\\a-zA-Z0-9_\x7f-\xff]*+@anonymous\x00.*?\.php(?:0x?|:[0-9]++\$)[0-9a-fA-F]++/', fn ($m) => class_exists($m[0], false) ? (get_parent_class($m[0]) ?: key(class_implements($m[0])) ?: 'class').'@anonymous' : $m[0], $message);
867 }
868
869 $width = $this->terminal->getWidth() ? $this->terminal->getWidth() - 1 : \PHP_INT_MAX;
870 $lines = [];
871 foreach ('' !== $message ? preg_split('/\r?\n/', $message) : [] as $line) {
872 foreach ($this->splitStringByWidth($line, $width - 4) as $line) {
873 // pre-format lines to get the right string length
874 $lineLength = Helper::width($line) + 4;
875 $lines[] = [$line, $lineLength];
876
877 $len = max($lineLength, $len);
878 }
879 }
880
881 $messages = [];
882 if (!$e instanceof ExceptionInterface || OutputInterface::VERBOSITY_VERBOSE <= $output->getVerbosity()) {
883 $messages[] = sprintf('<comment>%s</comment>', OutputFormatter::escape(sprintf('In %s line %s:', basename($e->getFile()) ?: 'n/a', $e->getLine() ?: 'n/a')));
884 }
885 $messages[] = $emptyLine = sprintf('<error>%s</error>', str_repeat(' ', $len));
886 if ('' === $message || OutputInterface::VERBOSITY_VERBOSE <= $output->getVerbosity()) {
887 $messages[] = sprintf('<error>%s%s</error>', $title, str_repeat(' ', max(0, $len - Helper::width($title))));
888 }
889 foreach ($lines as $line) {
890 $messages[] = sprintf('<error> %s %s</error>', OutputFormatter::escape($line[0]), str_repeat(' ', $len - $line[1]));
891 }
892 $messages[] = $emptyLine;
893 $messages[] = '';
894
895 $output->writeln($messages, OutputInterface::VERBOSITY_QUIET);

995 protected function doRunCommand(Command $command, InputInterface $input, OutputInterface $output)
996 {
997 foreach ($command->getHelperSet() as $helper) {
998 if ($helper instanceof InputAwareInterface) {
999 $helper->setInput($input);
1000 }
1001 }
1002
1003 $commandSignals = $command instanceof SignalableCommandInterface ? $command->getSubscribedSignals() : [];
1004 if ($commandSignals || $this->dispatcher && $this->signalsToDispatchEvent) {
1005 if (!$this->signalRegistry) {
1006 throw new RuntimeException('Unable to subscribe to signal events. Make sure that the "pcntl" extension is installed and that "pcntl_*" functions are not disabled by your php.ini\'s "disable_functions" directive.');
1007 }
1008
1009 if (Terminal::hasSttyAvailable()) {
1010 $sttyMode = shell_exec('stty -g');
1011
1012 foreach ([\SIGINT, \SIGTERM] as $signal) {
1013 $this->signalRegistry->register($signal, static fn () => shell_exec('stty '.$sttyMode));
1014 }
1015 }
1016
1017 if ($this->dispatcher) {
1018 // We register application signals, so that we can dispatch the event
1019 foreach ($this->signalsToDispatchEvent as $signal) {
1020 $event = new ConsoleSignalEvent($command, $input, $output, $signal);
1021
1022 $this->signalRegistry->register($signal, function ($signal) use ($event, $command, $commandSignals) {
1023 $this->dispatcher->dispatch($event, ConsoleEvents::SIGNAL);
1024 $exitCode = $event->getExitCode();
1025
1026 // If the command is signalable, we call the handleSignal() method
1027 if (\in_array($signal, $commandSignals, true)) {
1028 $exitCode = $command->handleSignal($signal, $exitCode);
1029 // BC layer for Symfony <= 5
1030 if (null === $exitCode) {
1031 trigger_deprecation('symfony/console', '6.3', 'Not returning an exit code from "%s::handleSignal()" is deprecated, return "false" to keep the command running or "0" to exit successfully.', get_debug_type($command));
1032 $exitCode = 0;
1033 }
1034 }
1035
1036 if (false !== $exitCode) {
1037 exit($exitCode);
1038 }
1039 });
1040 }
1041
1042 // then we register command signals, but not if already handled after the dispatcher
1043 $commandSignals = array_diff($commandSignals, $this->signalsToDispatchEvent);
1044 }
1045
1046 foreach ($commandSignals as $signal) {
1047 $this->signalRegistry->register($signal, function (int $signal) use ($command): void {
1048 $exitCode = $command->handleSignal($signal);
1049 // BC layer for Symfony <= 5
1050 if (null === $exitCode) {
1051 trigger_deprecation('symfony/console', '6.3', 'Not returning an exit code from "%s::handleSignal()" is deprecated, return "false" to keep the command running or "0" to exit successfully.', get_debug_type($command));
1052 $exitCode = 0;
1053 }
1054
1055 if (false !== $exitCode) {
1056 exit($exitCode);
1057 }
1058 });
1059 }
1060 }
1061
1062 if (null === $this->dispatcher) {
1063 return $command->run($input, $output);
1064 }
1065
1066 // bind before the console.command event, so the listeners have access to input options/arguments
1067 try {
1068 $command->mergeApplicationDefinition();
1069 $input->bind($command->getDefinition());
1070 } catch (ExceptionInterface) {
1071 // ignore invalid options/arguments for now, to allow the event listeners to customize the InputDefinition
1072 }
1073
1074 $event = new ConsoleCommandEvent($command, $input, $output);
1075 $e = null;
1076
1077 try {
1078 $this->dispatcher->dispatch($event, ConsoleEvents::COMMAND);
1079
1080 if ($event->commandShouldRun()) {
1081 $exitCode = $command->run($input, $output);
1082 } else {
1083 $exitCode = ConsoleCommandEvent::RETURN_CODE_DISABLED;
1084 }
1085 } catch (\Throwable $e) {
1086 $event = new ConsoleErrorEvent($input, $output, $e, $command);
1087 $this->dispatcher->dispatch($event, ConsoleEvents::ERROR);
1088 $e = $event->getError();
1089
1090 if (0 === $exitCode = $event->getExitCode()) {
1091 $e = null;
1092 }
1093 }
1094
1095 $event = new ConsoleTerminateEvent($command, $input, $output, $exitCode);
1096 $this->dispatcher->dispatch($event, ConsoleEvents::TERMINATE);
1097
1098 if (null !== $e) {
1099 throw $e;
1100 }
1101
1102 return $event->getExitCode();
1103 }

1178 private function findAlternatives(string $name, iterable $collection): array
1179 {
1180 $threshold = 1e3;
1181 $alternatives = [];
1182
1183 $collectionParts = [];
1184 foreach ($collection as $item) {
1185 $collectionParts[$item] = explode(':', $item);
1186 }
1187
1188 foreach (explode(':', $name) as $i => $subname) {
1189 foreach ($collectionParts as $collectionName => $parts) {
1190 $exists = isset($alternatives[$collectionName]);
1191 if (!isset($parts[$i]) && $exists) {
1192 $alternatives[$collectionName] += $threshold;
1193 continue;
1194 } elseif (!isset($parts[$i])) {
1195 continue;
1196 }
1197
1198 $lev = levenshtein($subname, $parts[$i]);
1199 if ($lev <= \strlen($subname) / 3 || '' !== $subname && str_contains($parts[$i], $subname)) {
1200 $alternatives[$collectionName] = $exists ? $alternatives[$collectionName] + $lev : $lev;
1201 } elseif ($exists) {
1202 $alternatives[$collectionName] += $threshold;
1203 }
1204 }
1205 }
1206
1207 foreach ($collection as $item) {
1208 $lev = levenshtein($name, $item);
1209 if ($lev <= \strlen($name) / 3 || str_contains($item, $name)) {
1210 $alternatives[$item] = isset($alternatives[$item]) ? $alternatives[$item] - $lev : $lev;
1211 }
1212 }
1213
1214 $alternatives = array_filter($alternatives, fn ($lev) => $lev < 2 * $threshold);
1215 ksort($alternatives, \SORT_NATURAL | \SORT_FLAG_CASE);
1216
1217 return array_keys($alternatives);
1218 }

1247 private function splitStringByWidth(string $string, int $width): array
1248 {
1249 // str_split is not suitable for multi-byte characters, we should use preg_split to get char array properly.
1250 // additionally, array_slice() is not enough as some character has doubled width.
1251 // we need a function to split string not by character count but by string width
1252 if (false === $encoding = mb_detect_encoding($string, null, true)) {
1253 return str_split($string, $width);
1254 }
1255
1256 $utf8String = mb_convert_encoding($string, 'utf8', $encoding);
1257 $lines = [];
1258 $line = '';
1259
1260 $offset = 0;
1261 while (preg_match('/.{1,10000}/u', $utf8String, $m, 0, $offset)) {
1262 $offset += \strlen($m[0]);
1263
1264 foreach (preg_split('//u', $m[0]) as $char) {
1265 // test if $char could be appended to current line
1266 if (mb_strwidth($line.$char, 'utf8') <= $width) {
1267 $line .= $char;
1268 continue;
1269 }
1270 // if not, push current line to array and make new line
1271 $lines[] = str_pad($line, $width);
1272 $line = $char;
1273 }
1274 }
1275
1276 $lines[] = \count($lines) ? str_pad($line, $width) : $line;
1277
1278 mb_convert_variables($encoding, 'utf8', $lines);
1279
1280 return $lines;
1281 }
