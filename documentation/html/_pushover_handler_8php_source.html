145 private function buildContent(LogRecord $record): string
146 {
147 // Pushover has a limit of 512 characters on title and message combined.
148 $maxMessageLength = 512 - strlen($this->title);
149
150 $message = ($this->useFormattedMessage) ? $record->formatted : $record->message;
151 $message = Utils::substr($message, 0, $maxMessageLength);
152
153 $timestamp = $record->datetime->getTimestamp();
154
155 $dataArray = [
156 'token' => $this->token,
157 'user' => $this->user,
158 'message' => $message,
159 'title' => $this->title,
160 'timestamp' => $timestamp,
161 ];
162
163 if ($record->level->value >= $this->emergencyLevel->value) {
164 $dataArray['priority'] = 2;
165 $dataArray['retry'] = $this->retry;
166 $dataArray['expire'] = $this->expire;
167 } elseif ($record->level->value >= $this->highPriorityLevel->value) {
168 $dataArray['priority'] = 1;
169 }
170
171 // First determine the available parameters
172 $context = array_intersect_key($record->context, $this->parameterNames);
173 $extra = array_intersect_key($record->extra, $this->parameterNames);
174
175 // Least important info should be merged with subsequent info
176 $dataArray = array_merge($extra, $context, $dataArray);
177
178 // Only pass sounds that are supported by the API
179 if (isset($dataArray['sound']) && !in_array($dataArray['sound'], $this->sounds, true)) {
180 unset($dataArray['sound']);
181 }
182
183 return http_build_query($dataArray);
184 }
