123 public static function tcNoChecksum($identityPrefix)
124 {
125 if (strlen((string) $identityPrefix) !== 9) {
126 throw new \InvalidArgumentException('Argument should be an integer and should be 9 digits.');
127 }
128
129 $oddSum = 0;
130 $evenSum = 0;
131
132 $identityArray = array_map('intval', str_split($identityPrefix)); // Creates array from int
133
134 foreach ($identityArray as $index => $digit) {
135 if ($index % 2 === 0) {
136 $evenSum += $digit;
137 } else {
138 $oddSum += $digit;
139 }
140 }
141
142 $tenthDigit = (7 * $evenSum - $oddSum) % 10;
