71 public function leaveNode(Node $node)
72 {
73 // Store a reference to every "use" statement, because we'll need them in a bit.
74 if ($node instanceof Use_) {
75 foreach ($node->uses as $use) {
76 $alias = $use->alias ?: \end($use->name->parts);
77 $this->aliases[\strtolower($alias)] = $use->name;
78 }
79
80 return NodeTraverser::REMOVE_NODE;
81 }
82
83 // Expand every "use" statement in the group into a full, standalone "use" and store 'em with the others.
84 if ($node instanceof GroupUse) {
85 foreach ($node->uses as $use) {
86 $alias = $use->alias ?: \end($use->name->parts);
87 $this->aliases[\strtolower($alias)] = Name::concat($node->prefix, $use->name, [
88 'startLine' => $node->prefix->getAttribute('startLine'),
89 'endLine' => $use->name->getAttribute('endLine'),
90 ]);
91 }
92
93 return NodeTraverser::REMOVE_NODE;
94 }
95
96 // Start fresh, since we're done with this namespace.
97 if ($node instanceof Namespace_) {
98 $this->lastNamespace = $node->name;
99 $this->lastAliases = $this->aliases;
100 $this->aliases = [];
101
102 return;
103 }
104
105 // Do nothing with UseUse; this an entry in the list of uses in the use statement.
106 if ($node instanceof UseUse) {
107 return;
108 }
109
110 // For everything else, we'll implicitly thunk all aliases into fully-qualified names.
111 foreach ($node as $name => $subNode) {
112 if ($subNode instanceof Name) {
113 if ($replacement = $this->findAlias($subNode)) {
114 $node->$name = $replacement;
115 }
116 }
117 }
118
119 return $node;
120 }

129 private function findAlias(Name $name)
130 {
131 $that = \strtolower($name);
132 foreach ($this->aliases as $alias => $prefix) {
133 if ($that === $alias) {
134 return new FullyQualifiedName($prefix->toString());
135 } elseif (\substr($that, 0, \strlen($alias) + 1) === $alias.'\\') {
136 return new FullyQualifiedName($prefix->toString().\substr($name, \strlen($alias)));
137 }
138 }
