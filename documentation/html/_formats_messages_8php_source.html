31 // custom message, otherwise we'll keep searching for a valid message.
32 if (! is_null($inlineMessage)) {
33 return $inlineMessage;
34 }

54 // specific error message for the type of attribute being validated such
55 // as a number, file or string which all have different message types.

99 {

102 $keys = ["{$attribute}.{$lowerRule}", $lowerRule, $attribute];

105 // message for the fields, then we will check for a general custom line
106 // that is not attribute specific. If we find either we'll return it.

144 protected function getCustomMessageFromTranslator($keys)
145 {
146 foreach (Arr::wrap($keys) as $key) {
147 if (($message = $this->translator->get($key)) !== $key) {
148 return $message;
149 }
150
151 // If an exact match was not found for the key, we will collapse all of these
152 // messages and loop through them and try to find a wildcard match for the
153 // given key. Otherwise, we will simply return the key's value back out.
154 $shortKey = preg_replace(
155 '/^validation\.custom\./', '', $key
156 );
157
158 $message = $this->getWildcardCustomMessages(Arr::dot(
159 (array) $this->translator->get('validation.custom')
160 ), $shortKey, $key);
161
162 if ($message !== $key) {
163 return $message;
164 }
165 }
166
167 return Arr::last(Arr::wrap($keys));

266 public function getDisplayableAttribute($attribute)
267 {
268 $primaryAttribute = $this->getPrimaryAttribute($attribute);
269
270 $expectedAttributes = $attribute != $primaryAttribute
271 ? [$attribute, $primaryAttribute] : [$attribute];
272
273 foreach ($expectedAttributes as $name) {
274 // The developer may dynamically specify the array of custom attributes on this
275 // validator instance. If the attribute exists in this array it is used over
276 // the other ways of pulling the attribute name for this given attributes.
277 if (isset($this->customAttributes[$name])) {

280

282 // application, which allows flexibility for displaying a unique displayable
283 // version of the attribute name instead of the name used in an HTTP POST.
284 if ($line = $this->getAttributeFromTranslations($name)) {

293 return ($formatter = $this->implicitAttributesFormatter)
