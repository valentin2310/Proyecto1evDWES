180 protected function breadcrumbs(AbstractNode $node): string
181 {
182 $breadcrumbs = '';
183 $path = $node->pathAsArray();
184 $pathToRoot = [];
185 $max = count($path);
186
187 if ($node instanceof FileNode) {
188 $max--;
189 }
190
191 for ($i = 0; $i < $max; $i++) {
192 $pathToRoot[] = str_repeat('../', $i);
193 }
194
195 foreach ($path as $step) {
196 if ($step !== $node) {
197 $breadcrumbs .= $this->inactiveBreadcrumb(
198 $step,
199 array_pop($pathToRoot)
200 );
201 } else {
202 $breadcrumbs .= $this->activeBreadcrumb($step);
203 }
204 }
205
206 return $breadcrumbs;
207 }

261 protected function colorLevel(float $percent): string
