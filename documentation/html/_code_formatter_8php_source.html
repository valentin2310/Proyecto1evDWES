149 private static function tokenizeSpans(string $code): \Generator
150 {
151 $spanType = null;
152 $buffer = '';
153
154 foreach (\token_get_all($code) as $token) {
155 $nextType = self::nextHighlightType($token, $spanType);
156 $spanType = $spanType ?: $nextType;
157
158 if ($spanType !== $nextType) {
159 yield [$spanType, $buffer];
160 $spanType = $nextType;
161 $buffer = '';
162 }
163
164 $buffer .= \is_array($token) ? $token[1] : $token;
165 }
166
167 if ($spanType !== null && $buffer !== '') {
168 yield [$spanType, $buffer];
169 }
170 }

180 private static function nextHighlightType($token, $currentType)
181 {
182 if ($token === '"') {
183 return self::HIGHLIGHT_STRING;
184 }
185
186 if (\is_array($token)) {
187 if ($token[0] === \T_WHITESPACE) {
188 return $currentType;
189 }
190
191 if (\array_key_exists($token[0], self::$tokenMap)) {
192 return self::$tokenMap[$token[0]];
193 }
194 }
195
196 return self::HIGHLIGHT_KEYWORD;
197 }

210 private static function splitLines(\Generator $spans, int $startLine = 1, int $endLine = null): \Generator
211 {
212 $lineNum = 1;
213 $buffer = [];
214
215 foreach ($spans as list($spanType, $spanText)) {
216 foreach (\preg_split('/(\r\n?|\n)/', $spanText) as $index => $spanLine) {
217 if ($index > 0) {
218 if ($lineNum >= $startLine) {
219 yield $lineNum => $buffer;
220 }
221
222 $lineNum++;
223 $buffer = [];
224
225 if ($endLine !== null && $lineNum > $endLine) {
226 return;
227 }
228 }
229
230 if ($spanLine !== '') {
231 $buffer[] = [$spanType, $spanLine];
232 }
233 }
234 }
235
236 if (!empty($buffer)) {
237 yield $lineNum => $buffer;
238 }
239 }

248 private static function formatLines(\Generator $spanLines): \Generator
249 {
250 foreach ($spanLines as $lineNum => $spanLine) {
251 $line = '';
252
253 foreach ($spanLine as list($spanType, $spanText)) {
254 if ($spanType === self::HIGHLIGHT_DEFAULT) {
255 $line .= OutputFormatter::escape($spanText);
256 } else {
257 $line .= \sprintf('<%s>%s</%s>', $spanType, OutputFormatter::escape($spanText), $spanType);
258 }
259 }
260
261 yield $lineNum => $line.\PHP_EOL;
262 }

277 private static function numberLines(\Generator $lines, int $markLine = null): \Generator
278 {
279 $lines = \iterator_to_array($lines);
280
281 // Figure out how much space to reserve for line numbers.
282 \end($lines);
283 $pad = \strlen(\key($lines));
284
285 // If $markLine is before or after our line range, don't bother reserving space for the marker.
286 if ($markLine !== null) {
287 if ($markLine > \key($lines)) {
288 $markLine = null;
289 }
290
291 \reset($lines);
292 if ($markLine < \key($lines)) {
293 $markLine = null;
294 }
295 }
296
297 foreach ($lines as $lineNum => $line) {
298 $mark = '';
299 if ($markLine !== null) {
300 $mark = ($markLine === $lineNum) ? self::LINE_MARKER : self::NO_LINE_MARKER;
301 }
302
303 yield \sprintf("%s<aside>%{$pad}s</aside>: %s", $mark, $lineNum, $line);
304 }
