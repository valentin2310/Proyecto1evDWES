264 $code = @highlight_file($file, true);
265 if (\PHP_VERSION_ID >= 80300) {
266 // remove main pre/code tags
267 $code = preg_replace('#^<pre.*?>\s*<code.*?>(.*)</code>\s*</pre>#s', '\\1', $code);

269 $code = preg_replace_callback('#<code ([^>]++)>((?:[^<]*+\\n)++[^<]*+)</code>#', fn ($m) => "<code $m[1]>".str_replace("\n", "</code>\n<code $m[1]>", $m[2]).'</code>', $code);
270 // Convert spaces to html entities to preserve indentation when rendered
271 $code = str_replace(' ', '&nbsp;', $code);
272 $content = explode("\n", $code);
273 } else {
274 // remove main code/span tags
275 $code = preg_replace('#^<code.*?>\s*<span.*?>(.*)</span>\s*</code>#s', '\\1', $code);
276 // split multiline spans
277 $code = preg_replace_callback('#<span ([^>]++)>((?:[^<]*+<br \/>)++[^<]*+)</span>#', fn ($m) => "<span $m[1]>".str_replace('<br />', "</span><br /><span $m[1]>", $m[2]).'</span>', $code);
278 $content = explode('<br />', $code);
279 }
280
281 $lines = [];
282 if (0 > $srcContext) {
283 $srcContext = \count($content);
284 }
285
286 for ($i = max($line - $srcContext, 1), $max = min($line + $srcContext, \count($content)); $i <= $max; ++$i) {
287 $lines[] = '<li'.($i == $line ? ' class="selected"' : '').'><code>'.$this->fixCodeMarkup($content[$i - 1]).'</code></li>';
288 }
289
290 return '<ol start="'.max($line - $srcContext, 1).'">'.implode("\n", $lines).'</ol>';
291 }
292
293 return '';
294 }

296 private function fixCodeMarkup(string $line): string
297 {
298 // </span> ending tag from previous line
299 $opening = strpos($line, '<span');
300 $closing = strpos($line, '</span>');
301 if (false !== $closing && (false === $opening || $closing < $opening)) {
302 $line = substr_replace($line, '', $closing, 7);
303 }
304
305 // missing </span> tag at the end of line
306 $opening = strrpos($line, '<span');
307 $closing = strrpos($line, '</span>');
308 if (false !== $opening && (false === $closing || $closing < $opening)) {
309 $line .= '</span>';
310 }
311
312 return trim($line);
313 }
