95 public function reorderTestsInSuite(Test $suite, int $order, bool $resolveDependencies, int $orderDefects, bool $isRootTestSuite = true): void
96 {
97 $allowedOrders = [
98 self::ORDER_DEFAULT,
99 self::ORDER_REVERSED,
100 self::ORDER_RANDOMIZED,
101 self::ORDER_DURATION,
102 self::ORDER_SIZE,
103 ];
104
105 if (!in_array($order, $allowedOrders, true)) {
106 throw new InvalidOrderException;
107 }
108
109 $allowedOrderDefects = [
110 self::ORDER_DEFAULT,
111 self::ORDER_DEFECTS_FIRST,
112 ];
113
114 if (!in_array($orderDefects, $allowedOrderDefects, true)) {
115 throw new InvalidOrderException;
116 }
117
118 if ($isRootTestSuite) {
119 $this->originalExecutionOrder = $this->calculateTestExecutionOrder($suite);
120 }
121
122 if ($suite instanceof TestSuite) {
123 foreach ($suite as $_suite) {
124 $this->reorderTestsInSuite($_suite, $order, $resolveDependencies, $orderDefects, false);
125 }
126

128 $this->addSuiteToDefectSortOrder($suite);
129 }
130
131 $this->sort($suite, $order, $resolveDependencies, $orderDefects);
132 }
133
134 if ($isRootTestSuite) {

136 }

149 private function sort(TestSuite $suite, int $order, bool $resolveDependencies, int $orderDefects): void
150 {
151 if (empty($suite->tests())) {
152 return;
153 }
154
155 if ($order === self::ORDER_REVERSED) {
156 $suite->setTests($this->reverse($suite->tests()));
157 } elseif ($order === self::ORDER_RANDOMIZED) {
158 $suite->setTests($this->randomize($suite->tests()));
159 } elseif ($order === self::ORDER_DURATION) {
160 $suite->setTests($this->sortByDuration($suite->tests()));
161 } elseif ($order === self::ORDER_SIZE) {
162 $suite->setTests($this->sortBySize($suite->tests()));
163 }
164
165 if ($orderDefects === self::ORDER_DEFECTS_FIRST) {
166 $suite->setTests($this->sortDefectsFirst($suite->tests()));
167 }
168
169 if ($resolveDependencies && !($suite instanceof DataProviderTestSuite)) {
170 $tests = $suite->tests();
171
172 $suite->setTests($this->resolveDependencies($tests));
173 }
174 }

176 private function addSuiteToDefectSortOrder(TestSuite $suite): void
177 {
178 $max = 0;
179
180 foreach ($suite->tests() as $test) {
181 if (!$test instanceof Reorderable) {
182 continue;
183 }
184
185 if (!isset($this->defectSortOrder[$test->sortId()])) {
186 $this->defectSortOrder[$test->sortId()] = $this->cache->status($test->sortId())->asInt();
187 $max = max($max, $this->defectSortOrder[$test->sortId()]);
188 }
189 }
190
191 $this->defectSortOrder[$suite->sortId()] = $max;

243 private function cmpDefectPriorityAndTime(Test $a, Test $b): int
244 {
245 if (!($a instanceof Reorderable && $b instanceof Reorderable)) {
246 return 0;
247 }
248
249 $priorityA = $this->defectSortOrder[$a->sortId()] ?? 0;
250 $priorityB = $this->defectSortOrder[$b->sortId()] ?? 0;
251
252 if ($priorityB <=> $priorityA) {
253 // Sort defect weight descending
254 return $priorityB <=> $priorityA;
255 }
256
257 if ($priorityA || $priorityB) {
258 return $this->cmpDuration($a, $b);
259 }
260
261 // do not change execution order
262 return 0;
263 }

268 private function cmpDuration(Test $a, Test $b): int
269 {
270 if (!($a instanceof Reorderable && $b instanceof Reorderable)) {
271 return 0;
272 }
273

307 private function resolveDependencies(array $tests): array
308 {
309 $newTestOrder = [];
310 $i = 0;
311 $provided = [];
312
313 do {
314 if ([] === array_diff($tests[$i]->requires(), $provided)) {
315 $provided = array_merge($provided, $tests[$i]->provides());
316 $newTestOrder = array_merge($newTestOrder, array_splice($tests, $i, 1));
317 $i = 0;
318 } else {
319 $i++;
320 }
321 } while (!empty($tests) && ($i < count($tests)));
322
323 return array_merge($newTestOrder, $tests);
324 }
