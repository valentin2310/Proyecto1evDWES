132 throw new \RuntimeException(sprintf('Failed to start the session because headers have already been sent by "%s" at line %d.', $file, $line));
133 }

135 $sessionId = $_COOKIE[session_name()] ?? null;
136 /*
137 * Explanation of the session ID regular expression: `/^[a-zA-Z0-9,-]{22,250}$/`.
138 *
139 * ---------- Part 1
140 *
141 * The part `[a-zA-Z0-9,-]` is related to the PHP ini directive `session.sid_bits_per_character` defined as 6.
142 * See https://www.php.net/manual/en/session.configuration.php#ini.session.sid-bits-per-character.
143 * Allowed values are integers such as:
144 * - 4 for range `a-f0-9`
145 * - 5 for range `a-v0-9`
146 * - 6 for range `a-zA-Z0-9,-`
147 *
148 * ---------- Part 2
149 *
150 * The part `{22,250}` is related to the PHP ini directive `session.sid_length`.
151 * See https://www.php.net/manual/en/session.configuration.php#ini.session.sid-length.
152 * Allowed values are integers between 22 and 256, but we use 250 for the max.
153 *
154 * Where does the 250 come from?
155 * - The length of Windows and Linux filenames is limited to 255 bytes. Then the max must not exceed 255.
156 * - The session filename prefix is `sess_`, a 5 bytes string. Then the max must not exceed 255 - 5 = 250.
157 *
158 * ---------- Conclusion
159 *
160 * The parts 1 and 2 prevent the warning below:
161 * `PHP Warning: SessionHandler::read(): Session ID is too long or contains illegal characters. Only the A-Z, a-z, 0-9, "-", and "," characters are allowed.`
162 *
163 * The part 2 prevents the warning below:
164 * `PHP Warning: SessionHandler::read(): open(filepath, O_RDWR) failed: No such file or directory (2).`
165 */
166 if ($sessionId && $this->saveHandler instanceof AbstractProxy && 'files' === $this->saveHandler->getSaveHandlerName() && !preg_match('/^[a-zA-Z0-9,-]{22,250}$/', $sessionId)) {
167 // the session ID in the header is invalid, create a new one
168 session_id(session_create_id());
169 }
170
171 // ok to try and start the session
172 if (!session_start()) {
173 throw new \RuntimeException('Failed to start the session.');
174 }
175
176 $this->loadSession();
177
178 return true;
179 }

234 public function save()
235 {
236 // Store a copy so we can restore the bags in case the session was not left empty
237 $session = $_SESSION;
238
239 foreach ($this->bags as $bag) {
240 if (empty($_SESSION[$key = $bag->getStorageKey()])) {
241 unset($_SESSION[$key]);
242 }
243 }
244 if ($_SESSION && [$key = $this->metadataBag->getStorageKey()] === array_keys($_SESSION)) {
245 unset($_SESSION[$key]);
246 }
247
248 // Register error handler to add information about the current save handler
249 $previousHandler = set_error_handler(function ($type, $msg, $file, $line) use (&$previousHandler) {
250 if (\E_WARNING === $type && str_starts_with($msg, 'session_write_close():')) {
251 $handler = $this->saveHandler instanceof SessionHandlerProxy ? $this->saveHandler->getHandler() : $this->saveHandler;
252 $msg = sprintf('session_write_close(): Failed to write session data with "%s" handler', $handler::class);
253 }
254
255 return $previousHandler ? $previousHandler($type, $msg, $file, $line) : false;
256 });
257
258 try {
259 session_write_close();
260 } finally {
261 restore_error_handler();
262
263 // Restore only if not empty
264 if ($_SESSION) {
265 $_SESSION = $session;
266 }
267 }
268
269 $this->closed = true;
270 $this->started = false;
271 }
