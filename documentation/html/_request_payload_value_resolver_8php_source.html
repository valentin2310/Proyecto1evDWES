84 public function onKernelControllerArguments(ControllerArgumentsEvent $event): void
85 {
86 $arguments = $event->getArguments();
87
88 foreach ($arguments as $i => $argument) {
89 if ($argument instanceof MapQueryString) {
90 $payloadMapper = 'mapQueryString';
91 $validationFailedCode = Response::HTTP_NOT_FOUND;
92 } elseif ($argument instanceof MapRequestPayload) {
93 $payloadMapper = 'mapRequestPayload';
94 $validationFailedCode = Response::HTTP_UNPROCESSABLE_ENTITY;
95 } else {
96 continue;
97 }
98 $request = $event->getRequest();
99
100 if (!$type = $argument->metadata->getType()) {
101 throw new \LogicException(sprintf('Could not resolve the "$%s" controller argument: argument should be typed.', $argument->metadata->getName()));
102 }
103
104 if ($this->validator) {
105 $violations = new ConstraintViolationList();
106 try {
107 $payload = $this->$payloadMapper($request, $type, $argument);
108 } catch (PartialDenormalizationException $e) {
109 $trans = $this->translator ? $this->translator->trans(...) : fn ($m, $p) => strtr($m, $p);
110 foreach ($e->getErrors() as $error) {
111 $parameters = ['{{ type }}' => implode('|', $error->getExpectedTypes())];
112 if ($error->canUseMessageForUser()) {
113 $parameters['hint'] = $error->getMessage();
114 }
115 $template = 'This value should be of type {{ type }}.';
116 $message = $trans($template, $parameters, 'validators');
117 $violations->add(new ConstraintViolation($message, $template, $parameters, null, $error->getPath(), null));
118 }
119 $payload = $e->getData();
120 }
121
122 if (null !== $payload) {
123 $violations->addAll($this->validator->validate($payload, null, $argument->validationGroups ?? null));
124 }
125
126 if (\count($violations)) {
127 throw new HttpException($validationFailedCode, implode("\n", array_map(static fn ($e) => $e->getMessage(), iterator_to_array($violations))), new ValidationFailedException($payload, $violations));
128 }
129 } else {
130 try {
131 $payload = $this->$payloadMapper($request, $type, $argument);
132 } catch (PartialDenormalizationException $e) {
133 throw new HttpException($validationFailedCode, implode("\n", array_map(static fn ($e) => $e->getMessage(), $e->getErrors())), $e);
134 }
135 }
136
137 if (null === $payload) {
138 $payload = match (true) {
139 $argument->metadata->hasDefaultValue() => $argument->metadata->getDefaultValue(),
140 $argument->metadata->isNullable() => null,
141 default => throw new HttpException($validationFailedCode)
142 };
143 }
144
145 $arguments[$i] = $payload;
146 }
147
148 $event->setArguments($arguments);
149 }

167 private function mapRequestPayload(Request $request, string $type, MapRequestPayload $attribute): ?object
168 {
169 if (null === $format = $request->getContentTypeFormat()) {
170 throw new HttpException(Response::HTTP_UNSUPPORTED_MEDIA_TYPE, 'Unsupported format.');
171 }
172
173 if ($attribute->acceptFormat && !\in_array($format, (array) $attribute->acceptFormat, true)) {
174 throw new HttpException(Response::HTTP_UNSUPPORTED_MEDIA_TYPE, sprintf('Unsupported format, expects "%s", but "%s" given.', implode('", "', (array) $attribute->acceptFormat), $format));
175 }
176
177 if ($data = $request->request->all()) {
178 return $this->serializer->denormalize($data, $type, null, $attribute->serializationContext + self::CONTEXT_DENORMALIZE);
179 }
180
181 if ('' === $data = $request->getContent()) {
182 return null;
183 }
184
185 if ('form' === $format) {
186 throw new HttpException(Response::HTTP_BAD_REQUEST, 'Request payload contains invalid "form" data.');
187 }
188
189 try {
190 return $this->serializer->deserialize($data, $type, $format, self::CONTEXT_DESERIALIZE + $attribute->serializationContext);
191 } catch (UnsupportedFormatException $e) {
192 throw new HttpException(Response::HTTP_UNSUPPORTED_MEDIA_TYPE, sprintf('Unsupported format: "%s".', $format), $e);
193 } catch (NotEncodableValueException $e) {
194 throw new HttpException(Response::HTTP_BAD_REQUEST, sprintf('Request payload contains invalid "%s" data.', $format), $e);
195 }
196 }
