121 public function dispatch(object $event, string $eventName = null): object
122 {
123 $eventName ??= $event::class;
124
125 $this->callStack ??= new \SplObjectStorage();
126
127 $currentRequestHash = $this->currentRequestHash = $this->requestStack && ($request = $this->requestStack->getCurrentRequest()) ? spl_object_hash($request) : '';
128
129 if (null !== $this->logger && $event instanceof StoppableEventInterface && $event->isPropagationStopped()) {
130 $this->logger->debug(sprintf('The "%s" event is already stopped. No listeners have been called.', $eventName));
131 }
132
133 $this->preProcess($eventName);
134 try {
135 $this->beforeDispatch($eventName, $event);
136 try {
137 $e = $this->stopwatch->start($eventName, 'section');
138 try {
139 $this->dispatcher->dispatch($event, $eventName);
140 } finally {
141 if ($e->isStarted()) {
142 $e->stop();
143 }
144 }
145 } finally {
146 $this->afterDispatch($eventName, $event);
147 }
148 } finally {
149 $this->currentRequestHash = $currentRequestHash;
150 $this->postProcess($eventName);
151 }
152
153 return $event;
154 }

174 public function getNotCalledListeners(Request $request = null): array
175 {
176 try {
177 $allListeners = $this->dispatcher instanceof EventDispatcher ? $this->getListenersWithPriority() : $this->getListenersWithoutPriority();
178 } catch (\Exception $e) {
179 $this->logger?->info('An exception was thrown while getting the uncalled listeners.', ['exception' => $e]);
180
181 // unable to retrieve the uncalled listeners
182 return [];
183 }
184
185 $hash = $request ? spl_object_hash($request) : null;
186 $calledListeners = [];
187
188 if (null !== $this->callStack) {
189 foreach ($this->callStack as $calledListener) {
190 [, $requestHash] = $this->callStack->getInfo();
191
192 if (null === $hash || $hash === $requestHash) {
193 $calledListeners[] = $calledListener->getWrappedListener();
194 }
195 }
196 }
197
198 $notCalled = [];
199
200 foreach ($allListeners as $eventName => $listeners) {
201 foreach ($listeners as [$listener, $priority]) {
202 if (!\in_array($listener, $calledListeners, true)) {
203 if (!$listener instanceof WrappedListener) {
204 $listener = new WrappedListener($listener, null, $this->stopwatch, $this, $priority);
205 }
206 $notCalled[] = $listener->getInfo($eventName);
207 }
208 }
209 }
210
211 uasort($notCalled, $this->sortNotCalledListeners(...));
212
213 return $notCalled;
214 }

268 private function preProcess(string $eventName): void
269 {
270 if (!$this->dispatcher->hasListeners($eventName)) {
271 $this->orphanedEvents[$this->currentRequestHash][] = $eventName;
272
273 return;
274 }
275
276 foreach ($this->dispatcher->getListeners($eventName) as $listener) {
277 $priority = $this->getListenerPriority($eventName, $listener);
278 $wrappedListener = new WrappedListener($listener instanceof WrappedListener ? $listener->getWrappedListener() : $listener, null, $this->stopwatch, $this);
279 $this->wrappedListeners[$eventName][] = $wrappedListener;
280 $this->dispatcher->removeListener($eventName, $listener);
281 $this->dispatcher->addListener($eventName, $wrappedListener, $priority);
282 $this->callStack->attach($wrappedListener, [$eventName, $this->currentRequestHash]);
283 }
284 }

286 private function postProcess(string $eventName): void
287 {
288 unset($this->wrappedListeners[$eventName]);
289 $skipped = false;
290 foreach ($this->dispatcher->getListeners($eventName) as $listener) {
291 if (!$listener instanceof WrappedListener) { // #12845: a new listener was added during dispatch.
292 continue;
293 }
294 // Unwrap listener
295 $priority = $this->getListenerPriority($eventName, $listener);
296 $this->dispatcher->removeListener($eventName, $listener);
297 $this->dispatcher->addListener($eventName, $listener->getWrappedListener(), $priority);
298
299 if (null !== $this->logger) {
300 $context = ['event' => $eventName, 'listener' => $listener->getPretty()];
301 }
302
303 if ($listener->wasCalled()) {
304 $this->logger?->debug('Notified event "{event}" to listener "{listener}".', $context);
305 } else {
306 $this->callStack->detach($listener);
307 }
308
309 if (null !== $this->logger && $skipped) {
310 $this->logger->debug('Listener "{listener}" was not called for event "{event}".', $context);
311 }
312
313 if ($listener->stoppedPropagation()) {
314 $this->logger?->debug('Listener "{listener}" stopped propagation of the event "{event}".', $context);
315
316 $skipped = true;
317 }
318 }

321 private function sortNotCalledListeners(array $a, array $b): int
322 {
323 if (0 !== $cmp = strcmp($a['event'], $b['event'])) {
324 return $cmp;
325 }
326
327 if (\is_int($a['priority']) && !\is_int($b['priority'])) {
328 return 1;
329 }
330
331 if (!\is_int($a['priority']) && \is_int($b['priority'])) {
332 return -1;
333 }
334
335 if ($a['priority'] === $b['priority']) {
336 return 0;
337 }
338
339 if ($a['priority'] > $b['priority']) {
340 return -1;
341 }
342
343 return 1;
344 }
