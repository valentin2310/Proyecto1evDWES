125 private function writeDiffHunks($output, array $diff): void
126 {
127 // detect "No newline at end of file" and insert into `$diff` if needed
128
129 $upperLimit = count($diff);
130
131 if (0 === $diff[$upperLimit - 1][1]) {
132 $lc = substr($diff[$upperLimit - 1][0], -1);
133
134 if ("\n" !== $lc) {
135 array_splice($diff, $upperLimit, 0, [["\n\\ No newline at end of file\n", Differ::NO_LINE_END_EOF_WARNING]]);
136 }
137 } else {
138 // search back for the last `+` and `-` line,
139 // check if it has a trailing linebreak, else add a warning under it
140 $toFind = [1 => true, 2 => true];
141
142 for ($i = $upperLimit - 1; $i >= 0; $i--) {
143 if (isset($toFind[$diff[$i][1]])) {
144 unset($toFind[$diff[$i][1]]);
145 $lc = substr($diff[$i][0], -1);
146
147 if ("\n" !== $lc) {
148 array_splice($diff, $i + 1, 0, [["\n\\ No newline at end of file\n", Differ::NO_LINE_END_EOF_WARNING]]);
149 }
150
151 if (!count($toFind)) {
152 break;
153 }
154 }
155 }
156 }
157
158 // write hunks to output buffer
159
160 $cutOff = max($this->commonLineThreshold, $this->contextLines);
161 $hunkCapture = false;
162 $sameCount = $toRange = $fromRange = 0;
163 $toStart = $fromStart = 1;
164 $i = 0;
165
167 foreach ($diff as $i => $entry) {
168 if (0 === $entry[1]) { // same
169 if (false === $hunkCapture) {
170 $fromStart++;
171 $toStart++;
172
173 continue;
174 }
175
176 $sameCount++;
177 $toRange++;
178 $fromRange++;
179
180 if ($sameCount === $cutOff) {
181 $contextStartOffset = ($hunkCapture - $this->contextLines) < 0
182 ? $hunkCapture
183 : $this->contextLines;
184
185 // note: $contextEndOffset = $this->contextLines;
186 //
187 // because we never go beyond the end of the diff.
188 // with the cutoff/contextlines here the follow is never true;
189 //
190 // if ($i - $cutOff + $this->contextLines + 1 > \count($diff)) {
191 // $contextEndOffset = count($diff) - 1;
192 // }
193 //
194 // ; that would be true for a trailing incomplete hunk case which is dealt with after this loop
195
196 $this->writeHunk(
197 $diff,
198 $hunkCapture - $contextStartOffset,
199 $i - $cutOff + $this->contextLines + 1,
200 $fromStart - $contextStartOffset,
201 $fromRange - $cutOff + $contextStartOffset + $this->contextLines,
202 $toStart - $contextStartOffset,
203 $toRange - $cutOff + $contextStartOffset + $this->contextLines,
204 $output
205 );
206
207 $fromStart += $fromRange;
208 $toStart += $toRange;
209
210 $hunkCapture = false;
211 $sameCount = $toRange = $fromRange = 0;
212 }
213
214 continue;
215 }
216
217 $sameCount = 0;
218
219 if ($entry[1] === Differ::NO_LINE_END_EOF_WARNING) {
220 continue;
221 }
222
223 $this->changed = true;
224
225 if (false === $hunkCapture) {
226 $hunkCapture = $i;
227 }
228
229 if (Differ::ADDED === $entry[1]) { // added
230 $toRange++;
231 }
232
233 if (Differ::REMOVED === $entry[1]) { // removed
234 $fromRange++;
235 }
236 }
237
238 if (false === $hunkCapture) {
239 return;
240 }
241
242 // we end here when cutoff (commonLineThreshold) was not reached, but we were capturing a hunk,
243 // do not render hunk till end automatically because the number of context lines might be less than the commonLineThreshold
244
245 $contextStartOffset = $hunkCapture - $this->contextLines < 0
246 ? $hunkCapture
247 : $this->contextLines;
248
249 // prevent trying to write out more common lines than there are in the diff _and_
250 // do not write more than configured through the context lines
251 $contextEndOffset = min($sameCount, $this->contextLines);
252
253 $fromRange -= $sameCount;
254 $toRange -= $sameCount;
255
256 $this->writeHunk(
257 $diff,
258 $hunkCapture - $contextStartOffset,
259 $i - $sameCount + $contextEndOffset + 1,
260 $fromStart - $contextStartOffset,
261 $fromRange + $contextStartOffset + $contextEndOffset,
262 $toStart - $contextStartOffset,
263 $toRange + $contextStartOffset + $contextEndOffset,
264 $output
265 );
266 }
