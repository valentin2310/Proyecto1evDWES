514 $parentKey = null, $relatedKey = null, $relation = null)

516 // If no relationship name was passed, we will pull backtraces to get the

579 public function morphToMany($related, $name, $table = null, $foreignPivotKey = null,
580 $relatedPivotKey = null, $parentKey = null,
581 $relatedKey = null, $relation = null, $inverse = false)
582 {
583 $relation = $relation ?: $this->guessBelongsToManyRelation();
584
585 // First, we will need to determine the foreign key and "other key" for the
586 // relationship. Once we have determined the keys we will make the query

588 $instance = $this->newRelatedInstance($related);

688 public function joiningTable($related, $instance = null)
689 {
690 // The joining table name, by convention, is simply the snake cased models
691 // sorted alphabetically and concatenated with an underscore, so we can
692 // just sort the models and join them together to get the table name.
693 $segments = [
694 $instance ? $instance->joiningTableSegment()
695 : Str::snake(class_basename($related)),
696 $this->joiningTableSegment(),
697 ];
698
699 // Now that we have the model names in an array we can just sort them and
700 // use the implode function to join them together with an underscores,
701 // which is typically used by convention within the database system.
702 sort($segments);
703
704 return strtolower(implode('_', $segments));
705 }
