56 {

58 $testWith = MetadataRegistry::parser()->forMethod($className, $methodName)->isTestWith();

235 private function dataProvidedByTestWithAnnotation(string $className, string $methodName): ?array
236 {
237 $docComment = (new ReflectionMethod($className, $methodName))->getDocComment();
238
239 if (!$docComment) {
240 return null;
241 }
242
243 $docComment = str_replace("\r\n", "\n", $docComment);
244 $docComment = preg_replace('/\n\s*\*\s?/', "\n", $docComment);
245 $docComment = substr($docComment, 0, -1);
246 $docComment = rtrim($docComment, "\n");
247
248 if (!preg_match('/@testWith\s+/', $docComment, $matches, PREG_OFFSET_CAPTURE)) {
249 return null;
250 }
251
252 $offset = strlen($matches[0][0]) + (int) $matches[0][1];
253 $annotationContent = substr($docComment, $offset);
254 $data = [];
255
256 foreach (explode("\n", $annotationContent) as $candidateRow) {
257 $candidateRow = trim($candidateRow);
258
259 if ($candidateRow[0] !== '[') {
260 break;
261 }
262
263 $dataSet = json_decode($candidateRow, true);
264
265 if (json_last_error() !== JSON_ERROR_NONE) {
266 throw new InvalidDataProviderException(
267 'The data set for the @testWith annotation cannot be parsed: ' . json_last_error_msg(),
268 );
269 }
270
271 $data[] = $dataSet;
272 }
273
274 if (!$data) {
275 throw new InvalidDataProviderException(
276 'The data set for the @testWith annotation cannot be parsed.',
277 );
278 }
279
280 return $data;
281 }
