298 public function send($view, array $data = [], $callback = null)
299 {
300 if ($view instanceof MailableContract) {
301 return $this->sendMailable($view);
302 }
303
304 $data['mailer'] = $this->name;
305
306 // First we need to parse the view, which could either be a string or an array
307 // containing both an HTML and plain text versions of the view which should
308 // be used when sending an e-mail. We will extract both of them out here.
309 [$view, $plain, $raw] = $this->parseView($view);
310
311 $data['message'] = $message = $this->createMessage();
312
313 // Once we have retrieved the view content for the e-mail we will set the body
314 // of this message using the HTML type, which will provide a simple wrapper
315 // to creating view based emails that are able to receive arrays of data.
316 if (! is_null($callback)) {
317 $callback($message);
318 }
319
320 $this->addContent($message, $view, $plain, $raw, $data);
321
322 // If a global "to" address has been set, we will set that address on the mail
323 // message. This is primarily useful during local development in which each
324 // message should be delivered into a single mail address for inspection.
325 if (isset($this->to['address'])) {
326 $this->setGlobalToAndRemoveCcAndBcc($message);
327 }
328
329 // Next we will determine if the message should be sent. We give the developer
330 // one final chance to stop this message and then we will send it to all of
331 // its recipients. We will then fire the sent event for the sent message.
332 $symfonyMessage = $message->getSymfonyMessage();
333
334 if ($this->shouldSendMessage($symfonyMessage, $data)) {
335 $symfonySentMessage = $this->sendSymfonyMessage($symfonyMessage);
336

340 $this->dispatchSentEvent($sentMessage, $data);

368 protected function parseView($view)
369 {
370 if (is_string($view) || $view instanceof Closure) {
371 return [$view, null, null];
372 }
373
374 // If the given view is an array with numeric keys, we will just assume that
375 // both a "pretty" and "plain" view were provided, so we will return this
376 // array as is, since it should contain both views with numerical keys.
377 if (is_array($view) && isset($view[0])) {
378 return [$view[0], $view[1], null];
379 }
380
381 // If this view is an array but doesn't contain numeric keys, we will assume
382 // the views are being explicitly specified and will extract them via the
383 // named keys instead, allowing the developers to use one or the other.
384 if (is_array($view)) {
385 return [
386 $view['html'] ?? null,
387 $view['text'] ?? null,
388 $view['raw'] ?? null,
389 ];
390 }
391
392 throw new InvalidArgumentException('Invalid view.');
393 }
