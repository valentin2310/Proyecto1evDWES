48 public function enterNode(Node $node)
49 {
50 if (!$node instanceof Assign) {
51 return;
52 }
53
54 if (!$node->var instanceof Array_ && !$node->var instanceof List_) {
55 return;
56 }
57
58 if (!$this->atLeastPhp71 && $node->var instanceof Array_) {
59 $msg = "syntax error, unexpected '='";
60 throw new ParseErrorException($msg, $node->expr->getLine());
61 }
62
63 // Polyfill for PHP-Parser 2.x
64 $items = isset($node->var->items) ? $node->var->items : $node->var->vars;
65
66 if ($items === [] || $items === [null]) {
67 throw new ParseErrorException('Cannot use empty list', $node->var->getLine());
68 }
69
70 $itemFound = false;
71 foreach ($items as $item) {
72 if ($item === null) {
73 continue;
74 }
75
76 $itemFound = true;
77
78 // List_->$vars in PHP-Parser 2.x is Variable instead of ArrayItem.
79 if (!$this->atLeastPhp71 && $item instanceof ArrayItem && $item->key !== null) {
80 $msg = 'Syntax error, unexpected T_CONSTANT_ENCAPSED_STRING, expecting \',\' or \')\'';
81 throw new ParseErrorException($msg, $item->key->getLine());
82 }
83
84 if (!self::isValidArrayItem($item)) {
85 $msg = 'Assignments can only happen to writable values';
86 throw new ParseErrorException($msg, $item->getLine());
87 }
88 }
89
90 if (!$itemFound) {
91 throw new ParseErrorException('Cannot use empty list');
92 }
93 }

100 private static function isValidArrayItem(Expr $item): bool
101 {
102 $value = ($item instanceof ArrayItem) ? $item->value : $item;
103
104 while ($value instanceof ArrayDimFetch || $value instanceof PropertyFetch) {
105 $value = $value->var;
106 }
107
108 // We just kind of give up if it's a method call. We can't tell if it's
109 // valid via static analysis.
110 return $value instanceof Variable || $value instanceof MethodCall || $value instanceof FuncCall;
111 }
