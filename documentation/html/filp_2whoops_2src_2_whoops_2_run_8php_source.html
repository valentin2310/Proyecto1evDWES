373 public function handleException($exception)
374 {
375 // Walk the registered handlers in the reverse order
376 // they were registered, and pass off the exception
377 $inspector = $this->getInspector($exception);
378
379 // Capture output produced while handling the exception,
380 // we might want to send it straight away to the client,
381 // or return it silently.
382 $this->system->startOutputBuffering();
383
384 // Just in case there are no handlers:
385 $handlerResponse = null;
386 $handlerContentType = null;
387
388 try {
389 foreach (array_reverse($this->handlerStack) as $handler) {
390 $handler->setRun($this);
391 $handler->setInspector($inspector);
392 $handler->setException($exception);
393
394 // The HandlerInterface does not require an Exception passed to handle()
395 // and neither of our bundled handlers use it.
396 // However, 3rd party handlers may have already relied on this parameter,
397 // and removing it would be possibly breaking for users.
398 $handlerResponse = $handler->handle($exception);
399
400 // Collect the content type for possible sending in the headers.
401 $handlerContentType = method_exists($handler, 'contentType') ? $handler->contentType() : null;
402
403 if (in_array($handlerResponse, [Handler::LAST_HANDLER, Handler::QUIT])) {
404 // The Handler has handled the exception in some way, and
405 // wishes to quit execution (Handler::QUIT), or skip any
406 // other handlers (Handler::LAST_HANDLER). If $this->allowQuit
407 // is false, Handler::QUIT behaves like Handler::LAST_HANDLER
408 break;
409 }
410 }
411
412 $willQuit = $handlerResponse == Handler::QUIT && $this->allowQuit();
413 } finally {
414 $output = $this->system->cleanOutputBuffer();
415 }
416
417 // If we're allowed to, send output generated by handlers directly
418 // to the output, otherwise, and if the script doesn't quit, return
419 // it so that it may be used by the caller
420 if ($this->writeToOutput()) {
421 // @todo Might be able to clean this up a bit better
422 if ($willQuit) {
423 // Cleanup all other output buffers before sending our output:
424 while ($this->system->getOutputBufferLevel() > 0) {
425 $this->system->endOutputBuffering();
426 }
427
428 // Send any headers if needed:
429 if (Misc::canSendHeaders() && $handlerContentType) {
430 header("Content-Type: {$handlerContentType}");
431 }
432 }
433
434 $this->writeToOutputNow($output);
435 }
436
437 if ($willQuit) {
438 // HHVM fix for https://github.com/facebook/hhvm/issues/4055
439 $this->system->flushOutputBuffer();
440
441 $this->system->stopExecution(
442 $this->sendExitCode()
443 );
444 }
445
446 return $output;
447 }

463 public function handleError($level, $message, $file = null, $line = null)
464 {
465 if ($level & $this->system->getErrorReportingLevel()) {
466 foreach ($this->silencedPatterns as $entry) {
467 $pathMatches = (bool) preg_match($entry["pattern"], $file);
468 $levelMatches = $level & $entry["levels"];
469 if ($pathMatches && $levelMatches) {
470 // Ignore the error, abort handling
471 // See https://github.com/filp/whoops/issues/418
472 return true;
473 }
474 }
475
476 // XXX we pass $level for the "code" param only for BC reasons.
477 // see https://github.com/filp/whoops/issues/267
478 $exception = new ErrorException($message, /*code*/ $level, /*severity*/ $level, $file, $line);
479 if ($this->canThrowExceptions) {
480 throw $exception;
481 } else {
482 $this->handleException($exception);
483 }
484 // Do not propagate errors which were already handled by Whoops.
485 return true;
486 }
487

489 // work on silenced errors.
490 return false;
