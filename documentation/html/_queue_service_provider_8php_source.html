244 protected function registerFailedJobServices()
245 {
246 $this->app->singleton('queue.failer', function ($app) {
247 $config = $app['config']['queue.failed'];
248
249 if (array_key_exists('driver', $config) &&
250 (is_null($config['driver']) || $config['driver'] === 'null')) {
251 return new NullFailedJobProvider;
252 }
253
254 if (isset($config['driver']) && $config['driver'] === 'file') {
255 return new FileFailedJobProvider(
256 $config['path'] ?? $this->app->storagePath('framework/cache/failed-jobs.json'),
257 $config['limit'] ?? 100,
258 fn () => $app['cache']->store('file'),
259 );
260 } elseif (isset($config['driver']) && $config['driver'] === 'dynamodb') {
261 return $this->dynamoFailedJobProvider($config);
262 } elseif (isset($config['driver']) && $config['driver'] === 'database-uuids') {
263 return $this->databaseUuidFailedJobProvider($config);
264 } elseif (isset($config['table'])) {
265 return $this->databaseFailedJobProvider($config);
266 } else {
267 return new NullFailedJobProvider;
268 }
269 });
