60 public function onKernelRequest(RequestEvent $event): void
61 {
62 if (!$event->isMainRequest()) {
63 return;
64 }
65
66 $request = $event->getRequest();
67 if (!$request->hasSession()) {
68 $request->setSessionFactory(function () use ($request) {
69 // Prevent calling `$this->getSession()` twice in case the Request (and the below factory) is cloned
70 static $sess;
71
72 if (!$sess) {
73 $sess = $this->getSession();
74 $request->setSession($sess);
75
76 /*
77 * For supporting sessions in php runtime with runners like roadrunner or swoole, the session
78 * cookie needs to be read from the cookie bag and set on the session storage.
79 *
80 * Do not set it when a native php session is active.
81 */

83 $sessionId = $sess->getId() ?: $request->cookies->get($sess->getName(), '');

93 public function onKernelResponse(ResponseEvent $event): void
94 {
95 if (!$event->isMainRequest() || (!$this->container->has('initialized_session') && !$event->getRequest()->hasSession())) {
96 return;
97 }
98
99 $response = $event->getResponse();
100 $autoCacheControl = !$response->headers->has(self::NO_AUTO_CACHE_CONTROL_HEADER);
101 // Always remove the internal header if present
102 $response->headers->remove(self::NO_AUTO_CACHE_CONTROL_HEADER);
103 if (!$event->getRequest()->hasSession(true)) {
104 return;
105 }
106 $session = $event->getRequest()->getSession();
107
108 if ($session->isStarted()) {
109 /*
110 * Saves the session, in case it is still open, before sending the response/headers.
111 *
112 * This ensures several things in case the developer did not save the session explicitly:
113 *
114 * * If a session save handler without locking is used, it ensures the data is available
115 * on the next request, e.g. after a redirect. PHPs auto-save at script end via
116 * session_register_shutdown is executed after fastcgi_finish_request. So in this case
117 * the data could be missing the next request because it might not be saved the moment
118 * the new request is processed.
119 * * A locking save handler (e.g. the native 'files') circumvents concurrency problems like
120 * the one above. But by saving the session before long-running things in the terminate event,
121 * we ensure the session is not blocked longer than needed.
122 * * When regenerating the session ID no locking is involved in PHPs session design. See
123 * https://bugs.php.net/61470 for a discussion. So in this case, the session must
124 * be saved anyway before sending the headers with the new session ID. Otherwise session
125 * data could get lost again for concurrent requests with the new ID. One result could be
126 * that you get logged out after just logging in.
127 *
128 * This listener should be executed as one of the last listeners, so that previous listeners
129 * can still operate on the open session. This prevents the overhead of restarting it.
130 * Listeners after closing the session can still work with the session as usual because
131 * Symfonys session implementation starts the session on demand. So writing to it after
132 * it is saved will just restart it.
133 */
134 $session->save();
135
136 /*
137 * For supporting sessions in php runtime with runners like roadrunner or swoole the session
138 * cookie need to be written on the response object and should not be written by PHP itself.
139 */
140 $sessionName = $session->getName();
141 $sessionId = $session->getId();
142 $sessionOptions = $this->getSessionOptions($this->sessionOptions);
143 $sessionCookiePath = $sessionOptions['cookie_path'] ?? '/';
144 $sessionCookieDomain = $sessionOptions['cookie_domain'] ?? null;
145 $sessionCookieSecure = $sessionOptions['cookie_secure'] ?? false;
146 $sessionCookieHttpOnly = $sessionOptions['cookie_httponly'] ?? true;
147 $sessionCookieSameSite = $sessionOptions['cookie_samesite'] ?? Cookie::SAMESITE_LAX;
148 $sessionUseCookies = $sessionOptions['use_cookies'] ?? true;
149
150 SessionUtils::popSessionCookie($sessionName, $sessionId);
151
152 if ($sessionUseCookies) {
153 $request = $event->getRequest();
154 $requestSessionCookieId = $request->cookies->get($sessionName);
155
156 $isSessionEmpty = ($session instanceof Session ? $session->isEmpty() : !$session->all()) && empty($_SESSION); // checking $_SESSION to keep compatibility with native sessions
157 if ($requestSessionCookieId && $isSessionEmpty) {
158 // PHP internally sets the session cookie value to "deleted" when setcookie() is called with empty string $value argument
159 // which happens in \Symfony\Component\HttpFoundation\Session\Storage\Handler\AbstractSessionHandler::destroy
160 // when the session gets invalidated (for example on logout) so we must handle this case here too
161 // otherwise we would send two Set-Cookie headers back with the response
162 SessionUtils::popSessionCookie($sessionName, 'deleted');
163 $response->headers->clearCookie(
164 $sessionName,
165 $sessionCookiePath,
166 $sessionCookieDomain,
167 $sessionCookieSecure,
168 $sessionCookieHttpOnly,
169 $sessionCookieSameSite
170 );
171 } elseif ($sessionId !== $requestSessionCookieId && !$isSessionEmpty) {
172 $expire = 0;
173 $lifetime = $sessionOptions['cookie_lifetime'] ?? null;
174 if ($lifetime) {
175 $expire = time() + $lifetime;
176 }
177
178 $response->headers->setCookie(
179 Cookie::create(
180 $sessionName,
181 $sessionId,
182 $expire,
183 $sessionCookiePath,
184 $sessionCookieDomain,
185 $sessionCookieSecure,
186 $sessionCookieHttpOnly,
187 false,
188 $sessionCookieSameSite
189 )
190 );
191 }
192 }
193 }
194
195 if ($session instanceof Session ? 0 === $session->getUsageIndex() : !$session->isStarted()) {
196 return;
197 }
198
199 if ($autoCacheControl) {
200 $maxAge = $response->headers->hasCacheControlDirective('public') ? 0 : (int) $response->getMaxAge();
201 $response
202 ->setExpires(new \DateTimeImmutable('+'.$maxAge.' seconds'))
203 ->setPrivate()
204 ->setMaxAge($maxAge)
205 ->headers->addCacheControlDirective('must-revalidate');
206 }
207
208 if (!$event->getRequest()->attributes->get('_stateless', false)) {
209 return;
210 }
211
212 if ($this->debug) {
213 throw new UnexpectedSessionUsageException('Session was used while the request was declared stateless.');
214 }
215
216 if ($this->container->has('logger')) {
217 $this->container->get('logger')->warning('Session was used while the request was declared stateless.');
218 }

221 public function onSessionUsage(): void
222 {
223 if (!$this->debug) {
224 return;
225 }
226
227 if ($this->container?->has('session_collector')) {
228 $this->container->get('session_collector')();
229 }
230
231 if (!$requestStack = $this->container?->has('request_stack') ? $this->container->get('request_stack') : null) {
232 return;
233 }
234
235 $stateless = false;
236 $clonedRequestStack = clone $requestStack;
237 while (null !== ($request = $clonedRequestStack->pop()) && !$stateless) {
238 $stateless = $request->attributes->get('_stateless');
239 }
240
241 if (!$stateless) {
242 return;
243 }
244
245 if (!$session = $requestStack->getCurrentRequest()->getSession()) {
246 return;
247 }
248
249 if ($session->isStarted()) {
250 $session->save();
251 }
252
253 throw new UnexpectedSessionUsageException('Session was used while the request was declared stateless.');
254 }
