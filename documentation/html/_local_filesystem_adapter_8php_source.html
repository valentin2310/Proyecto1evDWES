199 $location = $this->prefixer->prefixPath($path);

203 }
204

207
208 foreach ($iterator as $fileInfo) {

212 if ($fileInfo->isLink()) {

214 continue;
215 }
216 throw SymbolicLinkEncountered::atLocation($pathName);
217 }

219 $path = $this->prefixer->stripPrefix($pathName);
220 $lastModified = $fileInfo->getMTime();
221 $isDirectory = $fileInfo->isDir();

223 $visibility = $isDirectory ? $this->visibility->inverseForDirectory($permissions) : $this->visibility->inverseForFile($permissions);
224
225 yield $isDirectory ? new DirectoryAttributes(str_replace('\\', '/', $path), $visibility, $lastModified) : new FileAttributes(
226 str_replace('\\', '/', $path),

229 $lastModified
230 );
231 } catch (Throwable $exception) {
