679 public function copyDirectory($directory, $destination, $options = null)
680 {
681 if (! $this->isDirectory($directory)) {
682 return false;
683 }
684
685 $options = $options ?: FilesystemIterator::SKIP_DOTS;
686
687 // If the destination directory does not actually exist, we will go ahead and
688 // create it recursively, which just gets the destination prepared to copy
689 // the files over. Once we make the directory we'll proceed the copying.
690 $this->ensureDirectoryExists($destination, 0777);
691
692 $items = new FilesystemIterator($directory, $options);
693
694 foreach ($items as $item) {
695 // As we spin through items, we will check to see if the current file is actually
696 // a directory or a file. When it is actually a directory we will need to call
697 // back into this function recursively to keep copying these nested folders.
698 $target = $destination.'/'.$item->getBasename();
699
700 if ($item->isDir()) {
701 $path = $item->getPathname();
702
703 if (! $this->copyDirectory($path, $target, $options)) {
704 return false;
705 }
706 }
707
708 // If the current items is just a regular file, we will just copy this to the new
709 // location and keep looping. If for some reason the copy fails we'll bail out
710 // and return false, so the developer is aware that the copy process failed.
711 elseif (! $this->copy($item->getPathname(), $target)) {
712 return false;
713 }
714 }
715
716 return true;
717 }

728 public function deleteDirectory($directory, $preserve = false)
729 {
730 if (! $this->isDirectory($directory)) {
731 return false;
732 }
733
734 $items = new FilesystemIterator($directory);
735
736 foreach ($items as $item) {
737 // If the item is a directory, we can just recurse into the function and
738 // delete that sub-directory otherwise we'll just delete the file and
739 // keep iterating through each file until the directory is cleaned.
740 if ($item->isDir() && ! $item->isLink()) {
741 $this->deleteDirectory($item->getPathname());
742 }
743
744 // If the item is just a file, we can go ahead and delete it since we're
745 // just looping through and waxing all of the files in this directory
746 // and calling directories recursively, so we delete the real path.
747 else {
748 $this->delete($item->getPathname());
749 }
750 }
751
752 unset($items);
753
754 if (! $preserve) {
755 @rmdir($directory);
756 }
757
758 return true;
