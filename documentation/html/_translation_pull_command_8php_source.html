119 protected function execute(InputInterface $input, OutputInterface $output): int
120 {
121 $io = new SymfonyStyle($input, $output);
122
123 $provider = $this->providerCollection->get($input->getArgument('provider'));
124 $force = $input->getOption('force');
125 $intlIcu = $input->getOption('intl-icu');
126 $locales = $input->getOption('locales') ?: $this->enabledLocales;
127 $domains = $input->getOption('domains');
128 $format = $input->getOption('format');
129 $xliffVersion = '1.2';
130
131 if ($intlIcu && !$force) {
132 $io->note('--intl-icu option only has an effect when used with --force. Here, it will be ignored.');
133 }
134
135 switch ($format) {
136 case 'xlf20': $xliffVersion = '2.0';
137 // no break
138 case 'xlf12': $format = 'xlf';
139 }
140
141 $writeOptions = [
142 'path' => end($this->transPaths),
143 'xliff_version' => $xliffVersion,
144 'default_locale' => $this->defaultLocale,
145 ];
146
147 if (!$domains) {
148 $domains = $provider->getDomains();
149 }
150
151 $providerTranslations = $provider->read($domains, $locales);
152
153 if ($force) {
154 foreach ($providerTranslations->getCatalogues() as $catalogue) {
155 $operation = new TargetOperation(new MessageCatalogue($catalogue->getLocale()), $catalogue);
156 if ($intlIcu) {
157 $operation->moveMessagesToIntlDomainsIfPossible();
158 }
159 $this->writer->write($operation->getResult(), $format, $writeOptions);
160 }
161
162 $io->success(sprintf('Local translations has been updated from "%s" (for "%s" locale(s), and "%s" domain(s)).', parse_url($provider, \PHP_URL_SCHEME), implode(', ', $locales), implode(', ', $domains)));
163
164 return 0;
165 }
166
167 $localTranslations = $this->readLocalTranslations($locales, $domains, $this->transPaths);
168
169 // Append pulled translations to local ones.
170 $localTranslations->addBag($providerTranslations->diff($localTranslations));
171
172 foreach ($localTranslations->getCatalogues() as $catalogue) {
173 $this->writer->write($catalogue, $format, $writeOptions);
174 }
175
176 $io->success(sprintf('New translations from "%s" has been written locally (for "%s" locale(s), and "%s" domain(s)).', parse_url($provider, \PHP_URL_SCHEME), implode(', ', $locales), implode(', ', $domains)));
177
178 return 0;
179 }
