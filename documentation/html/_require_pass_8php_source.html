78 public static function resolve($file, $lineNumber = null): string
79 {
80 $file = (string) $file;
81
82 if ($file === '') {
83 // @todo Shell::handleError would be better here, because we could
84 // fake the file and line number, but we can't call it statically.
85 // So we're duplicating some of the logics here.
86 if (\E_WARNING & \error_reporting()) {
87 ErrorException::throwException(\E_WARNING, 'Filename cannot be empty', null, $lineNumber);
88 }
89 // @todo trigger an error as fallback? this is pretty ugly…
90 // trigger_error('Filename cannot be empty', E_USER_WARNING);
91 }
92
93 $resolvedPath = \stream_resolve_include_path($file);
94 if ($file === '' || !$resolvedPath) {
95 $msg = \sprintf("Failed opening required '%s'", $file);
96 throw new FatalErrorException($msg, 0, \E_ERROR, null, $lineNumber);
97 }
98
99 // Special case: if the path is not already relative or absolute, and it would resolve to
100 // something inside the currently running phar (e.g. `vendor/autoload.php`), we'll resolve
101 // it relative to the include path so PHP won't grab the phar version.
102 //
103 // Note that this only works if the phar has `psysh` in the path. We might want to lift this
104 // restriction and special case paths that would collide with any running phar?
105 if ($resolvedPath !== $file && $file[0] !== '.') {
106 $runningPhar = \Phar::running();
107 if (\strpos($runningPhar, 'psysh') !== false && \is_file($runningPhar.\DIRECTORY_SEPARATOR.$file)) {
108 foreach (self::getIncludePath() as $prefix) {
109 $resolvedPath = $prefix.\DIRECTORY_SEPARATOR.$file;
110 if (\is_file($resolvedPath)) {
111 return $resolvedPath;
112 }
113 }
114 }
115 }
116
117 return $file;
118 }
