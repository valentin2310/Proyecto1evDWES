132 public function compileForeign(Blueprint $blueprint, Fluent $command)
133 {
134 // We need to prepare several of the elements of the foreign key definition
135 // before we can create the SQL, such as wrapping the tables and convert
136 // an array of columns to comma-delimited strings for the SQL queries.
137 $sql = sprintf('alter table %s add constraint %s ',
138 $this->wrapTable($blueprint),
139 $this->wrap($command->index)
140 );
141
142 // Once we have the initial portion of the SQL statement we will add on the
143 // key name, table name, and referenced columns. These will complete the
144 // main portion of the SQL statement and this SQL will almost be done.
145 $sql .= sprintf('foreign key (%s) references %s (%s)',
146 $this->columnize($command->columns),
147 $this->wrapTable($command->on),
148 $this->columnize((array) $command->references)
149 );
150
151 // Once we have the basic foreign key creation statement constructed we can
152 // build out the syntax for what should happen on an update or delete of
153 // the affected columns, which will get something like "cascade", etc.
154 if (! is_null($command->onDelete)) {
155 $sql .= " on delete {$command->onDelete}";
156 }
157
158 if (! is_null($command->onUpdate)) {
159 $sql .= " on update {$command->onUpdate}";
160 }
161
162 return $sql;
163 }
