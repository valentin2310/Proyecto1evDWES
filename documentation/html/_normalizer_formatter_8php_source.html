162 protected function normalize(mixed $data, int $depth = 0): mixed
163 {
164 if ($depth > $this->maxNormalizeDepth) {
165 return 'Over ' . $this->maxNormalizeDepth . ' levels deep, aborting normalization';
166 }
167
168 if (null === $data || is_scalar($data)) {
169 if (is_float($data)) {
170 if (is_infinite($data)) {
171 return ($data > 0 ? '' : '-') . 'INF';
172 }
173 if (is_nan($data)) {
174 return 'NaN';
175 }
176 }
177
178 return $data;
179 }
180
181 if (is_array($data)) {
182 $normalized = [];
183
184 $count = 1;
185 foreach ($data as $key => $value) {
186 if ($count++ > $this->maxNormalizeItemCount) {
187 $normalized['...'] = 'Over ' . $this->maxNormalizeItemCount . ' items ('.count($data).' total), aborting normalization';
188 break;
189 }
190
191 $normalized[$key] = $this->normalize($value, $depth + 1);
192 }
193
194 return $normalized;
195 }
196
197 if ($data instanceof \DateTimeInterface) {
198 return $this->formatDate($data);
199 }
200
201 if (is_object($data)) {
202 if ($data instanceof Throwable) {
203 return $this->normalizeException($data, $depth);
204 }
205
206 if ($data instanceof \JsonSerializable) {
208 $value = $data->jsonSerialize();
209 } elseif (\get_class($data) === '__PHP_Incomplete_Class') {
210 $accessor = new \ArrayObject($data);
211 $value = (string) $accessor['__PHP_Incomplete_Class_Name'];
212 } elseif (method_exists($data, '__toString')) {
214 $value = $data->__toString();
215 } else {
216 // the rest is normalized by json encoding and decoding it
218 $value = json_decode($this->toJson($data, true), true);
219 }
220
221 return [Utils::getClass($data) => $value];
222 }
223
224 if (is_resource($data)) {
225 return sprintf('[resource(%s)]', get_resource_type($data));
226 }
227
228 return '[unknown('.gettype($data).')]';
229 }

234 protected function normalizeException(Throwable $e, int $depth = 0)
235 {
236 if ($depth > $this->maxNormalizeDepth) {
237 return ['Over ' . $this->maxNormalizeDepth . ' levels deep, aborting normalization'];
238 }
239
240 if ($e instanceof \JsonSerializable) {
241 return (array) $e->jsonSerialize();
242 }
243
244 $data = [
245 'class' => Utils::getClass($e),
246 'message' => $e->getMessage(),
247 'code' => (int) $e->getCode(),
248 'file' => $e->getFile().':'.$e->getLine(),
249 ];
250
251 if ($e instanceof \SoapFault) {
252 if (isset($e->faultcode)) {
253 $data['faultcode'] = $e->faultcode;
254 }
255
256 if (isset($e->faultactor)) {
257 $data['faultactor'] = $e->faultactor;
258 }
259
260 if (isset($e->detail)) {
261 if (is_string($e->detail)) {
262 $data['detail'] = $e->detail;
263 } elseif (is_object($e->detail) || is_array($e->detail)) {
264 $data['detail'] = $this->toJson($e->detail, true);
265 }
266 }
267 }
268
269 $trace = $e->getTrace();
270 foreach ($trace as $frame) {
271 if (isset($frame['file'], $frame['line'])) {
272 $data['trace'][] = $frame['file'].':'.$frame['line'];
273 }
274 }
275
276 if (($previous = $e->getPrevious()) instanceof \Throwable) {
277 $data['previous'] = $this->normalizeException($previous, $depth + 1);
278 }
279
280 return $data;
281 }
