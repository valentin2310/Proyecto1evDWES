182 public function handle(Request $request, int $type = HttpKernelInterface::MAIN_REQUEST, bool $catch = true): Response
183 {
184 if (!$this->booted) {
185 $container = $this->container ?? $this->preBoot();
186
187 if ($container->has('http_cache')) {
188 return $container->get('http_cache')->handle($request, $type, $catch);
189 }
190 }
191
192 $this->boot();
193 ++$this->requestStackSize;
194 $this->resetServices = true;
195
196 try {
197 return $this->getHttpKernel()->handle($request, $type, $catch);
198 } finally {
199 --$this->requestStackSize;
200 }

778 public static function stripComments(string $source): string
779 {
780 if (!\function_exists('token_get_all')) {
781 return $source;
782 }
783
784 $rawChunk = '';
785 $output = '';
786 $tokens = token_get_all($source);
787 $ignoreSpace = false;
788 for ($i = 0; isset($tokens[$i]); ++$i) {
789 $token = $tokens[$i];
790 if (!isset($token[1]) || 'b"' === $token) {
791 $rawChunk .= $token;
792 } elseif (\T_START_HEREDOC === $token[0]) {
793 $output .= $rawChunk.$token[1];
794 do {
795 $token = $tokens[++$i];
796 $output .= isset($token[1]) && 'b"' !== $token ? $token[1] : $token;
797 } while (\T_END_HEREDOC !== $token[0]);
798 $rawChunk = '';
799 } elseif (\T_WHITESPACE === $token[0]) {
800 if ($ignoreSpace) {
801 $ignoreSpace = false;
802
803 continue;
804 }
805
806 // replace multiple new lines with a single newline
807 $rawChunk .= preg_replace(['/\n{2,}/S'], "\n", $token[1]);
