294 public function start(callable $callback = null, array $env = [])
295 {
296 if ($this->isRunning()) {
297 throw new RuntimeException('Process is already running.');
298 }
299
300 $this->resetProcessData();
301 $this->starttime = $this->lastOutputTime = microtime(true);
302 $this->callback = $this->buildCallback($callback);
303 $this->hasCallback = null !== $callback;
304 $descriptors = $this->getDescriptors();
305
306 if ($this->env) {
307 $env += '\\' === \DIRECTORY_SEPARATOR ? array_diff_ukey($this->env, $env, 'strcasecmp') : $this->env;
308 }
309
310 $env += '\\' === \DIRECTORY_SEPARATOR ? array_diff_ukey($this->getDefaultEnv(), $env, 'strcasecmp') : $this->getDefaultEnv();
311
312 if (\is_array($commandline = $this->commandline)) {
313 $commandline = implode(' ', array_map($this->escapeArgument(...), $commandline));
314
315 if ('\\' !== \DIRECTORY_SEPARATOR) {
316 // exec is mandatory to deal with sending a signal to the process
317 $commandline = 'exec '.$commandline;
318 }
319 } else {
320 $commandline = $this->replacePlaceholders($commandline, $env);
321 }
322
323 if ('\\' === \DIRECTORY_SEPARATOR) {
324 $commandline = $this->prepareWindowsCommandLine($commandline, $env);
325 } elseif (!$this->useFileHandles && $this->isSigchildEnabled()) {
326 // last exit code is output on the fourth pipe and caught to work around --enable-sigchild
327 $descriptors[3] = ['pipe', 'w'];
328
329 // See https://unix.stackexchange.com/questions/71205/background-process-pipe-input
330 $commandline = '{ ('.$commandline.') <&3 3<&- 3>/dev/null & } 3<&0;';
331 $commandline .= 'pid=$!; echo $pid >&3; wait $pid 2>/dev/null; code=$?; echo $code >&3; exit $code';
332
333 // Workaround for the bug, when PTS functionality is enabled.
334 // @see : https://bugs.php.net/69442
335 $ptsWorkaround = fopen(__FILE__, 'r');
336 }
337
338 $envPairs = [];
339 foreach ($env as $k => $v) {
340 if (false !== $v && false === \in_array($k, ['argc', 'argv', 'ARGC', 'ARGV'], true)) {
341 $envPairs[] = $k.'='.$v;
342 }
343 }
344
345 if (!is_dir($this->cwd)) {
346 throw new RuntimeException(sprintf('The provided cwd "%s" does not exist.', $this->cwd));
347 }
348
349 $this->process = @proc_open($commandline, $descriptors, $this->processPipes->pipes, $this->cwd, $envPairs, $this->options);
350
351 if (!\is_resource($this->process)) {
352 throw new RuntimeException('Unable to launch a new process.');
353 }
354 $this->status = self::STATUS_STARTED;
355
356 if (isset($descriptors[3])) {
357 $this->fallbackStatus['pid'] = (int) fgets($this->processPipes->pipes[3]);
358 }
359
360 if ($this->tty) {
361 return;
362 }
363
364 $this->updateStatus(false);
365 $this->checkTimeout();
366 }

410 public function wait(callable $callback = null): int
411 {
412 $this->requireProcessIsStarted(__FUNCTION__);
413
414 $this->updateStatus(false);
415
416 if (null !== $callback) {
417 if (!$this->processPipes->haveReadSupport()) {
418 $this->stop(0);
419 throw new LogicException('Pass the callback to the "Process::start" method or call enableOutput to use a callback with "Process::wait".');
420 }
421 $this->callback = $this->buildCallback($callback);
422 }
423
424 do {
425 $this->checkTimeout();
426 $running = $this->isRunning() && ('\\' === \DIRECTORY_SEPARATOR || $this->processPipes->areOpen());
427 $this->readPipes($running, '\\' !== \DIRECTORY_SEPARATOR || !$running);
428 } while ($running);
429
430 while ($this->isRunning()) {
431 $this->checkTimeout();
432 usleep(1000);
433 }
434
435 if ($this->processInformation['signaled'] && $this->processInformation['termsig'] !== $this->latestSignal) {
436 throw new ProcessSignaledException($this);
437 }
438
439 return $this->exitcode;
440 }

453 public function waitUntil(callable $callback): bool
454 {
455 $this->requireProcessIsStarted(__FUNCTION__);
456 $this->updateStatus(false);
457
458 if (!$this->processPipes->haveReadSupport()) {
459 $this->stop(0);
460 throw new LogicException('Pass the callback to the "Process::start" method or call enableOutput to use a callback with "Process::waitUntil".');
461 }
462 $callback = $this->buildCallback($callback);
463
464 $ready = false;
465 while (true) {
466 $this->checkTimeout();
467 $running = '\\' === \DIRECTORY_SEPARATOR ? $this->isRunning() : $this->processPipes->areOpen();
468 $output = $this->processPipes->readAndWrite($running, '\\' !== \DIRECTORY_SEPARATOR || !$running);
469
470 foreach ($output as $type => $data) {
471 if (3 !== $type) {
472 $ready = $callback(self::STDOUT === $type ? self::OUT : self::ERR, $data) || $ready;
473 } elseif (!isset($this->fallbackStatus['signaled'])) {
474 $this->fallbackStatus['exitcode'] = (int) $data;
475 }
476 }
477 if ($ready) {
478 return true;
479 }
480 if (!$running) {
481 return false;
482 }
483
484 usleep(1000);
485 }

614 public function getIterator(int $flags = 0): \Generator
615 {
616 $this->readPipesForOutput(__FUNCTION__, false);
617
618 $clearOutput = !(self::ITER_KEEP_OUTPUT & $flags);
619 $blocking = !(self::ITER_NON_BLOCKING & $flags);
620 $yieldOut = !(self::ITER_SKIP_OUT & $flags);
621 $yieldErr = !(self::ITER_SKIP_ERR & $flags);
622
623 while (null !== $this->callback || ($yieldOut && !feof($this->stdout)) || ($yieldErr && !feof($this->stderr))) {
624 if ($yieldOut) {
625 $out = stream_get_contents($this->stdout, -1, $this->incrementalOutputOffset);
626
627 if (isset($out[0])) {
628 if ($clearOutput) {
629 $this->clearOutput();
630 } else {
631 $this->incrementalOutputOffset = ftell($this->stdout);
632 }
633
634 yield self::OUT => $out;
635 }
636 }
637
638 if ($yieldErr) {
639 $err = stream_get_contents($this->stderr, -1, $this->incrementalErrorOutputOffset);
640
641 if (isset($err[0])) {
642 if ($clearOutput) {
643 $this->clearErrorOutput();
644 } else {
645 $this->incrementalErrorOutputOffset = ftell($this->stderr);
646 }
647
648 yield self::ERR => $err;
649 }
650 }
651
652 if (!$blocking && !isset($out[0]) && !isset($err[0])) {
653 yield self::OUT => '';
654 }
655
656 $this->checkTimeout();
657 $this->readPipesForOutput(__FUNCTION__, $blocking);
658 }
659 }

883 public function stop(float $timeout = 10, int $signal = null): ?int
884 {
885 $timeoutMicro = microtime(true) + $timeout;
886 if ($this->isRunning()) {
887 // given SIGTERM may not be defined and that "proc_terminate" uses the constant value and not the constant itself, we use the same here
888 $this->doSignal(15, false);
889 do {
890 usleep(1000);
891 } while ($this->isRunning() && microtime(true) < $timeoutMicro);
892
893 if ($this->isRunning()) {
894 // Avoid exception here: process is supposed to be running, but it might have stopped just
895 // after this line. In any case, let's silently discard the error, we cannot do anything.
896 $this->doSignal($signal ?: 9, false);
897 }
898 }
899
900 if ($this->isRunning()) {
901 if (isset($this->fallbackStatus['pid'])) {
902 unset($this->fallbackStatus['pid']);
903
904 return $this->stop(0, $signal);
905 }
906 $this->close();
907 }
908
909 return $this->exitcode;
910 }
