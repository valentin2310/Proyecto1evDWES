65 : Yaml::parse(file_get_contents(__DIR__ . '/../../../stubs/docker-compose.stub'));
66
67 // Adds the new services as dependencies of the laravel.test service...
68 if (! array_key_exists('laravel.test', $compose['services'])) {
69 $this->warn('Couldn\'t find the laravel.test service. Make sure you add ['.implode(',', $services).'] to the depends_on config.');
70 } else {
71 $compose['services']['laravel.test']['depends_on'] = collect($compose['services']['laravel.test']['depends_on'] ?? [])
72 ->merge($services)

74 ->values()

82 })->each(function ($service) use (&$compose) {
83 $compose['services'][$service] = Yaml::parseFile(__DIR__ . "/../../../stubs/{$service}.stub")[$service];
84 });
85
86 // Merge volumes...
87 collect($services)
88 ->filter(function ($service) {
89 return in_array($service, ['mysql', 'pgsql', 'mariadb', 'redis', 'meilisearch', 'minio']);
90 })->filter(function ($service) use ($compose) {
91 return ! array_key_exists($service, $compose['volumes'] ?? []);
92 })->each(function ($service) use (&$compose) {
93 $compose['volumes']["sail-{$service}"] = ['driver' => 'local'];
94 });
95
96 // If the list of volumes is empty, we can remove it...
97 if (empty($compose['volumes'])) {
98 unset($compose['volumes']);
99 }

101 // Replace Selenium with ARM base container on Apple Silicon...
102 if (in_array('selenium', $services) && in_array(php_uname('m'), ['arm64', 'aarch64'])) {

104 }

106 file_put_contents($this->laravel->basePath('docker-compose.yml'), Yaml::dump($compose, Yaml::DUMP_OBJECT_AS_MAP));
