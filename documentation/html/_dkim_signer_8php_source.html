130 private function canonicalizeHeader(string $header, string $headerCanon): string
131 {
132 if (self::CANON_RELAXED !== $headerCanon) {
133 return $header."\r\n";
134 }
135
136 $exploded = explode(':', $header, 2);
137 $name = strtolower(trim($exploded[0]));
138 $value = str_replace("\r\n", '', $exploded[1]);
139 $value = trim(preg_replace("/[ \t][ \t]+/", ' ', $value));
140
141 return $name.':'.$value."\r\n";

144 private function hashBody(AbstractPart $body, string $bodyCanon, int $maxLength): array
145 {
146 $hash = hash_init('sha256');
147 $relaxed = self::CANON_RELAXED === $bodyCanon;
148 $currentLine = '';
149 $emptyCounter = 0;
150 $isSpaceSequence = false;
151 $length = 0;
152 foreach ($body->bodyToIterable() as $chunk) {
153 $canon = '';
154 for ($i = 0, $len = \strlen($chunk); $i < $len; ++$i) {
155 switch ($chunk[$i]) {
156 case "\r":
157 break;
158 case "\n":
159 // previous char is always \r
160 if ($relaxed) {
161 $isSpaceSequence = false;
162 }
163 if ('' === $currentLine) {
164 ++$emptyCounter;
165 } else {
166 $currentLine = '';
167 $canon .= "\r\n";
168 }
169 break;
170 case ' ':
171 case "\t":
172 if ($relaxed) {
173 $isSpaceSequence = true;
174 break;
175 }
176 // no break
177 default:
178 if ($emptyCounter > 0) {
179 $canon .= str_repeat("\r\n", $emptyCounter);
180 $emptyCounter = 0;
181 }
182 if ($isSpaceSequence) {
183 $currentLine .= ' ';
184 $canon .= ' ';
185 $isSpaceSequence = false;
186 }
187 $currentLine .= $chunk[$i];
188 $canon .= $chunk[$i];
189 }
190 }
191
192 if ($length + \strlen($canon) >= $maxLength) {
193 $canon = substr($canon, 0, $maxLength - $length);
194 $length += \strlen($canon);
195 hash_update($hash, $canon);
196
197 break;
198 }
199
200 $length += \strlen($canon);
201 hash_update($hash, $canon);
202 }
203
204 // Add trailing Line return if last line is non empty
205 if ('' !== $currentLine) {
206 hash_update($hash, "\r\n");
207 $length += \strlen("\r\n");
208 }
209
210 if (!$relaxed && 0 === $length) {
211 hash_update($hash, "\r\n");
212 $length = 2;
213 }
214
215 return [hash_final($hash, true), $length];
216 }
