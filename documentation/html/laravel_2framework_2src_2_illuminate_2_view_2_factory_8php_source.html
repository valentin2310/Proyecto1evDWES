215 public function renderEach($view, $data, $iterator, $empty = 'raw|')
216 {
217 $result = '';
218
219 // If is actually data in the array, we will loop through the data and append
220 // an instance of the partial view to the final result HTML passing in the
221 // iterated value of this data array, allowing the views to access them.
222 if (count($data) > 0) {
223 foreach ($data as $key => $value) {
224 $result .= $this->make(
225 $view, ['key' => $key, $iterator => $value]
226 )->render();
227 }
228 }
229
230 // If there is no data in the array, we will render the contents of the empty
231 // view. Alternatively, the "empty view" could be a raw string that begins
232 // with "raw|" for convenience and to let this know that it is a string.
233 else {
234 $result = str_starts_with($empty, 'raw|')
235 ? substr($empty, 4)
236 : $this->make($empty)->render();
237 }
238
239 return $result;
