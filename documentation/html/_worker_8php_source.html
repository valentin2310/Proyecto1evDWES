131 public function daemon($connectionName, $queue, WorkerOptions $options)
132 {
133 if ($supportsAsyncSignals = $this->supportsAsyncSignals()) {
134 $this->listenForSignals();
135 }
136
137 $lastRestart = $this->getTimestampOfLastQueueRestart();
138
139 [$startTime, $jobsProcessed] = [hrtime(true) / 1e9, 0];
140
141 while (true) {
142 // Before reserving any jobs, we will make sure this queue is not paused and
143 // if it is we will just pause this worker for a given amount of time and
144 // make sure we do not need to kill this worker process off completely.
145 if (! $this->daemonShouldRun($options, $connectionName, $queue)) {
146 $status = $this->pauseWorker($options, $lastRestart);
147
148 if (! is_null($status)) {
149 return $this->stop($status, $options);
150 }
151
152 continue;
153 }
154
155 if (isset($this->resetScope)) {
156 ($this->resetScope)();
157 }
158
159 // First, we will attempt to get the next job off of the queue. We will also
160 // register the timeout handler and reset the alarm for this job so it is
161 // not stuck in a frozen state forever. Then, we can fire off this job.
162 $job = $this->getNextJob(
163 $this->manager->connection($connectionName), $queue
164 );
165
166 if ($supportsAsyncSignals) {
167 $this->registerTimeoutHandler($job, $options);
168 }
169
170 // If the daemon should run (not in maintenance mode, etc.), then we can run
171 // fire off this job for processing. Otherwise, we will need to sleep the
172 // worker so no more jobs are processed until they should be processed.
173 if ($job) {
174 $jobsProcessed++;
175
176 $this->runJob($job, $connectionName, $options);
177
178 if ($options->rest > 0) {
179 $this->sleep($options->rest);
180 }
181 } else {
182 $this->sleep($options->sleep);
183 }
184
185 if ($supportsAsyncSignals) {
186 $this->resetTimeoutHandler();
187 }
188
189 // Finally, we will check to see if we have exceeded our memory limits or if
190 // the queue should restart based on other indications. If so, we'll stop
191 // this worker and let whatever is "monitoring" it restart the process.
192 $status = $this->stopIfNecessary(
193 $options, $lastRestart, $startTime, $jobsProcessed, $job
194 );
195
196 if (! is_null($status)) {
197 return $this->stop($status, $options);
198 }
199 }

346 protected function getNextJob($connection, $queue)
347 {
348 $popJobCallback = function ($queue) use ($connection) {
349 return $connection->pop($queue);
350 };
351
352 $this->raiseBeforeJobPopEvent($connection->getConnectionName());
353
354 try {
355 if (isset(static::$popCallbacks[$this->name])) {
356 return tap(
357 (static::$popCallbacks[$this->name])($popJobCallback, $queue),
358 fn ($job) => $this->raiseAfterJobPopEvent($connection->getConnectionName(), $job)
359 );
360 }
361
362 foreach (explode(',', $queue) as $queue) {
363 if (! is_null($job = $popJobCallback($queue))) {
364 $this->raiseAfterJobPopEvent($connection->getConnectionName(), $job);
365
366 return $job;
367 }
368 }
369 } catch (Throwable $e) {
370 $this->exceptions->report($e);
371
372 $this->stopWorkerIfLostConnection($e);
373
374 $this->sleep(1);
375 }

420 public function process($connectionName, $job, WorkerOptions $options)
421 {
422 try {
423 // First we will raise the before job event and determine if the job has already run
424 // over its maximum attempt limits, which could primarily happen when this job is
425 // continually timing out and not actually throwing any exceptions from itself.
426 $this->raiseBeforeJobEvent($connectionName, $job);
427
428 $this->markJobAsFailedIfAlreadyExceedsMaxAttempts(
429 $connectionName, $job, (int) $options->maxTries
430 );
431
432 if ($job->isDeleted()) {
433 return $this->raiseAfterJobEvent($connectionName, $job);
434 }
435
436 // Here we will fire off the job and let it process. We will catch any exceptions, so
437 // they can be reported to the developer's logs, etc. Once the job is finished the
438 // proper events will be fired to let any listeners know this job has completed.
439 $job->fire();
440
441 $this->raiseAfterJobEvent($connectionName, $job);
442 } catch (Throwable $e) {
443 $this->handleJobException($connectionName, $job, $options, $e);
444 }
445 }

458 protected function handleJobException($connectionName, $job, WorkerOptions $options, Throwable $e)
459 {
460 try {
461 // First, we will go ahead and mark the job as failed if it will exceed the maximum
462 // attempts it is allowed to run the next time we process it. If so we will just
463 // go ahead and mark it as failed now so we do not have to release this again.
464 if (! $job->hasFailed()) {
465 $this->markJobAsFailedIfWillExceedMaxAttempts(
466 $connectionName, $job, (int) $options->maxTries, $e
467 );
468
469 $this->markJobAsFailedIfWillExceedMaxExceptions(
470 $connectionName, $job, $e
471 );
472 }
473
474 $this->raiseExceptionOccurredJobEvent(
475 $connectionName, $job, $e
476 );
477 } finally {
478 // If we catch an exception, we will attempt to release the job back onto the queue
479 // so it is not lost entirely. This'll let the job be retried at a later time by
480 // another listener (or this same one). We will re-throw this exception after.
481 if (! $job->isDeleted() && ! $job->isReleased() && ! $job->hasFailed()) {
482 $job->release($this->calculateBackoff($job, $options));
483
484 $this->events->dispatch(new JobReleasedAfterException(
485 $connectionName, $job
486 ));
487 }
488 }
489
490 throw $e;
491 }

505 protected function markJobAsFailedIfAlreadyExceedsMaxAttempts($connectionName, $job, $maxTries)
506 {
507 $maxTries = ! is_null($job->maxTries()) ? $job->maxTries() : $maxTries;
