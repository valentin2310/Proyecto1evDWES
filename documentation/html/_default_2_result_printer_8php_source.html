360 private function printIssueList(string $type, array $issues): void
361 {
362 if (empty($issues)) {
363 return;
364 }
365
366 $numberOfUniqueIssues = count($issues);
367 $triggeringTests = [];
368
369 foreach ($issues as $issue) {
370 $triggeringTests = array_merge($triggeringTests, array_keys($issue->triggeringTests()));
371 }
372
373 $numberOfTests = count(array_unique($triggeringTests));
374 unset($triggeringTests);
375
376 $this->printListHeader(
377 sprintf(
378 '%d test%s triggered %d %s%s:' . PHP_EOL . PHP_EOL,
379 $numberOfTests,
380 $numberOfTests !== 1 ? 's' : '',
381 $numberOfUniqueIssues,
382 $type,
383 $numberOfUniqueIssues !== 1 ? 's' : '',
384 ),
385 );
386
387 $i = 1;
388
389 foreach ($issues as $issue) {
390 $title = sprintf(
391 '%s:%d',
392 $issue->file(),
393 $issue->line(),
394 );
395
396 $body = trim($issue->description()) . PHP_EOL . PHP_EOL . 'Triggered by:';
397
398 $triggeringTests = $issue->triggeringTests();
399
400 ksort($triggeringTests);
401
402 foreach ($triggeringTests as $triggeringTest) {
403 $body .= PHP_EOL . PHP_EOL . '* ' . $triggeringTest['test']->id();
404
405 if ($triggeringTest['count'] > 1) {
406 $body .= sprintf(
407 ' (%d times)',
408 $triggeringTest['count'],
409 );
410 }
411
412 if ($triggeringTest['test']->isTestMethod()) {
413 $body .= PHP_EOL . ' ' . $triggeringTest['test']->file() . ':' . $triggeringTest['test']->line();
414 }
415 }
416
417 $this->printIssueListElement($i++, $title, $body);
418
419 $this->printer->print(PHP_EOL);
420 }
421 }

529 private function mapTestsWithIssuesEventsToElements(array $events): array
530 {
531 $elements = [];
532 $issues = 0;
533
534 foreach ($events as $reasons) {
535 $test = $reasons[0]->test();
536 $testLocation = $this->testLocation($test);
537 $title = $this->name($test);
538 $body = '';
539 $first = true;
540 $single = count($reasons) === 1;
541
542 foreach ($reasons as $reason) {
543 if ($first) {
544 $first = false;
545 } else {
546 $body .= PHP_EOL;
547 }
548
549 $body .= $this->reasonMessage($reason, $single);
550 $body .= $this->reasonLocation($reason, $single);
551
552 $issues++;
553 }
554
555 if (!empty($testLocation)) {
556 $body .= $testLocation;
557 }
558
559 $elements[] = [
560 'title' => $title,
561 'body' => $body,
562 ];
563 }
564
565 return [
566 'numberOfTestsWithIssues' => count($events),
567 'numberOfIssues' => $issues,
568 'elements' => $elements,
569 ];
570 }
