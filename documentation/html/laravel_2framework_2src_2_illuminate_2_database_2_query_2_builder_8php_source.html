755 public function where($column, $operator = null, $value = null, $boolean = 'and')
756 {
757 if ($column instanceof ConditionExpression) {
758 $type = 'Expression';
759
760 $this->wheres[] = compact('type', 'column', 'boolean');
761
762 return $this;
763 }
764
765 // If the column is an array, we will assume it is an array of key-value pairs
766 // and can add them each as a where clause. We will maintain the boolean we
767 // received when the method was called and pass it into the nested where.
768 if (is_array($column)) {
769 return $this->addArrayOfWheres($column, $boolean);
770 }
771
772 // Here we will make some assumptions about the operator. If only 2 values are
773 // passed to the method, we will assume that the operator is an equals sign
774 // and keep going. Otherwise, we'll require the operator to be passed in.
775 [$value, $operator] = $this->prepareValueAndOperator(
776 $value, $operator, func_num_args() === 2
777 );
778
779 // If the column is actually a Closure instance, we will assume the developer
780 // wants to begin a nested where statement which is wrapped in parentheses.
781 // We will add that Closure to the query and return back out immediately.
782 if ($column instanceof Closure && is_null($operator)) {
783 return $this->whereNested($column, $boolean);
784 }
785
786 // If the column is a Closure instance and there is an operator value, we will
787 // assume the developer wants to run a subquery and then compare the result
788 // of that subquery with the given value that was provided to the method.
789 if ($this->isQueryable($column) && ! is_null($operator)) {
790 [$sub, $bindings] = $this->createSub($column);
791
792 return $this->addBinding($bindings, 'where')
793 ->where(new Expression('('.$sub.')'), $operator, $value, $boolean);
794 }
795
796 // If the given operator is not found in the list of valid operators we will
797 // assume that the developer is just short-cutting the '=' operators and
798 // we will set the operators to '=' and set the values appropriately.
799 if ($this->invalidOperator($operator)) {
800 [$value, $operator] = [$operator, '='];
801 }
802
803 // If the value is a Closure, it means the developer is performing an entire
804 // sub-select within the query and we will need to compile the sub-select
805 // within the where clause to get the appropriate query record results.
806 if ($this->isQueryable($value)) {
807 return $this->whereSub($column, $operator, $value, $boolean);
808 }
809
810 // If the value is "null", we will just assume the developer wants to add a
811 // where null clause to the query. So, we will allow a short-cut here to
812 // that method for convenience so the developer doesn't have to check.
813 if (is_null($value)) {
814 return $this->whereNull($column, $boolean, $operator !== '=');
815 }
816
817 $type = 'Basic';
818
819 $columnString = ($column instanceof ExpressionContract)
820 ? $this->grammar->getValue($column)
821 : $column;
822
823 // If the column is making a JSON reference we'll check to see if the value
824 // is a boolean. If it is, we'll add the raw boolean string as an actual
825 // value to the query to ensure this is properly handled by the query.
826 if (str_contains($columnString, '->') && is_bool($value)) {
827 $value = new Expression($value ? 'true' : 'false');
828
829 if (is_string($column)) {
830 $type = 'JsonBoolean';
831 }
832 }
833
834 if ($this->isBitwiseOperator($operator)) {
835 $type = 'Bitwise';
836 }
837
838 // Now that we are working with just a simple query we can put the elements
839 // in our array and add the query binding to our array of bindings that
840 // will be bound to each SQL statements when it is finally executed.
841 $this->wheres[] = compact(
842 'type', 'column', 'operator', 'value', 'boolean'
843 );
844
845 if (! $value instanceof ExpressionContract) {
846 $this->addBinding($this->flattenValue($value), 'where');
847 }
848
849 return $this;
850 }

1964 public function dynamicWhere($method, $parameters)
1965 {
1966 $finder = substr($method, 5);
1967
1968 $segments = preg_split(
1969 '/(And|Or)(?=[A-Z])/', $finder, -1, PREG_SPLIT_DELIM_CAPTURE
1970 );
1971
1972 // The connector variable will determine which connector will be used for the
1973 // query condition. We will change it as we come across new boolean values
1974 // in the dynamic method strings, which could contain a number of these.
1975 $connector = 'and';
1976
1977 $index = 0;
1978
1979 foreach ($segments as $segment) {
1980 // If the segment is not a boolean connector, we can assume it is a column's name
1981 // and we will add it to the query as a new constraint as a where clause, then
1982 // we can keep iterating through the dynamic method string's segments again.
1983 if ($segment !== 'And' && $segment !== 'Or') {
1984 $this->addDynamic($segment, $connector, $parameters, $index);

1986 $index++;

1990 // find in the query should be connected to the previous ones, meaning we will
1991 // have the proper boolean connector to connect the next where clause found.
1992 else {
1993 $connector = $segment;
1994 }
1995 }
