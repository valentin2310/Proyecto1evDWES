138 private function format(string $level, string $message, array $context, bool $prefixDate = true): string
139 {
140 if (str_contains($message, '{')) {
141 $replacements = [];
142 foreach ($context as $key => $val) {
143 if (null === $val || \is_scalar($val) || $val instanceof \Stringable) {
144 $replacements["{{$key}}"] = $val;
145 } elseif ($val instanceof \DateTimeInterface) {
146 $replacements["{{$key}}"] = $val->format(\DateTimeInterface::RFC3339);
147 } elseif (\is_object($val)) {
148 $replacements["{{$key}}"] = '[object '.$val::class.']';
149 } else {
150 $replacements["{{$key}}"] = '['.\gettype($val).']';
151 }
152 }
153
154 $message = strtr($message, $replacements);
155 }
156
157 $log = sprintf('[%s] %s', $level, $message);
158 if ($prefixDate) {
159 $log = date(\DateTimeInterface::RFC3339).' '.$log;
160 }
