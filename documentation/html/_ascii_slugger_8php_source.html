107 public function slug(string $string, string $separator = '-', string $locale = null): AbstractUnicodeString
108 {
109 $locale ??= $this->defaultLocale;
110
111 $transliterator = [];
112 if ($locale && ('de' === $locale || str_starts_with($locale, 'de_'))) {
113 // Use the shortcut for German in UnicodeString::ascii() if possible (faster and no requirement on intl)
114 $transliterator = ['de-ASCII'];
115 } elseif (\function_exists('transliterator_transliterate') && $locale) {
116 $transliterator = (array) $this->createTransliterator($locale);
117 }
118
119 if ($emojiTransliterator = $this->createEmojiTransliterator($locale)) {
120 $transliterator[] = $emojiTransliterator;
121 }
122
123 if ($this->symbolsMap instanceof \Closure) {
124 // If the symbols map is passed as a closure, there is no need to fallback to the parent locale
125 // as the closure can just provide substitutions for all locales of interest.
126 $symbolsMap = $this->symbolsMap;
127 array_unshift($transliterator, static fn ($s) => $symbolsMap($s, $locale));
128 }
129
130 $unicodeString = (new UnicodeString($string))->ascii($transliterator);
131
132 if (\is_array($this->symbolsMap)) {
133 $map = null;
134 if (isset($this->symbolsMap[$locale])) {
135 $map = $this->symbolsMap[$locale];
136 } else {
137 $parent = self::getParentLocale($locale);
138 if ($parent && isset($this->symbolsMap[$parent])) {
139 $map = $this->symbolsMap[$parent];
140 }
141 }
142 if ($map) {
143 foreach ($map as $char => $replace) {
144 $unicodeString = $unicodeString->replace($char, ' '.$replace.' ');
145 }
146 }
147 }
148
149 return $unicodeString
150 ->replaceMatches('/[^A-Za-z0-9]++/', $separator)
151 ->trim($separator)
152 ;
153 }

155 private function createTransliterator(string $locale): ?\Transliterator
156 {
157 if (\array_key_exists($locale, $this->transliterators)) {
158 return $this->transliterators[$locale];
159 }
160
161 // Exact locale supported, cache and return
162 if ($id = self::LOCALE_TO_TRANSLITERATOR_ID[$locale] ?? null) {
163 return $this->transliterators[$locale] = \Transliterator::create($id.'/BGN') ?? \Transliterator::create($id);
164 }
165
166 // Locale not supported and no parent, fallback to any-latin
167 if (!$parent = self::getParentLocale($locale)) {
168 return $this->transliterators[$locale] = null;
169 }
170
171 // Try to use the parent locale (ie. try "de" for "de_AT") and cache both locales
172 if ($id = self::LOCALE_TO_TRANSLITERATOR_ID[$parent] ?? null) {
173 $transliterator = \Transliterator::create($id.'/BGN') ?? \Transliterator::create($id);
174 }
175
176 return $this->transliterators[$locale] = $this->transliterators[$parent] = $transliterator ?? null;
177 }

179 private function createEmojiTransliterator(?string $locale): ?EmojiTransliterator
180 {

188 try {
189 return EmojiTransliterator::create("emoji-$locale");
190 } catch (\IntlException) {

193 }

198 private static function getParentLocale(?string $locale): ?string

201 return null;
202 }
