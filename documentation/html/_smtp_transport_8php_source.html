206 protected function doSend(SentMessage $message): void
207 {
208 if (microtime(true) - $this->lastMessageTime > $this->pingThreshold) {
209 $this->ping();
210 }
211
212 if (!$this->started) {
213 $this->start();
214 }
215
216 try {
217 $envelope = $message->getEnvelope();
218 $this->doMailFromCommand($envelope->getSender()->getEncodedAddress());
219 foreach ($envelope->getRecipients() as $recipient) {
220 $this->doRcptToCommand($recipient->getEncodedAddress());
221 }
222
223 $this->executeCommand("DATA\r\n", [354]);
224 try {
225 foreach (AbstractStream::replace("\r\n.", "\r\n..", $message->toIterable()) as $chunk) {
226 $this->stream->write($chunk, false);
227 }
228 $this->stream->flush();
229 } catch (TransportExceptionInterface $e) {
230 throw $e;
231 } catch (\Exception $e) {
232 $this->stream->terminate();
233 $this->started = false;
234 $this->getLogger()->debug(sprintf('Email transport "%s" stopped', __CLASS__));
235 throw $e;
236 }
237 $this->mtaResult = $this->executeCommand("\r\n.\r\n", [250]);
238 $message->appendDebug($this->stream->getDebug());
239 $this->lastMessageTime = microtime(true);
240 } catch (TransportExceptionInterface $e) {
241 $e->appendDebug($this->stream->getDebug());

325 private function assertResponseCode(string $response, array $codes): void
326 {
327 if (!$codes) {
328 throw new LogicException('You must set the expected response code.');
329 }
330
331 [$code] = sscanf($response, '%3d');
332 $valid = \in_array($code, $codes);
333
334 if (!$valid || !$response) {
335 $codeStr = $code ? sprintf('code "%s"', $code) : 'empty code';
336 $responseStr = $response ? sprintf(', with message "%s"', trim($response)) : '';
337
338 throw new TransportException(sprintf('Expected response code "%s" but got ', implode('/', $codes)).$codeStr.$responseStr.'.', $code ?: 0);
339 }

353 private function checkRestartThreshold(): void
354 {
355 // when using sendmail via non-interactive mode, the transport is never "started"
356 if (!$this->started) {
