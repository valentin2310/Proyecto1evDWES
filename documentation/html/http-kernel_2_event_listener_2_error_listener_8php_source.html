59 $throwable = $event->getThrowable();
60 $logLevel = $this->resolveLogLevel($throwable);
61
62 foreach ($this->exceptionsMapping as $class => $config) {

64 continue;
65 }

72 }

98 public function onKernelException(ExceptionEvent $event)
99 {
100 if (null === $this->controller) {
101 return;
102 }
103
104 $throwable = $event->getThrowable();
105 $request = $this->duplicateRequest($throwable, $event->getRequest());

107 try {
108 $response = $event->getKernel()->handle($request, HttpKernelInterface::SUB_REQUEST, false);

191 private function resolveLogLevel(\Throwable $throwable): string
192 {
193 foreach ($this->exceptionsMapping as $class => $config) {
194 if ($throwable instanceof $class && $config['log_level']) {
195 return $config['log_level'];
196 }
197 }
198
199 $class = new \ReflectionClass($throwable);
200
201 do {
202 if ($attributes = $class->getAttributes(WithLogLevel::class)) {
204 $instance = $attributes[0]->newInstance();
205
206 return $instance->level;
207 }
208 } while ($class = $class->getParentClass());
209
210 if (!$throwable instanceof HttpExceptionInterface || $throwable->getStatusCode() >= 500) {
211 return LogLevel::CRITICAL;
212 }
213
214 return LogLevel::ERROR;
