440 private function generateBody(): AbstractPart

474 private function prepareParts(): ?array
475 {
476 $names = [];
477 $htmlPart = null;
478 $html = $this->html;
479 if (null !== $html) {
480 $htmlPart = new TextPart($html, $this->htmlCharset, 'html');
481 $html = $htmlPart->getBody();
482
483 $regexes = [
484 '<img\s+[^>]*src\s*=\s*(?:([\'"])cid:(.+?)\\1|cid:([^>\s]+))',
485 '<\w+\s+[^>]*background\s*=\s*(?:([\'"])cid:(.+?)\\1|cid:([^>\s]+))',
486 ];
487 $tmpMatches = [];
488 foreach ($regexes as $regex) {
489 preg_match_all('/'.$regex.'/i', $html, $tmpMatches);
490 $names = array_merge($names, $tmpMatches[2], $tmpMatches[3]);
491 }
492 $names = array_filter(array_unique($names));
493 }
494
495 $otherParts = $relatedParts = [];
496 foreach ($this->attachments as $part) {
497 foreach ($names as $name) {
498 if ($name !== $part->getName() && (!$part->hasContentId() || $name !== $part->getContentId())) {
499 continue;
500 }
501 if (isset($relatedParts[$name])) {
502 continue 2;
503 }
504
505 if ($name !== $part->getContentId()) {
506 $html = str_replace('cid:'.$name, 'cid:'.$part->getContentId(), $html, $count);
507 }
508 $relatedParts[$name] = $part;
509 $part->setName($part->getContentId())->asInline();
510
511 continue 2;
512 }
513
514 $otherParts[] = $part;
515 }
516 if (null !== $htmlPart) {
517 $htmlPart = new TextPart($html, $this->htmlCharset, 'html');
518 }
