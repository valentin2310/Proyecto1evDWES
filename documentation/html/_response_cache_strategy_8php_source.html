60 public function add(Response $response)
61 {
62 ++$this->embeddedResponses;
63
64 foreach (self::OVERRIDE_DIRECTIVES as $directive) {
65 if ($response->headers->hasCacheControlDirective($directive)) {
66 $this->flagDirectives[$directive] = true;
67 }
68 }
69
70 foreach (self::INHERIT_DIRECTIVES as $directive) {
71 if (false !== $this->flagDirectives[$directive]) {
72 $this->flagDirectives[$directive] = $response->headers->hasCacheControlDirective($directive);
73 }
74 }
75
76 $age = $response->getAge();
77 $this->age = max($this->age, $age);
78
79 if ($this->willMakeFinalResponseUncacheable($response)) {
80 $this->isNotCacheableResponseEmbedded = true;
81
82 return;
83 }
84
85 $isHeuristicallyCacheable = $response->headers->hasCacheControlDirective('public');
86 $maxAge = $response->headers->hasCacheControlDirective('max-age') ? (int) $response->headers->getCacheControlDirective('max-age') : null;
87 $this->storeRelativeAgeDirective('max-age', $maxAge, $age, $isHeuristicallyCacheable);
88 $sharedMaxAge = $response->headers->hasCacheControlDirective('s-maxage') ? (int) $response->headers->getCacheControlDirective('s-maxage') : $maxAge;
89 $this->storeRelativeAgeDirective('s-maxage', $sharedMaxAge, $age, $isHeuristicallyCacheable);
90
91 $expires = $response->getExpires();
92 $expires = null !== $expires ? (int) $expires->format('U') - (int) $response->getDate()->format('U') : null;
93 $this->storeRelativeAgeDirective('expires', $expires >= 0 ? $expires : null, 0, $isHeuristicallyCacheable);
94
95 if (false !== $this->lastModified) {
96 $lastModified = $response->getLastModified();
97 $this->lastModified = $lastModified ? max($this->lastModified, $lastModified) : false;
98 }

104 public function update(Response $response)
105 {
106 // if we have no embedded Response, do nothing
107 if (0 === $this->embeddedResponses) {
108 return;
109 }
110
111 // Remove Etag since it cannot be merged from embedded responses.
112 $response->setEtag(null);
113
114 $this->add($response);
115
116 $response->headers->set('Age', $this->age);
117
118 if ($this->isNotCacheableResponseEmbedded) {
119 $response->setLastModified(null);
120
121 if ($this->flagDirectives['no-store']) {
122 $response->headers->set('Cache-Control', 'no-cache, no-store, must-revalidate');
123 } else {
124 $response->headers->set('Cache-Control', 'no-cache, must-revalidate');
125 }
126
127 return;
128 }
129
130 $response->setLastModified($this->lastModified ?: null);
131
132 $flags = array_filter($this->flagDirectives);
133
134 if (isset($flags['must-revalidate'])) {
135 $flags['no-cache'] = true;
136 }
137
138 $response->headers->set('Cache-Control', implode(', ', array_keys($flags)));
139
140 $maxAge = null;
141
142 if (is_numeric($this->ageDirectives['max-age'])) {
143 $maxAge = $this->ageDirectives['max-age'] + $this->age;
144 $response->headers->addCacheControlDirective('max-age', $maxAge);
145 }
146
147 if (is_numeric($this->ageDirectives['s-maxage'])) {
148 $sMaxage = $this->ageDirectives['s-maxage'] + $this->age;
149
150 if ($maxAge !== $sMaxage) {
151 $response->headers->addCacheControlDirective('s-maxage', $sMaxage);
152 }
153 }
154
155 if (is_numeric($this->ageDirectives['expires'])) {
156 $date = clone $response->getDate();
157 $date = $date->modify('+'.($this->ageDirectives['expires'] + $this->age).' seconds');
158 $response->setExpires($date);
159 }

167 private function willMakeFinalResponseUncacheable(Response $response): bool
168 {
169 // RFC2616: A response received with a status code of 200, 203, 300, 301 or 410
170 // MAY be stored by a cache [โฆ] unless a cache-control directive prohibits caching.
171 if ($response->headers->hasCacheControlDirective('no-cache')
172 || $response->headers->hasCacheControlDirective('no-store')
173 ) {
174 return true;
175 }
176
177 // Etag headers cannot be merged, they render the response uncacheable
178 // by default (except if the response also has max-age etc.).
179 if (null === $response->getEtag() && \in_array($response->getStatusCode(), [200, 203, 300, 301, 410])) {
180 return false;
181 }
182
183 // RFC2616: A response received with any other status code (e.g. status codes 302 and 307)
184 // MUST NOT be returned in a reply to a subsequent request unless there are
185 // cache-control directives or another header(s) that explicitly allow it.
186 $cacheControl = ['max-age', 's-maxage', 'must-revalidate', 'proxy-revalidate', 'public', 'private'];
187 foreach ($cacheControl as $key) {
188 if ($response->headers->hasCacheControlDirective($key)) {
189 return false;
190 }
191 }
192
193 if ($response->headers->has('Expires')) {
194 return false;
195 }
196
197 return true;
198 }

218 private function storeRelativeAgeDirective(string $directive, ?int $value, int $age, bool $isHeuristicallyCacheable): void
219 {
220 if (null === $value) {
221 if ($isHeuristicallyCacheable) {
222 /*
223 * See https://datatracker.ietf.org/doc/html/rfc7234#section-4.2.2
224 * This particular response does not require maximum lifetime; heuristics might be applied.
225 * Other responses, however, might have more stringent requirements on maximum lifetime.
226 * So, return early here so that the final response can have the more limiting value set.
227 */
228 return;
229 }
230 $this->ageDirectives[$directive] = false;
231 }
232
233 if (false !== $this->ageDirectives[$directive]) {
234 $value -= $age;
235 $this->ageDirectives[$directive] = null !== $this->ageDirectives[$directive] ? min($this->ageDirectives[$directive], $value) : $value;
236 }
