139 private function detectUnmatchedLineEndings(array $diff): bool
140 {
141 $newLineBreaks = ['' => true];
142 $oldLineBreaks = ['' => true];
143
144 foreach ($diff as $entry) {
145 if (self::OLD === $entry[1]) {
146 $ln = $this->getLinebreak($entry[0]);
147 $oldLineBreaks[$ln] = true;
148 $newLineBreaks[$ln] = true;
149 } elseif (self::ADDED === $entry[1]) {
150 $newLineBreaks[$this->getLinebreak($entry[0])] = true;
151 } elseif (self::REMOVED === $entry[1]) {
152 $oldLineBreaks[$this->getLinebreak($entry[0])] = true;
153 }
154 }
155
156 // if either input or output is a single line without breaks than no warning should be raised
157 if (['' => true] === $newLineBreaks || ['' => true] === $oldLineBreaks) {
158 return false;
159 }
160
161 // two-way compare
162 foreach ($newLineBreaks as $break => $set) {
163 if (!isset($oldLineBreaks[$break])) {
164 return true;
165 }
166 }
167
168 foreach ($oldLineBreaks as $break => $set) {
169 if (!isset($newLineBreaks[$break])) {
170 return true;
171 }
172 }
173
174 return false;
175 }

200 private static function getArrayDiffParted(array &$from, array &$to): array
201 {
202 $start = [];
203 $end = [];
204
205 reset($to);
206
207 foreach ($from as $k => $v) {
208 $toK = key($to);
209
210 if ($toK === $k && $v === $to[$k]) {
211 $start[$k] = $v;
212
213 unset($from[$k], $to[$k]);
214 } else {
215 break;
216 }
217 }
218
219 end($from);
220 end($to);
221
222 do {
223 $fromK = key($from);
224 $toK = key($to);
225
226 if (null === $fromK || null === $toK || current($from) !== current($to)) {
227 break;
228 }
229
230 prev($from);
231 prev($to);
232
233 $end = [$fromK => $from[$fromK]] + $end;
234 unset($from[$fromK], $to[$toK]);
235 } while (true);
236
237 return [$from, $to, $start, $end];
238 }
