266 // step which compiles the component Blade tags into @component directives
267 // that may be used by Blade. Then we should call any other precompilers.
268 $value = $this->compileComponentTags(

519 protected function compileStatements($template)
520 {

522
523 $offset = 0;
524
525 for ($i = 0; isset($matches[0][$i]); $i++) {
526 $match = [
527 $matches[0][$i],
528 $matches[1][$i],
529 $matches[2][$i],
530 $matches[3][$i] ?: null,
531 $matches[4][$i] ?: null,
532 ];
533
534 // Here we check to see if we have properly found the closing parenthesis by
535 // regex pattern or not, and will recursively continue on to the next ")"
536 // then check again until the tokenizer confirms we find the right one.
537 while (isset($match[4]) &&
538 Str::endsWith($match[0], ')') &&
539 ! $this->hasEvenNumberOfParentheses($match[0])) {
540 if (($after = Str::after($template, $match[0])) === $template) {
541 break;
542 }
543
544 $rest = Str::before($after, ')');
545
546 if (isset($matches[0][$i + 1]) && Str::contains($rest.')', $matches[0][$i + 1])) {
547 unset($matches[0][$i + 1]);
548 $i++;
549 }
550
551 $match[0] = $match[0].$rest.')';
552 $match[3] = $match[3].$rest.')';
553 $match[4] = $match[4].$rest;
554 }
555
556 [$template, $offset] = $this->replaceFirstStatement(
557 $match[0],
558 $this->compileStatement($match),
559 $template,
560 $offset

602 protected function hasEvenNumberOfParentheses(string $expression)
603 {
604 $tokens = token_get_all('<?php '.$expression);
605
606 if (Arr::last($tokens) !== ')') {
607 return false;
608 }
609
610 $opening = 0;
611 $closing = 0;
612
613 foreach ($tokens as $token) {
614 if ($token == ')') {
615 $closing++;
616 } elseif ($token == '(') {
617 $opening++;
618 }
619 }
620
621 return $opening === $closing;

630 protected function compileStatement($match)
631 {
632 if (str_contains($match[1], '@')) {
633 $match[0] = isset($match[3]) ? $match[1].$match[3] : $match[1];
634 } elseif (isset($this->customDirectives[$match[1]])) {
