33 public static function unescapeAndEncode(string $uri): string
34 {
35 // Optimization: if the URL only includes characters we know will be kept as-is, then just return the URL as-is.
36 if (\preg_match('/^[A-Za-z0-9~!@#$&*()\-_=+;:,.\/?]+$/', $uri)) {
37 return $uri;
38 }
39
40 if (! \mb_check_encoding($uri, 'UTF-8')) {
41 throw new UnexpectedEncodingException('Unexpected encoding - UTF-8 or ASCII was expected');
42 }
43
44 $result = '';
45
46 $chars = \mb_str_split($uri, 1, 'UTF-8');
47
48 $l = \count($chars);
49 for ($i = 0; $i < $l; $i++) {
50 $code = $chars[$i];
51 if ($code === '%' && $i + 2 < $l) {
52 if (\preg_match('/^[0-9a-f]{2}$/i', $chars[$i + 1] . $chars[$i + 2]) === 1) {
53 $result .= '%' . $chars[$i + 1] . $chars[$i + 2];
54 $i += 2;
55 continue;
56 }
57 }
58
59 if (\ord($code) < 128) {

62 }
