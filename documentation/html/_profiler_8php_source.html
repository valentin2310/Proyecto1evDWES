138 public function collect(Request $request, Response $response, \Throwable $exception = null): ?Profile
139 {
140 if (false === $this->enabled) {
141 return null;
142 }
143
144 $profile = new Profile(substr(hash('sha256', uniqid(mt_rand(), true)), 0, 6));
145 $profile->setTime(time());
146 $profile->setUrl($request->getUri());
147 $profile->setMethod($request->getMethod());
148 $profile->setStatusCode($response->getStatusCode());
149 try {
150 $profile->setIp($request->getClientIp());
151 } catch (ConflictingHeadersException) {
152 $profile->setIp('Unknown');
153 }
154
155 if ($prevToken = $response->headers->get('X-Debug-Token')) {
156 $response->headers->set('X-Previous-Debug-Token', $prevToken);
157 }
158
159 $response->headers->set('X-Debug-Token', $profile->getToken());
160
161 foreach ($this->collectors as $collector) {
162 $collector->collect($request, $response, $exception);
163
164 // we need to clone for sub-requests
165 $profile->addCollector(clone $collector);
166 }
167
168 return $profile;
169 }
