109 protected function resolve($name)
110 {
111 $config = $this->getConfig($name);
112
113 if (is_null($config)) {
114 throw new InvalidArgumentException("Mailer [{$name}] is not defined.");
115 }
116
117 // Once we have created the mailer instance we will set a container instance
118 // on the mailer. This allows us to resolve mailer classes via containers
119 // for maximum testability on said classes instead of passing Closures.
120 $mailer = new Mailer(
121 $name,
122 $this->app['view'],
123 $this->createSymfonyTransport($config),
124 $this->app['events']
125 );
126
127 if ($this->app->bound('queue')) {
128 $mailer->setQueue($this->app['queue']);
129 }
130
131 // Next we will set all of the global addresses on this mailer, which allows
132 // for easy unification of all "from" addresses as well as easy debugging
133 // of sent messages since these will be sent to a single email address.
134 foreach (['from', 'reply_to', 'to', 'return_path'] as $type) {
135 $this->setGlobalAddress($mailer, $config, $type);
136 }
137
138 return $mailer;

153 // format and use the "mail.driver" configuration option instead for BC.
154 $transport = $config['transport'] ?? $this->app['config']['mail.driver'];
