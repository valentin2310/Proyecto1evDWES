230 protected function loadSchemaState()
231 {
232 $connection = $this->migrator->resolveConnection($this->option('database'));
233
234 // First, we will make sure that the connection supports schema loading and that
235 // the schema file exists before we proceed any further. If not, we will just
236 // continue with the standard migration operation as normal without errors.
237 if ($connection instanceof SqlServerConnection ||
238 ! is_file($path = $this->schemaPath($connection))) {
239 return;
240 }
241
242 $this->components->info('Loading stored database schemas.');
243
244 $this->components->task($path, function () use ($connection, $path) {
245 // Since the schema file will create the "migrations" table and reload it to its
246 // proper state, we need to delete it here so we don't get an error that this
247 // table already exists when the stored database schema file gets executed.
248 $this->migrator->deleteRepository();
249
250 $connection->getSchemaState()->handleOutputUsing(function ($type, $buffer) {
251 $this->output->write($buffer);
252 })->load($path);
253 });
254
255 $this->newLine();
256
257 // Finally, we will fire an event that this schema has been loaded so developers
258 // can perform any post schema load tasks that are necessary in listeners for
259 // this event, which may seed the database tables with some necessary data.
260 $this->dispatcher->dispatch(
261 new SchemaLoaded($connection, $path)
262 );
263 }
