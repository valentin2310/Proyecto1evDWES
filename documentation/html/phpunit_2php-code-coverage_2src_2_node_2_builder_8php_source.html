202 private function reducePaths(ProcessedCodeCoverageData $coverage): string
203 {
204 if (empty($coverage->coveredFiles())) {
205 return '.';
206 }
207
208 $commonPath = '';
209 $paths = $coverage->coveredFiles();
210
211 if (count($paths) === 1) {
212 $commonPath = dirname($paths[0]) . DIRECTORY_SEPARATOR;
213 $coverage->renameFile($paths[0], basename($paths[0]));
214
215 return $commonPath;
216 }
217
218 $max = count($paths);
219
220 for ($i = 0; $i < $max; $i++) {
221 // strip phar:// prefixes
222 if (str_starts_with($paths[$i], 'phar://')) {
223 $paths[$i] = substr($paths[$i], 7);
224 $paths[$i] = str_replace('/', DIRECTORY_SEPARATOR, $paths[$i]);
225 }
226 $paths[$i] = explode(DIRECTORY_SEPARATOR, $paths[$i]);
227
228 if (empty($paths[$i][0])) {
229 $paths[$i][0] = DIRECTORY_SEPARATOR;
230 }
231 }
232
233 $done = false;
234 $max = count($paths);
235
236 while (!$done) {
237 for ($i = 0; $i < $max - 1; $i++) {
238 if (!isset($paths[$i][0]) ||
239 !isset($paths[$i + 1][0]) ||
240 $paths[$i][0] !== $paths[$i + 1][0]) {
241 $done = true;
242
243 break;
244 }
245 }
246
247 if (!$done) {
248 $commonPath .= $paths[0][0];
249
250 if ($paths[0][0] !== DIRECTORY_SEPARATOR) {
251 $commonPath .= DIRECTORY_SEPARATOR;
252 }
253
254 for ($i = 0; $i < $max; $i++) {
255 array_shift($paths[$i]);
256 }
257 }
258 }
259
260 $original = $coverage->coveredFiles();
261 $max = count($original);
262
263 for ($i = 0; $i < $max; $i++) {
264 $coverage->renameFile($original[$i], implode(DIRECTORY_SEPARATOR, $paths[$i]));
265 }
266
267 return substr($commonPath, 0, -1);
