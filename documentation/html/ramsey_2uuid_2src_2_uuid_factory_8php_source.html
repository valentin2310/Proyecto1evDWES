374 public function uuid6(?Hexadecimal $node = null, ?int $clockSeq = null): UuidInterface
375 {
376 $nodeHex = $node ? $node->toString() : null;
377 $bytes = $this->timeGenerator->generate($nodeHex, $clockSeq);
378
379 // Rearrange the bytes, according to the UUID version 6 specification.
380 $v6 = $bytes[6] . $bytes[7] . $bytes[4] . $bytes[5]
381 . $bytes[0] . $bytes[1] . $bytes[2] . $bytes[3];
382 $v6 = bin2hex($v6);
383
384 // Drop the first four bits, while adding an empty four bits for the
385 // version field. This allows us to reconstruct the correct time from
386 // the bytes of this UUID.
387 $v6Bytes = hex2bin(substr($v6, 1, 12) . '0' . substr($v6, -3));
388 $v6Bytes .= substr($bytes, 8);
389
390 return $this->uuidFromBytesAndVersion($v6Bytes, Uuid::UUID_TYPE_REORDERED_TIME);
