34 if (!$remoteAddr || !IpUtils::checkIp($remoteAddr, $trustedProxies)) {

36 'FORWARDED' => $trustedHeaderSet & Request::HEADER_FORWARDED,
37 'X_FORWARDED_FOR' => $trustedHeaderSet & Request::HEADER_X_FORWARDED_FOR,

42 ];
43 foreach (array_filter($trustedHeaders) as $name => $key) {
44 $request->headers->remove($name);
45 $request->server->remove('HTTP_'.$name);

49 // compute trusted values, taking any trusted proxies into account
50 $trustedIps = [];

52 foreach (array_reverse($request->getClientIps()) as $ip) {
53 $trustedIps[] = $ip;
54 $trustedValues[] = sprintf('for="%s"', $ip);
55 }
56 if ($ip !== $remoteAddr) {
57 $trustedIps[] = $remoteAddr;

66 }
67 if (Request::HEADER_X_FORWARDED_FOR & $trustedHeaderSet) {

70 } elseif (!(Request::HEADER_FORWARDED & $trustedHeaderSet)) {
71 Request::setTrustedProxies($trustedProxies, $trustedHeaderSet | Request::HEADER_X_FORWARDED_FOR);
