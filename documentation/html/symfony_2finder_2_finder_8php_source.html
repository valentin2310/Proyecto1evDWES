629 public function in(string|array $dirs): static
630 {
631 $resolvedDirs = [];
632
633 foreach ((array) $dirs as $dir) {
634 if (is_dir($dir)) {
635 $resolvedDirs[] = [$this->normalizeDir($dir)];
636 } elseif ($glob = glob($dir, (\defined('GLOB_BRACE') ? \GLOB_BRACE : 0) | \GLOB_ONLYDIR | \GLOB_NOSORT)) {
637 sort($glob);
638 $resolvedDirs[] = array_map($this->normalizeDir(...), $glob);
639 } else {
640 throw new DirectoryNotFoundException(sprintf('The "%s" directory does not exist.', $dir));
641 }
642 }
643
644 $this->dirs = array_merge($this->dirs, ...$resolvedDirs);
645
646 return $this;
647 }

658 public function getIterator(): \Iterator
659 {
660 if (0 === \count($this->dirs) && 0 === \count($this->iterators)) {
661 throw new \LogicException('You must call one of in() or append() methods before iterating over a Finder.');
662 }
663
664 if (1 === \count($this->dirs) && 0 === \count($this->iterators)) {
665 $iterator = $this->searchInDirectory($this->dirs[0]);
666
667 if ($this->sort || $this->reverseSorting) {
668 $iterator = (new Iterator\SortableIterator($iterator, $this->sort, $this->reverseSorting))->getIterator();
669 }
670
671 return $iterator;
672 }
673
674 $iterator = new \AppendIterator();
675 foreach ($this->dirs as $dir) {
676 $iterator->append(new \IteratorIterator(new LazyIterator(fn () => $this->searchInDirectory($dir))));
677 }
678
679 foreach ($this->iterators as $it) {

739 private function searchInDirectory(string $dir): \Iterator
740 {
741 $exclude = $this->exclude;
742 $notPaths = $this->notPaths;
743
744 if (static::IGNORE_VCS_FILES === (static::IGNORE_VCS_FILES & $this->ignore)) {
745 $exclude = array_merge($exclude, self::$vcsPatterns);
746 }
747
748 if (static::IGNORE_DOT_FILES === (static::IGNORE_DOT_FILES & $this->ignore)) {
749 $notPaths[] = '#(^|/)\..+(/|$)#';
750 }
751
752 $minDepth = 0;
753 $maxDepth = \PHP_INT_MAX;
754
755 foreach ($this->depths as $comparator) {
756 switch ($comparator->getOperator()) {
757 case '>':
758 $minDepth = $comparator->getTarget() + 1;
759 break;
760 case '>=':
761 $minDepth = $comparator->getTarget();
762 break;
763 case '<':
764 $maxDepth = $comparator->getTarget() - 1;
765 break;
766 case '<=':
767 $maxDepth = $comparator->getTarget();
768 break;
769 default:
770 $minDepth = $maxDepth = $comparator->getTarget();
771 }
772 }
773
774 $flags = \RecursiveDirectoryIterator::SKIP_DOTS;
775
776 if ($this->followLinks) {
777 $flags |= \RecursiveDirectoryIterator::FOLLOW_SYMLINKS;
778 }
779
780 $iterator = new Iterator\RecursiveDirectoryIterator($dir, $flags, $this->ignoreUnreadableDirs);
781
782 if ($exclude) {
783 $iterator = new Iterator\ExcludeDirectoryFilterIterator($iterator, $exclude);
784 }
785
786 $iterator = new \RecursiveIteratorIterator($iterator, \RecursiveIteratorIterator::SELF_FIRST);
787
788 if ($minDepth > 0 || $maxDepth < \PHP_INT_MAX) {
789 $iterator = new Iterator\DepthRangeFilterIterator($iterator, $minDepth, $maxDepth);
790 }
791
792 if ($this->mode) {
793 $iterator = new Iterator\FileTypeFilterIterator($iterator, $this->mode);
794 }
795
796 if ($this->names || $this->notNames) {
797 $iterator = new Iterator\FilenameFilterIterator($iterator, $this->names, $this->notNames);
798 }
799
800 if ($this->contains || $this->notContains) {
801 $iterator = new Iterator\FilecontentFilterIterator($iterator, $this->contains, $this->notContains);
802 }
803
804 if ($this->sizes) {
805 $iterator = new Iterator\SizeRangeFilterIterator($iterator, $this->sizes);
806 }
807
808 if ($this->dates) {
809 $iterator = new Iterator\DateRangeFilterIterator($iterator, $this->dates);
810 }
811
812 if ($this->filters) {
813 $iterator = new Iterator\CustomFilterIterator($iterator, $this->filters);
814 }
815
816 if ($this->paths || $notPaths) {
817 $iterator = new Iterator\PathFilterIterator($iterator, $this->paths, $notPaths);
818 }
819
820 if (static::IGNORE_VCS_IGNORED_FILES === (static::IGNORE_VCS_IGNORED_FILES & $this->ignore)) {
821 $iterator = new Iterator\VcsIgnoredFilterIterator($iterator, $dir);
822 }
823
824 return $iterator;
825 }
