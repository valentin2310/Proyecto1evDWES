223 private static function methodParametersForDeclaration(ReflectionMethod $method): string
224 {
225 $parameters = [];
226 $types = (new ReflectionMapper)->fromParameterTypes($method);
227
228 foreach ($method->getParameters() as $i => $parameter) {
229 $name = '$' . $parameter->getName();
230
231 /* Note: PHP extensions may use empty names for reference arguments
232 * or "..." for methods taking a variable number of arguments.
233 */
234 if ($name === '$' || $name === '$...') {
235 $name = '$arg' . $i;
236 }
237
238 $default = '';
239 $reference = '';
240 $typeDeclaration = '';
241
242 if (!$types[$i]->type()->isUnknown()) {
243 $typeDeclaration = $types[$i]->type()->asString() . ' ';
244 }
245
246 if ($parameter->isPassedByReference()) {
247 $reference = '&';
248 }
249
250 if ($parameter->isVariadic()) {
251 $name = '...' . $name;
252 } elseif ($parameter->isDefaultValueAvailable()) {
253 $default = ' = ' . self::exportDefaultValue($parameter);
254 } elseif ($parameter->isOptional()) {
255 $default = ' = null';
256 }
257
258 $parameters[] = $typeDeclaration . $reference . $name . $default;
259 }

261 return implode(', ', $parameters);
