118 public function get(string $name): ?Route
119 {
120 $visited = [];
121 while (null !== $alias = $this->aliases[$name] ?? null) {
122 if (false !== $searchKey = array_search($name, $visited)) {
123 $visited[] = $name;
124
125 throw new RouteCircularReferenceException($name, \array_slice($visited, $searchKey));
126 }
127
128 if ($alias->isDeprecated()) {

131 trigger_deprecation($deprecation['package'], $deprecation['version'], $deprecation['message']);
132 }
133
134 $visited[] = $name;
135 $name = $alias->getId();
136 }
