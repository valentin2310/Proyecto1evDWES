218 private static function parseFilesize(string $size): int|float
219 {
220 if ('' === $size) {
221 return 0;
222 }
223
224 $size = strtolower($size);
225
226 $max = ltrim($size, '+');
227 if (str_starts_with($max, '0x')) {
228 $max = \intval($max, 16);
229 } elseif (str_starts_with($max, '0')) {
230 $max = \intval($max, 8);
231 } else {
232 $max = (int) $max;
233 }
234
235 switch (substr($size, -1)) {
236 case 't': $max *= 1024;
237 // no break
238 case 'g': $max *= 1024;
239 // no break
240 case 'm': $max *= 1024;
241 // no break
242 case 'k': $max *= 1024;
243 }
244
