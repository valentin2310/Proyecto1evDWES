508 public function fill(array $attributes)
509 {
510 $totallyGuarded = $this->totallyGuarded();
511
512 $fillable = $this->fillableFromArray($attributes);
513
514 foreach ($fillable as $key => $value) {
515 // The developers may choose to place some attributes in the "fillable" array
516 // which means only those attributes may be set through mass assignment to
517 // the model, and all others will just get ignored for security reasons.
518 if ($this->isFillable($key)) {
519 $this->setAttribute($key, $value);
520 } elseif ($totallyGuarded || static::preventsSilentlyDiscardingAttributes()) {
521 if (isset(static::$discardedAttributeViolationCallback)) {
522 call_user_func(static::$discardedAttributeViolationCallback, $this, [$key]);
523 } else {
524 throw new MassAssignmentException(sprintf(
525 'Add [%s] to fillable property to allow mass assignment on [%s].',
526 $key, get_class($this)

528 }
529 }
530 }
