112 public function addContent(string $input, bool $newline = true): int
113 {
114 $width = $this->terminal->getWidth();
115 $lines = explode(\PHP_EOL, $input);
116 $linesAdded = 0;
117 $count = \count($lines) - 1;
118 foreach ($lines as $i => $lineContent) {
119 // re-add the line break (that has been removed in the above `explode()` for
120 // - every line that is not the last line
121 // - if $newline is required, also add it to the last line
122 if ($i < $count || $newline) {
123 $lineContent .= \PHP_EOL;
124 }
125
126 // skip line if there is no text (or newline for that matter)
127 if ('' === $lineContent) {
128 continue;
129 }
130
131 // For the first line, check if the previous line (last entry of `$this->content`)
132 // needs to be continued (i.e. does not end with a line break).
133 if (0 === $i
134 && (false !== $lastLine = end($this->content))
135 && !str_ends_with($lastLine, \PHP_EOL)
136 ) {
137 // deduct the line count of the previous line
138 $this->lines -= (int) ceil($this->getDisplayLength($lastLine) / $width) ?: 1;
139 // concatenate previous and new line
140 $lineContent = $lastLine.$lineContent;
141 // replace last entry of `$this->content` with the new expanded line
142 array_splice($this->content, -1, 1, $lineContent);
143 } else {
144 // otherwise just add the new content
145 $this->content[] = $lineContent;
146 }
147
148 $linesAdded += (int) ceil($this->getDisplayLength($lineContent) / $width) ?: 1;
149 }
150
151 $this->lines += $linesAdded;
152
153 return $linesAdded;
154 }

168 protected function doWrite(string $message, bool $newline)
169 {
170 // Simulate newline behavior for consistent output formatting, avoiding extra logic
171 if (!$newline && str_ends_with($message, \PHP_EOL)) {
172 $message = substr($message, 0, -\strlen(\PHP_EOL));
173 $newline = true;
174 }
175
176 if (!$this->isDecorated()) {
177 parent::doWrite($message, $newline);
178
179 return;
180 }
181
182 // Check if the previous line (last entry of `$this->content`) needs to be continued
183 // (i.e. does not end with a line break). In which case, it needs to be erased first.
184 $linesToClear = $deleteLastLine = ($lastLine = end($this->content) ?: '') && !str_ends_with($lastLine, \PHP_EOL) ? 1 : 0;
185
186 $linesAdded = $this->addContent($message, $newline);
187
188 if ($lineOverflow = $this->maxHeight > 0 && $this->lines > $this->maxHeight) {
189 // on overflow, clear the whole section and redraw again (to remove the first lines)
190 $linesToClear = $this->maxHeight;
191 }
192
193 $erasedContent = $this->popStreamContentUntilCurrentSection($linesToClear);
194
195 if ($lineOverflow) {
196 // redraw existing lines of the section
197 $previousLinesOfSection = \array_slice($this->content, $this->lines - $this->maxHeight, $this->maxHeight - $linesAdded);
198 parent::doWrite(implode('', $previousLinesOfSection), false);
199 }
200
201 // if the last line was removed, re-print its content together with the new content.
202 // otherwise, just print the new content.
203 parent::doWrite($deleteLastLine ? $lastLine.$message : $message, true);
204 parent::doWrite($erasedContent, false);
205 }

211 private function popStreamContentUntilCurrentSection(int $numberOfLinesToClearFromCurrentSection = 0): string
212 {
213 $numberOfLinesToClear = $numberOfLinesToClearFromCurrentSection;
214 $erasedContent = [];
215
216 foreach ($this->sections as $section) {
217 if ($section === $this) {
218 break;
219 }
220
221 $numberOfLinesToClear += $section->maxHeight ? min($section->lines, $section->maxHeight) : $section->lines;
222 if ('' !== $sectionContent = $section->getVisibleContent()) {
223 if (!str_ends_with($sectionContent, \PHP_EOL)) {
224 $sectionContent .= \PHP_EOL;
225 }
226 $erasedContent[] = $sectionContent;
227 }
228 }
229
230 if ($numberOfLinesToClear > 0) {
231 // move cursor up n lines
232 parent::doWrite(sprintf("\x1b[%dA", $numberOfLinesToClear), false);
233 // erase to end of screen
234 parent::doWrite("\x1b[0J", false);
235 }
236
237 return implode('', array_reverse($erasedContent));
238 }
