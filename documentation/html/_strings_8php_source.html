497 private static function pos(string $haystack, string $needle, int $nth = 1): ?int
498 {
499 if (!$nth) {
500 return null;
501 } elseif ($nth > 0) {
502 if ($needle === '') {
503 return 0;
504 }
505
506 $pos = 0;
507 while (($pos = strpos($haystack, $needle, $pos)) !== false && --$nth) {
508 $pos++;
509 }
510 } else {
511 $len = strlen($haystack);
512 if ($needle === '') {
513 return $len;
514 } elseif ($len === 0) {
515 return null;
516 }
517
518 $pos = $len - 1;
519 while (($pos = strrpos($haystack, $needle, $pos - $len)) !== false && ++$nth) {
520 $pos--;
521 }
522 }
523
524 return Helpers::falseToNull($pos);
525 }

672 {
