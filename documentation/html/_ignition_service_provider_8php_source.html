247 protected function configureQueue(): void
248 {
249 if (! $this->app->bound('queue')) {
250 return;
251 }
252
253 $queue = $this->app->get('queue');
254
255 // Reset before executing a queue job to make sure the job's log/query/dump recorders are empty.
256 // When using a sync queue this also reports the queued reports from previous exceptions.
257 $queue->before(function () {
258 $this->resetFlareAndLaravelIgnition();
259 app(Flare::class)->sendReportsImmediately();
260 });
261
262 // Send queued reports (and reset) after executing a queue job.
263 $queue->after(function () {
264 $this->resetFlareAndLaravelIgnition();
265 });
266
267 // Note: the $queue->looping() event can't be used because it's not triggered on Vapor
268 }
