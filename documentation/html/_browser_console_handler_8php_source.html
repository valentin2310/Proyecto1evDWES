169 private static function generateScript(): string
170 {
171 $script = [];
172 foreach (static::$records as $record) {
173 $context = self::dump('Context', $record->context);
174 $extra = self::dump('Extra', $record->extra);
175
176 if (\count($context) === 0 && \count($extra) === 0) {
177 $script[] = self::call_array(self::getConsoleMethodForLevel($record->level), self::handleStyles($record->formatted));
178 } else {
179 $script = array_merge(
180 $script,
181 [self::call_array('groupCollapsed', self::handleStyles($record->formatted))],
182 $context,
183 $extra,
184 [self::call('groupEnd')]
185 );
186 }
187 }
188
189 return "(function (c) {if (c && c.groupCollapsed) {\n" . implode("\n", $script) . "\n}})(console);";
190 }

205 private static function handleStyles(string $formatted): array
206 {
207 $args = [];
208 $format = '%c' . $formatted;
209 preg_match_all('/\[\[(.*?)\]\]\{([^}]*)\}/s', $format, $matches, PREG_OFFSET_CAPTURE | PREG_SET_ORDER);
210
211 foreach (array_reverse($matches) as $match) {
212 $args[] = '"font-weight: normal"';
213 $args[] = self::quote(self::handleCustomStyles($match[2][0], $match[1][0]));
214
215 $pos = $match[0][1];
216 $format = Utils::substr($format, 0, $pos) . '%c' . $match[1][0] . '%c' . Utils::substr($format, $pos + strlen($match[0][0]));
217 }
218
219 $args[] = self::quote('font-weight: normal');
220 $args[] = self::quote($format);
221
222 return array_reverse($args);
223 }

225 private static function handleCustomStyles(string $style, string $string): string
226 {
227 static $colors = ['blue', 'green', 'red', 'magenta', 'orange', 'black', 'grey'];
228 static $labels = [];
229
230 $style = preg_replace_callback('/macro\s*:(.*?)(?:;|$)/', function (array $m) use ($string, &$colors, &$labels) {
231 if (trim($m[1]) === 'autolabel') {
232 // Format the string as a label with consistent auto assigned background color
233 if (!isset($labels[$string])) {
234 $labels[$string] = $colors[count($labels) % count($colors)];
235 }
236 $color = $labels[$string];
237
238 return "background-color: $color; color: white; border-radius: 3px; padding: 0 2px 0 2px";
239 }
240
241 return $m[1];
242 }, $style);
243
244 if (null === $style) {
245 $pcreErrorCode = preg_last_error();
246
247 throw new \RuntimeException('Failed to run preg_replace_callback: ' . $pcreErrorCode . ' / ' . Utils::pcreLastErrorMessage($pcreErrorCode));
248 }
249
250 return $style;

257 private static function dump(string $title, array $dict): array
258 {
259 $script = [];
260 $dict = array_filter($dict);
261 if (\count($dict) === 0) {
262 return $script;
263 }
264 $script[] = self::call('log', self::quote('%c%s'), self::quote('font-weight: bold'), self::quote($title));
265 foreach ($dict as $key => $value) {
266 $value = json_encode($value);
267 if (false === $value) {
268 $value = self::quote('');
269 }
270 $script[] = self::call('log', self::quote('%s: %o'), self::quote((string) $key), $value);
271 }
