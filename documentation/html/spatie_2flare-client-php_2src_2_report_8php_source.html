328 protected function cleanupStackTraceForError(array $frames): array
329 {
330 if ($this->throwable === null || get_class($this->throwable) !== ErrorException::class) {
331 return $frames;
332 }
333
334 $firstErrorFrameIndex = null;
335
336 $restructuredFrames = array_values(array_slice($frames, 1)); // remove the first frame where error was created
337
338 foreach ($restructuredFrames as $index => $frame) {
339 if ($frame->file === $this->throwable->getFile()) {
340 $firstErrorFrameIndex = $index;
341
342 break;
343 }
344 }
345
346 if ($firstErrorFrameIndex === null) {
347 return $frames;
348 }
349
350 $restructuredFrames[$firstErrorFrameIndex]->arguments = null; // Remove error arguments
351
352 return array_values(array_slice($restructuredFrames, $firstErrorFrameIndex));
353 }

388 $data = random_bytes(16);
