122 public function write($messages, $newline = false, $type = 0)
123 {
124 if ($this->getVerbosity() === self::VERBOSITY_QUIET) {
125 return;
126 }
127
128 $messages = (array) $messages;
129
130 if ($type & self::NUMBER_LINES) {
131 $pad = \strlen((string) \count($messages));
132 $template = $this->isDecorated() ? "<aside>%{$pad}s</aside>: %s" : "%{$pad}s: %s";
133
134 if ($type & self::OUTPUT_RAW) {
135 $messages = \array_map([OutputFormatter::class, 'escape'], $messages);
136 }
137
138 foreach ($messages as $i => $line) {
139 $messages[$i] = \sprintf($template, $i, $line);
140 }
141
142 // clean this up for super.
143 $type = $type & ~self::NUMBER_LINES & ~self::OUTPUT_RAW;
144 }
