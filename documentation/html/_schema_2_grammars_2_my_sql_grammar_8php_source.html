162 protected function compileCreateEncoding($sql, Connection $connection, Blueprint $blueprint)
163 {
164 // First we will set the character set if one has been set on either the create
165 // blueprint itself or on the root configuration for the connection that the
166 // table is being created on. We will add these to the create table query.
167 if (isset($blueprint->charset)) {
168 $sql .= ' default character set '.$blueprint->charset;
169 } elseif (! is_null($charset = $connection->getConfig('charset'))) {
170 $sql .= ' default character set '.$charset;
171 }
172
173 // Next we will add the collation to the create table statement if one has been
174 // added to either this create table blueprint or the configuration for this
175 // connection that the query is targeting. We'll add it to this SQL query.
176 if (isset($blueprint->collation)) {
177 $sql .= " collate '{$blueprint->collation}'";
178 } elseif (! is_null($collation = $connection->getConfig('collation'))) {
179 $sql .= " collate '{$collation}'";
180 }
181
182 return $sql;
183 }
