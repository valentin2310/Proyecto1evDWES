356 private function flattenArgs(array $args, int $level = 0, int &$count = 0): array
357 {
358 $result = [];
359 foreach ($args as $key => $value) {
360 if (++$count > 1e4) {
361 return ['array', '*SKIPPED over 10000 entries*'];
362 }
363 if ($value instanceof \__PHP_Incomplete_Class) {
364 $result[$key] = ['incomplete-object', $this->getClassNameFromIncomplete($value)];
365 } elseif (\is_object($value)) {
366 $result[$key] = ['object', get_debug_type($value)];
367 } elseif (\is_array($value)) {
368 if ($level > 10) {
369 $result[$key] = ['array', '*DEEP NESTED ARRAY*'];
370 } else {
371 $result[$key] = ['array', $this->flattenArgs($value, $level + 1, $count)];
372 }
373 } elseif (null === $value) {
374 $result[$key] = ['null', null];
375 } elseif (\is_bool($value)) {
376 $result[$key] = ['boolean', $value];
377 } elseif (\is_int($value)) {
378 $result[$key] = ['integer', $value];
379 } elseif (\is_float($value)) {
380 $result[$key] = ['float', $value];
381 } elseif (\is_resource($value)) {
382 $result[$key] = ['resource', get_resource_type($value)];
383 } else {
384 $result[$key] = ['string', (string) $value];
385 }
386 }
387
388 return $result;
389 }
