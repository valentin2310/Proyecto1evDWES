173 private function getEventFromTypeDeclaration(ContainerBuilder $container, string $id, string $method): string
174 {
175 if (
176 null === ($class = $container->getDefinition($id)->getClass())
177 || !($r = $container->getReflectionClass($class, false))
178 || !$r->hasMethod($method)
179 || 1 > ($m = $r->getMethod($method))->getNumberOfParameters()
180 || !($type = $m->getParameters()[0]->getType()) instanceof \ReflectionNamedType
181 || $type->isBuiltin()
182 || Event::class === ($name = $type->getName())
183 ) {
184 throw new InvalidArgumentException(sprintf('Service "%s" must define the "event" attribute on "kernel.event_listener" tags.', $id));
185 }
186
187 return $name;
188 }
