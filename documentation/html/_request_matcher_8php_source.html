160 public function matches(Request $request): bool
161 {
162 if ($this->schemes && !\in_array($request->getScheme(), $this->schemes, true)) {
163 return false;
164 }
165
166 if ($this->methods && !\in_array($request->getMethod(), $this->methods, true)) {
167 return false;
168 }
169
170 foreach ($this->attributes as $key => $pattern) {
171 $requestAttribute = $request->attributes->get($key);
172 if (!\is_string($requestAttribute)) {
173 return false;
174 }
175 if (!preg_match('{'.$pattern.'}', $requestAttribute)) {
176 return false;
177 }
178 }
179
180 if (null !== $this->path && !preg_match('{'.$this->path.'}', rawurldecode($request->getPathInfo()))) {
181 return false;
182 }
183
184 if (null !== $this->host && !preg_match('{'.$this->host.'}i', $request->getHost())) {
185 return false;
186 }
187
188 if (null !== $this->port && 0 < $this->port && $request->getPort() !== $this->port) {
189 return false;
190 }
191
192 if (IpUtils::checkIp($request->getClientIp() ?? '', $this->ips)) {
193 return true;
194 }
195
196 // Note to future implementors: add additional checks above the
197 // foreach above or else your check might not be run!
198 return 0 === \count($this->ips);
