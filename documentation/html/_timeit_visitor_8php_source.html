51 public function enterNode(Node $node)
52 {
53 // keep track of nested function-like nodes, because they can have
54 // returns statements... and we don't want to call markEnd for those.
55 if ($node instanceof FunctionLike) {
56 $this->functionDepth++;
57
58 return;
59 }
60
61 // replace any top-level `return` statements with a `markEnd` call
62 if ($this->functionDepth === 0 && $node instanceof Return_) {
63 return new Return_($this->getEndCall($node->expr), $node->getAttributes());
64 }
65 }
66
72 public function leaveNode(Node $node)
73 {
74 if ($node instanceof FunctionLike) {
75 $this->functionDepth--;
76 }
77 }

84 public function afterTraverse(array $nodes)
85 {
86 // prepend a `markStart` call
87 \array_unshift($nodes, $this->maybeExpression($this->getStartCall()));
88
89 // append a `markEnd` call (wrapping the final node, if it's an expression)
90 $last = $nodes[\count($nodes) - 1];
91 if ($last instanceof Expr) {
92 \array_pop($nodes);
93 $nodes[] = $this->getEndCall($last);
94 } elseif ($last instanceof Expression) {
95 \array_pop($nodes);
96 $nodes[] = new Expression($this->getEndCall($last->expr), $last->getAttributes());
97 } elseif ($last instanceof Return_) {
98 // nothing to do here, we're already ending with a return call
99 } else {
100 $nodes[] = $this->maybeExpression($this->getEndCall());
101 }
102
103 return $nodes;
104 }
