96 private static function formatClass(\ReflectionClass $reflector): string
97 {
98 $chunks = [];
99
100 if ($modifiers = self::formatModifiers($reflector)) {
101 $chunks[] = $modifiers;
102 }
103
104 if ($reflector->isTrait()) {
105 $chunks[] = 'trait';
106 } else {
107 $chunks[] = $reflector->isInterface() ? 'interface' : 'class';
108 }
109
110 $chunks[] = \sprintf('<class>%s</class>', self::formatName($reflector));
111
112 if ($parent = $reflector->getParentClass()) {
113 $chunks[] = 'extends';
114 $chunks[] = \sprintf('<class>%s</class>', $parent->getName());
115 }
116
117 $interfaces = $reflector->getInterfaceNames();
118 if (!empty($interfaces)) {
119 \sort($interfaces);
120
121 $chunks[] = $reflector->isInterface() ? 'extends' : 'implements';
122 $chunks[] = \implode(', ', \array_map(function ($name) {
123 return \sprintf('<class>%s</class>', $name);
124 }, $interfaces));
125 }
126
127 return \implode(' ', $chunks);
128 }

263 private static function formatFunctionParams(\ReflectionFunctionAbstract $reflector): array
264 {
265 $params = [];
266 foreach ($reflector->getParameters() as $param) {
267 $hint = '';
268 try {
269 if (\method_exists($param, 'getType')) {
270 $hint = self::formatReflectionType($param->getType());
271 } else {
272 if ($param->isArray()) {
273 $hint = '<keyword>array</keyword>';
274 } elseif ($class = $param->getClass()) {
275 $hint = \sprintf('<class>%s</class>', $class->getName());
276 }
277 }
278 } catch (\Throwable $e) {
279 // sometimes we just don't know...
280 // bad class names, or autoloaded classes that haven't been loaded yet, or whathaveyou.
281 // come to think of it, the only time I've seen this is with the intl extension.
282
283 // Hax: we'll try to extract it :P
284
285 // @codeCoverageIgnoreStart
286 $chunks = \explode('$'.$param->getName(), (string) $param);
287 $chunks = \explode(' ', \trim($chunks[0]));
288 $guess = \end($chunks);
289
290 $hint = \sprintf('<urgent>%s</urgent>', OutputFormatter::escape($guess));
291 // @codeCoverageIgnoreEnd
292 }
293
294 if ($param->isOptional()) {
295 if (!$param->isDefaultValueAvailable()) {
296 $value = 'unknown';
297 $typeStyle = 'urgent';
298 } else {
299 $value = $param->getDefaultValue();
300 $typeStyle = self::getTypeStyle($value);
301 $value = \is_array($value) ? '[]' : ($value === null ? 'null' : \var_export($value, true));
302 }
303 $default = \sprintf(' = <%s>%s</%s>', $typeStyle, OutputFormatter::escape($value), $typeStyle);
304 } else {
305 $default = '';
306 }
307
308 $params[] = \sprintf(
309 '%s%s%s<strong>$%s</strong>%s',
310 $param->isPassedByReference() ? '&' : '',
311 $hint,
312 $hint !== '' ? ' ' : '',
313 $param->getName(),
314 $default
315 );
316 }
317
318 return $params;
319 }

326 private static function formatReflectionType(\ReflectionType $type = null): string
327 {
328 if ($type === null) {
329 return '';
330 }
331
332 $types = $type instanceof \ReflectionUnionType ? $type->getTypes() : [$type];
333 $formattedTypes = [];
334
335 foreach ($types as $type) {
336 $typeStyle = $type->isBuiltin() ? 'keyword' : 'class';
337
338 // PHP 7.0 didn't have `getName` on reflection types, so wheee!
339 $typeName = \method_exists($type, 'getName') ? $type->getName() : (string) $type;
340
341 // @todo Do we want to include the ? for nullable types? Maybe only sometimes?
342 $formattedTypes[] = \sprintf('<%s>%s</%s>', $typeStyle, OutputFormatter::escape($typeName), $typeStyle);
343 }
344
345 return \implode('|', $formattedTypes);
