135 private static function castFFIStructLike(CType $type, CData $data = null): array
136 {
137 $isUnion = ($type->getAttributes() & CType::ATTR_UNION) === CType::ATTR_UNION;
138
139 $result = [];
140
141 foreach ($type->getStructFieldNames() as $name) {
142 $field = $type->getStructFieldType($name);
143
144 // Retrieving the value of a field from a union containing
145 // a pointer is not a safe operation, because may contain
146 // incorrect data.
147 $isUnsafe = $isUnion && CType::TYPE_POINTER === $field->getKind();
148
149 if ($isUnsafe) {
150 $result[Caster::PREFIX_VIRTUAL.$name.'?'] = $field;
151 } elseif (null === $data) {
152 $result[Caster::PREFIX_VIRTUAL.$name] = $field;
153 } else {
154 $fieldName = $data->{$name} instanceof CData ? '' : $field->getName().' ';
155 $result[Caster::PREFIX_VIRTUAL.$fieldName.$name] = $data->{$name};
156 }
157 }
158
159 return $result;
