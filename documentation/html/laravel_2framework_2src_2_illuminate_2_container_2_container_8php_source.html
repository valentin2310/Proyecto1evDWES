763 protected function resolve($abstract, $parameters = [], $raiseEvents = true)
764 {
765 $abstract = $this->getAlias($abstract);
766
767 // First we'll fire any event handlers which handle the "before" resolving of
768 // specific types. This gives some hooks the chance to add various extends
769 // calls to change the resolution of objects that they're interested in.
770 if ($raiseEvents) {
771 $this->fireBeforeResolvingCallbacks($abstract, $parameters);
772 }
773
774 $concrete = $this->getContextualConcrete($abstract);
775
776 $needsContextualBuild = ! empty($parameters) || ! is_null($concrete);
777
778 // If an instance of the type is currently being managed as a singleton we'll
779 // just return an existing instance instead of instantiating new instances
780 // so the developer can keep using the same objects instance every time.
781 if (isset($this->instances[$abstract]) && ! $needsContextualBuild) {
782 return $this->instances[$abstract];
783 }
784
785 $this->with[] = $parameters;
786
787 if (is_null($concrete)) {
788 $concrete = $this->getConcrete($abstract);
789 }
790
791 // We're ready to instantiate an instance of the concrete type registered for
792 // the binding. This will instantiate the types, as well as resolve any of
793 // its "nested" dependencies recursively until all have gotten resolved.
794 $object = $this->isBuildable($concrete, $abstract)
795 ? $this->build($concrete)
796 : $this->make($concrete);
797
798 // If we defined any extenders for this type, we'll need to spin through them
799 // and apply them to the object being built. This allows for the extension
800 // of services, such as changing configuration or decorating the object.
801 foreach ($this->getExtenders($abstract) as $extender) {
802 $object = $extender($object, $this);
803 }
804
805 // If the requested type is registered as a singleton we'll want to cache off
806 // the instances in "memory" so we can return it later without creating an
807 // entirely new instance of an object on each subsequent request for it.
808 if ($this->isShared($abstract) && ! $needsContextualBuild) {
809 $this->instances[$abstract] = $object;
810 }
811
812 if ($raiseEvents) {
813 $this->fireResolvingCallbacks($abstract, $object);
814 }
815
816 // Before returning, we will also set the resolved flag to "true" and pop off
817 // the parameter overrides for this build. After those two things are done
818 // we will be ready to return back the fully constructed class instance.
819 $this->resolved[$abstract] = true;
820
821 array_pop($this->with);
822
823 return $object;
824 }

902 public function build($concrete)
903 {
904 // If the concrete type is actually a Closure, we will just execute it and
905 // hand back the results of the functions, which allows functions to be
906 // used as resolvers for more fine-tuned resolution of these objects.
907 if ($concrete instanceof Closure) {
908 return $concrete($this, $this->getLastParameterOverride());
909 }
910
911 try {
912 $reflector = new ReflectionClass($concrete);
913 } catch (ReflectionException $e) {
914 throw new BindingResolutionException("Target class [$concrete] does not exist.", 0, $e);
915 }
916
917 // If the type is not instantiable, the developer is attempting to resolve
918 // an abstract type such as an Interface or Abstract Class and there is
919 // no binding registered for the abstractions so we need to bail out.
920 if (! $reflector->isInstantiable()) {
921 return $this->notInstantiable($concrete);
922 }
923
924 $this->buildStack[] = $concrete;
925
926 $constructor = $reflector->getConstructor();
927
928 // If there are no constructors, that means there are no dependencies then
929 // we can just resolve the instances of the objects right away, without
930 // resolving any other types or dependencies out of these containers.
931 if (is_null($constructor)) {
932 array_pop($this->buildStack);
933
934 return new $concrete;
935 }
936
937 $dependencies = $constructor->getParameters();
938
939 // Once we have all the constructor's parameters we can create each of the
940 // dependency instances and then use the reflection instances to make a
941 // new instance of this class, injecting the created dependencies in.
942 try {
943 $instances = $this->resolveDependencies($dependencies);
944 } catch (BindingResolutionException $e) {
945 array_pop($this->buildStack);
946
947 throw $e;
948 }
949
950 array_pop($this->buildStack);
951
952 return $reflector->newInstanceArgs($instances);
953 }

963 protected function resolveDependencies(array $dependencies)
964 {
965 $results = [];
966
967 foreach ($dependencies as $dependency) {
968 // If the dependency has an override for this particular build we will use
969 // that instead as the value. Otherwise, we will continue with this run
970 // of resolutions and let reflection attempt to determine the result.
971 if ($this->hasParameterOverride($dependency)) {
972 $results[] = $this->getParameterOverride($dependency);
973
974 continue;
975 }
976
977 // If the class is null, it means the dependency is a string or some other
978 // primitive type which we can not resolve since it is not a class and
979 // we will just bomb out with an error since we have no-where to go.
980 $result = is_null(Util::getParameterClassName($dependency))
981 ? $this->resolvePrimitive($dependency)
982 : $this->resolveClass($dependency);
983
984 if ($dependency->isVariadic()) {
985 $results = array_merge($results, $result);
986 } else {
987 $results[] = $result;
988 }
989 }
990
991 return $results;
992 }
