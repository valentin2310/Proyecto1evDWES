75 private function setHeaders(Message $message): void
76 {
77 if (!$this->headers) {
78 return;
79 }
80
81 $headers = $message->getHeaders();
82 foreach ($this->headers->all() as $name => $header) {
83 if (!$headers->has($name)) {
84 $headers->add($header);
85
86 continue;
87 }
88
89 switch ($this->headerRules[$name] ?? self::HEADER_SET_IF_EMPTY) {
90 case self::HEADER_SET_IF_EMPTY:
91 break;
92
93 case self::HEADER_REPLACE:
94 $headers->remove($name);
95 $headers->add($header);
96
97 break;
98
99 case self::HEADER_ADD:
100 if (!Headers::isUniqueHeader($name)) {
101 $headers->add($header);
102
103 break;
104 }
105
106 $h = $headers->get($name);
107 if (!$h instanceof MailboxListHeader) {
108 throw new RuntimeException(sprintf('Unable to set header "%s".', $name));
109 }
110
111 Headers::checkHeaderClass($header);
112 foreach ($header->getAddresses() as $address) {
113 $h->addAddress($address);
114 }
115 }
116 }
