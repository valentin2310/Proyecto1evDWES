82 protected function execute(InputInterface $input, OutputInterface $output): int
83 {
84 try {
85 // "symfony" must be kept for compat with the shell scripts generated by Symfony Console 5.4 - 6.1
86 $version = $input->getOption('symfony') ? '1' : $input->getOption('api-version');
87 if ($version && version_compare($version, self::COMPLETION_API_VERSION, '<')) {
88 $message = sprintf('Completion script version is not supported ("%s" given, ">=%s" required).', $version, self::COMPLETION_API_VERSION);
89 $this->log($message);
90
91 $output->writeln($message.' Install the Symfony completion script again by using the "completion" command.');
92
93 return 126;
94 }
95
96 $shell = $input->getOption('shell');
97 if (!$shell) {
98 throw new \RuntimeException('The "--shell" option must be set.');
99 }
100
101 if (!$completionOutput = $this->completionOutputs[$shell] ?? false) {
102 throw new \RuntimeException(sprintf('Shell completion is not supported for your shell: "%s" (supported: "%s").', $shell, implode('", "', array_keys($this->completionOutputs))));
103 }
104
105 $completionInput = $this->createCompletionInput($input);
106 $suggestions = new CompletionSuggestions();
107
108 $this->log([
109 '',
110 '<comment>'.date('Y-m-d H:i:s').'</>',
111 '<info>Input:</> <comment>("|" indicates the cursor position)</>',
112 ' '.(string) $completionInput,
113 '<info>Command:</>',
114 ' '.(string) implode(' ', $_SERVER['argv']),
115 '<info>Messages:</>',
116 ]);
117
118 $command = $this->findCommand($completionInput, $output);
119 if (null === $command) {
120 $this->log(' No command found, completing using the Application class.');
121
122 $this->getApplication()->complete($completionInput, $suggestions);
123 } elseif (
124 $completionInput->mustSuggestArgumentValuesFor('command')
125 && $command->getName() !== $completionInput->getCompletionValue()
126 && !\in_array($completionInput->getCompletionValue(), $command->getAliases(), true)
127 ) {
128 $this->log(' No command found, completing using the Application class.');
129
130 // expand shortcut names ("cache:cl<TAB>") into their full name ("cache:clear")
131 $suggestions->suggestValues(array_filter(array_merge([$command->getName()], $command->getAliases())));
132 } else {
133 $command->mergeApplicationDefinition();
134 $completionInput->bind($command->getDefinition());
135
136 if (CompletionInput::TYPE_OPTION_NAME === $completionInput->getCompletionType()) {
137 $this->log(' Completing option names for the <comment>'.($command instanceof LazyCommand ? $command->getCommand() : $command)::class.'</> command.');
138
139 $suggestions->suggestOptions($command->getDefinition()->getOptions());
140 } else {
141 $this->log([
142 ' Completing using the <comment>'.($command instanceof LazyCommand ? $command->getCommand() : $command)::class.'</> class.',
143 ' Completing <comment>'.$completionInput->getCompletionType().'</> for <comment>'.$completionInput->getCompletionName().'</>',
144 ]);
145 if (null !== $compval = $completionInput->getCompletionValue()) {
146 $this->log(' Current value: <comment>'.$compval.'</>');
147 }
148
149 $command->complete($completionInput, $suggestions);
150 }
151 }
152
154 $completionOutput = new $completionOutput();
155
156 $this->log('<info>Suggestions:</>');
157 if ($options = $suggestions->getOptionSuggestions()) {
158 $this->log(' --'.implode(' --', array_map(fn ($o) => $o->getName(), $options)));
159 } elseif ($values = $suggestions->getValueSuggestions()) {
160 $this->log(' '.implode(' ', $values));
161 } else {
162 $this->log(' <comment>No suggestions were provided</>');
163 }
164
165 $completionOutput->write($suggestions, $output);
166 } catch (\Throwable $e) {
167 $this->log([
168 '<error>Error!</error>',
169 (string) $e,
170 ]);
171
172 if ($output->isDebug()) {
173 throw $e;
174 }
175
176 return 2;
177 }
178
179 return 0;
180 }

182 private function createCompletionInput(InputInterface $input): CompletionInput
183 {
184 $currentIndex = $input->getOption('current');
185 if (!$currentIndex || !ctype_digit($currentIndex)) {
186 throw new \RuntimeException('The "--current" option must be set and it must be an integer.');
187 }
188
189 $completionInput = CompletionInput::fromTokens($input->getOption('input'), (int) $currentIndex);
190
191 try {
192 $completionInput->bind($this->getApplication()->getDefinition());
193 } catch (ExceptionInterface) {
194 }
195
196 return $completionInput;
197 }
