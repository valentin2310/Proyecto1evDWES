352 public function put($path, $contents, $options = [])
353 {
354 $options = is_string($options)
355 ? ['visibility' => $options]
356 : (array) $options;
357
358 // If the given contents is actually a file or uploaded file instance than we will
359 // automatically store the file using a stream. This provides a convenient path
360 // for the developer to store streams without managing them manually in code.
361 if ($contents instanceof File ||
362 $contents instanceof UploadedFile) {
363 return $this->putFile($path, $contents, $options);
364 }
365
366 try {
367 if ($contents instanceof StreamInterface) {
368 $this->driver->writeStream($path, $contents->detach(), $options);
369
370 return true;
371 }
372
373 is_resource($contents)
374 ? $this->driver->writeStream($path, $contents, $options)
375 : $this->driver->write($path, $contents, $options);
376 } catch (UnableToWriteFile|UnableToSetVisibility $e) {
377 throw_if($this->throwsExceptions(), $e);
378
379 return false;
380 }
381
382 return true;
383 }

413 public function putFileAs($path, $file, $name = null, $options = [])
414 {
415 if (is_null($name) || is_array($name)) {
416 [$path, $file, $name, $options] = ['', $path, $file, $name ?? []];
417 }
418
419 $stream = fopen(is_string($file) ? $file : $file->getRealPath(), 'r');
420
421 // Next, we will format the path of the file and store the file using a stream since
422 // they provide better performance than alternatives. Once we write the file this
423 // stream will get closed automatically by us so the developer doesn't have to.
424 $result = $this->put(
425 $path = trim($path.'/'.$name, '/'), $stream, $options
426 );
427
428 if (is_resource($stream)) {
429 fclose($stream);
430 }
431
432 return $result ? $path : false;
433 }

702 protected function getLocalUrl($path)
703 {
704 // If an explicit base URL has been set on the disk configuration then we will use
705 // it as the base URL instead of the default path. This allows the developer to
706 // have full control over the base path for this filesystem's generated URLs.
707 if (isset($this->config['url'])) {
708 return $this->concatPathToUrl($this->config['url'], $path);
709 }
710
711 $path = '/storage/'.$path;
712
713 // If the path contains "storage/public", it probably means the developer is using
714 // the default disk to generate the path instead of the "public" disk like they
715 // are really supposed to use. We will remove the public from this path here.
716 if (str_contains($path, '/storage/public/')) {
717 return Str::replaceFirst('/public/', '/', $path);
718 }
719
720 return $path;
721 }
