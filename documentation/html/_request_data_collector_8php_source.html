432 public function collectSessionUsage(): void
433 {
434 $trace = debug_backtrace(\DEBUG_BACKTRACE_IGNORE_ARGS);
435
436 $traceEndIndex = \count($trace) - 1;
437 for ($i = $traceEndIndex; $i > 0; --$i) {
438 if (null !== ($class = $trace[$i]['class'] ?? null) && (is_subclass_of($class, SessionInterface::class) || is_subclass_of($class, SessionBagInterface::class))) {
439 $traceEndIndex = $i;
440 break;
441 }
442 }
443
444 if ((\count($trace) - 1) === $traceEndIndex) {
445 return;
446 }
447
448 // Remove part of the backtrace that belongs to session only
449 array_splice($trace, 0, $traceEndIndex);
450
451 // Merge identical backtraces generated by internal call reports
452 $name = sprintf('%s:%s', $trace[1]['class'] ?? $trace[0]['file'], $trace[0]['line']);
453 if (!\array_key_exists($name, $this->sessionUsages)) {
454 $this->sessionUsages[$name] = [
455 'name' => $name,
456 'file' => $trace[0]['file'],
457 'line' => $trace[0]['line'],
458 'trace' => $trace,
459 ];
460 }

466 private function parseController(array|object|string|null $controller): array|string
467 {
468 if (\is_string($controller) && str_contains($controller, '::')) {
469 $controller = explode('::', $controller);
470 }
471
472 if (\is_array($controller)) {
473 try {
474 $r = new \ReflectionMethod($controller[0], $controller[1]);
475
476 return [
477 'class' => \is_object($controller[0]) ? get_debug_type($controller[0]) : $controller[0],
478 'method' => $controller[1],
479 'file' => $r->getFileName(),
480 'line' => $r->getStartLine(),
481 ];
482 } catch (\ReflectionException) {
483 if (\is_callable($controller)) {
484 // using __call or __callStatic
485 return [
486 'class' => \is_object($controller[0]) ? get_debug_type($controller[0]) : $controller[0],
487 'method' => $controller[1],
488 'file' => 'n/a',
489 'line' => 'n/a',
490 ];
491 }
492 }
493 }
494
495 if ($controller instanceof \Closure) {
496 $r = new \ReflectionFunction($controller);
497
498 $controller = [
499 'class' => $r->getName(),
500 'method' => null,
501 'file' => $r->getFileName(),
502 'line' => $r->getStartLine(),
503 ];
504
505 if (str_contains($r->name, '{closure}')) {
506 return $controller;
507 }
508 $controller['method'] = $r->name;
509
510 if ($class = \PHP_VERSION_ID >= 80111 ? $r->getClosureCalledClass() : $r->getClosureScopeClass()) {
511 $controller['class'] = $class->name;
512 } else {
513 return $r->name;
514 }
515
516 return $controller;
517 }
518
519 if (\is_object($controller)) {
520 $r = new \ReflectionClass($controller);
521
522 return [
523 'class' => $r->getName(),
524 'method' => null,
525 'file' => $r->getFileName(),
526 'line' => $r->getStartLine(),
527 ];
528 }
529
530 return \is_string($controller) ? $controller : 'n/a';
531 }
