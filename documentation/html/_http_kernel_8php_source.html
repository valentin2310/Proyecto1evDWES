69 public function handle(Request $request, int $type = HttpKernelInterface::MAIN_REQUEST, bool $catch = true): Response
70 {
71 $request->headers->set('X-Php-Ob-Level', (string) ob_get_level());
72
73 $this->requestStack->push($request);
74 $response = null;
75 try {
76 return $response = $this->handleRaw($request, $type);
77 } catch (\Throwable $e) {
78 if ($e instanceof \Error && !$this->handleAllThrowables) {
79 throw $e;
80 }
81
82 if ($e instanceof RequestExceptionInterface) {
83 $e = new BadRequestHttpException($e->getMessage(), $e);
84 }
85 if (false === $catch) {
86 $this->finishRequest($request, $type);
87
88 throw $e;
89 }

93 $this->requestStack->pop();

121 public function terminateWithException(\Throwable $exception, Request $request = null): void
122 {
123 if (!$request ??= $this->requestStack->getMainRequest()) {
124 throw $exception;
125 }
126
127 if ($pop = $request !== $this->requestStack->getMainRequest()) {
128 $this->requestStack->push($request);
129 }

132 $response = $this->handleThrowable($exception, $request, self::MAIN_REQUEST);
133 } finally {

153 private function handleRaw(Request $request, int $type = self::MAIN_REQUEST): Response
154 {
155 // request
156 $event = new RequestEvent($this, $request, $type);
157 $this->dispatcher->dispatch($event, KernelEvents::REQUEST);
158
159 if ($event->hasResponse()) {
160 return $this->filterResponse($event->getResponse(), $request, $type);
161 }
162
163 // load controller
164 if (false === $controller = $this->resolver->getController($request)) {
165 throw new NotFoundHttpException(sprintf('Unable to find the controller for path "%s". The route is wrongly configured.', $request->getPathInfo()));
166 }
167
168 $event = new ControllerEvent($this, $controller, $request, $type);
169 $this->dispatcher->dispatch($event, KernelEvents::CONTROLLER);
170 $controller = $event->getController();
171
172 // controller arguments
173 $arguments = $this->argumentResolver->getArguments($request, $controller, $event->getControllerReflector());
174
175 $event = new ControllerArgumentsEvent($this, $event, $arguments, $request, $type);
176 $this->dispatcher->dispatch($event, KernelEvents::CONTROLLER_ARGUMENTS);
177 $controller = $event->getController();
178 $arguments = $event->getArguments();
179
180 // call controller
181 $response = $controller(...$arguments);
182
183 // view
184 if (!$response instanceof Response) {
185 $event = new ViewEvent($this, $request, $type, $response, $event);
186 $this->dispatcher->dispatch($event, KernelEvents::VIEW);
187
188 if ($event->hasResponse()) {
189 $response = $event->getResponse();
190 } else {
191 $msg = sprintf('The controller must return a "Symfony\Component\HttpFoundation\Response" object but it returned %s.', $this->varToString($response));
192
193 // the user may have forgotten to return something
194 if (null === $response) {
195 $msg .= ' Did you forget to add a return statement somewhere in your controller?';
196 }
197
198 throw new ControllerDoesNotReturnResponseException($msg, $controller, __FILE__, __LINE__ - 17);
199 }
200 }
201
202 return $this->filterResponse($response, $request, $type);
203 }

236 private function handleThrowable(\Throwable $e, Request $request, int $type): Response
237 {
238 $event = new ExceptionEvent($this, $request, $type, $e);
239 $this->dispatcher->dispatch($event, KernelEvents::EXCEPTION);
240
241 // a listener might have replaced the exception
242 $e = $event->getThrowable();
243
244 if (!$event->hasResponse()) {
245 $this->finishRequest($request, $type);
246
247 throw $e;
248 }
249
250 $response = $event->getResponse();
251
252 // the developer asked for a specific status code
253 if (!$event->isAllowingCustomResponseCode() && !$response->isClientError() && !$response->isServerError() && !$response->isRedirect()) {
254 // ensure that we actually have an error response
255 if ($e instanceof HttpExceptionInterface) {
256 // keep the HTTP status code and headers
257 $response->setStatusCode($e->getStatusCode());
258 $response->headers->add($e->getHeaders());
259 } else {
260 $response->setStatusCode(500);
261 }
262 }
263
264 try {
265 return $this->filterResponse($response, $request, $type);
266 } catch (\Throwable $e) {
267 if ($e instanceof \Error && !$this->handleAllThrowables) {
268 throw $e;
269 }
270
271 return $response;
272 }

278 private function varToString(mixed $var): string
279 {
280 if (\is_object($var)) {
281 return sprintf('an object of type %s', $var::class);
282 }
283
284 if (\is_array($var)) {
285 $a = [];
286 foreach ($var as $k => $v) {
287 $a[] = sprintf('%s => ...', $k);
288 }
289
290 return sprintf('an array ([%s])', mb_substr(implode(', ', $a), 0, 255));
291 }
292
293 if (\is_resource($var)) {
294 return sprintf('a resource (%s)', get_resource_type($var));
295 }
296
297 if (null === $var) {
298 return 'null';
299 }
300
301 if (false === $var) {
302 return 'a boolean value (false)';
303 }
304
305 if (true === $var) {
306 return 'a boolean value (true)';
307 }
308
309 if (\is_string($var)) {
310 return sprintf('a string ("%s%s")', mb_substr($var, 0, 255), mb_strlen($var) > 255 ? '...' : '');
311 }
312
313 if (is_numeric($var)) {
314 return sprintf('a number (%s)', (string) $var);
315 }
316
317 return (string) $var;
318 }
