318 public function render()
319 {
320 $divider = new TableSeparator();
321 $isCellWithColspan = static fn ($cell) => $cell instanceof TableCell && $cell->getColspan() >= 2;
322
323 $horizontal = self::DISPLAY_ORIENTATION_HORIZONTAL === $this->displayOrientation;
324 $vertical = self::DISPLAY_ORIENTATION_VERTICAL === $this->displayOrientation;
325
326 $rows = [];
327 if ($horizontal) {
328 foreach ($this->headers[0] ?? [] as $i => $header) {
329 $rows[$i] = [$header];
330 foreach ($this->rows as $row) {
331 if ($row instanceof TableSeparator) {
332 continue;
333 }
334 if (isset($row[$i])) {
335 $rows[$i][] = $row[$i];
336 } elseif ($isCellWithColspan($rows[$i][0])) {
337 // Noop, there is a "title"
338 } else {
339 $rows[$i][] = null;
340 }
341 }
342 }
343 } elseif ($vertical) {
344 $formatter = $this->output->getFormatter();
345 $maxHeaderLength = array_reduce($this->headers[0] ?? [], static fn ($max, $header) => max($max, Helper::width(Helper::removeDecoration($formatter, $header))), 0);
346
347 foreach ($this->rows as $row) {
348 if ($row instanceof TableSeparator) {
349 continue;
350 }
351
352 if ($rows) {
353 $rows[] = [$divider];
354 }
355
356 $containsColspan = false;
357 foreach ($row as $cell) {
358 if ($containsColspan = $isCellWithColspan($cell)) {
359 break;
360 }
361 }
362
363 $headers = $this->headers[0] ?? [];
364 $maxRows = max(\count($headers), \count($row));
365 for ($i = 0; $i < $maxRows; ++$i) {
366 $cell = (string) ($row[$i] ?? '');
367 if ($headers && !$containsColspan) {
368 $rows[] = [sprintf(
369 '<comment>%s</>: %s',
370 str_pad($headers[$i] ?? '', $maxHeaderLength, ' ', \STR_PAD_LEFT),
371 $cell
372 )];
373 } elseif ('' !== $cell) {
374 $rows[] = [$cell];
375 }
376 }
377 }
378 } else {
379 $rows = array_merge($this->headers, [$divider], $this->rows);
380 }
381
382 $this->calculateNumberOfColumns($rows);
383
384 $rowGroups = $this->buildTableRows($rows);
385 $this->calculateColumnsWidth($rowGroups);
386
387 $isHeader = !$horizontal;
388 $isFirstRow = $horizontal;
389 $hasTitle = (bool) $this->headerTitle;
390
391 foreach ($rowGroups as $rowGroup) {
392 $isHeaderSeparatorRendered = false;
393
394 foreach ($rowGroup as $row) {
395 if ($divider === $row) {
396 $isHeader = false;
397 $isFirstRow = true;
398
399 continue;
400 }
401
402 if ($row instanceof TableSeparator) {
403 $this->renderRowSeparator();
404
405 continue;
406 }
407
408 if (!$row) {
409 continue;
410 }
411
412 if ($isHeader && !$isHeaderSeparatorRendered) {
413 $this->renderRowSeparator(
414 $isHeader ? self::SEPARATOR_TOP : self::SEPARATOR_TOP_BOTTOM,
415 $hasTitle ? $this->headerTitle : null,
416 $hasTitle ? $this->style->getHeaderTitleFormat() : null
417 );
418 $hasTitle = false;
419 $isHeaderSeparatorRendered = true;
420 }
421
422 if ($isFirstRow) {
423 $this->renderRowSeparator(
424 $isHeader ? self::SEPARATOR_TOP : self::SEPARATOR_TOP_BOTTOM,
425 $hasTitle ? $this->headerTitle : null,
426 $hasTitle ? $this->style->getHeaderTitleFormat() : null
427 );
428 $isFirstRow = false;
429 $hasTitle = false;
430 }
431
432 if ($vertical) {
433 $isHeader = false;
434 $isFirstRow = false;
435 }
436
437 if ($horizontal) {
438 $this->renderRow($row, $this->style->getCellRowFormat(), $this->style->getCellHeaderFormat());
439 } else {
440 $this->renderRow($row, $isHeader ? $this->style->getCellHeaderFormat() : $this->style->getCellRowFormat());
441 }
442 }
443 }
444 $this->renderRowSeparator(self::SEPARATOR_BOTTOM, $this->footerTitle, $this->style->getFooterTitleFormat());
445
446 $this->cleanup();
447 $this->rendered = true;
448 }

457 private function renderRowSeparator(int $type = self::SEPARATOR_MID, string $title = null, string $titleFormat = null): void
458 {
459 if (!$count = $this->numberOfColumns) {
460 return;
461 }
462
463 $borders = $this->style->getBorderChars();
464 if (!$borders[0] && !$borders[2] && !$this->style->getCrossingChar()) {
465 return;
466 }
467
468 $crossings = $this->style->getCrossingChars();
469 if (self::SEPARATOR_MID === $type) {
470 [$horizontal, $leftChar, $midChar, $rightChar] = [$borders[2], $crossings[8], $crossings[0], $crossings[4]];
471 } elseif (self::SEPARATOR_TOP === $type) {
472 [$horizontal, $leftChar, $midChar, $rightChar] = [$borders[0], $crossings[1], $crossings[2], $crossings[3]];
473 } elseif (self::SEPARATOR_TOP_BOTTOM === $type) {
474 [$horizontal, $leftChar, $midChar, $rightChar] = [$borders[0], $crossings[9], $crossings[10], $crossings[11]];
475 } else {
476 [$horizontal, $leftChar, $midChar, $rightChar] = [$borders[0], $crossings[7], $crossings[6], $crossings[5]];
477 }
478
479 $markup = $leftChar;
480 for ($column = 0; $column < $count; ++$column) {
481 $markup .= str_repeat($horizontal, $this->effectiveColumnWidths[$column]);
482 $markup .= $column === $count - 1 ? $rightChar : $midChar;
483 }
484
485 if (null !== $title) {
486 $titleLength = Helper::width(Helper::removeDecoration($formatter = $this->output->getFormatter(), $formattedTitle = sprintf($titleFormat, $title)));
487 $markupLength = Helper::width($markup);
488 if ($titleLength > $limit = $markupLength - 4) {
489 $titleLength = $limit;
490 $formatLength = Helper::width(Helper::removeDecoration($formatter, sprintf($titleFormat, '')));
491 $formattedTitle = sprintf($titleFormat, Helper::substr($title, 0, $limit - $formatLength - 3).'...');
492 }
493
494 $titleStart = intdiv($markupLength - $titleLength, 2);
495 if (false === mb_detect_encoding($markup, null, true)) {
496 $markup = substr_replace($markup, $formattedTitle, $titleStart, $titleLength);
497 } else {
498 $markup = mb_substr($markup, 0, $titleStart).$formattedTitle.mb_substr($markup, $titleStart + $titleLength);
499 }
500 }
501
502 $this->output->writeln(sprintf($this->style->getBorderFormat(), $markup));
503 }

522 private function renderRow(array $row, string $cellFormat, string $firstCellFormat = null): void
523 {
524 $rowContent = $this->renderColumnSeparator(self::BORDER_OUTSIDE);
525 $columns = $this->getRowColumns($row);
526 $last = \count($columns) - 1;
527 foreach ($columns as $i => $column) {
528 if ($firstCellFormat && 0 === $i) {
529 $rowContent .= $this->renderCell($row, $column, $firstCellFormat);

541 private function renderCell(array $row, int $column, string $cellFormat): string
542 {
543 $cell = $row[$column] ?? '';
544 $width = $this->effectiveColumnWidths[$column];
545 if ($cell instanceof TableCell && $cell->getColspan() > 1) {
546 // add the width of the following columns(numbers of colspan).
547 foreach (range($column + 1, $column + $cell->getColspan() - 1) as $nextColumn) {
548 $width += $this->getColumnSeparatorWidth() + $this->effectiveColumnWidths[$nextColumn];
549 }
550 }
551
552 // str_pad won't work properly with multi-byte strings, we need to fix the padding
553 if (false !== $encoding = mb_detect_encoding($cell, null, true)) {
554 $width += \strlen($cell) - mb_strwidth($cell, $encoding);
555 }
556
557 $style = $this->getColumnStyle($column);
558
559 if ($cell instanceof TableSeparator) {
560 return sprintf($style->getBorderFormat(), str_repeat($style->getBorderChars()[2], $width));
561 }
562
563 $width += Helper::length($cell) - Helper::length(Helper::removeDecoration($this->output->getFormatter(), $cell));
564 $content = sprintf($style->getCellRowContentFormat(), $cell);
565
566 $padType = $style->getPadType();
567 if ($cell instanceof TableCell && $cell->getStyle() instanceof TableCellStyle) {
568 $isNotStyledByTag = !preg_match('/^<(\w+|(\w+=[\w,]+;?)*)>.+<\/(\w+|(\w+=\w+;?)*)?>$/', $cell);
569 if ($isNotStyledByTag) {
570 $cellFormat = $cell->getStyle()->getCellFormat();
571 if (!\is_string($cellFormat)) {
572 $tag = http_build_query($cell->getStyle()->getTagOptions(), '', ';');
573 $cellFormat = '<'.$tag.'>%s</>';
574 }
575
576 if (str_contains($content, '</>')) {
577 $content = str_replace('</>', '', $content);
578 $width -= 3;
579 }
580 if (str_contains($content, '<fg=default;bg=default>')) {
581 $content = str_replace('<fg=default;bg=default>', '', $content);
582 $width -= \strlen('<fg=default;bg=default>');
583 }
584 }
585
586 $padType = $cell->getStyle()->getPadByAlign();
587 }
588
589 return sprintf($cellFormat, str_pad($content, $width, $style->getPaddingChar(), $padType));
590 }

609 private function buildTableRows(array $rows): TableRows
610 {
612 $formatter = $this->output->getFormatter();
613 $unmergedRows = [];
614 for ($rowKey = 0; $rowKey < \count($rows); ++$rowKey) {
615 $rows = $this->fillNextRows($rows, $rowKey);
616
617 // Remove any new line breaks and replace it with a new line
618 foreach ($rows[$rowKey] as $column => $cell) {
619 $colspan = $cell instanceof TableCell ? $cell->getColspan() : 1;
620
621 if (isset($this->columnMaxWidths[$column]) && Helper::width(Helper::removeDecoration($formatter, $cell)) > $this->columnMaxWidths[$column]) {
622 $cell = $formatter->formatAndWrap($cell, $this->columnMaxWidths[$column] * $colspan);
623 }
624 if (!str_contains($cell ?? '', "\n")) {
625 continue;
626 }
627 $escaped = implode("\n", array_map(OutputFormatter::escapeTrailingBackslash(...), explode("\n", $cell)));
628 $cell = $cell instanceof TableCell ? new TableCell($escaped, ['colspan' => $cell->getColspan()]) : $escaped;
629 $lines = explode("\n", str_replace("\n", "<fg=default;bg=default></>\n", $cell));
630 foreach ($lines as $lineKey => $line) {
631 if ($colspan > 1) {
632 $line = new TableCell($line, ['colspan' => $colspan]);
633 }
634 if (0 === $lineKey) {
635 $rows[$rowKey][$column] = $line;
636 } else {
637 if (!\array_key_exists($rowKey, $unmergedRows) || !\array_key_exists($lineKey, $unmergedRows[$rowKey])) {
638 $unmergedRows[$rowKey][$lineKey] = $this->copyRow($rows, $rowKey);
639 }
640 $unmergedRows[$rowKey][$lineKey][$column] = $line;
641 }
642 }
643 }
644 }
645
646 return new TableRows(function () use ($rows, $unmergedRows): \Traversable {
647 foreach ($rows as $rowKey => $row) {
648 $rowGroup = [$row instanceof TableSeparator ? $row : $this->fillCells($row)];
649
650 if (isset($unmergedRows[$rowKey])) {
651 foreach ($unmergedRows[$rowKey] as $row) {
652 $rowGroup[] = $row instanceof TableSeparator ? $row : $this->fillCells($row);
653 }
654 }
655 yield $rowGroup;
656 }
657 });
658 }

680 private function fillNextRows(array $rows, int $line): array
681 {
682 $unmergedRows = [];
683 foreach ($rows[$line] as $column => $cell) {
684 if (null !== $cell && !$cell instanceof TableCell && !\is_scalar($cell) && !$cell instanceof \Stringable) {
685 throw new InvalidArgumentException(sprintf('A cell must be a TableCell, a scalar or an object implementing "__toString()", "%s" given.', get_debug_type($cell)));
686 }
687 if ($cell instanceof TableCell && $cell->getRowspan() > 1) {
688 $nbLines = $cell->getRowspan() - 1;
689 $lines = [$cell];
690 if (str_contains($cell, "\n")) {
691 $lines = explode("\n", str_replace("\n", "<fg=default;bg=default>\n</>", $cell));
692 $nbLines = \count($lines) > $nbLines ? substr_count($cell, "\n") : $nbLines;
693
694 $rows[$line][$column] = new TableCell($lines[0], ['colspan' => $cell->getColspan(), 'style' => $cell->getStyle()]);
695 unset($lines[0]);
696 }
697
698 // create a two dimensional array (rowspan x colspan)
699 $unmergedRows = array_replace_recursive(array_fill($line + 1, $nbLines, []), $unmergedRows);
700 foreach ($unmergedRows as $unmergedRowKey => $unmergedRow) {
701 $value = $lines[$unmergedRowKey - $line] ?? '';
702 $unmergedRows[$unmergedRowKey][$column] = new TableCell($value, ['colspan' => $cell->getColspan(), 'style' => $cell->getStyle()]);
703 if ($nbLines === $unmergedRowKey - $line) {
704 break;
705 }
706 }
707 }
708 }
709
710 foreach ($unmergedRows as $unmergedRowKey => $unmergedRow) {
711 // we need to know if $unmergedRow will be merged or inserted into $rows
712 if (isset($rows[$unmergedRowKey]) && \is_array($rows[$unmergedRowKey]) && ($this->getNumberOfColumns($rows[$unmergedRowKey]) + $this->getNumberOfColumns($unmergedRows[$unmergedRowKey]) <= $this->numberOfColumns)) {
713 foreach ($unmergedRow as $cellKey => $cell) {
714 // insert cell into row at cellKey position
715 array_splice($rows[$unmergedRowKey], $cellKey, 0, [$cell]);
716 }
717 } else {
718 $row = $this->copyRow($rows, $unmergedRowKey - 1);
719 foreach ($unmergedRow as $column => $cell) {
720 if (!empty($cell)) {
721 $row[$column] = $unmergedRow[$column];
722 }
723 }
724 array_splice($rows, $unmergedRowKey, 0, [$row]);
725 }
726 }
727
728 return $rows;
729 }

797 {
