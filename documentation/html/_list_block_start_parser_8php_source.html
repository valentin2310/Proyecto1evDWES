43 public function tryStart(Cursor $cursor, MarkdownParserStateInterface $parserState): ?BlockStart
44 {
45 if ($cursor->isIndented()) {
46 return BlockStart::none();
47 }
48
49 $listData = $this->parseList($cursor, $parserState->getParagraphContent() !== null);

52 }

60 // We start out with assuming a list is tight. If we find a blank line, we set it to loose later.

62
63 return BlockStart::of($listBlockParser, $listItemParser)->at($cursor);
64 }

69 private function parseList(Cursor $cursor, bool $inParagraph): ?ListData
70 {
71 $indent = $cursor->getIndent();
72
73 $tmpCursor = clone $cursor;
74 $tmpCursor->advanceToNextNonSpaceOrTab();
75 $rest = $tmpCursor->getRemainder();
76
77 if (\preg_match($this->listMarkerRegex ?? $this->generateListMarkerRegex(), $rest) === 1) {
78 $data = new ListData();
79 $data->markerOffset = $indent;
80 $data->type = ListBlock::TYPE_BULLET;
81 $data->delimiter = null;
82 $data->bulletChar = $rest[0];
83 $markerLength = 1;
84 } elseif (($matches = RegexHelper::matchFirst('/^(\d{1,9})([.)])/', $rest)) && (! $inParagraph || $matches[1] === '1')) {
85 $data = new ListData();
86 $data->markerOffset = $indent;
87 $data->type = ListBlock::TYPE_ORDERED;
88 $data->start = (int) $matches[1];
89 $data->delimiter = $matches[2] === '.' ? ListBlock::DELIM_PERIOD : ListBlock::DELIM_PAREN;
90 $data->bulletChar = null;
91 $markerLength = \strlen($matches[0]);
92 } else {
93 return null;
94 }
95
96 // Make sure we have spaces after
97 $nextChar = $tmpCursor->peek($markerLength);
98 if (! ($nextChar === null || $nextChar === "\t" || $nextChar === ' ')) {
99 return null;
100 }
101
102 // If it interrupts paragraph, make sure first line isn't blank
103 if ($inParagraph && ! RegexHelper::matchAt(RegexHelper::REGEX_NON_SPACE, $rest, $markerLength)) {
104 return null;
105 }
106
107 $cursor->advanceToNextNonSpaceOrTab(); // to start of marker
108 $cursor->advanceBy($markerLength, true); // to end of marker
109 $data->padding = self::calculateListMarkerPadding($cursor, $markerLength);
110
111 return $data;
112 }

114 private static function calculateListMarkerPadding(Cursor $cursor, int $markerLength): int
115 {
116 $start = $cursor->saveState();
117 $spacesStartCol = $cursor->getColumn();
118
119 while ($cursor->getColumn() - $spacesStartCol < 5) {
120 if (! $cursor->advanceBySpaceOrTab()) {
121 break;
122 }
123 }
124
125 $blankItem = $cursor->peek() === null;
126 $spacesAfterMarker = $cursor->getColumn() - $spacesStartCol;
127
128 if ($spacesAfterMarker >= 5 || $spacesAfterMarker < 1 || $blankItem) {
129 $cursor->restoreState($start);
130 $cursor->advanceBySpaceOrTab();
131
132 return $markerLength + 1;
133 }
134
135 return $markerLength + $spacesAfterMarker;
136 }
