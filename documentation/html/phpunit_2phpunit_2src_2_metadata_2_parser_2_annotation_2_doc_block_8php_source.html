124 public function requirements(): array
125 {
126 if ($this->parsedRequirements !== null) {
127 return $this->parsedRequirements;
128 }
129
130 $offset = $this->startLine;
131 $requires = [];
132 $recordedSettings = [];
133 $extensionVersions = [];
134 $recordedOffsets = [
135 '__FILE' => realpath($this->fileName),
136 ];
137
138 // Trim docblock markers, split it into lines and rewind offset to start of docblock
139 $lines = preg_replace(['#^/\*{2}#', '#\*/$#'], '', preg_split('/\r\n|\r|\n/', $this->docComment));
140 $offset -= count($lines);
141
142 foreach ($lines as $line) {
143 if (preg_match(self::REGEX_REQUIRES_OS, $line, $matches)) {
144 $requires[$matches['name']] = $matches['value'];
145 $recordedOffsets[$matches['name']] = $offset;
146 }
147
148 if (preg_match(self::REGEX_REQUIRES_VERSION, $line, $matches)) {
149 $requires[$matches['name']] = [
150 'version' => $matches['version'],
151 'operator' => $matches['operator'],
152 ];
153
154 $recordedOffsets[$matches['name']] = $offset;
155 }
156
157 if (preg_match(self::REGEX_REQUIRES_VERSION_CONSTRAINT, $line, $matches)) {
158 if (!empty($requires[$matches['name']])) {
159 $offset++;
160
161 continue;
162 }
163
164 try {
165 $versionConstraintParser = new VersionConstraintParser;
166
167 $requires[$matches['name'] . '_constraint'] = [
168 'constraint' => $versionConstraintParser->parse(trim($matches['constraint'])),
169 ];
170
171 $recordedOffsets[$matches['name'] . '_constraint'] = $offset;
172 } catch (PharIoVersionException $e) {
173 throw new InvalidVersionRequirementException(
174 $e->getMessage(),
175 $e->getCode(),
176 $e,
177 );
178 }
179 }
180
181 if (preg_match(self::REGEX_REQUIRES_SETTING, $line, $matches)) {
182 $recordedSettings[$matches['setting']] = $matches['value'];
183 $recordedOffsets['__SETTING_' . $matches['setting']] = $offset;
184 }
185
186 if (preg_match(self::REGEX_REQUIRES, $line, $matches)) {
187 $name = $matches['name'] . 's';
188
189 if (!isset($requires[$name])) {
190 $requires[$name] = [];
191 }
192
193 $requires[$name][] = $matches['value'];
194 $recordedOffsets[$matches['name'] . '_' . $matches['value']] = $offset;
195
196 if ($name === 'extensions' && !empty($matches['version'])) {
197 $extensionVersions[$matches['value']] = [
198 'version' => $matches['version'],
199 'operator' => $matches['operator'],
200 ];
201 }
202 }
203
204 $offset++;
205 }
206
207 return $this->parsedRequirements = array_merge(
208 $requires,
209 ['__OFFSET' => $recordedOffsets],
210 array_filter(
211 [
212 'setting' => $recordedSettings,
213 'extension_versions' => $extensionVersions,
214 ],
215 ),
216 );
217 }

227 private static function parseDocBlock(string $docBlock): array
228 {
229 // Strip away the docblock header and footer to ease parsing of one line annotations
230 $docBlock = substr($docBlock, 3, -2);
231 $annotations = [];
232
233 if (preg_match_all('/@(?P<name>[A-Za-z_-]+)(?:[ \t]+(?P<value>.*?))?[ \t]*\r?$/m', $docBlock, $matches)) {
234 $numMatches = count($matches[0]);
235
236 for ($i = 0; $i < $numMatches; $i++) {
237 $annotations[$matches['name'][$i]][] = $matches['value'][$i];
238 }
239 }
240
241 return $annotations;
242 }

244 private static function extractAnnotationsFromReflector(ReflectionClass|ReflectionFunctionAbstract $reflector): array
245 {

256 }
