192 public function handle(Request $request, int $type = HttpKernelInterface::MAIN_REQUEST, bool $catch = true): Response
193 {
194 // FIXME: catch exceptions and implement a 500 error page here? -> in Varnish, there is a built-in error page mechanism
195 if (HttpKernelInterface::MAIN_REQUEST === $type) {
196 $this->traces = [];
197 // Keep a clone of the original request for surrogates so they can access it.
198 // We must clone here to get a separate instance because the application will modify the request during
199 // the application flow (we know it always does because we do ourselves by setting REMOTE_ADDR to 127.0.0.1
200 // and adding the X-Forwarded-For header, see HttpCache::forward()).
201 $this->request = clone $request;
202 if (null !== $this->surrogate) {
203 $this->surrogateCacheStrategy = $this->surrogate->createCacheStrategy();
204 }
205 }
206
207 $this->traces[$this->getTraceKey($request)] = [];
208
209 if (!$request->isMethodSafe()) {
210 $response = $this->invalidate($request, $catch);
211 } elseif ($request->headers->has('expect') || !$request->isMethodCacheable()) {
212 $response = $this->pass($request, $catch);
213 } elseif ($this->options['allow_reload'] && $request->isNoCache()) {
214 /*
215 If allow_reload is configured and the client requests "Cache-Control: no-cache",
216 reload the cache by fetching a fresh response and caching it (if possible).
217 */
218 $this->record($request, 'reload');
219 $response = $this->fetch($request, $catch);
220 } else {
221 $response = $this->lookup($request, $catch);
222 }
223
224 $this->restoreResponseBody($request, $response);
225
226 if (HttpKernelInterface::MAIN_REQUEST === $type) {
227 $this->addTraces($response);
228 }
229
230 if (null !== $this->surrogate) {
231 if (HttpKernelInterface::MAIN_REQUEST === $type) {
232 $this->surrogateCacheStrategy->update($response);
233 } else {
234 $this->surrogateCacheStrategy->add($response);
235 }
236 }
237
238 $response->prepare($request);
239
240 $response->isNotModified($request);
241
242 return $response;
243 }

286 {
287 $response = $this->pass($request, $catch);
288
289 // invalidate only when the response is successful
290 if ($response->isSuccessful() || $response->isRedirect()) {
291 try {
292 $this->store->invalidate($request);
293

295 foreach (['Location', 'Content-Location'] as $header) {
296 if ($uri = $response->headers->get($header)) {
297 $subRequest = Request::create($uri, 'get', [], [], [], $request->server->all());

300 }
301 }
302
303 $this->record($request, 'invalidate');
304 } catch (\Exception $e) {

306
307 if ($this->options['debug']) {

330 {
331 try {

374 protected function validate(Request $request, Response $entry, bool $catch = false): Response
375 {
376 $subRequest = clone $request;
377
378 // send no head requests because we want content
379 if ('HEAD' === $request->getMethod()) {
380 $subRequest->setMethod('GET');
381 }
382
383 // add our cached last-modified validator
384 if ($entry->headers->has('Last-Modified')) {
385 $subRequest->headers->set('If-Modified-Since', $entry->headers->get('Last-Modified'));
386 }
387
388 // Add our cached etag validator to the environment.
389 // We keep the etags from the client to handle the case when the client
390 // has a different private valid entry which is not cached here.
391 $cachedEtags = $entry->getEtag() ? [$entry->getEtag()] : [];
392 $requestEtags = $request->getETags();
393 if ($etags = array_unique(array_merge($cachedEtags, $requestEtags))) {
394 $subRequest->headers->set('If-None-Match', implode(', ', $etags));
395 }
396
397 $response = $this->forward($subRequest, $catch, $entry);
398
399 if (304 == $response->getStatusCode()) {
400 $this->record($request, 'valid');
401
402 // return the response and not the cache entry if the response is valid but not cached
403 $etag = $response->getEtag();
404 if ($etag && \in_array($etag, $requestEtags) && !\in_array($etag, $cachedEtags)) {
405 return $response;
406 }
407
408 $entry = clone $entry;
409 $entry->headers->remove('Date');
410
411 foreach (['Date', 'Expires', 'Cache-Control', 'ETag', 'Last-Modified'] as $name) {
412 if ($response->headers->has($name)) {
413 $entry->headers->set($name, $response->headers->get($name));
414 }
415 }
416
417 $response = $entry;
418 } else {
419 $this->record($request, 'invalid');
420 }
421
422 if ($response->isCacheable()) {
423 $this->store($request, $response);
424 }
425
426 return $response;
427 }

475 /*
476 * Support stale-if-error given on Responses or as a config option.
477 * RFC 7234 summarizes in Section 4.2.4 (but also mentions with the individual
478 * Cache-Control directives) that
479 *
480 * A cache MUST NOT generate a stale response if it is prohibited by an
481 * explicit in-protocol directive (e.g., by a "no-store" or "no-cache"
482 * cache directive, a "must-revalidate" cache-response-directive, or an
483 * applicable "s-maxage" or "proxy-revalidate" cache-response-directive;
484 * see Section 5.2.2).
485 *
486 * https://tools.ietf.org/html/rfc7234#section-4.2.4
487 *
488 * We deviate from this in one detail, namely that we *do* serve entries in the
489 * stale-if-error case even if they have a `s-maxage` Cache-Control directive.
490 */
491 if (null !== $entry
492 && \in_array($response->getStatusCode(), [500, 502, 503, 504])
493 && !$entry->headers->hasCacheControlDirective('no-cache')
494 && !$entry->mustRevalidate()
495 ) {
496 if (null === $age = $entry->headers->getCacheControlDirective('stale-if-error')) {
497 $age = $this->options['stale_if_error'];
498 }
499
500 /*
501 * stale-if-error gives the (extra) time that the Response may be used *after* it has become stale.
502 * So we compare the time the $entry has been sitting in the cache already with the
503 * time it was fresh plus the allowed grace period.
504 */
505 if ($entry->getAge() <= $entry->getMaxAge() + $age) {
506 $this->record($request, 'stale-if-error');
507
508 return $entry;
509 }
510 }

512 /*
513 RFC 7231 Sect. 7.1.1.2 says that a server that does not have a reasonably accurate
514 clock MUST NOT send a "Date" header, although it MUST send one in most other cases
515 except for 1xx or 5xx responses where it MAY do so.
516
517 Anyway, a client that received a message without a "Date" header MUST add it.
518 */
519 if (!$response->headers->has('Date')) {
520 $response->setDate(\DateTimeImmutable::createFromFormat('U', time()));
521 }
522
523 $this->processResponseBody($request, $response);
524
525 if ($this->isPrivateRequest($request) && !$response->headers->hasCacheControlDirective('public')) {
526 $response->setPrivate();
527 } elseif ($this->options['default_ttl'] > 0 && null === $response->getTtl() && !$response->headers->getCacheControlDirective('must-revalidate')) {
528 $response->setTtl($this->options['default_ttl']);
529 }
530
531 return $response;

537 protected function isFreshEnough(Request $request, Response $entry): bool
538 {
539 if (!$entry->isFresh()) {
540 return $this->lock($request, $entry);
541 }
542

555 protected function lock(Request $request, Response $entry): bool
556 {
557 // try to acquire a lock to call the backend
558 $lock = $this->store->lock($request);
559
560 if (true === $lock) {
561 // we have the lock, call the backend
562 return false;
563 }
564
565 // there is already another process calling the backend
566
567 // May we serve a stale response?
568 if ($this->mayServeStaleWhileRevalidate($entry)) {
569 $this->record($request, 'stale-while-revalidate');
570
571 return true;
572 }
573
574 // wait for the lock to be released
575 if ($this->waitForLock($request)) {
576 // replace the current entry with the fresh one
577 $new = $this->lookup($request);
578 $entry->headers = $new->headers;
579 $entry->setContent($new->getContent());
580 $entry->setStatusCode($new->getStatusCode());
581 $entry->setProtocolVersion($new->getProtocolVersion());
582 foreach ($new->headers->getCookies() as $cookie) {
583 $entry->headers->setCookie($cookie);
584 }
585 } else {
586 // backend is slow as hell, send a 503 response (to avoid the dog pile effect)
587 $entry->setStatusCode(503);
588 $entry->setContent('503 Service Unavailable');
589 $entry->headers->set('Retry-After', 10);
590 }
591
592 return true;
593 }

602 protected function store(Request $request, Response $response)
603 {
604 try {
605 $restoreHeaders = [];
606 foreach ($this->options['skip_response_headers'] as $header) {
607 if (!$response->headers->has($header)) {
608 continue;
609 }
610
611 $restoreHeaders[$header] = $response->headers->all($header);
612 $response->headers->remove($header);
613 }
614
615 $this->store->write($request, $response);
616 $this->record($request, 'store');
617
618 $response->headers->set('Age', $response->getAge());
619 } catch (\Exception $e) {
620 $this->record($request, 'store-failed');
621
622 if ($this->options['debug']) {
623 throw $e;
624 }
625 } finally {
626 foreach ($restoreHeaders as $header => $values) {
627 $response->headers->set($header, $values);
628 }
629 }
630
631 // now that the response is cached, release the lock
632 $this->store->unlock($request);
633 }

638 private function restoreResponseBody(Request $request, Response $response): void
639 {
640 if ($response->headers->has('X-Body-Eval')) {
641 \assert(self::BODY_EVAL_BOUNDARY_LENGTH === 24);
642
643 ob_start();
644
645 $content = $response->getContent();
646 $boundary = substr($content, 0, 24);
647 $j = strpos($content, $boundary, 24);
648 echo substr($content, 24, $j - 24);
649 $i = $j + 24;
650
651 while (false !== $j = strpos($content, $boundary, $i)) {
652 [$uri, $alt, $ignoreErrors, $part] = explode("\n", substr($content, $i, $j - $i), 4);
653 $i = $j + 24;
654
655 echo $this->surrogate->handle($this, $uri, $alt, $ignoreErrors);
656 echo $part;
657 }
658
659 $response->setContent(ob_get_clean());
660 $response->headers->remove('X-Body-Eval');
661 if (!$response->headers->has('Transfer-Encoding')) {
662 $response->headers->set('Content-Length', \strlen($response->getContent()));
663 }
664 } elseif ($response->headers->has('X-Body-File')) {
665 // Response does not include possibly dynamic content (ESI, SSI), so we need
666 // not handle the content for HEAD requests
667 if (!$request->isMethod('HEAD')) {
668 $response->setContent(file_get_contents($response->headers->get('X-Body-File')));
669 }
670 } else {
671 return;
672 }
673
674 $response->headers->remove('X-Body-File');
675 }
