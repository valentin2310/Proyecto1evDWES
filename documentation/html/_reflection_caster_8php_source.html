398 public static function getSignature(array $a)
399 {
400 $prefix = Caster::PREFIX_VIRTUAL;
401 $signature = '';
402
403 if (isset($a[$prefix.'parameters'])) {
404 foreach ($a[$prefix.'parameters']->value as $k => $param) {
405 $signature .= ', ';
406 if ($type = $param->getType()) {
407 if (!$type instanceof \ReflectionNamedType) {
408 $signature .= $type.' ';
409 } else {
410 if (!$param->isOptional() && $param->allowsNull() && 'mixed' !== $type->getName()) {
411 $signature .= '?';
412 }
413 $signature .= substr(strrchr('\\'.$type->getName(), '\\'), 1).' ';
414 }
415 }
416 $signature .= $k;
417
418 if (!$param->isDefaultValueAvailable()) {
419 continue;
420 }
421 $v = $param->getDefaultValue();
422 $signature .= ' = ';
423
424 if ($param->isDefaultValueConstant()) {
425 $signature .= substr(strrchr('\\'.$param->getDefaultValueConstantName(), '\\'), 1);
426 } elseif (null === $v) {
427 $signature .= 'null';
428 } elseif (\is_array($v)) {
429 $signature .= $v ? '[โฆ'.\count($v).']' : '[]';
430 } elseif (\is_string($v)) {
431 $signature .= 10 > \strlen($v) && !str_contains($v, '\\') ? "'{$v}'" : "'โฆ".\strlen($v)."'";
432 } elseif (\is_bool($v)) {
433 $signature .= $v ? 'true' : 'false';
434 } elseif (\is_object($v)) {
435 $signature .= 'new '.substr(strrchr('\\'.get_debug_type($v), '\\'), 1);
436 } else {
437 $signature .= $v;
438 }
439 }
440 }
441 $signature = (empty($a[$prefix.'returnsReference']) ? '' : '&').'('.substr($signature, 2).')';
442
443 if (isset($a[$prefix.'returnType'])) {
444 $signature .= ': '.substr(strrchr('\\'.$a[$prefix.'returnType'], '\\'), 1);
445 }
446
447 return $signature;
