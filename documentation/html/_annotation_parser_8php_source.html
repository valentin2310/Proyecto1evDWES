54
55 break;
56
57 case 'backupStaticAttributes':
58 case 'backupStaticProperties':
59 $result[] = Metadata::backupStaticPropertiesOnClass($this->stringToBool($values[0]));
60
61 break;

65 $value = $this->cleanUpCoversOrUsesTarget($value);
66
67 $result[] = Metadata::coversOnClass($value);

70 break;
71
72 case 'coversDefaultClass':

74 $result[] = Metadata::coversDefaultClass($value);

90 case 'ticket':

169 public function forMethod(string $className, string $methodName): MetadataCollection
170 {
171 $result = [];
172
173 foreach (AnnotationRegistry::getInstance()->forMethod($className, $methodName)->symbolAnnotations() as $annotation => $values) {
174 switch ($annotation) {
175 case 'after':
176 $result[] = Metadata::after();
177
178 break;
179
180 case 'afterClass':
181 $result[] = Metadata::afterClass();
182
183 break;
184
185 case 'backupGlobals':
186 $result[] = Metadata::backupGlobalsOnMethod($this->stringToBool($values[0]));
187
188 break;
189
190 case 'backupStaticAttributes':
191 case 'backupStaticProperties':
192 $result[] = Metadata::backupStaticPropertiesOnMethod($this->stringToBool($values[0]));
193
194 break;
195
196 case 'before':
197 $result[] = Metadata::before();
198
199 break;
200
201 case 'beforeClass':
202 $result[] = Metadata::beforeClass();
203
204 break;
205
206 case 'covers':
207 foreach ($values as $value) {
208 $value = $this->cleanUpCoversOrUsesTarget($value);
209
210 $result[] = Metadata::coversOnMethod($value);
211 }
212
213 break;
214
215 case 'coversNothing':
216 $result[] = Metadata::coversNothingOnMethod();
217
218 break;
219
220 case 'dataProvider':
221 foreach ($values as $value) {
222 $value = rtrim($value, " ()\n\r\t\v\x00");
223
224 if (str_contains($value, '::')) {
225 $result[] = Metadata::dataProvider(...explode('::', $value));
226
227 continue;
228 }
229
230 $result[] = Metadata::dataProvider($className, $value);
231 }
232
233 break;
234
235 case 'depends':
236 foreach ($values as $value) {
237 $deepClone = false;
238 $shallowClone = false;
239
240 if (str_starts_with($value, 'clone ')) {
241 $deepClone = true;
242 $value = substr($value, strlen('clone '));
243 } elseif (str_starts_with($value, '!clone ')) {
244 $value = substr($value, strlen('!clone '));
245 } elseif (str_starts_with($value, 'shallowClone ')) {
246 $shallowClone = true;
247 $value = substr($value, strlen('shallowClone '));
248 } elseif (str_starts_with($value, '!shallowClone ')) {
249 $value = substr($value, strlen('!shallowClone '));
250 }
251
252 if (str_contains($value, '::')) {
253 [$_className, $_methodName] = explode('::', $value);
254
255 assert($_className !== '');
256 assert($_methodName !== '');
257
258 if ($_methodName === 'class') {
259 $result[] = Metadata::dependsOnClass($_className, $deepClone, $shallowClone);
260
261 continue;
262 }
263
264 $result[] = Metadata::dependsOnMethod($_className, $_methodName, $deepClone, $shallowClone);
265
266 continue;
267 }
268
269 $result[] = Metadata::dependsOnMethod($className, $value, $deepClone, $shallowClone);
270 }
271
272 break;
273
274 case 'doesNotPerformAssertions':
275 $result[] = Metadata::doesNotPerformAssertionsOnMethod();
276
277 break;
278
279 case 'excludeGlobalVariableFromBackup':
280 foreach ($values as $value) {
281 $result[] = Metadata::excludeGlobalVariableFromBackupOnMethod($value);
282 }
283
284 break;
285
286 case 'excludeStaticPropertyFromBackup':
287 foreach ($values as $value) {
288 $tmp = explode(' ', $value);
289
290 if (count($tmp) !== 2) {
291 continue;
292 }
293
294 $result[] = Metadata::excludeStaticPropertyFromBackupOnMethod(
295 trim($tmp[0]),
296 trim($tmp[1]),
297 );
298 }
299
300 break;
301
302 case 'group':
303 case 'ticket':
304 foreach ($values as $value) {
305 $result[] = Metadata::groupOnMethod($value);
306 }
307
308 break;
309
310 case 'large':
311 $result[] = Metadata::groupOnMethod('large');
312
313 break;
314
315 case 'medium':
316 $result[] = Metadata::groupOnMethod('medium');
317
318 break;
319
320 case 'postCondition':
321 $result[] = Metadata::postCondition();
322
323 break;
324
325 case 'preCondition':
326 $result[] = Metadata::preCondition();
327
328 break;
329
330 case 'preserveGlobalState':
331 $result[] = Metadata::preserveGlobalStateOnMethod($this->stringToBool($values[0]));
332
333 break;
334
335 case 'runInSeparateProcess':
336 $result[] = Metadata::runInSeparateProcess();
337
338 break;
339
340 case 'small':
341 $result[] = Metadata::groupOnMethod('small');
342
343 break;
344
345 case 'test':
346 $result[] = Metadata::test();
347
348 break;
349
350 case 'testdox':
351 $result[] = Metadata::testDoxOnMethod($values[0]);
352
353 break;
354
355 case 'uses':
356 foreach ($values as $value) {
357 $value = $this->cleanUpCoversOrUsesTarget($value);
358
359 $result[] = Metadata::usesOnMethod($value);
360 }
361
362 break;
363 }
364 }
365
366 if (method_exists($className, $methodName)) {
367 $result = array_merge(
368 $result,
369 $this->parseRequirements(
370 AnnotationRegistry::getInstance()->forMethod($className, $methodName)->requirements(),
371 'method',
372 ),
373 );
374 }
375
376 return MetadataCollection::fromArray($result);
377 }

415 private function parseRequirements(array $requirements, string $level): array
416 {
417 $result = [];
418
419 if (!empty($requirements['PHP'])) {
420 $versionRequirement = new ComparisonRequirement(
421 $requirements['PHP']['version'],
422 new VersionComparisonOperator(empty($requirements['PHP']['operator']) ? '>=' : $requirements['PHP']['operator']),
423 );
424
425 if ($level === 'class') {
426 $result[] = Metadata::requiresPhpOnClass($versionRequirement);
427 } else {
428 $result[] = Metadata::requiresPhpOnMethod($versionRequirement);
429 }
430 } elseif (!empty($requirements['PHP_constraint'])) {
431 $versionRequirement = new ConstraintRequirement($requirements['PHP_constraint']['constraint']);
432
433 if ($level === 'class') {
434 $result[] = Metadata::requiresPhpOnClass($versionRequirement);
435 } else {
436 $result[] = Metadata::requiresPhpOnMethod($versionRequirement);
437 }
438 }
439
440 if (!empty($requirements['extensions'])) {
441 foreach ($requirements['extensions'] as $extension) {
442 if (isset($requirements['extension_versions'][$extension])) {
443 continue;
444 }
445
446 if ($level === 'class') {
447 $result[] = Metadata::requiresPhpExtensionOnClass($extension, null);
448 } else {
449 $result[] = Metadata::requiresPhpExtensionOnMethod($extension, null);
450 }
451 }
452 }
453
454 if (!empty($requirements['extension_versions'])) {
455 foreach ($requirements['extension_versions'] as $extension => $version) {
456 $versionRequirement = new ComparisonRequirement(
457 $version['version'],
458 new VersionComparisonOperator(empty($version['operator']) ? '>=' : $version['operator']),
459 );
460
461 if ($level === 'class') {
462 $result[] = Metadata::requiresPhpExtensionOnClass($extension, $versionRequirement);
463 } else {
464 $result[] = Metadata::requiresPhpExtensionOnMethod($extension, $versionRequirement);
465 }
466 }
467 }
468
469 if (!empty($requirements['PHPUnit'])) {
470 $versionRequirement = new ComparisonRequirement(
471 $requirements['PHPUnit']['version'],
472 new VersionComparisonOperator(empty($requirements['PHPUnit']['operator']) ? '>=' : $requirements['PHPUnit']['operator']),
473 );
474
475 if ($level === 'class') {
476 $result[] = Metadata::requiresPhpunitOnClass($versionRequirement);
477 } else {
478 $result[] = Metadata::requiresPhpunitOnMethod($versionRequirement);
479 }
480 } elseif (!empty($requirements['PHPUnit_constraint'])) {
481 $versionRequirement = new ConstraintRequirement($requirements['PHPUnit_constraint']['constraint']);
482
483 if ($level === 'class') {
484 $result[] = Metadata::requiresPhpunitOnClass($versionRequirement);
485 } else {
486 $result[] = Metadata::requiresPhpunitOnMethod($versionRequirement);
487 }
488 }
489
490 if (!empty($requirements['OSFAMILY'])) {
491 if ($level === 'class') {
492 $result[] = Metadata::requiresOperatingSystemFamilyOnClass($requirements['OSFAMILY']);
493 } else {
494 $result[] = Metadata::requiresOperatingSystemFamilyOnMethod($requirements['OSFAMILY']);
495 }
496 }
497
498 if (!empty($requirements['OS'])) {
499 if ($level === 'class') {
500 $result[] = Metadata::requiresOperatingSystemOnClass($requirements['OS']);
501 } else {
502 $result[] = Metadata::requiresOperatingSystemOnMethod($requirements['OS']);
503 }
504 }
505
506 if (!empty($requirements['functions'])) {
507 foreach ($requirements['functions'] as $function) {
508 $pieces = explode('::', $function);
509
510 if (count($pieces) === 2) {
511 if ($level === 'class') {
512 $result[] = Metadata::requiresMethodOnClass($pieces[0], $pieces[1]);
513 } else {
514 $result[] = Metadata::requiresMethodOnMethod($pieces[0], $pieces[1]);
515 }
516 } elseif ($level === 'class') {
517 $result[] = Metadata::requiresFunctionOnClass($function);
518 } else {
519 $result[] = Metadata::requiresFunctionOnMethod($function);
520 }
521 }
522 }
523
524 if (!empty($requirements['setting'])) {
525 foreach ($requirements['setting'] as $setting => $value) {
526 if ($level === 'class') {
527 $result[] = Metadata::requiresSettingOnClass($setting, $value);
528 } else {
529 $result[] = Metadata::requiresSettingOnMethod($setting, $value);
530 }
531 }
532 }
533
534 return $result;
