192 public function compileChange(Blueprint $blueprint, Fluent $command, Connection $connection)
193 {
194 if (! $connection->usingNativeSchemaOperations()) {
195 return parent::compileChange($blueprint, $command, $connection);
196 }
197
198 $columns = [];
199
200 foreach ($blueprint->getChangedColumns() as $column) {
201 $changes = ['type '.$this->getType($column).$this->modifyCollate($blueprint, $column)];
202
203 foreach ($this->modifiers as $modifier) {
204 if ($modifier === 'Collate') {
205 continue;
206 }
207
208 if (method_exists($this, $method = "modify{$modifier}")) {
209 $constraints = (array) $this->{$method}($blueprint, $column);

211 foreach ($constraints as $constraint) {
212 $changes[] = $constraint;
213 }
214 }
215 }
216
217 $columns[] = implode(', ', $this->prefixArray('alter column '.$this->wrap($column), $changes));
218 }
