243 public function dispatch($event, $payload = [], $halt = false)
244 {
245 // When the given "event" is actually an object we will assume it is an event
246 // object and use the class as the event name and this event itself as the
247 // payload to the handler, which makes object based events quite simple.
248 [$isEventObject, $event, $payload] = [
249 is_object($event),
250 ...$this->parseEventAndPayload($event, $payload),
251 ];
252
253 // If the event is not intended to be dispatched unless the current database
254 // transaction is successful, we'll register a callback which will handle
255 // dispatching this event on the next successful DB transaction commit.
256 if ($isEventObject &&
257 $payload[0] instanceof ShouldDispatchAfterCommit &&
258 ! is_null($transactions = $this->resolveTransactionManager())) {
259 $transactions->addCallback(
260 fn () => $this->invokeListeners($event, $payload, $halt)
261 );
262
263 return null;
264 }
265
266 return $this->invokeListeners($event, $payload, $halt);
267 }

277 protected function invokeListeners($event, $payload, $halt = false)
278 {
279 if ($this->shouldBroadcast($payload)) {
280 $this->broadcastEvent($payload[0]);
281 }
282
283 $responses = [];
284
285 foreach ($this->getListeners($event) as $listener) {
286 $response = $listener($event, $payload);
287
288 // If a response is returned from the listener and event halting is enabled
289 // we will just return this response, and not call the rest of the event
290 // listeners. Otherwise we will add the response on the response list.
291 if ($halt && ! is_null($response)) {
292 return $response;
293 }
294
295 // If a boolean false is returned from a listener, we will stop propagating
296 // the event to any further listeners down in the chain, else we keep on
297 // looping through the listeners and firing every one in our sequence.
298 if ($response === false) {
299 break;
300 }
301
302 $responses[] = $response;
303 }
304
305 return $halt ? null : $responses;
306 }
