95 public static function fromClassReflector(ReflectionClass $class): static
96 {
97 $testSuite = new static($class->getName());
98
99 $constructor = $class->getConstructor();
100
101 if ($constructor !== null && !$constructor->isPublic()) {
102 Event\Facade::emitter()->testRunnerTriggeredWarning(
103 sprintf(
104 'Class "%s" has no public constructor.',
105 $class->getName(),
106 ),
107 );
108
109 return $testSuite;
110 }
111
112 foreach (Reflection::publicMethodsInTestClass($class) as $method) {
113 if ($method->getDeclaringClass()->getName() === Assert::class) {
114 continue;
115 }
116
117 if ($method->getDeclaringClass()->getName() === TestCase::class) {
118 continue;
119 }
120
121 if (!TestUtil::isTestMethod($method)) {
122 continue;
123 }
124
125 $testSuite->addTestMethod($class, $method);
126 }
127
128 if (count($testSuite) === 0) {
129 Event\Facade::emitter()->testRunnerTriggeredWarning(
130 sprintf(
131 'No tests found in class "%s".',
132 $class->getName(),
133 ),
134 );
135 }
136
137 return $testSuite;
138 }

160 {

164 $this->tests[] = $test;

177 $this->groups[$group] = [$test];

181 }
182
183 if ($test instanceof TestCase) {
184 $test->setGroups($groups);

318 public function run(): void
319 {
320 if (count($this) === 0) {
321 return;
322 }
323
324 $emitter = Event\Facade::emitter();
325 $testSuiteValueObjectForEvents = Event\TestSuite\TestSuiteBuilder::from($this);
326
327 $emitter->testSuiteStarted($testSuiteValueObjectForEvents);
328
329 if (!$this->invokeMethodsBeforeFirstTest($emitter, $testSuiteValueObjectForEvents)) {
330 return;
331 }
332
333 foreach ($this as $test) {
334 if (TestResultFacade::shouldStop()) {
335 $emitter->testRunnerExecutionAborted();
336
337 break;
338 }
339
340 $test->run();
341 }
342
343 $this->invokeMethodsAfterLastTest($emitter);
344
345 $emitter->testSuiteFinished($testSuiteValueObjectForEvents);

572 {

584 continue;
