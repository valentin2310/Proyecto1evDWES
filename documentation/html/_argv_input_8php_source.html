78 protected function parseToken(string $token, bool $parseOptions): bool
79 {
80 if ($parseOptions && '' == $token) {
81 $this->parseArgument($token);
82 } elseif ($parseOptions && '--' == $token) {
83 return false;
84 } elseif ($parseOptions && str_starts_with($token, '--')) {
85 $this->parseLongOption($token);
86 } elseif ($parseOptions && '-' === $token[0] && '-' !== $token) {
87 $this->parseShortOption($token);
88 } else {
89 $this->parseArgument($token);
90 }
91
92 return $parseOptions;

98 private function parseShortOption(string $token): void

119 private function parseShortOptionSet(string $name): void
120 {
121 $len = \strlen($name);
122 for ($i = 0; $i < $len; ++$i) {
123 if (!$this->definition->hasShortcut($name[$i])) {
124 $encoding = mb_detect_encoding($name, null, true);
125 throw new RuntimeException(sprintf('The "-%s" option does not exist.', false === $encoding ? $name[$i] : mb_substr($name, $i, 1, $encoding)));
126 }
127
128 $option = $this->definition->getOptionForShortcut($name[$i]);
129 if ($option->acceptValue()) {
130 $this->addLongOption($option->getName(), $i === $len - 1 ? null : substr($name, $i + 1));

133 } else {

161 private function parseArgument(string $token): void
162 {
163 $c = \count($this->arguments);
164
165 // if input is expecting another argument, add it
166 if ($this->definition->hasArgument($c)) {
167 $arg = $this->definition->getArgument($c);
168 $this->arguments[$arg->getName()] = $arg->isArray() ? [$token] : $token;
169
170 // if last argument isArray(), append token to last argument
171 } elseif ($this->definition->hasArgument($c - 1) && $this->definition->getArgument($c - 1)->isArray()) {
172 $arg = $this->definition->getArgument($c - 1);
173 $this->arguments[$arg->getName()][] = $token;
174
175 // unexpected argument
176 } else {
177 $all = $this->definition->getArguments();
178 $symfonyCommandName = null;
179 if (($inputArgument = $all[$key = array_key_first($all)] ?? null) && 'command' === $inputArgument->getName()) {
180 $symfonyCommandName = $this->arguments['command'] ?? null;
181 unset($all[$key]);
182 }
183
184 if (\count($all)) {
185 if ($symfonyCommandName) {
186 $message = sprintf('Too many arguments to "%s" command, expected arguments "%s".', $symfonyCommandName, implode('" "', array_keys($all)));
187 } else {
188 $message = sprintf('Too many arguments, expected arguments "%s".', implode('" "', array_keys($all)));
189 }
190 } elseif ($symfonyCommandName) {
191 $message = sprintf('No arguments expected for "%s" command, got "%s".', $symfonyCommandName, $token);
192 } else {
193 $message = sprintf('No arguments expected, got "%s".', $token);

219 private function addLongOption(string $name, mixed $value): void
220 {
221 if (!$this->definition->hasOption($name)) {
222 if (!$this->definition->hasNegation($name)) {
223 throw new RuntimeException(sprintf('The "--%s" option does not exist.', $name));
224 }
225
226 $optionName = $this->definition->negationToName($name);
227 if (null !== $value) {
228 throw new RuntimeException(sprintf('The "--%s" option does not accept a value.', $name));
229 }
230 $this->options[$optionName] = false;
231
232 return;
233 }
234
235 $option = $this->definition->getOption($name);
236
237 if (null !== $value && !$option->acceptValue()) {
238 throw new RuntimeException(sprintf('The "--%s" option does not accept a value.', $name));
239 }
240
241 if (\in_array($value, ['', null], true) && $option->acceptValue() && \count($this->parsed)) {
242 // if option accepts an optional or mandatory argument
243 // let's see if there is one provided
244 $next = array_shift($this->parsed);
245 if ((isset($next[0]) && '-' !== $next[0]) || \in_array($next, ['', null], true)) {
246 $value = $next;
247 } else {
248 array_unshift($this->parsed, $next);
249 }
250 }
251
252 if (null === $value) {
253 if ($option->isValueRequired()) {
254 throw new RuntimeException(sprintf('The "--%s" option requires a value.', $name));
255 }
256
257 if (!$option->isArray() && !$option->isValueOptional()) {
258 $value = true;
259 }
260 }
261
262 if ($option->isArray()) {
263 $this->options[$name][] = $value;
264 } else {
265 $this->options[$name] = $value;
266 }
267 }

269 public function getFirstArgument(): ?string
270 {
271 $isOption = false;
272 foreach ($this->tokens as $i => $token) {
273 if ($token && '-' === $token[0]) {
274 if (str_contains($token, '=') || !isset($this->tokens[$i + 1])) {
275 continue;
276 }
277
278 // If it's a long option, consider that everything after "--" is the option name.
279 // Otherwise, use the last char (if it's a short option set, only the last one can take a value with space separator)
280 $name = '-' === $token[1] ? substr($token, 2) : substr($token, -1);
281 if (!isset($this->options[$name]) && !$this->definition->hasShortcut($name)) {
282 // noop
283 } elseif ((isset($this->options[$name]) || isset($this->options[$name = $this->definition->shortcutToName($name)])) && $this->tokens[$i + 1] === $this->options[$name]) {
284 $isOption = true;
285 }
286
287 continue;
288 }
289
290 if ($isOption) {
291 $isOption = false;
292 continue;
293 }
294
295 return $token;
296 }
297
298 return null;

301 public function hasParameterOption(string|array $values, bool $onlyParams = false): bool
302 {
303 $values = (array) $values;
304
305 foreach ($this->tokens as $token) {
306 if ($onlyParams && '--' === $token) {
307 return false;
308 }
309 foreach ($values as $value) {
310 // Options with values:
311 // For long options, test for '--option=' at beginning
312 // For short options, test for '-o' at beginning
313 $leading = str_starts_with($value, '--') ? $value.'=' : $value;
314 if ($token === $value || '' !== $leading && str_starts_with($token, $leading)) {
315 return true;
316 }
317 }
318 }
319
320 return false;
321 }

323 public function getParameterOption(string|array $values, string|bool|int|float|array|null $default = false, bool $onlyParams = false): mixed
324 {
325 $values = (array) $values;
326 $tokens = $this->tokens;
327
328 while (0 < \count($tokens)) {
329 $token = array_shift($tokens);
330 if ($onlyParams && '--' === $token) {
331 return $default;
332 }
333
334 foreach ($values as $value) {
335 if ($token === $value) {
336 return array_shift($tokens);
337 }
338 // Options with values:
339 // For long options, test for '--option=' at beginning
340 // For short options, test for '-o' at beginning
341 $leading = str_starts_with($value, '--') ? $value.'=' : $value;
342 if ('' !== $leading && str_starts_with($token, $leading)) {
343 return substr($token, \strlen($leading));
344 }
345 }
346 }
