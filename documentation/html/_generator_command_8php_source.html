149 public function handle()
150 {
151 // First we need to ensure that the given name is not a reserved word within the PHP
152 // language and that the class name will actually be valid. If it is not valid we
153 // can error now and prevent from polluting the filesystem using invalid files.
154 if ($this->isReservedName($this->getNameInput())) {
155 $this->components->error('The name "'.$this->getNameInput().'" is reserved by PHP.');
156
157 return false;
158 }
159
160 $name = $this->qualifyClass($this->getNameInput());
161
162 $path = $this->getPath($name);
163
164 // Next, We will check to see if the class already exists. If it does, we don't want
165 // to create the class and overwrite the user's code. So, we will bail out so the
166 // code is untouched. Otherwise, we will continue generating this class' files.
167 if ((! $this->hasOption('force') ||
168 ! $this->option('force')) &&
169 $this->alreadyExists($this->getNameInput())) {
170 $this->components->error($this->type.' already exists.');
171
172 return false;
173 }
174
175 // Next, we will generate the path to the location where this class' file should get
176 // written. Then, we will build the class and make the proper replacements on the
177 // stub files so that it gets the correctly formatted namespace and class name.
178 $this->makeDirectory($path);
179
180 $this->files->put($path, $this->sortImports($this->buildClass($name)));
181
182 $info = $this->type;
183
184 if (in_array(CreatesMatchingTest::class, class_uses_recursive($this))) {
185 if ($this->handleTestCreation($path)) {
186 $info .= ' and test';
187 }
188 }
189
190 $this->components->info(sprintf('%s [%s] created successfully.', $info, $path));
191 }
