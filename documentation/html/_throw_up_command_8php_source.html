113 private function prepareArgs(string $code = null): array
114 {
115 if (!$code) {
116 // Default to last exception if nothing else was supplied
117 return [new Arg(new Variable('_e'))];
118 }
119
120 $nodes = $this->parser->parse($code);
121 if (\count($nodes) !== 1) {
122 throw new \InvalidArgumentException('No idea how to throw this');
123 }
124
125 $node = $nodes[0];
126
127 // Make this work for PHP Parser v3.x
128 $expr = isset($node->expr) ? $node->expr : $node;
129
130 $args = [new Arg($expr, false, false, $node->getAttributes())];
131
132 // Allow throwing via a string, e.g. `throw-up "SUP"`
133 if ($expr instanceof String_) {
134 return [new New_(new FullyQualifiedName(\Exception::class), $args)];
135 }
136
