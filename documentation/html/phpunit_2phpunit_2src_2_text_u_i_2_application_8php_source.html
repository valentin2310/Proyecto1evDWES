271 private function loadBootstrapScript(string $filename): void
272 {
273 if (!is_readable($filename)) {
274 $this->exitWithErrorMessage(
275 sprintf(
276 'Cannot open bootstrap script "%s"',
277 $filename,
278 ),
279 );
280 }
281
282 try {
283 include_once $filename;
284 } catch (Throwable $t) {
285 $message = sprintf(
286 'Error in bootstrap script: %s:%s%s%s%s',
287 $t::class,
288 PHP_EOL,
289 $t->getMessage(),
290 PHP_EOL,
291 $t->getTraceAsString(),
292 );
293
294 while ($t = $t->getPrevious()) {
295 $message .= sprintf(
296 '%s%sPrevious error: %s:%s%s%s%s',
297 PHP_EOL,
298 PHP_EOL,
299 $t::class,
300 PHP_EOL,
301 $t->getMessage(),
302 PHP_EOL,
303 $t->getTraceAsString(),
304 );
305 }
306
307 $this->exitWithErrorMessage($message);
308 }
309
310 EventFacade::emitter()->testRunnerBootstrapFinished($filename);
311 }
