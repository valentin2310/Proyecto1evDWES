383 private function writeToSocket(string $data): void
384 {
385 $length = strlen($data);
386 $sent = 0;
387 $this->lastSentBytes = $sent;
388 while ($this->isConnected() && $sent < $length) {
389 if (0 == $sent) {
390 $chunk = $this->fwrite($data);
391 } else {
392 $chunk = $this->fwrite(substr($data, $sent));
393 }
394 if ($chunk === false) {
395 throw new \RuntimeException("Could not write to socket");
396 }
397 $sent += $chunk;
398 $socketInfo = $this->streamGetMetadata();
399 if (is_array($socketInfo) && (bool) $socketInfo['timed_out']) {
400 throw new \RuntimeException("Write timed-out");
401 }
402
403 if ($this->writingIsTimedOut($sent)) {
404 throw new \RuntimeException("Write timed-out, no data sent for `{$this->writingTimeout}` seconds, probably we got disconnected (sent $sent of $length)");
405 }
406 }
407 if (!$this->isConnected() && $sent < $length) {
408 throw new \RuntimeException("End-of-file reached, probably we got disconnected (sent $sent of $length)");
409 }
410 }

412 private function writingIsTimedOut(int $sent): bool
413 {
414 // convert to ms

418

422

424 } else {

426 }
