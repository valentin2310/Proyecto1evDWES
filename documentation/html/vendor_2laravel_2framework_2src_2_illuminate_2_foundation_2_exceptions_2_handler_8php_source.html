338 protected function shouldntReport(Throwable $e)
339 {
340 if ($this->withoutDuplicates && ($this->reportedExceptionMap[$e] ?? false)) {
341 return true;

344 $dontReport = array_merge($this->dontReport, $this->internalDontReport);

347 return true;
