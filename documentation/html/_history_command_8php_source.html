96 protected function execute(InputInterface $input, OutputInterface $output)
97 {
98 $this->validateOnlyOne($input, ['show', 'head', 'tail']);
99 $this->validateOnlyOne($input, ['save', 'replay', 'clear']);
100
101 $history = $this->getHistorySlice(
102 $input->getOption('show'),
103 $input->getOption('head'),
104 $input->getOption('tail')
105 );
106 $highlighted = false;
107
108 $this->filter->bind($input);
109 if ($this->filter->hasFilter()) {
110 $matches = [];
111 $highlighted = [];
112 foreach ($history as $i => $line) {
113 if ($this->filter->match($line, $matches)) {
114 if (isset($matches[0])) {
115 $chunks = \explode($matches[0], $history[$i]);
116 $chunks = \array_map([__CLASS__, 'escape'], $chunks);
117 $glue = \sprintf('<urgent>%s</urgent>', self::escape($matches[0]));
118
119 $highlighted[$i] = \implode($glue, $chunks);
120 }
121 } else {
122 unset($history[$i]);
123 }
124 }
125 }
126
127 if ($save = $input->getOption('save')) {
128 $output->writeln(\sprintf('Saving history in %s...', $save));
129 \file_put_contents($save, \implode(\PHP_EOL, $history).\PHP_EOL);
130 $output->writeln('<info>History saved.</info>');
131 } elseif ($input->getOption('replay')) {
132 if (!($input->getOption('show') || $input->getOption('head') || $input->getOption('tail'))) {
133 throw new \InvalidArgumentException('You must limit history via --head, --tail or --show before replaying');
134 }
135
136 $count = \count($history);
137 $output->writeln(\sprintf('Replaying %d line%s of history', $count, ($count !== 1) ? 's' : ''));
138 $this->getApplication()->addInput($history);
139 } elseif ($input->getOption('clear')) {
140 $this->clearHistory();
141 $output->writeln('<info>History cleared.</info>');
142 } else {
143 $type = $input->getOption('no-numbers') ? 0 : ShellOutput::NUMBER_LINES;
144 if (!$highlighted) {
145 $type = $type | OutputInterface::OUTPUT_RAW;
146 }
147
148 $output->page($highlighted ?: $history, $type);
149 }
150
151 return 0;
152 }

161 private function extractRange(string $range): array
162 {
163 if (\preg_match('/^\d+$/', $range)) {
164 return [$range, $range + 1];
165 }
166
167 $matches = [];
168 if ($range !== '..' && \preg_match('/^(\d*)\.\.(\d*)$/', $range, $matches)) {
169 $start = $matches[1] ? (int) $matches[1] : 0;
170 $end = $matches[2] ? (int) $matches[2] + 1 : \PHP_INT_MAX;
171
172 return [$start, $end];
173 }
174
175 throw new \InvalidArgumentException('Unexpected range: '.$range);
176 }

187 private function getHistorySlice($show, $head, $tail): array
188 {
189 $history = $this->readline->listHistory();
190
191 // don't show the current `history` invocation
192 \array_pop($history);
193
194 if ($show) {
195 list($start, $end) = $this->extractRange($show);
196 $length = $end - $start;
197 } elseif ($head) {
198 if (!\preg_match('/^\d+$/', $head)) {
199 throw new \InvalidArgumentException('Please specify an integer argument for --head');
200 }
201
202 $start = 0;
203 $length = (int) $head;
204 } elseif ($tail) {
205 if (!\preg_match('/^\d+$/', $tail)) {
206 throw new \InvalidArgumentException('Please specify an integer argument for --tail');
207 }
208
209 $start = \count($history) - $tail;
210 $length = (int) $tail + 1;
211 } else {
212 return $history;
213 }
214
215 return \array_slice($history, $start, $length, true);
216 }
