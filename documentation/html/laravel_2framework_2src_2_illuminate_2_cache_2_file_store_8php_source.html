285 protected function getPayload($key)
286 {
287 $path = $this->path($key);
288
289 // If the file doesn't exist, we obviously cannot return the cache so we will
290 // just return null. Otherwise, we'll get the contents of the file and get
291 // the expiration UNIX timestamps from the start of the file's contents.
292 try {
293 $expire = substr(
294 $contents = $this->files->get($path, true), 0, 10
295 );
296 } catch (Exception) {
297 return $this->emptyPayload();
298 }
299
300 // If the current time is greater than expiration timestamps we will delete
301 // the file and return null. This helps clean up the old files and keeps
302 // this directory much cleaner for us as old files aren't hanging out.
303 if ($this->currentTime() >= $expire) {
304 $this->forget($key);
305
306 return $this->emptyPayload();
307 }
308
309 try {
310 $data = unserialize(substr($contents, 10));
311 } catch (Exception) {
312 $this->forget($key);
313
314 return $this->emptyPayload();
315 }
316
317 // Next, we'll extract the number of seconds that are remaining for a cache
318 // so that we can properly retain the time for things like the increment
319 // operation that may be performed on this cache on a later operation.
320 $time = $expire - $this->currentTime();

322 return compact('data', 'time');
