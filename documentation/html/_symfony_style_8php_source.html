227 public function definitionList(string|array|TableSeparator ...$list)
228 {
229 $headers = [];
230 $row = [];
231 foreach ($list as $value) {
232 if ($value instanceof TableSeparator) {
233 $headers[] = $value;
234 $row[] = $value;
235 continue;
236 }
237 if (\is_string($value)) {
238 $headers[] = new TableCell($value, ['colspan' => 2]);
239 $row[] = null;
240 continue;
241 }
242 if (!\is_array($value)) {
243 throw new InvalidArgumentException('Value should be an array, string, or an instance of TableSeparator.');
244 }
245 $headers[] = key($value);
246 $row[] = current($value);
247 }
248
249 $this->horizontalTable($headers, [$row]);
250 }

338 public function askQuestion(Question $question): mixed
339 {
340 if ($this->input->isInteractive()) {
341 $this->autoPrependBlock();
342 }
343
344 $this->questionHelper ??= new SymfonyQuestionHelper();
345
346 $answer = $this->questionHelper->ask($this->input, $this, $question);
347
348 if ($this->input->isInteractive()) {
349 if ($this->output instanceof ConsoleSectionOutput) {
350 // add the new line of the `return` to submit the input to ConsoleSectionOutput, because ConsoleSectionOutput is holding all it's lines.
351 // this is relevant when a `ConsoleSectionOutput::clear` is called.
352 $this->output->addNewLineOfInputSubmit();
353 }
354 $this->newLine();
355 $this->bufferedOutput->write("\n");
356 }
357
358 return $answer;
359 }

423 private function autoPrependBlock(): void
424 {
425 $chars = substr(str_replace(\PHP_EOL, "\n", $this->bufferedOutput->fetch()), -2);
426
427 if (!isset($chars[0])) {
428 $this->newLine(); // empty history, so we should start with a new line.
429
430 return;
431 }
432 // Prepend new line for each non LF chars (This means no blank line was output before)
433 $this->newLine(2 - substr_count($chars, "\n"));
434 }

436 private function autoPrependText(): void
437 {
438 $fetched = $this->bufferedOutput->fetch();
439 // Prepend new line if last char isn't EOL:
440 if ($fetched && !str_ends_with($fetched, "\n")) {
441 $this->newLine();
442 }
443 }

451 private function createBlock(iterable $messages, string $type = null, string $style = null, string $prefix = ' ', bool $padding = false, bool $escape = false): array
452 {
453 $indentLength = 0;
454 $prefixLength = Helper::width(Helper::removeDecoration($this->getFormatter(), $prefix));
455 $lines = [];
456
457 if (null !== $type) {
458 $type = sprintf('[%s] ', $type);
459 $indentLength = Helper::width($type);
460 $lineIndentation = str_repeat(' ', $indentLength);
461 }
462
463 // wrap and add newlines for each element
464 $outputWrapper = new OutputWrapper();
465 foreach ($messages as $key => $message) {
466 if ($escape) {
467 $message = OutputFormatter::escape($message);
468 }
469
470 $lines = array_merge(
471 $lines,
472 explode(\PHP_EOL, $outputWrapper->wrap(
473 $message,
474 $this->lineLength - $prefixLength - $indentLength,
475 \PHP_EOL
476 ))
477 );
478
479 if (\count($messages) > 1 && $key < \count($messages) - 1) {
480 $lines[] = '';
481 }
482 }
483
484 $firstLineIndex = 0;
485 if ($padding && $this->isDecorated()) {
486 $firstLineIndex = 1;
487 array_unshift($lines, '');
488 $lines[] = '';
489 }
490
491 foreach ($lines as $i => &$line) {
492 if (null !== $type) {
493 $line = $firstLineIndex === $i ? $type.$line : $lineIndentation.$line;
494 }
495
496 $line = $prefix.$line;
497 $line .= str_repeat(' ', max($this->lineLength - Helper::width(Helper::removeDecoration($this->getFormatter(), $line)), 0));
498
499 if ($style) {
500 $line = sprintf('<%s>%s</>', $style, $line);
501 }
502 }
503
504 return $lines;
505 }
