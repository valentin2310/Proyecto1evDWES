157 private function editFile(string $filePath, bool $shouldRemoveFile): string
158 {
159 $escapedFilePath = \escapeshellarg($filePath);
160 $editor = (isset($_SERVER['EDITOR']) && $_SERVER['EDITOR']) ? $_SERVER['EDITOR'] : 'nano';
161
162 $pipes = [];
163 $proc = \proc_open("{$editor} {$escapedFilePath}", [\STDIN, \STDOUT, \STDERR], $pipes);
164 \proc_close($proc);
165
166 $editedContent = @\file_get_contents($filePath);
167
168 if ($shouldRemoveFile) {
169 @\unlink($filePath);
170 }
171
172 if ($editedContent === false) {
173 throw new \UnexpectedValueException("Reading {$filePath} returned false");
174 }
175
176 return $editedContent;
