66 protected function execute(InputInterface $input, OutputInterface $output)
67 {
68 $value = $input->getArgument('target');
69 if (ReflectionLanguageConstruct::isLanguageConstruct($value)) {
70 $reflector = new ReflectionLanguageConstruct($value);
71 $doc = $this->getManualDocById($value);
72 } else {
73 list($target, $reflector) = $this->getTargetAndReflector($value);
74 $doc = $this->getManualDoc($reflector) ?: DocblockFormatter::format($reflector);
75 }
76
77 $db = $this->getApplication()->getManualDb();
78
79 if ($output instanceof ShellOutput) {
80 $output->startPaging();
81 }
82
83 // Maybe include the declaring class
84 if ($reflector instanceof \ReflectionMethod || $reflector instanceof \ReflectionProperty) {
85 $output->writeln(SignatureFormatter::format($reflector->getDeclaringClass()));
86 }
87
88 $output->writeln(SignatureFormatter::format($reflector));
89 $output->writeln('');
90
91 if (empty($doc) && !$db) {

93 $output->writeln(' To document core PHP functionality, download the PHP reference manual:');

95 } else {
96 $output->writeln($doc);

99 // Implicit --all if the original docblock has an {@inheritdoc} tag.
100 if ($input->getOption('all') || \stripos($doc, self::INHERIT_DOC_TAG) !== false) {
101 $parent = $reflector;
102 foreach ($this->getParentReflectors($reflector) as $parent) {
103 $output->writeln('');
104 $output->writeln('---');
105 $output->writeln('');
106

116 $output->writeln($doc);
117 }
118 }
119 }

121 if ($output instanceof ShellOutput) {
122 $output->stopPaging();
123 }

131 private function getManualDoc($reflector)
132 {
133 switch (\get_class($reflector)) {
134 case \ReflectionClass::class:
135 case \ReflectionObject::class:
136 case \ReflectionFunction::class:
137 $id = $reflector->name;
138 break;
139
140 case \ReflectionMethod::class:
141 $id = $reflector->class.'::'.$reflector->name;
142 break;
143
144 case \ReflectionProperty::class:
145 $id = $reflector->class.'::$'.$reflector->name;
146 break;
147
148 case \ReflectionClassConstant::class:
149 case ReflectionClassConstant::class:
150 // @todo this is going to collide with ReflectionMethod ids
151 // someday... start running the query by id + type if the DB
152 // supports it.
153 $id = $reflector->class.'::'.$reflector->name;
154 break;
155
156 case ReflectionConstant_::class:
157 $id = $reflector->name;
158 break;
159
160 default:
161 return false;
162 }
163
164 return $this->getManualDocById($id);
165 }

177 private function getParentReflectors($reflector): \Generator
178 {
179 $seenClasses = [];
180
181 switch (\get_class($reflector)) {
182 case \ReflectionClass::class:
183 case \ReflectionObject::class:
184 foreach ($reflector->getTraits() as $trait) {
185 if (!\in_array($trait->getName(), $seenClasses)) {
186 $seenClasses[] = $trait->getName();
187 yield $trait;
188 }
189 }
190
191 foreach ($reflector->getInterfaces() as $interface) {
192 if (!\in_array($interface->getName(), $seenClasses)) {
193 $seenClasses[] = $interface->getName();
194 yield $interface;
195 }
196 }
197
198 while ($reflector = $reflector->getParentClass()) {
199 yield $reflector;
200
201 foreach ($reflector->getTraits() as $trait) {
202 if (!\in_array($trait->getName(), $seenClasses)) {
203 $seenClasses[] = $trait->getName();
204 yield $trait;
205 }
206 }
207
208 foreach ($reflector->getInterfaces() as $interface) {
209 if (!\in_array($interface->getName(), $seenClasses)) {
210 $seenClasses[] = $interface->getName();
211 yield $interface;
212 }
213 }
214 }
215
216 return;
217
218 case \ReflectionMethod::class:
219 foreach ($this->getParentReflectors($reflector->getDeclaringClass()) as $parent) {
220 if ($parent->hasMethod($reflector->getName())) {
221 $parentMethod = $parent->getMethod($reflector->getName());
222 if (!\in_array($parentMethod->getDeclaringClass()->getName(), $seenClasses)) {
223 $seenClasses[] = $parentMethod->getDeclaringClass()->getName();
224 yield $parentMethod;
225 }
226 }
227 }
228
229 return;
230
231 case \ReflectionProperty::class:
232 foreach ($this->getParentReflectors($reflector->getDeclaringClass()) as $parent) {
233 if ($parent->hasProperty($reflector->getName())) {
234 $parentProperty = $parent->getProperty($reflector->getName());
235 if (!\in_array($parentProperty->getDeclaringClass()->getName(), $seenClasses)) {
236 $seenClasses[] = $parentProperty->getDeclaringClass()->getName();
237 yield $parentProperty;
238 }
239 }
240 }
241 break;
242 }
