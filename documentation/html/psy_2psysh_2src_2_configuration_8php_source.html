186 public static function fromInput(InputInterface $input): self
187 {
188 $config = new self(['configFile' => self::getConfigFileFromInput($input)]);
189
190 // Handle --color and --no-color (and --ansi and --no-ansi aliases)
191 if (self::getOptionFromInput($input, ['color', 'ansi'])) {
192 $config->setColorMode(self::COLOR_MODE_FORCED);
193 } elseif (self::getOptionFromInput($input, ['no-color', 'no-ansi'])) {
194 $config->setColorMode(self::COLOR_MODE_DISABLED);
195 }
196
197 // Handle verbosity options
198 if ($verbosity = self::getVerbosityFromInput($input)) {
199 $config->setVerbosity($verbosity);
200 }
201
202 // Handle interactive mode
203 if (self::getOptionFromInput($input, ['interactive', 'interaction'], ['-a', '-i'])) {
204 $config->setInteractiveMode(self::INTERACTIVE_MODE_FORCED);
205 } elseif (self::getOptionFromInput($input, ['no-interactive', 'no-interaction'], ['-n'])) {
206 $config->setInteractiveMode(self::INTERACTIVE_MODE_DISABLED);
207 }
208
209 // Handle --compact
210 if (self::getOptionFromInput($input, ['compact'])) {
211 $config->setTheme('compact');
212 }
213
214 // Handle --raw-output
215 // @todo support raw output with interactive input?
216 if (!$config->getInputInteractive()) {
217 if (self::getOptionFromInput($input, ['raw-output'], ['-r'])) {
218 $config->setRawOutput(true);
219 }
220 }
221
222 // Handle --yolo
223 if (self::getOptionFromInput($input, ['yolo'])) {
224 $config->setYolo(true);
225 }
226
227 return $config;
228 }

254 private static function getOptionFromInput(InputInterface $input, array $names, array $otherParams = []): bool
255 {
256 // Best case, input is properly bound and validated.
257 foreach ($names as $name) {
258 if ($input->hasOption($name) && $input->getOption($name)) {
259 return true;
260 }
261 }
262
263 foreach ($names as $name) {
264 $otherParams[] = '--'.$name;
265 }
266
267 foreach ($otherParams as $name) {
268 if ($input->hasParameterOption($name, true)) {
269 return true;
270 }
271 }
272
273 return false;
274 }

284 private static function getVerbosityFromInput(InputInterface $input)
285 {
286 // --quiet wins!
287 if (self::getOptionFromInput($input, ['quiet'], ['-q'])) {
288 return self::VERBOSITY_QUIET;
289 }
290
291 // Best case, input is properly bound and validated.
292 //
293 // Note that if the `--verbose` option is incorrectly defined as `VALUE_NONE` rather than
294 // `VALUE_OPTIONAL` (as it is in Symfony Console by default) it doesn't actually work with
295 // multiple verbosity levels as it claims.
296 //
297 // We can detect this by checking whether the the value === true, and fall back to unbound
298 // parsing for this option.
299 if ($input->hasOption('verbose') && $input->getOption('verbose') !== true) {
300 switch ($input->getOption('verbose')) {
301 case '-1':
302 return self::VERBOSITY_QUIET;
303 case '0': // explicitly normal, overrides config file default
304 return self::VERBOSITY_NORMAL;
305 case '1':
306 case null: // `--verbose` and `-v`
307 return self::VERBOSITY_VERBOSE;
308 case '2':
309 case 'v': // `-vv`
310 return self::VERBOSITY_VERY_VERBOSE;
311 case '3':
312 case 'vv': // `-vvv`
313 return self::VERBOSITY_DEBUG;
314 default: // implicitly normal, config file default wins
315 return;
316 }
317 }
318
319 // quiet and normal have to come before verbose, because it eats everything else.
320 if ($input->hasParameterOption('--verbose=-1', true) || $input->getParameterOption('--verbose', false, true) === '-1') {
321 return self::VERBOSITY_QUIET;
322 }
323
324 if ($input->hasParameterOption('--verbose=0', true) || $input->getParameterOption('--verbose', false, true) === '0') {
325 return self::VERBOSITY_NORMAL;
326 }
327
328 // `-vvv`, `-vv` and `-v` have to come in descending length order, because `hasParameterOption` matches prefixes.
329 if ($input->hasParameterOption('-vvv', true) || $input->hasParameterOption('--verbose=3', true) || $input->getParameterOption('--verbose', false, true) === '3') {
330 return self::VERBOSITY_DEBUG;
331 }
332
333 if ($input->hasParameterOption('-vv', true) || $input->hasParameterOption('--verbose=2', true) || $input->getParameterOption('--verbose', false, true) === '2') {
334 return self::VERBOSITY_VERY_VERBOSE;
335 }
336
337 if ($input->hasParameterOption('-v', true) || $input->hasParameterOption('--verbose=1', true) || $input->hasParameterOption('--verbose', true)) {
338 return self::VERBOSITY_VERBOSE;
339 }

462 public function loadConfig(array $options)
463 {
464 foreach (self::$AVAILABLE_OPTIONS as $option) {
465 if (isset($options[$option])) {
466 $method = 'set'.\ucfirst($option);
467 $this->$method($options[$option]);
468 }
469 }
470
471 // legacy `tabCompletion` option
472 if (isset($options['tabCompletion'])) {
473 $msg = '`tabCompletion` is deprecated; use `useTabCompletion` instead.';
474 @\trigger_error($msg, \E_USER_DEPRECATED);
475
476 $this->setUseTabCompletion($options['tabCompletion']);
477 }
478
479 foreach (['commands', 'matchers', 'casters'] as $option) {
480 if (isset($options[$option])) {
481 $method = 'add'.\ucfirst($option);
482 $this->$method($options[$option]);
483 }
484 }
485
486 // legacy `tabCompletionMatchers` option
487 if (isset($options['tabCompletionMatchers'])) {
488 $msg = '`tabCompletionMatchers` is deprecated; use `matchers` instead.';
489 @\trigger_error($msg, \E_USER_DEPRECATED);
490
491 $this->addMatchers($options['tabCompletionMatchers']);
492 }
493 }

506 public function loadConfigFile(string $file)
507 {
508 if (!\is_file($file)) {
509 throw new \InvalidArgumentException(\sprintf('Invalid configuration file specified, %s does not exist', $file));
510 }
511
512 $__psysh_config_file__ = $file;
513 $load = function ($config) use ($__psysh_config_file__) {
514 $result = require $__psysh_config_file__;
515 if ($result !== 1) {
516 return $result;
517 }
518 };
519 $result = $load($this);
520
521 if (!empty($result)) {
522 if (\is_array($result)) {
523 $this->loadConfig($result);
524 } else {
525 throw new \InvalidArgumentException('Psy Shell configuration must return an array of options');
526 }
527 }
528 }
