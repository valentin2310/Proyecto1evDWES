122 public static function filter(array $a, int $filter, array $listedProperties = [], ?int &$count = 0): array
123 {
124 $count = 0;
125
126 foreach ($a as $k => $v) {
127 $type = self::EXCLUDE_STRICT & $filter;
128
129 if (null === $v) {
130 $type |= self::EXCLUDE_NULL & $filter;
131 $type |= self::EXCLUDE_EMPTY & $filter;
132 } elseif (false === $v || '' === $v || '0' === $v || 0 === $v || 0.0 === $v || [] === $v) {
133 $type |= self::EXCLUDE_EMPTY & $filter;
134 }
135 if ((self::EXCLUDE_NOT_IMPORTANT & $filter) && !\in_array($k, $listedProperties, true)) {
136 $type |= self::EXCLUDE_NOT_IMPORTANT;
137 }
138 if ((self::EXCLUDE_VERBOSE & $filter) && \in_array($k, $listedProperties, true)) {
139 $type |= self::EXCLUDE_VERBOSE;
140 }
141
142 if (!isset($k[1]) || "\0" !== $k[0]) {
143 $type |= self::EXCLUDE_PUBLIC & $filter;
144 } elseif ('~' === $k[1]) {
145 $type |= self::EXCLUDE_VIRTUAL & $filter;
146 } elseif ('+' === $k[1]) {
147 $type |= self::EXCLUDE_DYNAMIC & $filter;
148 } elseif ('*' === $k[1]) {
149 $type |= self::EXCLUDE_PROTECTED & $filter;
150 } else {
151 $type |= self::EXCLUDE_PRIVATE & $filter;
152 }
153
154 if ((self::EXCLUDE_STRICT & $filter) ? $type === $filter : $type) {
155 unset($a[$k]);
156 ++$count;
157 }
158 }
159
160 return $a;
