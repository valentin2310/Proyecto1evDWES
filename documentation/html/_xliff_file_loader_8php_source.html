31 public function load(mixed $resource, string $locale, string $domain = 'messages'): MessageCatalogue
32 {
33 if (!class_exists(XmlUtils::class)) {
34 throw new RuntimeException('Loading translations from the Xliff format requires the Symfony Config component.');
35 }
36
37 if (!$this->isXmlString($resource)) {
38 if (!stream_is_local($resource)) {
39 throw new InvalidResourceException(sprintf('This is not a local file "%s".', $resource));
40 }
41
42 if (!file_exists($resource)) {
43 throw new NotFoundResourceException(sprintf('File "%s" not found.', $resource));
44 }
45
46 if (!is_file($resource)) {
47 throw new InvalidResourceException(sprintf('This is neither a file nor an XLIFF string "%s".', $resource));
48 }
49 }
50
51 try {
52 if ($this->isXmlString($resource)) {
53 $dom = XmlUtils::parse($resource);
54 } else {
55 $dom = XmlUtils::loadFile($resource);
56 }
57 } catch (\InvalidArgumentException|XmlParsingException|InvalidXmlException $e) {
58 throw new InvalidResourceException(sprintf('Unable to load "%s": ', $resource).$e->getMessage(), $e->getCode(), $e);
59 }
60
61 if ($errors = XliffUtils::validateSchema($dom)) {
62 throw new InvalidResourceException(sprintf('Invalid resource provided: "%s"; Errors: ', $resource).XliffUtils::getErrorsAsString($errors));
63 }
64
65 $catalogue = new MessageCatalogue($locale);
66 $this->extract($dom, $catalogue, $domain);
67
68 if (is_file($resource) && class_exists(FileResource::class)) {
69 $catalogue->addResource(new FileResource($resource));
70 }
71
72 return $catalogue;

91 private function extractXliff1(\DOMDocument $dom, MessageCatalogue $catalogue, string $domain): void
92 {
93 $xml = simplexml_import_dom($dom);
94 $encoding = $dom->encoding ? strtoupper($dom->encoding) : null;
95
96 $namespace = 'urn:oasis:names:tc:xliff:document:1.2';
97 $xml->registerXPathNamespace('xliff', $namespace);
98
99 foreach ($xml->xpath('//xliff:file') as $file) {
100 $fileAttributes = $file->attributes();
101
102 $file->registerXPathNamespace('xliff', $namespace);
103
104 foreach ($file->xpath('.//xliff:prop') as $prop) {
105 $catalogue->setCatalogueMetadata($prop->attributes()['prop-type'], (string) $prop, $domain);
106 }
107
108 foreach ($file->xpath('.//xliff:trans-unit') as $translation) {
109 $attributes = $translation->attributes();
110
111 if (!(isset($attributes['resname']) || isset($translation->source))) {
112 continue;
113 }
114
115 $source = isset($attributes['resname']) && $attributes['resname'] ? $attributes['resname'] : $translation->source;
116 // If the xlf file has another encoding specified, try to convert it because
117 // simple_xml will always return utf-8 encoded values
118 $target = $this->utf8ToCharset((string) ($translation->target ?? $translation->source), $encoding);
119
120 $catalogue->set((string) $source, $target, $domain);
121
122 $metadata = [
123 'source' => (string) $translation->source,
124 'file' => [
125 'original' => (string) $fileAttributes['original'],
126 ],
127 ];
128 if ($notes = $this->parseNotesMetadata($translation->note, $encoding)) {
129 $metadata['notes'] = $notes;
130 }
131
132 if (isset($translation->target) && $translation->target->attributes()) {
133 $metadata['target-attributes'] = [];
134 foreach ($translation->target->attributes() as $key => $value) {
135 $metadata['target-attributes'][$key] = (string) $value;
136 }
137 }
138
139 if (isset($attributes['id'])) {
140 $metadata['id'] = (string) $attributes['id'];
141 }
142
143 $catalogue->setMetadata((string) $source, $metadata, $domain);
144 }
145 }

148 private function extractXliff2(\DOMDocument $dom, MessageCatalogue $catalogue, string $domain): void
149 {
150 $xml = simplexml_import_dom($dom);
151 $encoding = $dom->encoding ? strtoupper($dom->encoding) : null;
152
153 $xml->registerXPathNamespace('xliff', 'urn:oasis:names:tc:xliff:document:2.0');
154
155 foreach ($xml->xpath('//xliff:unit') as $unit) {
156 foreach ($unit->segment as $segment) {
157 $attributes = $unit->attributes();
158 $source = $attributes['name'] ?? $segment->source;
159
160 // If the xlf file has another encoding specified, try to convert it because
161 // simple_xml will always return utf-8 encoded values
162 $target = $this->utf8ToCharset((string) ($segment->target ?? $segment->source), $encoding);
163
164 $catalogue->set((string) $source, $target, $domain);
165
166 $metadata = [];
167 if (isset($segment->target) && $segment->target->attributes()) {
168 $metadata['target-attributes'] = [];
169 foreach ($segment->target->attributes() as $key => $value) {
170 $metadata['target-attributes'][$key] = (string) $value;
171 }
172 }
173
174 if (isset($unit->notes)) {
175 $metadata['notes'] = [];
176 foreach ($unit->notes->note as $noteNode) {
177 $note = [];
178 foreach ($noteNode->attributes() as $key => $value) {
179 $note[$key] = (string) $value;
180 }
181 $note['content'] = (string) $noteNode;
182 $metadata['notes'][] = $note;
183 }
184 }
185
186 $catalogue->setMetadata((string) $source, $metadata, $domain);
187 }
188 }

204 {
