110 public function run($paths = [], array $options = [])
111 {
112 // Once we grab all of the migration files for the path, we will compare them
113 // against the migrations that have already been run for this package then
114 // run each of the outstanding migrations against a database connection.
115 $files = $this->getMigrationFiles($paths);
116
117 $this->requireFiles($migrations = $this->pendingMigrations(
118 $files, $this->repository->getRan()
119 ));
120
121 // Once we have all these migrations that are outstanding we are ready to run
122 // we will go ahead and run them "up". This will execute each migration as
123 // an operation against a database. Then we'll return this list of them.
124 $this->runPending($migrations, $options);
125
126 return $migrations;
127 }

151 public function runPending(array $migrations, array $options = [])
152 {
153 // First we will just make sure that there are any migrations to run. If there
154 // aren't, we will just make a note of it to the developer so they're aware
155 // that all of the migrations have been run against this database system.
156 if (count($migrations) === 0) {
157 $this->fireMigrationEvent(new NoPendingMigrations('up'));
158
159 $this->write(Info::class, 'Nothing to migrate');
160
161 return;
162 }
163
164 // Next, we will get the next batch number for the migrations so we can insert
165 // correct batch number in the database migrations repository when we store
166 // each migration's execution. We will also extract a few of the options.
167 $batch = $this->repository->getNextBatchNumber();
168
169 $pretend = $options['pretend'] ?? false;
170
171 $step = $options['step'] ?? false;
172
173 $this->fireMigrationEvent(new MigrationsStarted('up'));
174
175 $this->write(Info::class, 'Running migrations.');
176
177 // Once we have the array of migrations, we will spin through them and run the
178 // migrations "up" so the changes are made to the databases. We'll then log
179 // that the migration was run so we don't repeat it next time we execute.
180 foreach ($migrations as $file) {
181 $this->runUp($file, $batch, $pretend);
182
183 if ($step) {
184 $batch++;
185 }
186 }
187
188 $this->fireMigrationEvent(new MigrationsEnded('up'));
189
190 if ($this->output) {
191 $this->output->writeln('');
192 }
193 }

203 protected function runUp($file, $batch, $pretend)
204 {
205 // First we will resolve a "real" instance of the migration class from this
206 // migration file name. Once we have the instances we can run the actual
207 // command such as "up" or "down", or we can just simulate the action.
208 $migration = $this->resolvePath($file);
209
210 $name = $this->getMigrationName($file);
211
212 if ($pretend) {
213 return $this->pretendToRun($migration, 'up');
214 }
215
216 $this->write(Task::class, $name, fn () => $this->runMigration($migration, 'up'));
217
218 // Once we have run a migrations class, we will log that it was run in this
219 // repository so that we don't try to run it next time we do a migration
220 // in the application. A migration repository keeps the migrate order.
221 $this->repository->log($name, $batch);
222 }

231 public function rollback($paths = [], array $options = [])
232 {
233 // We want to pull in the last batch of migrations that ran on the previous
234 // migration operation. We'll then reverse those migrations and run each
235 // of them "down" to reverse the last migration "operation" which ran.
236 $migrations = $this->getMigrationsForRollback($options);
237
238 if (count($migrations) === 0) {
239 $this->fireMigrationEvent(new NoPendingMigrations('down'));
240
241 $this->write(Info::class, 'Nothing to rollback.');
242
243 return [];
244 }
245
246 return tap($this->rollbackMigrations($migrations, $paths, $options), function () {
247 if ($this->output) {
248 $this->output->writeln('');
249 }
250 });
251 }

280 protected function rollbackMigrations(array $migrations, $paths, array $options)
281 {
282 $rolledBack = [];
283
284 $this->requireFiles($files = $this->getMigrationFiles($paths));
285
286 $this->fireMigrationEvent(new MigrationsStarted('down'));
287
288 $this->write(Info::class, 'Rolling back migrations.');
289
290 // Next we will run through all of the migrations and call the "down" method
291 // which will reverse each migration in order. This getLast method on the
292 // repository already returns these migration's names in reverse order.
293 foreach ($migrations as $migration) {
294 $migration = (object) $migration;
295
296 if (! $file = Arr::get($files, $migration->migration)) {
297 $this->write(TwoColumnDetail::class, $migration->migration, '<fg=yellow;options=bold>Migration not found</>');
298
299 continue;
300 }
301
302 $rolledBack[] = $file;
303
304 $this->runDown(
305 $file, $migration,
306 $options['pretend'] ?? false
307 );
308 }
309
310 $this->fireMigrationEvent(new MigrationsEnded('down'));
311
312 return $rolledBack;
313 }

322 public function reset($paths = [], $pretend = false)
323 {
324 // Next, we will reverse the migration list so we can run them back in the
325 // correct order for resetting this database. This will allow us to get
326 // the database back into its "empty" state ready for the migrations.
327 $migrations = array_reverse($this->repository->getRan());

350 protected function resetMigrations(array $migrations, array $paths, $pretend = false)
351 {
352 // Since the getRan method that retrieves the migration name just gives us the
353 // migration name, we will format the names into objects with the name as a
354 // property on the objects so that we can pass it to the rollback method.
355 $migrations = collect($migrations)->map(function ($m) {
356 return (object) ['migration' => $m];

372 protected function runDown($file, $migration, $pretend)
373 {
374 // First we will get the file name of the migration so we can resolve out an
375 // instance of the migration. Once we get an instance we can either run a
376 // pretend execution of the migration or we can run the real migration.
377 $instance = $this->resolvePath($file);
378
379 $name = $this->getMigrationName($file);
380
381 if ($pretend) {
382 return $this->pretendToRun($instance, 'down');
383 }
384
385 $this->write(Task::class, $name, fn () => $this->runMigration($instance, 'down'));
386
387 // Once we have successfully run the migration "down" we will remove it from
388 // the migration repository so it will be considered to have not been run
389 // by the application then will be able to fire by any later operation.
390 $this->repository->delete($migration);
391 }

400 protected function runMigration($migration, $method)
401 {
402 $connection = $this->resolveConnection(
403 $migration->getConnection()
404 );
405
406 $callback = function () use ($connection, $migration, $method) {
407 if (method_exists($migration, $method)) {
408 $this->fireMigrationEvent(new MigrationStarted($migration, $method));
409
410 $this->runMethod($connection, $migration, $method);
411
412 $this->fireMigrationEvent(new MigrationEnded($migration, $method));
413 }
414 };
415
416 $this->getSchemaGrammar($connection)->supportsSchemaTransactions()
417 && $migration->withinTransaction
418 ? $connection->transaction($callback)
419 : $callback();

429 protected function pretendToRun($migration, $method)
430 {
431 try {
432 $name = get_class($migration);

518 protected function resolvePath(string $path)
519 {
520 $class = $this->getMigrationClass($this->getMigrationName($path));
521
522 if (class_exists($class) && realpath($path) == (new ReflectionClass($class))->getFileName()) {
523 return new $class;
524 }
525
526 $migration = static::$requiredPathCache[$path] ??= $this->files->getRequire($path);
527
528 if (is_object($migration)) {
529 return method_exists($migration, '__construct')
530 ? $this->files->getRequire($path)
531 : clone $migration;
532 }
533
534 return new $class;
