115 public function encodeString(string $string, ?string $charset = 'utf-8', int $firstLineOffset = 0, int $maxLineLength = 0): string
116 {
117 if ($maxLineLength > 76 || $maxLineLength <= 0) {
118 $maxLineLength = 76;
119 }
120
121 $thisLineLength = $maxLineLength - $firstLineOffset;
122
123 $lines = [];
124 $lNo = 0;
125 $lines[$lNo] = '';
126 $currentLine = &$lines[$lNo++];
127 $size = $lineLen = 0;
128 $charStream = new CharacterStream($string, $charset);
129
130 // Fetching more than 4 chars at one is slower, as is fetching fewer bytes
131 // Conveniently 4 chars is the UTF-8 safe number since UTF-8 has up to 6
132 // bytes per char and (6 * 4 * 3 = 72 chars per line) * =NN is 3 bytes
133 while (null !== $bytes = $charStream->readBytes(4)) {
134 $enc = $this->encodeByteSequence($bytes, $size);
135
136 $i = strpos($enc, '=0D=0A');
137 $newLineLength = $lineLen + (false === $i ? $size : $i);
138
139 if ($currentLine && $newLineLength >= $thisLineLength) {
140 $lines[$lNo] = '';
141 $currentLine = &$lines[$lNo++];
142 $thisLineLength = $maxLineLength;
143 $lineLen = 0;
144 }
145
146 $currentLine .= $enc;
147
148 if (false === $i) {
149 $lineLen += $size;
150 } else {
151 // 6 is the length of '=0D=0A'.
152 $lineLen = $size - strrpos($enc, '=0D=0A') - 6;
153 }
154 }
155
156 return $this->standardize(implode("=\r\n", $lines));
157 }

162 private function encodeByteSequence(array $bytes, int &$size): string
163 {
164 $ret = '';
165 $size = 0;
166 foreach ($bytes as $b) {
167 if (isset($this->safeMap[$b])) {
168 $ret .= $this->safeMap[$b];
169 ++$size;
170 } else {
171 $ret .= self::QP_MAP[$b];
172 $size += 3;
173 }
174 }
175
176 return $ret;
177 }

182 private function standardize(string $string): string
183 {
184 $string = str_replace(["\t=0D=0A", ' =0D=0A', '=0D=0A'], ["=09\r\n", "=20\r\n", "\r\n"], $string);
185
186 return match ($end = \ord(substr($string, -1))) {
187 0x09,
188 0x20 => substr_replace($string, self::QP_MAP[$end], -1),
189 default => $string,
190 };
