66 protected function execute(InputInterface $input, OutputInterface $output): int
67 {
68 $io = new SymfonyStyle($input, $output instanceof ConsoleOutputInterface ? $output->getErrorOutput() : $output);
69
70 if (null !== $time = $input->getOption('time')) {
71 try {
72 $time = new \DateTimeImmutable($time);

74 $io->error(sprintf('Invalid timestamp "%s": %s', $time, str_replace('DateTimeImmutable::__construct(): ', '', $e->getMessage())));

76 return 1;

78 }
79
80 $formatOption = $input->getOption('format');
81
82 if (\in_array($formatOption, $this->getAvailableFormatOptions())) {
83 $format = 'to'.ucfirst($formatOption);
84 } else {
85 $io->error(sprintf('Invalid format "%s", supported formats are "%s".', $formatOption, implode('", "', $this->getAvailableFormatOptions())));
86
87 return 1;
88 }
89
90 $count = (int) $input->getOption('count');
91 try {
92 for ($i = 0; $i < $count; ++$i) {
93 $output->writeln($this->factory->create($time)->$format());

95 } catch (\Exception $e) {
