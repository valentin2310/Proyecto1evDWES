436 {
437 if ($gender === null) {
438 $gender = $this->generator->boolean() ? static::GENDER_MALE : static::GENDER_FEMALE;
439 }
440
441 $startTimestamp = strtotime(sprintf('-%d year', $maxAge));
442 $endTimestamp = strtotime(sprintf('-%d year', $minAge));
443 $randTimestamp = self::numberBetween($startTimestamp, $endTimestamp);
444
445 $year = (int) (date('Y', $randTimestamp));
446 $month = (int) (date('n', $randTimestamp));
447 $day = (int) (date('j', $randTimestamp));
448 $suffix = self::numberBetween(0, 999);
449
450 // women has +50 to month

452 $month += 50;
453 }
454 // from year 2004 everyone has +20 to month when birth numbers in one day are exhausted
455 if ($year >= 2004 && $this->generator->boolean(10)) {
456 $month += 20;
457 }
458

472 if ($this->generator->boolean($slashProbability)) {
