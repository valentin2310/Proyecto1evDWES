68 public function run(TestCase $test): void

71
72 if ($this->configuration->registerMockObjectsFromTestArgumentsRecursively()) {
73 $test->registerMockObjectsFromTestArgumentsRecursively();
74 }
75
76 $shouldCodeCoverageBeCollected = (new CodeCoverageMetadataApi)->shouldCodeCoverageBeCollectedFor(
77 $test::class,
78 $test->name(),
79 );
80
81 $error = false;
82 $failure = false;
83 $incomplete = false;
84 $risky = false;
85 $skipped = false;
86
87 if ($this->shouldErrorHandlerBeUsed($test)) {
88 ErrorHandler::instance()->enable();
89 }

92 $shouldCodeCoverageBeCollected;

98 try {
99 if ($this->canTimeLimitBeEnforced() &&
100 $this->shouldTimeLimitBeEnforced($test)) {
101 $risky = $this->runTestWithTimeout($test);
102 } else {
103 $test->runBare();
104 }
105 } catch (AssertionFailedError $e) {
106 $failure = true;
107
108 if ($e instanceof IncompleteTestError) {
109 $incomplete = true;
110 } elseif ($e instanceof SkippedTest) {
111 $skipped = true;
112 }
113 } catch (AssertionError $e) {
114 $test->addToAssertionCount(1);
115
116 $failure = true;
117 $frame = $e->getTrace()[0];
118
119 assert(isset($frame['file']));
120 assert(isset($frame['line']));
121
122 $e = new AssertionFailedError(
123 sprintf(
124 '%s in %s:%s',
125 $e->getMessage(),
126 $frame['file'],
127 $frame['line'],
128 ),
129 );
130 } catch (Throwable $e) {
131 $error = true;
132 }
133
134 $test->addToAssertionCount(Assert::getCount());
135
136 if ($this->configuration->reportUselessTests() &&
137 !$test->doesNotPerformAssertions() &&
138 $test->numberOfAssertionsPerformed() === 0) {
139 $risky = true;
140 }
141
142 if (!$error && !$failure && !$incomplete && !$skipped && !$risky &&
143 $this->configuration->requireCoverageMetadata() &&
144 !$this->hasCoverageMetadata($test::class, $test->name())) {
145 Event\Facade::emitter()->testConsideredRisky(
146 $test->valueObjectForEvents(),
147 'This test does not define a code coverage target but is expected to do so',
148 );
149
150 $risky = true;
151 }
152
153 if ($collectCodeCoverage) {
154 $append = !$risky && !$incomplete && !$skipped;
155 $linesToBeCovered = [];
156 $linesToBeUsed = [];
157
158 if ($append) {
159 try {
160 $linesToBeCovered = (new CodeCoverageMetadataApi)->linesToBeCovered(
161 $test::class,
162 $test->name(),
163 );
164
165 $linesToBeUsed = (new CodeCoverageMetadataApi)->linesToBeUsed(
166 $test::class,
167 $test->name(),
168 );
169 } catch (InvalidCoversTargetException $cce) {
170 Event\Facade::emitter()->testTriggeredPhpunitWarning(
171 $test->valueObjectForEvents(),
172 $cce->getMessage(),
173 );
174 }
175 }
176
177 try {
178 CodeCoverage::instance()->stop(
179 $append,
180 $linesToBeCovered,
181 $linesToBeUsed,
182 );
183 } catch (UnintentionallyCoveredCodeException $cce) {
184 Event\Facade::emitter()->testConsideredRisky(
185 $test->valueObjectForEvents(),
186 'This test executed code that is not listed as code to be covered or used:' .
187 PHP_EOL .
188 $cce->getMessage(),
189 );
190 } catch (OriginalCodeCoverageException $cce) {
191 $error = true;
192
193 $e = $e ?? $cce;
194 }
195 }
196
197 ErrorHandler::instance()->disable();
198
199 if (!$error &&
200 !$incomplete &&
201 !$skipped &&
202 $this->configuration->reportUselessTests() &&
203 !$test->doesNotPerformAssertions() &&
204 $test->numberOfAssertionsPerformed() === 0) {
205 Event\Facade::emitter()->testConsideredRisky(
206 $test->valueObjectForEvents(),
207 'This test did not perform any assertions',
208 );
209 }
210
211 if ($test->doesNotPerformAssertions() &&
212 $test->numberOfAssertionsPerformed() > 0) {
213 Event\Facade::emitter()->testConsideredRisky(
214 $test->valueObjectForEvents(),
215 sprintf(
216 'This test is not expected to perform assertions but performed %d assertion%s',
217 $test->numberOfAssertionsPerformed(),
218 $test->numberOfAssertionsPerformed() > 1 ? 's' : '',
219 ),
220 );
221 }
222
223 if ($test->hasUnexpectedOutput()) {
224 Event\Facade::emitter()->testPrintedUnexpectedOutput($test->output());
225 }
226
227 if ($this->configuration->disallowTestOutput() && $test->hasUnexpectedOutput()) {
228 Event\Facade::emitter()->testConsideredRisky(
229 $test->valueObjectForEvents(),
230 sprintf(
231 'This test printed output: %s',
232 $test->output(),
233 ),
234 );
235 }
236
237 if ($test->wasPrepared()) {
238 Event\Facade::emitter()->testFinished(
239 $test->valueObjectForEvents(),
240 $test->numberOfAssertionsPerformed(),
241 );
242 }
243 }

356 private function hasCoverageMetadata(string $className, string $methodName): bool
357 {
358 $metadata = MetadataRegistry::parser()->forClassAndMethod($className, $methodName);
359
360 if ($metadata->isCovers()->isNotEmpty()) {
361 return true;
362 }
363
364 if ($metadata->isCoversClass()->isNotEmpty()) {
365 return true;
366 }
367
368 if ($metadata->isCoversFunction()->isNotEmpty()) {
369 return true;
370 }
371
372 if ($metadata->isCoversNothing()->isNotEmpty()) {
373 return true;
374 }

379 private function canTimeLimitBeEnforced(): bool
380 {
381 if ($this->timeLimitCanBeEnforced !== null) {
382 return $this->timeLimitCanBeEnforced;
383 }
384
385 if (!class_exists(Invoker::class)) {
386 $this->timeLimitCanBeEnforced = false;
387
388 return $this->timeLimitCanBeEnforced;
389 }

393 return $this->timeLimitCanBeEnforced;

396 private function shouldTimeLimitBeEnforced(TestCase $test): bool
397 {
398 if (!$this->configuration->enforceTimeLimit()) {
399 return false;
400 }
401
402 if (!(($this->configuration->defaultTimeLimit() || $test->size()->isKnown()))) {
403 return false;
404 }
405
406 if (extension_loaded('xdebug') && xdebug_is_debugger_active()) {
407 return false;
408 }
409
410 return true;
411 }

416 private function runTestWithTimeout(TestCase $test): bool
417 {
418 $_timeout = $this->configuration->defaultTimeLimit();
419
420 if ($test->size()->isSmall()) {
421 $_timeout = $this->configuration->timeoutForSmallTests();
422 } elseif ($test->size()->isMedium()) {
423 $_timeout = $this->configuration->timeoutForMediumTests();
424 } elseif ($test->size()->isLarge()) {
425 $_timeout = $this->configuration->timeoutForLargeTests();
426 }
427
428 try {
429 (new Invoker)->invoke([$test, 'runBare'], [], $_timeout);
430 } catch (TimeoutException) {
431 Event\Facade::emitter()->testConsideredRisky(
432 $test->valueObjectForEvents(),
433 sprintf(
434 'This test was aborted after %d second%s',
435 $_timeout,
436 $_timeout !== 1 ? 's' : '',
437 ),
438 );
439
440 return true;
441 }
442
443 return false;
444 }
