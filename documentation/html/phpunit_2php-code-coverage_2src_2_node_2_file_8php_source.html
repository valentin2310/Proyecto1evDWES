364 $this->processClasses($classes);

366 $this->processFunctions($functions);
367
368 foreach (range(1, $this->linesOfCode['linesOfCode']) as $lineNumber) {
369 if (isset($this->lineCoverageData[$lineNumber])) {
370 foreach ($this->codeUnitsByLine[$lineNumber] as &$codeUnit) {
371 $codeUnit['executableLines']++;
372 }
373
374 unset($codeUnit);
375
376 $this->numExecutableLines++;
377
378 if (count($this->lineCoverageData[$lineNumber]) > 0) {
