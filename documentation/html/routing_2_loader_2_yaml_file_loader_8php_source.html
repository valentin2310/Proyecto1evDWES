44 public function load(mixed $file, string $type = null): RouteCollection
45 {
46 $path = $this->locator->locate($file);
47
48 if (!stream_is_local($path)) {
49 throw new \InvalidArgumentException(sprintf('This is not a local file "%s".', $path));
50 }
51
52 if (!file_exists($path)) {
53 throw new \InvalidArgumentException(sprintf('File "%s" not found.', $path));
54 }
55
56 $this->yamlParser ??= new YamlParser();
57
58 try {
59 $parsedConfig = $this->yamlParser->parseFile($path, Yaml::PARSE_CONSTANT);
60 } catch (ParseException $e) {
61 throw new \InvalidArgumentException(sprintf('The file "%s" does not contain valid YAML: ', $path).$e->getMessage(), 0, $e);
62 }
63
64 $collection = new RouteCollection();
65 $collection->addResource(new FileResource($path));
66
67 // empty file
68 if (null === $parsedConfig) {
69 return $collection;
70 }
71
72 // not an array
73 if (!\is_array($parsedConfig)) {
74 throw new \InvalidArgumentException(sprintf('The file "%s" must contain a YAML array.', $path));
75 }
76
77 foreach ($parsedConfig as $name => $config) {
78 if (str_starts_with($name, 'when@')) {
79 if (!$this->env || 'when@'.$this->env !== $name) {
80 continue;
81 }
82
83 foreach ($config as $name => $config) {
84 $this->validate($config, $name.'" when "@'.$this->env, $path);
85
86 if (isset($config['resource'])) {
87 $this->parseImport($collection, $config, $path, $file);
88 } else {
89 $this->parseRoute($collection, $name, $config, $path);
90 }
91 }
92
93 continue;
94 }
95
96 $this->validate($config, $name, $path);
97
98 if (isset($config['resource'])) {
99 $this->parseImport($collection, $config, $path, $file);
100 } else {
101 $this->parseRoute($collection, $name, $config, $path);
102 }
103 }
104
105 return $collection;
106 }

250 protected function validate(mixed $config, string $name, string $path)
251 {
252 if (!\is_array($config)) {
253 throw new \InvalidArgumentException(sprintf('The definition of "%s" in "%s" must be a YAML array.', $name, $path));
254 }
255 if (isset($config['alias'])) {
256 $this->validateAlias($config, $name, $path);
257
258 return;
259 }
260 if ($extraKeys = array_diff(array_keys($config), self::AVAILABLE_KEYS)) {
261 throw new \InvalidArgumentException(sprintf('The routing file "%s" contains unsupported keys for "%s": "%s". Expected one of: "%s".', $path, $name, implode('", "', $extraKeys), implode('", "', self::AVAILABLE_KEYS)));
262 }
263 if (isset($config['resource']) && isset($config['path'])) {
264 throw new \InvalidArgumentException(sprintf('The routing file "%s" must not specify both the "resource" key and the "path" key for "%s". Choose between an import and a route definition.', $path, $name));
265 }
266 if (!isset($config['resource']) && isset($config['type'])) {
267 throw new \InvalidArgumentException(sprintf('The "type" key for the route definition "%s" in "%s" is unsupported. It is only available for imports in combination with the "resource" key.', $name, $path));
268 }
269 if (!isset($config['resource']) && !isset($config['path'])) {
270 throw new \InvalidArgumentException(sprintf('You must define a "path" for the route "%s" in file "%s".', $name, $path));
271 }
272 if (isset($config['controller']) && isset($config['defaults']['_controller'])) {
273 throw new \InvalidArgumentException(sprintf('The routing file "%s" must not specify both the "controller" key and the defaults key "_controller" for "%s".', $path, $name));
274 }
275 if (isset($config['stateless']) && isset($config['defaults']['_stateless'])) {
276 throw new \InvalidArgumentException(sprintf('The routing file "%s" must not specify both the "stateless" key and the defaults key "_stateless" for "%s".', $path, $name));
277 }

284 private function validateAlias(array $config, string $name, string $path): void
285 {
286 foreach ($config as $key => $value) {
287 if (!\in_array($key, ['alias', 'deprecated'], true)) {
288 throw new \InvalidArgumentException(sprintf('The routing file "%s" must not specify other keys than "alias" and "deprecated" for "%s".', $path, $name));
289 }
290
291 if ('deprecated' === $key) {
292 if (!isset($value['package'])) {
293 throw new \InvalidArgumentException(sprintf('The routing file "%s" must specify the attribute "package" of the "deprecated" option for "%s".', $path, $name));
294 }
295
296 if (!isset($value['version'])) {
297 throw new \InvalidArgumentException(sprintf('The routing file "%s" must specify the attribute "version" of the "deprecated" option for "%s".', $path, $name));
298 }
299 }
