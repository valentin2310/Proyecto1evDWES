174 public function compileChange(Blueprint $blueprint, Fluent $command, Connection $connection)
175 {
176 if (! $connection->usingNativeSchemaOperations()) {
177 return parent::compileChange($blueprint, $command, $connection);
178 }
179
180 $changes = [$this->compileDropDefaultConstraint($blueprint, $command)];
181
182 foreach ($blueprint->getChangedColumns() as $column) {
183 $sql = sprintf('alter table %s alter column %s %s',
184 $this->wrapTable($blueprint),
185 $this->wrap($column),
186 $this->getType($column)
187 );
188
189 foreach ($this->modifiers as $modifier) {

343 public function compileDropDefaultConstraint(Blueprint $blueprint, Fluent $command)
344 {
345 $columns = $command->name === 'change'
346 ? "'".collect($blueprint->getChangedColumns())->pluck('name')->implode("','")."'"
347 : "'".implode("','", $command->columns)."'";
348
349 $tableName = $this->getTablePrefix().$blueprint->getTable();
350
351 $sql = "DECLARE @sql NVARCHAR(MAX) = '';";
352 $sql .= "SELECT @sql += 'ALTER TABLE [dbo].[{$tableName}] DROP CONSTRAINT ' + OBJECT_NAME([default_object_id]) + ';' ";
353 $sql .= 'FROM sys.columns ';
354 $sql .= "WHERE [object_id] = OBJECT_ID('[dbo].[{$tableName}]') AND [name] in ({$columns}) AND [default_object_id] <> 0;";
355 $sql .= 'EXEC(@sql)';
356
357 return $sql;
