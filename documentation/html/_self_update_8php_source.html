100 public function run(InputInterface $input, OutputInterface $output): int
101 {
102 $currentVersion = Shell::VERSION;
103
104 // already have the latest version?
105 if ($this->checker->isLatest()) {
106 // current version is latest version...
107 $output->writeln('<info>Current version is up-to-date.</info>');
108
109 return self::SUCCESS;
110 }
111
112 // can overwrite current version?
113 if (!$this->installer->isInstallLocationWritable()) {
114 $output->writeln('<error>Installed version is not writable.</error>');
115
116 return self::FAILURE;
117 }
118 // can download to, and create a backup in the temp directory?
119 if (!$this->installer->isTempDirectoryWritable()) {
120 $output->writeln('<error>Temporary directory is not writable.</error>');
121
122 return self::FAILURE;
123 }
124
125 $latestVersion = $this->checker->getLatest();
126 $downloadUrl = $this->getAssetUrl($latestVersion);
127
128 $output->write("Downloading PsySH $latestVersion ...");
129
130 try {
131 $downloader = $this->getDownloader();
132 $downloader->setTempDir($this->installer->getTempDirectory());
133 $downloaded = $downloader->download($downloadUrl);
134 } catch (ErrorException $e) {
135 $output->write(' <error>Failed.</error>');
136 $output->writeln(\sprintf('<error>%s</error>', $e->getMessage()));
137
138 return self::FAILURE;
139 }
140
141 if (!$downloaded) {
142 $output->writeln('<error>Download failed.</error>');
143 $downloader->cleanup();
144
145 return self::FAILURE;
146 } else {
147 $output->write(' <info>OK</info>'.\PHP_EOL);
148 }
149
150 $downloadedFile = $downloader->getFilename();
151
152 if (!$this->installer->isValidSource($downloadedFile)) {
153 $downloader->cleanup();
154 $output->writeln('<error>Downloaded file is not a valid archive.</error>');
155
156 return self::FAILURE;
157 }
158
159 // create backup as bin.old-version in the temporary directory
160 $backupCreated = $this->installer->createBackup($currentVersion);
161 if (!$backupCreated) {
162 $downloader->cleanup();
163 $output->writeln('<error>Failed to create a backup of the current version.</error>');
164
165 return self::FAILURE;
166 } elseif ($input->getOption('verbose')) {
167 $backupFilename = $this->installer->getBackupFilename($currentVersion);
168 $output->writeln('Created backup of current version: '.$backupFilename);
169 }
170
171 if (!$this->installer->install($downloadedFile)) {
172 $this->installer->restoreFromBackup($currentVersion);
173 $downloader->cleanup();
174 $output->writeln("<error>Failed to install new PsySH version $latestVersion.</error>");
175
176 return self::FAILURE;
177 }
178
179 // Remove the downloaded archive file from the temporary directory
180 $downloader->cleanup();
181
182 $output->writeln("Updated PsySH from $currentVersion to <info>$latestVersion</info>");
183
184 return self::SUCCESS;
