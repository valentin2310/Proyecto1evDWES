211 protected function addFluentIndexes()
212 {
213 foreach ($this->columns as $column) {
214 foreach (['primary', 'unique', 'index', 'fulltext', 'fullText', 'spatialIndex'] as $index) {
215 // If the index has been specified on the given column, but is simply equal
216 // to "true" (boolean), no name has been specified for this index so the
217 // index method can be called without a name and it will generate one.
218 if ($column->{$index} === true) {
219 $this->{$index}($column->name);
220 $column->{$index} = null;
221
222 continue 2;
223 }
224
225 // If the index has been specified on the given column, but it equals false
226 // and the column is supposed to be changed, we will call the drop index
227 // method with an array of column to drop it by its conventional name.
228 elseif ($column->{$index} === false && $column->change) {
229 $this->{'drop'.ucfirst($index)}([$column->name]);
230 $column->{$index} = null;
231
232 continue 2;
233 }
234
235 // If the index has been specified on the given column, and it has a string
236 // value, we'll go ahead and call the index method and pass the name for
237 // the index since the developer specified the explicit name for this.
238 elseif (isset($column->{$index})) {
239 $this->{$index}($column->name, $column->{$index});
