116 private function parseLine(string $line): void
117 {
118 $this->cursor = new Cursor($line);
119
120 $matches = $this->parseBlockContinuation();
121 if ($matches === null) {
122 return;
123 }
124
125 $unmatchedBlocks = \count($this->activeBlockParsers) - $matches;
126 $blockParser = $this->activeBlockParsers[$matches - 1];
127 $startedNewBlock = false;
128
129 // Unless last matched container is a code block, try new container starts,
130 // adding children to the last matched container:
131 $tryBlockStarts = $blockParser->getBlock() instanceof Paragraph || $blockParser->isContainer();
132 while ($tryBlockStarts) {
133 // this is a little performance optimization
134 if ($this->cursor->isBlank()) {
135 $this->cursor->advanceToEnd();
136 break;
137 }
138
139 if ($blockParser->getBlock()->getDepth() >= $this->maxNestingLevel) {
140 break;
141 }
142
143 $blockStart = $this->findBlockStart($blockParser);
144 if ($blockStart === null || $blockStart->isAborting()) {
145 $this->cursor->advanceToNextNonSpaceOrTab();
146 break;
147 }
148
149 if (($state = $blockStart->getCursorState()) !== null) {
150 $this->cursor->restoreState($state);
151 }
152
153 $startedNewBlock = true;
154
155 // We're starting a new block. If we have any previous blocks that need to be closed, we need to do it now.
156 if ($unmatchedBlocks > 0) {
157 $this->closeBlockParsers($unmatchedBlocks, $this->lineNumber - 1);
158 $unmatchedBlocks = 0;
159 }
160
161 if ($blockStart->isReplaceActiveBlockParser()) {
162 $this->prepareActiveBlockParserForReplacement();
163 }
164
165 foreach ($blockStart->getBlockParsers() as $newBlockParser) {
166 $blockParser = $this->addChild($newBlockParser);
167 $tryBlockStarts = $newBlockParser->isContainer();
168 }
169 }
170
171 // What remains at the offset is a text line. Add the text to the appropriate block.
172
173 // First check for a lazy paragraph continuation:
174 if (! $startedNewBlock && ! $this->cursor->isBlank() && $this->getActiveBlockParser()->canHaveLazyContinuationLines()) {
175 $this->getActiveBlockParser()->addLine($this->cursor->getRemainder());
176 } else {
177 // finalize any blocks not matched
178 if ($unmatchedBlocks > 0) {
179 $this->closeBlockParsers($unmatchedBlocks, $this->lineNumber);
180 }
181
182 if (! $blockParser->isContainer()) {
183 $this->getActiveBlockParser()->addLine($this->cursor->getRemainder());
184 } elseif (! $this->cursor->isBlank()) {
185 $this->addChild(new ParagraphParser());
186 $this->getActiveBlockParser()->addLine($this->cursor->getRemainder());
187 }
188 }
189 }

191 private function parseBlockContinuation(): ?int
192 {
193 // For each containing block, try to parse the associated line start.
194 // The document will always match, so we can skip the first block parser and start at 1 matches
195 $matches = 1;
196 for ($i = 1; $i < \count($this->activeBlockParsers); $i++) {
197 $blockParser = $this->activeBlockParsers[$i];
198 $blockContinue = $blockParser->tryContinue(clone $this->cursor, $this->getActiveBlockParser());
199 if ($blockContinue === null) {
200 break;
201 }
202
203 if ($blockContinue->isFinalize()) {
204 $this->closeBlockParsers(\count($this->activeBlockParsers) - $i, $this->lineNumber);
205
206 return null;
207 }
208
209 if (($state = $blockContinue->getCursorState()) !== null) {
210 $this->cursor->restoreState($state);
211 }
212
213 $matches++;
214 }
215
216 return $matches;
217 }
