57 public function compileSelect(Builder $query)
58 {
59 if (($query->unions || $query->havings) && $query->aggregate) {
60 return $this->compileUnionAggregate($query);
61 }
62
63 // If the query does not have any columns set, we'll set the columns to the
64 // * character to just get all of the columns from the database. Then we
65 // can build the query and concatenate all the pieces together as one.
66 $original = $query->columns;
67
68 if (is_null($query->columns)) {
69 $query->columns = ['*'];
70 }
71
72 // To compile the query, we'll spin through each component of the query and
73 // see if that component exists. If it does we'll just call the compiler
74 // function for the component which is responsible for making the SQL.
75 $sql = trim($this->concatenate(
76 $this->compileComponents($query))
77 );
78
79 if ($query->unions) {
80 $sql = $this->wrapUnion($sql).' '.$this->compileUnions($query);
81 }
82
83 $query->columns = $original;
84
85 return $sql;

1020 public function compileInsert(Builder $query, array $values)
1021 {
1022 // Essentially we will force every insert to be treated as a batch insert which
1023 // simply makes creating the SQL easier for us since we can utilize the same
1024 // basic routine regardless of an amount of records given to us to insert.
1025 $table = $this->wrapTable($query->from);
1026
1027 if (empty($values)) {
1028 return "insert into {$table} default values";
1029 }
1030
1031 if (! is_array(reset($values))) {
1032 $values = [$values];
1033 }
1034
1035 $columns = $this->columnize(array_keys(reset($values)));
1036
1037 // We need to build a list of parameter place-holders of values that are bound
1038 // to the query. Each insert should have the exact same number of parameter
1039 // bindings so we will loop through the record and parameterize them all.
1040 $parameters = collect($values)->map(function ($record) {
1041 return '('.$this->parameterize($record).')';
1042 })->implode(', ');

1361 public function substituteBindingsIntoRawSql($sql, $bindings)
1362 {
1363 $bindings = array_map(fn ($value) => $this->escape($value), $bindings);
1364
1365 $query = '';
1366
1367 $isStringLiteral = false;
1368
1369 for ($i = 0; $i < strlen($sql); $i++) {
1370 $char = $sql[$i];
1371 $nextChar = $sql[$i + 1] ?? null;
1372
1373 // Single quotes can be escaped as '' according to the SQL standard while
1374 // MySQL uses \'. Postgres has operators like ?| that must get encoded
1375 // in PHP like ??|. We should skip over the escaped characters here.
1376 if (in_array($char.$nextChar, ["\'", "''", '??'])) {
1377 $query .= $char.$nextChar;
1378 $i += 1;
1379 } elseif ($char === "'") { // Starting / leaving string literal...
1380 $query .= $char;
1381 $isStringLiteral = ! $isStringLiteral;
1382 } elseif ($char === '?' && ! $isStringLiteral) { // Substitutable binding...
1383 $query .= array_shift($bindings) ?? '?';
1384 } else { // Normal character...
1385 $query .= $char;
1386 }
1387 }
1388
1389 return $query;
1390 }
