120 public static function checkIp6(string $requestIp, string $ip): bool
121 {
122 $cacheKey = $requestIp.'-'.$ip.'-v6';
123 if (null !== $cacheValue = self::getCacheResult($cacheKey)) {
124 return $cacheValue;
125 }
126
127 if (!((\extension_loaded('sockets') && \defined('AF_INET6')) || @inet_pton('::1'))) {
128 throw new \RuntimeException('Unable to check Ipv6. Check that PHP was not compiled with option "disable-ipv6".');
129 }
130
131 // Check to see if we were given a IP4 $requestIp or $ip by mistake
132 if (!filter_var($requestIp, \FILTER_VALIDATE_IP, \FILTER_FLAG_IPV6)) {
133 return self::setCacheResult($cacheKey, false);
134 }
135
136 if (str_contains($ip, '/')) {
137 [$address, $netmask] = explode('/', $ip, 2);
138
139 if (!filter_var($address, \FILTER_VALIDATE_IP, \FILTER_FLAG_IPV6)) {
140 return self::setCacheResult($cacheKey, false);
141 }
142
143 if ('0' === $netmask) {
144 return (bool) unpack('n*', @inet_pton($address));
145 }
146
147 if ($netmask < 1 || $netmask > 128) {
148 return self::setCacheResult($cacheKey, false);
149 }
150 } else {
151 if (!filter_var($ip, \FILTER_VALIDATE_IP, \FILTER_FLAG_IPV6)) {
152 return self::setCacheResult($cacheKey, false);
153 }
154
155 $address = $ip;
156 $netmask = 128;
157 }
158
159 $bytesAddr = unpack('n*', @inet_pton($address));
160 $bytesTest = unpack('n*', @inet_pton($requestIp));
161
162 if (!$bytesAddr || !$bytesTest) {
163 return self::setCacheResult($cacheKey, false);
164 }
165
166 for ($i = 1, $ceil = ceil($netmask / 16); $i <= $ceil; ++$i) {
167 $left = $netmask - 16 * ($i - 1);
168 $left = ($left <= 16) ? $left : 16;
169 $mask = ~(0xFFFF >> $left) & 0xFFFF;
170 if (($bytesAddr[$i] & $mask) != ($bytesTest[$i] & $mask)) {
171 return self::setCacheResult($cacheKey, false);
172 }
173 }
174
175 return self::setCacheResult($cacheKey, true);
176 }

183 public static function anonymize(string $ip): string
184 {
185 $wrappedIPv6 = false;
186 if (str_starts_with($ip, '[') && str_ends_with($ip, ']')) {
187 $wrappedIPv6 = true;
188 $ip = substr($ip, 1, -1);
189 }
190
191 $packedAddress = inet_pton($ip);
192 if (4 === \strlen($packedAddress)) {
193 $mask = '255.255.255.0';
194 } elseif ($ip === inet_ntop($packedAddress & inet_pton('::ffff:ffff:ffff'))) {
195 $mask = '::ffff:ffff:ff00';
196 } elseif ($ip === inet_ntop($packedAddress & inet_pton('::ffff:ffff'))) {
197 $mask = '::ffff:ff00';
198 } else {
199 $mask = 'ffff:ffff:ffff:ffff:0000:0000:0000:0000';
200 }
201 $ip = inet_ntop($packedAddress & inet_pton($mask));
202
203 if ($wrappedIPv6) {
204 $ip = '['.$ip.']';
205 }
206
207 return $ip;
208 }
