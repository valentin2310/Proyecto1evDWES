384 private function applyCoversAndUsesFilter(RawCodeCoverageData $rawData, array|false $linesToBeCovered, array $linesToBeUsed, TestSize $size): void
385 {
386 if ($linesToBeCovered === false) {
387 $rawData->clear();
388
389 return;
390 }
391
392 if (empty($linesToBeCovered)) {
393 return;
394 }
395
396 if ($this->checkForUnintentionallyCoveredCode && !$size->isMedium() && !$size->isLarge()) {
397 $this->performUnintentionallyCoveredCodeCheck($rawData, $linesToBeCovered, $linesToBeUsed);
398 }
399
400 $rawLineData = $rawData->lineCoverage();
401 $filesWithNoCoverage = array_diff_key($rawLineData, $linesToBeCovered);
402
403 foreach (array_keys($filesWithNoCoverage) as $fileWithNoCoverage) {
404 $rawData->removeCoverageDataForFile($fileWithNoCoverage);
405 }
406
407 if (is_array($linesToBeCovered)) {
408 foreach ($linesToBeCovered as $fileToBeCovered => $includedLines) {
409 $rawData->keepLineCoverageDataOnlyForLines($fileToBeCovered, $includedLines);
410 $rawData->keepFunctionCoverageDataOnlyForLines($fileToBeCovered, $includedLines);
411 }
412 }

501 private function performUnintentionallyCoveredCodeCheck(RawCodeCoverageData $data, array $linesToBeCovered, array $linesToBeUsed): void
502 {
503 $allowedLines = $this->getAllowedLines(
504 $linesToBeCovered,
505 $linesToBeUsed
506 );
507
508 $unintentionallyCoveredUnits = [];
509
510 foreach ($data->lineCoverage() as $file => $_data) {
511 foreach ($_data as $line => $flag) {
512 if ($flag === 1 && !isset($allowedLines[$file][$line])) {
513 $unintentionallyCoveredUnits[] = $this->wizard->lookup($file, $line);
514 }
515 }
516 }
517
518 $unintentionallyCoveredUnits = $this->processUnintentionallyCoveredUnits($unintentionallyCoveredUnits);
519
520 if (!empty($unintentionallyCoveredUnits)) {
521 throw new UnintentionallyCoveredCodeException(
522 $unintentionallyCoveredUnits
523 );
524 }

527 private function getAllowedLines(array $linesToBeCovered, array $linesToBeUsed): array
528 {
529 $allowedLines = [];
530
531 foreach (array_keys($linesToBeCovered) as $file) {
532 if (!isset($allowedLines[$file])) {
533 $allowedLines[$file] = [];
534 }
535
536 $allowedLines[$file] = array_merge(
537 $allowedLines[$file],
538 $linesToBeCovered[$file]
539 );
540 }
541
542 foreach (array_keys($linesToBeUsed) as $file) {
543 if (!isset($allowedLines[$file])) {
544 $allowedLines[$file] = [];
545 }
546
547 $allowedLines[$file] = array_merge(
548 $allowedLines[$file],
549 $linesToBeUsed[$file]
550 );
551 }
552
553 foreach (array_keys($allowedLines) as $file) {
554 $allowedLines[$file] = array_flip(
555 array_unique($allowedLines[$file])
556 );
557 }
558

569 private function processUnintentionallyCoveredUnits(array $unintentionallyCoveredUnits): array
570 {
571 $unintentionallyCoveredUnits = array_unique($unintentionallyCoveredUnits);
572 $processed = [];
573
574 foreach ($unintentionallyCoveredUnits as $unintentionallyCoveredUnit) {
575 $tmp = explode('::', $unintentionallyCoveredUnit);
576
577 if (count($tmp) !== 2) {
578 $processed[] = $unintentionallyCoveredUnit;
579
580 continue;
581 }
582
583 try {
584 $class = new ReflectionClass($tmp[0]);
585
586 foreach ($this->parentClassesExcludedFromUnintentionallyCoveredCodeCheck as $parentClass) {
587 if ($class->isSubclassOf($parentClass)) {
588 continue 2;
589 }
590 }
591 } catch (\ReflectionException $e) {
592 throw new ReflectionException(
593 $e->getMessage(),
594 $e->getCode(),
595 $e
596 );
597 }
598
599 $processed[] = $tmp[0];
600 }
601
602 $processed = array_unique($processed);
603
604 sort($processed);
605
606 return $processed;
607 }
