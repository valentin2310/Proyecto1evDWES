48 public function __construct(array $map = [])
49 {
50 foreach ($map as $mimeType => $extensions) {
51 $this->extensions[$mimeType] = $extensions;
52
53 foreach ($extensions as $extension) {
54 $this->mimeTypes[$extension][] = $mimeType;
55 }
56 }
57 $this->registerGuesser(new FileBinaryMimeTypeGuesser());
58 $this->registerGuesser(new FileinfoMimeTypeGuesser());
59 }

64 }

69 }

76 public function registerGuesser(MimeTypeGuesserInterface $guesser): void
77 {
78 array_unshift($this->guessers, $guesser);
79 }
80
81 public function getExtensions(string $mimeType): array
82 {
83 if ($this->extensions) {
84 $extensions = $this->extensions[$mimeType] ?? $this->extensions[$lcMimeType = strtolower($mimeType)] ?? null;
85 }
86
87 return $extensions ?? self::MAP[$mimeType] ?? self::MAP[$lcMimeType ?? strtolower($mimeType)] ?? [];
88 }
89
90 public function getMimeTypes(string $ext): array
91 {
92 if ($this->mimeTypes) {
93 $mimeTypes = $this->mimeTypes[$ext] ?? $this->mimeTypes[$lcExt = strtolower($ext)] ?? null;
94 }
95
96 return $mimeTypes ?? self::REVERSE_MAP[$ext] ?? self::REVERSE_MAP[$lcExt ?? strtolower($ext)] ?? [];
97 }
98
99 public function isGuesserSupported(): bool
100 {
101 foreach ($this->guessers as $guesser) {
102 if ($guesser->isGuesserSupported()) {
103 return true;
104 }
105 }
106
107 return false;
108 }
109
116 public function guessMimeType(string $path): ?string
117 {
118 foreach ($this->guessers as $guesser) {
119 if (!$guesser->isGuesserSupported()) {
120 continue;
121 }
122
123 if (null !== $mimeType = $guesser->guessMimeType($path)) {
124 return $mimeType;
125 }
126 }
127
128 if (!$this->isGuesserSupported()) {
129 throw new LogicException('Unable to guess the MIME type as no guessers are available (have you enabled the php_fileinfo extension?).');
130 }
131
132 return null;
133 }
