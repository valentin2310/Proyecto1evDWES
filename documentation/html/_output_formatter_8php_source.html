129 public function formatAndWrap(?string $message, int $width)
130 {
131 if (null === $message) {
132 return '';
133 }
134
135 $offset = 0;
136 $output = '';
137 $openTagRegex = '[a-z](?:[^\\\\<>]*+ | \\\\.)*';
138 $closeTagRegex = '[a-z][^<>]*+';
139 $currentLineLength = 0;
140 preg_match_all("#<(($openTagRegex) | /($closeTagRegex)?)>#ix", $message, $matches, \PREG_OFFSET_CAPTURE);
141 foreach ($matches[0] as $i => $match) {
142 $pos = $match[1];
143 $text = $match[0];
144
145 if (0 != $pos && '\\' == $message[$pos - 1]) {
146 continue;
147 }
148
149 // add the text up to the next tag
150 $output .= $this->applyCurrentStyle(substr($message, $offset, $pos - $offset), $output, $width, $currentLineLength);
151 $offset = $pos + \strlen($text);
152
153 // opening tag?
154 if ($open = '/' !== $text[1]) {
155 $tag = $matches[1][$i][0];
156 } else {
157 $tag = $matches[3][$i][0] ?? '';
158 }
159
160 if (!$open && !$tag) {
161 // </>
162 $this->styleStack->pop();
163 } elseif (null === $style = $this->createStyleFromString($tag)) {
164 $output .= $this->applyCurrentStyle($text, $output, $width, $currentLineLength);
165 } elseif ($open) {
166 $this->styleStack->push($style);
167 } else {
168 $this->styleStack->pop($style);
169 }
170 }
171
172 $output .= $this->applyCurrentStyle(substr($message, $offset), $output, $width, $currentLineLength);
173
174 return strtr($output, ["\0" => '\\', '\\<' => '<', '\\>' => '>']);

224 private function applyCurrentStyle(string $text, string $current, int $width, int &$currentLineLength): string
225 {
226 if ('' === $text) {
227 return '';
228 }
229
230 if (!$width) {
231 return $this->isDecorated() ? $this->styleStack->getCurrent()->apply($text) : $text;
232 }
233
234 if (!$currentLineLength && '' !== $current) {
235 $text = ltrim($text);
236 }
237
238 if ($currentLineLength) {
239 $prefix = substr($text, 0, $i = $width - $currentLineLength)."\n";
240 $text = substr($text, $i);
241 } else {
242 $prefix = '';
243 }
244
245 preg_match('~(\\n)$~', $text, $matches);
246 $text = $prefix.$this->addLineBreaks($text, $width);
247 $text = rtrim($text, "\n").($matches[1] ?? '');
248
249 if (!$currentLineLength && '' !== $current && !str_ends_with($current, "\n")) {
250 $text = "\n".$text;
251 }
252
253 $lines = explode("\n", $text);
254
255 foreach ($lines as $line) {
256 $currentLineLength += \strlen($line);
257 if ($width <= $currentLineLength) {
258 $currentLineLength = 0;
259 }
260 }
261
262 if ($this->isDecorated()) {
263 foreach ($lines as $i => $line) {
264 $lines[$i] = $this->styleStack->getCurrent()->apply($line);
265 }
266 }
267
268 return implode("\n", $lines);
269 }
