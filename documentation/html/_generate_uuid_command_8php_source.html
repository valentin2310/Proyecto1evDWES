87 protected function execute(InputInterface $input, OutputInterface $output): int
88 {
89 $io = new SymfonyStyle($input, $output instanceof ConsoleOutputInterface ? $output->getErrorOutput() : $output);
90
91 $time = $input->getOption('time-based');
92 $node = $input->getOption('node');
93 $name = $input->getOption('name-based');
94 $namespace = $input->getOption('namespace');
95 $random = $input->getOption('random-based');
96
97 if (false !== ($time ?? $name ?? $random) && 1 < ((null !== $time) + (null !== $name) + $random)) {
98 $io->error('Only one of "--time-based", "--name-based" or "--random-based" can be provided at a time.');
99
100 return 1;
101 }
102
103 if (null === $time && null !== $node) {
104 $io->error('Option "--node" can only be used with "--time-based".');
105
106 return 1;
107 }
108
109 if (null === $name && null !== $namespace) {
110 $io->error('Option "--namespace" can only be used with "--name-based".');
111
112 return 1;
113 }
114
115 switch (true) {
116 case null !== $time:
117 if (null !== $node) {
118 try {
119 $node = Uuid::fromString($node);
120 } catch (\InvalidArgumentException $e) {
121 $io->error(sprintf('Invalid node "%s": %s', $node, $e->getMessage()));
122
123 return 1;
124 }
125 }
126
127 try {
128 new \DateTimeImmutable($time);
129 } catch (\Exception $e) {
130 $io->error(sprintf('Invalid timestamp "%s": %s', $time, str_replace('DateTimeImmutable::__construct(): ', '', $e->getMessage())));
131
132 return 1;
133 }
134
135 $create = fn (): Uuid => $this->factory->timeBased($node)->create(new \DateTimeImmutable($time));
136 break;
137
138 case null !== $name:
139 if ($namespace && !\in_array($namespace, ['dns', 'url', 'oid', 'x500'], true)) {
140 try {
141 $namespace = Uuid::fromString($namespace);
142 } catch (\InvalidArgumentException $e) {
143 $io->error(sprintf('Invalid namespace "%s": %s', $namespace, $e->getMessage()));
144
145 return 1;
146 }
147 }
148
149 $create = function () use ($namespace, $name): Uuid {
150 try {
151 $factory = $this->factory->nameBased($namespace);
152 } catch (\LogicException) {
153 throw new \InvalidArgumentException('Missing namespace: use the "--namespace" option or configure a default namespace in the underlying factory.');
154 }
155
156 return $factory->create($name);
157 };
158 break;
159
160 case $random:
161 $create = $this->factory->randomBased()->create(...);
162 break;
163
164 default:
165 $create = $this->factory->create(...);
166 break;
167 }
168
169 $formatOption = $input->getOption('format');
170
171 if (\in_array($formatOption, $this->getAvailableFormatOptions())) {
172 $format = 'to'.ucfirst($formatOption);
173 } else {
174 $io->error(sprintf('Invalid format "%s", supported formats are "%s".', $formatOption, implode('", "', $this->getAvailableFormatOptions())));
175
176 return 1;
177 }
178
179 $count = (int) $input->getOption('count');
180 try {
181 for ($i = 0; $i < $count; ++$i) {
182 $output->writeln($create()->$format());
183 }
184 } catch (\Exception $e) {
185 $io->error($e->getMessage());
186
187 return 1;
188 }
189
190 return 0;
191 }
