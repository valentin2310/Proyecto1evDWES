30 // checking which of the given ID/records is in the list of current records

44 // records into the intermediate table. Then, we will add those attaches to
45 // this change list and get ready to return these results to the callers.
46 $attach = array_diff_key($records, array_flip($detach));

55 // have done any attaching or detaching, and if we have we will touch these
56 // relationships if they are configured to touch on any database updates.

58 count($changes['detached']))) {

83 public function sync($ids, $detaching = true)
84 {
85 $changes = [
86 'attached' => [], 'detached' => [], 'updated' => [],
87 ];
88
89 // First we need to attach any of the associated models that are not currently
90 // in this joining table. We'll spin through the given IDs, checking to see
91 // if they exist in the array of current ones, and if not we will insert.
92 $current = $this->getCurrentlyAttachedPivots()
93 ->pluck($this->relatedPivotKey)->all();
94
95 $records = $this->formatRecordsList($this->parseIds($ids));
96
97 // Next, we will take the differences of the currents and given IDs and detach
98 // all of the entities that exist in the "current" array but are not in the
99 // array of the new IDs given to the method which will complete the sync.
100 if ($detaching) {
101 $detach = array_diff($current, array_keys($records));
102
103 if (count($detach) > 0) {
104 $this->detach($detach);
105
106 $changes['detached'] = $this->castKeys($detach);
107 }
108 }
109
110 // Now we are finally ready to attach the new records. Note that we'll disable
111 // touching until after the entire operation is complete so we don't fire a
112 // ton of touch operations until we are totally done syncing the records.
113 $changes = array_merge(
114 $changes, $this->attachNew($records, $current, false)
115 );
116
117 // Once we have finished attaching or detaching the records, we will see if we
118 // have done any attaching or detaching, and if we have we will touch these
119 // relationships if they are configured to touch on any database updates.
120 if (count($changes['attached']) ||
121 count($changes['updated']) ||
122 count($changes['detached'])) {
123 $this->touchIfTouching();
124 }
125
126 return $changes;
127 }

169 protected function attachNew(array $records, array $current, $touch = true)
170 {
171 $changes = ['attached' => [], 'updated' => []];
172
173 foreach ($records as $id => $attributes) {
174 // If the ID is not in the list of existing pivot IDs, we will insert a new pivot
175 // record, otherwise, we will just update this existing record on this joining
176 // table, so that the developers will easily update these records pain free.
177 if (! in_array($id, $current)) {
178 $this->attach($id, $attributes, $touch);
179
180 $changes['attached'][] = $this->castKey($id);
181 }
182
183 // Now we'll try to update an existing pivot record with the attributes that were
184 // given to the method. If the model is actually updated we will add it to the
185 // list of updated pivot records so we return them back out to the consumer.
186 elseif (count($attributes) > 0 &&
187 $this->updateExistingPivot($id, $attributes, $touch)) {
188 $changes['updated'][] = $this->castKey($id);
189 }
190 }
191
192 return $changes;

432 {

435 empty($this->pivotWheres) &&
436 empty($this->pivotWhereIns) &&
437 empty($this->pivotWhereNulls)) {
438 $results = $this->detachUsingCustomClass($ids);

441
442 // If associated IDs were passed to the method we will only delete those
443 // associations, otherwise all of the association ties will be broken.
444 // We'll return the numbers of affected rows when we do the deletes.
445 if (! is_null($ids)) {

453 }

459 }
