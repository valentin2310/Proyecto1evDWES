117 public function format(LogRecord $record): string
118 {
119 $vars = parent::format($record);
120
121 if ($this->maxLevelNameLength !== null) {
122 $vars['level_name'] = substr($vars['level_name'], 0, $this->maxLevelNameLength);
123 }
124
125 $output = $this->format;
126 foreach ($vars['extra'] as $var => $val) {
127 if (false !== strpos($output, '%extra.'.$var.'%')) {
128 $output = str_replace('%extra.'.$var.'%', $this->stringify($val), $output);
129 unset($vars['extra'][$var]);
130 }
131 }
132
133 foreach ($vars['context'] as $var => $val) {
134 if (false !== strpos($output, '%context.'.$var.'%')) {
135 $output = str_replace('%context.'.$var.'%', $this->stringify($val), $output);
136 unset($vars['context'][$var]);
137 }
138 }
139
140 if ($this->ignoreEmptyContextAndExtra) {
141 if (\count($vars['context']) === 0) {
142 unset($vars['context']);
143 $output = str_replace('%context%', '', $output);
144 }
145
146 if (\count($vars['extra']) === 0) {
147 unset($vars['extra']);
148 $output = str_replace('%extra%', '', $output);
149 }
150 }
151
152 foreach ($vars as $var => $val) {
153 if (false !== strpos($output, '%'.$var.'%')) {
154 $output = str_replace('%'.$var.'%', $this->stringify($val), $output);
155 }
156 }
157
158 // remove leftover %extra.xxx% and %context.xxx% if any
159 if (false !== strpos($output, '%')) {
160 $output = preg_replace('/%(?:extra|context)\..+?%/', '', $output);
161 if (null === $output) {
162 $pcreErrorCode = preg_last_error();
163
164 throw new \RuntimeException('Failed to run preg_replace: ' . $pcreErrorCode . ' / ' . Utils::pcreLastErrorMessage($pcreErrorCode));
165 }
166 }
167
