88 public function parse(InlineParserContext $inlineContext): bool
89 {
90 $cursor = $inlineContext->getCursor();
91
92 // Autolinks can only come at the beginning of a line, after whitespace, or certain delimiting characters
93 $previousChar = $cursor->peek(-1);
94 if (! \in_array($previousChar, self::ALLOWED_AFTER, true)) {
95 return false;
96 }
97
98 // Check if we have a valid URL
99 if (! \preg_match($this->finalRegex, $cursor->getRemainder(), $matches)) {
100 return false;
101 }
102
103 $url = $matches[0];
104
105 // Does the URL end with punctuation that should be stripped?
106 if (\preg_match('/(.+?)([?!.,:*_~]+)$/', $url, $matches)) {
107 // Add the punctuation later
108 $url = $matches[1];
109 }
110
111 // Does the URL end with something that looks like an entity reference?
112 if (\preg_match('/(.+)(&[A-Za-z0-9]+;)$/', $url, $matches)) {
113 $url = $matches[1];
114 }
115
116 // Does the URL need unmatched parens chopped off?
117 if (\substr($url, -1) === ')' && ($diff = self::diffParens($url)) > 0) {
118 $url = \substr($url, 0, -$diff);
119 }
120
121 $cursor->advanceBy(\mb_strlen($url, 'UTF-8'));
122
123 // Auto-prefix 'http://' onto 'www' URLs
124 if (\substr($url, 0, 4) === 'www.') {
125 $inlineContext->getContainer()->appendChild(new Link('http://' . $url, $url));
126
127 return true;
128 }
129
130 $inlineContext->getContainer()->appendChild(new Link($url, $url));
131
132 return true;
133 }

138 private static function diffParens(string $content): int
139 {
140 // Scan the entire autolink for the total number of parentheses.
141 // If there is a greater number of closing parentheses than opening ones,
142 // we don’t consider ANY of the last characters as part of the autolink,
143 // in order to facilitate including an autolink inside a parenthesis.
144 \preg_match_all('/[()]/', $content, $matches);
145
146 $charCount = ['(' => 0, ')' => 0];
147 foreach ($matches[0] as $char) {
148 $charCount[$char]++;
149 }
150
151 return $charCount[')'] - $charCount['('];
152 }
