175 public function isExcluded(string $file): bool
176 {

192 private static function initialize(): void
193 {
194 if (self::$initialized) {
195 return;
196 }
197
198 foreach (self::EXCLUDED_CLASS_NAMES as $className => $parent) {
199 if (!class_exists($className)) {
200 continue;
201 }
202
203 $directory = (new ReflectionClass($className))->getFileName();
204
205 for ($i = 0; $i < $parent; $i++) {
206 $directory = dirname($directory);
207 }
208
209 self::$directories[] = $directory;
210 }
211
212 // Hide process isolation workaround on Windows.
213 if (PHP_OS_FAMILY === 'Windows') {
214 // tempnam() prefix is limited to first 3 chars.
215 // @see https://php.net/manual/en/function.tempnam.php
