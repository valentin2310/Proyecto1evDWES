429 public function raw($ability, $arguments = [])
430 {
431 $arguments = Arr::wrap($arguments);
432
433 $user = $this->resolveUser();
434
435 // First we will call the "before" callbacks for the Gate. If any of these give
436 // back a non-null response, we will immediately return that result in order
437 // to let the developers override all checks for some authorization cases.
438 $result = $this->callBeforeCallbacks(
439 $user, $ability, $arguments
440 );
441
442 if (is_null($result)) {
443 $result = $this->callAuthCallback($user, $ability, $arguments);
444 }
445
446 // After calling the authorization callback, we will call the "after" callbacks
447 // that are registered with the Gate, which allows a developer to do logging
448 // if that is required for this application. Then we'll return the result.

450 $user, $ability, $arguments, $result

741 protected function resolvePolicyCallback($user, $ability, array $arguments, $policy)
742 {
743 if (! is_callable([$policy, $this->formatAbilityToMethod($ability)])) {
744 return false;
745 }
746
747 return function () use ($user, $ability, $arguments, $policy) {
748 // This callback will be responsible for calling the policy's before method and
749 // running this policy method if necessary. This is used to when objects are
750 // mapped to policy objects in the user's configurations or on this class.
751 $result = $this->callPolicyBefore(
752 $policy, $user, $ability, $arguments
753 );
754
755 // When we receive a non-null result from this before method, we will return it
756 // as the "final" results. This will allow developers to override the checks
757 // in this policy to return the result for all rules defined in the class.
758 if (! is_null($result)) {
759 return $result;
760 }
761
