44 public function render(string|ControllerReference $uri, Request $request, array $options = []): Response
45 {
46 $reference = null;
47 if ($uri instanceof ControllerReference) {
48 $reference = $uri;
49
50 // Remove attributes from the generated URI because if not, the Symfony
51 // routing system will use them to populate the Request attributes. We don't
52 // want that as we want to preserve objects (so we manually set Request attributes
53 // below instead)
54 $attributes = $reference->attributes;
55 $reference->attributes = [];
56
57 // The request format and locale might have been overridden by the user
58 foreach (['_format', '_locale'] as $key) {
59 if (isset($attributes[$key])) {
60 $reference->attributes[$key] = $attributes[$key];
61 }
62 }
63
64 $uri = $this->generateFragmentUri($uri, $request, false, false);
65
66 $reference->attributes = array_merge($attributes, $reference->attributes);
67 }
68
69 $subRequest = $this->createSubRequest($uri, $request);
70
71 // override Request attributes as they can be objects (which are not supported by the generated URI)
72 if (null !== $reference) {
73 $subRequest->attributes->add($reference->attributes);
74 }
75
76 $level = ob_get_level();
77 try {
78 return SubRequestHandler::handle($this->kernel, $subRequest, HttpKernelInterface::SUB_REQUEST, false);
79 } catch (\Exception $e) {
80 // we dispatch the exception event to trigger the logging
81 // the response that comes back is ignored
82 if (isset($options['ignore_errors']) && $options['ignore_errors'] && $this->dispatcher) {
83 $event = new ExceptionEvent($this->kernel, $request, HttpKernelInterface::SUB_REQUEST, $e);
84
85 $this->dispatcher->dispatch($event, KernelEvents::EXCEPTION);
86 }
87
88 // let's clean up the output buffers that were created by the sub-request
89 Response::closeOutputBuffers($level, false);
90
91 if (isset($options['alt'])) {
92 $alt = $options['alt'];
93 unset($options['alt']);
94
95 return $this->render($alt, $request, $options);
96 }
97
98 if (!isset($options['ignore_errors']) || !$options['ignore_errors']) {
99 throw $e;
100 }

102 return new Response();
103 }
