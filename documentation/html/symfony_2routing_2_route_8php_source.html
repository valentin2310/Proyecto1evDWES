413 private function extractInlineDefaultsAndRequirements(string $pattern): string
414 {
415 if (false === strpbrk($pattern, '?<')) {
416 return $pattern;
417 }
418
419 return preg_replace_callback('#\{(!?)([\w\x80-\xFF]++)(<.*?>)?(\?[^\}]*+)?\}#', function ($m) {
420 if (isset($m[4][0])) {
421 $this->setDefault($m[2], '?' !== $m[4] ? substr($m[4], 1) : null);
422 }
423 if (isset($m[3][0])) {
424 $this->setRequirement($m[2], substr($m[3], 1, -1));
425 }
426
427 return '{'.$m[1].$m[2].'}';
428 }, $pattern);
429 }

431 private function sanitizeRequirement(string $key, string $regex): string
432 {
433 if ('' !== $regex) {
434 if ('^' === $regex[0]) {
435 $regex = substr($regex, 1);
436 } elseif (str_starts_with($regex, '\\A')) {
437 $regex = substr($regex, 2);
438 }
439 }
440
441 if (str_ends_with($regex, '$')) {
442 $regex = substr($regex, 0, -1);
443 } elseif (\strlen($regex) - 2 === strpos($regex, '\\z')) {
444 $regex = substr($regex, 0, -2);
445 }
446
447 if ('' === $regex) {
448 throw new \InvalidArgumentException(sprintf('Routing requirement for "%s" cannot be empty.', $key));
449 }
450
451 return $regex;
452 }
