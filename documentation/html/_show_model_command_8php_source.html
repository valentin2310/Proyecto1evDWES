154 protected function getVirtualAttributes($model, $columns)
155 {
156 $class = new ReflectionClass($model);
157
158 return collect($class->getMethods())
159 ->reject(
160 fn (ReflectionMethod $method) => $method->isStatic()
161 || $method->isAbstract()
162 || $method->getDeclaringClass()->getName() === Model::class
163 )
164 ->mapWithKeys(function (ReflectionMethod $method) use ($model) {
165 if (preg_match('/^get(.+)Attribute$/', $method->getName(), $matches) === 1) {
166 return [Str::snake($matches[1]) => 'accessor'];
167 } elseif ($model->hasAttributeMutator($method->getName())) {
168 return [Str::snake($method->getName()) => 'attribute'];

170 return [];

173 ->reject(fn ($cast, $name) => collect($columns)->has($name))

176 'type' => null,

195 protected function getRelations($model)
196 {
197 return collect(get_class_methods($model))
198 ->map(fn ($method) => new ReflectionMethod($model, $method))
199 ->reject(
200 fn (ReflectionMethod $method) => $method->isStatic()
201 || $method->isAbstract()
202 || $method->getDeclaringClass()->getName() === Model::class
203 )
204 ->filter(function (ReflectionMethod $method) {
205 $file = new SplFileObject($method->getFileName());

239 protected function getObservers($model)
240 {
241 $listeners = $this->getLaravel()->make('events')->getRawListeners();
242
243 // Get the Eloquent observers for this model...
244 $listeners = array_filter($listeners, function ($v, $key) use ($model) {
245 return Str::startsWith($key, 'eloquent.') && Str::endsWith($key, $model::class);
246 }, ARRAY_FILTER_USE_BOTH);
247
248 // Format listeners Eloquent verb => Observer methods...
249 $extractVerb = function ($key) {
250 preg_match('/eloquent.([a-zA-Z]+)\: /', $key, $matches);
251
252 return $matches[1] ?? '?';
253 };
254
255 $formatted = [];
256
257 foreach ($listeners as $key => $observerMethods) {
258 $formatted[] = [
259 'event' => $extractVerb($key),
260 'observer' => array_map(fn ($obs) => is_string($obs) ? $obs : 'Closure', $observerMethods),
261 ];
262 }
263
264 return collect($formatted);
