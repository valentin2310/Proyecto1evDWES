105 // If we have a cache record we will check the expiration time against current
106 // time on the system and see if the record has expired. If it has, we will
107 // remove the records from the database table so it isn't returned again.

109 return;

212 protected function incrementOrDecrement($key, $value, Closure $callback)
213 {
214 return $this->connection->transaction(function () use ($key, $value, $callback) {
215 $prefixed = $this->prefix.$key;
216
217 $cache = $this->table()->where('key', $prefixed)
218 ->lockForUpdate()->first();
219
220 // If there is no value in the cache, we will return false here. Otherwise the
221 // value will be decrypted and we will proceed with this function to either
222 // increment or decrement this value based on the given action callbacks.
223 if (is_null($cache)) {
224 return false;
225 }
226
227 $cache = is_array($cache) ? (object) $cache : $cache;
228
229 $current = $this->unserialize($cache->value);
230
231 // Here we'll call this callback function that was given to the function which
232 // is used to either increment or decrement the function. We use a callback
233 // so we do not have to recreate all this logic in each of the functions.
234 $new = $callback((int) $current, $value);
235
236 if (! is_numeric($current)) {
237 return false;
238 }
239
240 // Here we will update the values in the table. We will also encrypt the value
241 // since database cache values are encrypted by default with secure storage

244 'value' => $this->serialize($new),
