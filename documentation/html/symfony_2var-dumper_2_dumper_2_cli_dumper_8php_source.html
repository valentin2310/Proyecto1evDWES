366 protected function dumpKey(Cursor $cursor)
367 {
368 if (null !== $key = $cursor->hashKey) {
369 if ($cursor->hashKeyIsBinary) {
370 $key = $this->utf8Encode($key);
371 }
372 $attr = ['binary' => $cursor->hashKeyIsBinary];
373 $bin = $cursor->hashKeyIsBinary ? 'b' : '';
374 $style = 'key';
375 switch ($cursor->hashType) {
376 default:
377 case Cursor::HASH_INDEXED:
378 if (self::DUMP_LIGHT_ARRAY & $this->flags) {
379 break;
380 }
381 $style = 'index';
382 // no break
383 case Cursor::HASH_ASSOC:
384 if (\is_int($key)) {
385 $this->line .= $this->style($style, $key).' => ';
386 } else {
387 $this->line .= $bin.'"'.$this->style($style, $key).'" => ';
388 }
389 break;
390
391 case Cursor::HASH_RESOURCE:
392 $key = "\0~\0".$key;
393 // no break
394 case Cursor::HASH_OBJECT:
395 if (!isset($key[0]) || "\0" !== $key[0]) {
396 $this->line .= '+'.$bin.$this->style('public', $key).': ';
397 } elseif (0 < strpos($key, "\0", 1)) {
398 $key = explode("\0", substr($key, 1), 2);
399
400 switch ($key[0][0]) {
401 case '+': // User inserted keys
402 $attr['dynamic'] = true;
403 $this->line .= '+'.$bin.'"'.$this->style('public', $key[1], $attr).'": ';
404 break 2;
405 case '~':
406 $style = 'meta';
407 if (isset($key[0][1])) {
408 parse_str(substr($key[0], 1), $attr);
409 $attr += ['binary' => $cursor->hashKeyIsBinary];
410 }
411 break;
412 case '*':
413 $style = 'protected';
414 $bin = '#'.$bin;
415 break;
416 default:
417 $attr['class'] = $key[0];
418 $style = 'private';
419 $bin = '-'.$bin;
420 break;
421 }
422
423 if (isset($attr['collapse'])) {
424 if ($attr['collapse']) {
425 $this->collapseNextHash = true;
426 } else {
427 $this->expandNextHash = true;
428 }
429 }
430
431 $this->line .= $bin.$this->style($style, $key[1], $attr).($attr['separator'] ?? ': ');
432 } else {
433 // This case should not happen
434 $this->line .= '-'.$bin.'"'.$this->style('private', $key, ['class' => '']).'": ';
435 }
436 break;
437 }
438
439 if ($cursor->hardRefTo) {
440 $this->line .= $this->style('ref', '&'.($cursor->hardRefCount ? $cursor->hardRefTo : ''), ['count' => $cursor->hardRefCount]).' ';
