139 protected function ensureCanDefine(Stmt $stmt, string $scopeType = self::CLASS_TYPE)
140 {
141 // Anonymous classes don't have a name, and uniqueness shouldn't be enforced.
142 if ($stmt->name === null) {
143 return;
144 }
145
146 $name = $this->getFullyQualifiedName($stmt->name);
147
148 // check for name collisions
149 $errorType = null;
150 if ($this->classExists($name)) {
151 $errorType = self::CLASS_TYPE;
152 } elseif ($this->interfaceExists($name)) {
153 $errorType = self::INTERFACE_TYPE;
154 } elseif ($this->traitExists($name)) {
155 $errorType = self::TRAIT_TYPE;
156 }
157
158 if ($errorType !== null) {
159 throw $this->createError(\sprintf('%s named %s already exists', \ucfirst($errorType), $name), $stmt);
160 }
161
162 // Store creation for the rest of this code snippet so we can find local
163 // issue too
164 $this->currentScope[\strtolower($name)] = $scopeType;
165 }

221 protected function ensureMethodExists(string $class, string $name, Stmt $stmt)
222 {
223 $this->ensureClassOrTraitExists($class, $stmt);
224
225 // let's pretend all calls to self, parent and static are valid
226 if (\in_array(\strtolower($class), ['self', 'parent', 'static'])) {
227 return;
228 }
229
230 // ... and all calls to classes defined right now
231 if ($this->findInScope($class) === self::CLASS_TYPE) {
232 return;
233 }
234
235 // if method name is an expression, give it a pass for now
236 if ($name instanceof Expr) {
237 return;
238 }
239
240 if (!\method_exists($class, $name) && !\method_exists($class, '__callStatic')) {
241 throw $this->createError(\sprintf('Call to undefined method %s::%s()', $class, $name), $stmt);
242 }
243 }

295 protected function classExists(string $name): bool
296 {
297 // Give `self`, `static` and `parent` a pass. This will actually let
298 // some errors through, since we're not checking whether the keyword is
299 // being used in a class scope.
300 if (\in_array(\strtolower($name), ['self', 'static', 'parent'])) {
301 return true;
302 }
303
304 return \class_exists($name) || $this->findInScope($name) === self::CLASS_TYPE;
305 }
