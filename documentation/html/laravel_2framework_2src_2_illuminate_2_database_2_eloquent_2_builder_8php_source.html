769 protected function eagerLoadRelation(array $models, $name, Closure $constraints)
770 {
771 // First we will "back up" the existing where conditions on the query so we can
772 // add our eager constraints. Then we will merge the wheres that were on the
773 // query back to it in order that any where conditions might be specified.
774 $relation = $this->getRelation($name);
775
776 $relation->addEagerConstraints($models);
777
778 $constraints($relation);
779
780 // Once we have the results, we just match those back up to their parent models
781 // using the relationship instance. Then we just return the finished arrays
782 // of models which have been eagerly hydrated and are readied for return.
783 return $relation->match(
784 $relation->initRelation($models, $name),
785 $relation->getEager(), $name
786 );
787 }

795 public function getRelation($name)
796 {
797 // We want to run a relationship query without any constrains so that we will
798 // not have to remove these where clauses manually which gets really hacky
799 // and error prone. We don't want constraints because we add eager ones.
800 $relation = Relation::noConstraints(function () use ($name) {
801 try {
802 return $this->getModel()->newInstance()->$name();
803 } catch (BadMethodCallException) {
804 throw RelationNotFoundException::make($this->getModel(), $name);
805 }
806 });
807
808 $nested = $this->relationsNestedUnder($name);
809
810 // If there are nested relationships set on the query, we will put those onto
811 // the query instances so that they can be handled after this relationship
812 // is loaded. In this way they will all trickle down as they are loaded.
813 if (count($nested) > 0) {
814 $relation->getQuery()->with($nested);
815 }
816
817 return $relation;
818 }

1342 public function applyScopes()
1343 {
1344 if (! $this->scopes) {
1345 return $this;
1346 }
1347
1348 $builder = clone $this;
1349
1350 foreach ($this->scopes as $identifier => $scope) {
1351 if (! isset($builder->scopes[$identifier])) {
1352 continue;
1353 }
1354
1355 $builder->callScope(function (self $builder) use ($scope) {
1356 // If the scope is a Closure we will just go ahead and call the scope with the
1357 // builder instance. The "callScope" method will properly group the clauses
1358 // that are added to this query so "where" clauses maintain proper logic.
1359 if ($scope instanceof Closure) {
1360 $scope($builder);
1361 }
1362
1363 // If the scope is a scope object, we will call the apply method on this scope
1364 // passing in the builder and the model instance. After we run all of these
1365 // scopes we will return back the builder instance to the outside caller.
1366 if ($scope instanceof Scope) {
1367 $scope->apply($builder, $this->getModel());
1368 }
1369 });
1370 }

1372 return $builder;

1382 protected function callScope(callable $scope, array $parameters = [])
1383 {
1384 array_unshift($parameters, $this);
1385
1386 $query = $this->getQuery();
1387
1388 // We will keep track of how many wheres are on the query before running the
1389 // scope so that we can properly group the added scope constraints in the
1390 // query as their own isolated nested where statement and avoid issues.
1391 $originalWhereCount = is_null($query->wheres)
1392 ? 0 : count($query->wheres);
1393
1394 $result = $scope(...$parameters) ?? $this;
1395
1396 if (count((array) $query->wheres) > $originalWhereCount) {
1397 $this->addNewWheresWithinGroup($query, $originalWhereCount);
1398 }
1399
1400 return $result;

1425 {

1427 // rebuild them as nested queries by slicing the groups of wheres into
1428 // their own sections. This is to prevent any confusing logic order.

1431 $query->wheres = [];

1575 protected function prepareNestedWithRelationships($relations, $prefix = '')
1576 {
1577 $preparedRelationships = [];
1578
1579 if ($prefix !== '') {
1580 $prefix .= '.';
1581 }
1582
1583 // If any of the relationships are formatted with the [$attribute => array()]
1584 // syntax, we shall loop over the nested relations and prepend each key of
1585 // this array while flattening into the traditional dot notation format.
1586 foreach ($relations as $key => $value) {
1587 if (! is_string($key) || ! is_array($value)) {
1588 continue;
1589 }
1590
1591 [$attribute, $attributeSelectConstraint] = $this->parseNameAndAttributeSelectionConstraint($key);
1592
1593 $preparedRelationships = array_merge(
1594 $preparedRelationships,
1595 ["{$prefix}{$attribute}" => $attributeSelectConstraint],
1596 $this->prepareNestedWithRelationships($value, "{$prefix}{$attribute}"),
1597 );
1598
1599 unset($relations[$key]);
1600 }
1601
1602 // We now know that the remaining relationships are in a dot notation format
1603 // and may be a string or Closure. We'll loop over them and ensure all of
1604 // the present Closures are merged + strings are made into constraints.
1605 foreach ($relations as $key => $value) {
1606 if (is_numeric($key) && is_string($value)) {
1607 [$key, $value] = $this->parseNameAndAttributeSelectionConstraint($value);
1608 }
1609
1610 $preparedRelationships[$prefix.$key] = $this->combineConstraints([
1611 $value,
1612 $preparedRelationships[$prefix.$key] ?? static function () {
1613 //
1614 },
1615 ]);
1616 }
1617
1618 return $preparedRelationships;
1619 }
