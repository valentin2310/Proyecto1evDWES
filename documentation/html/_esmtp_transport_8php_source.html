166 private function parseCapabilities(string $ehloResponse): array
167 {
168 $capabilities = [];
169 $lines = explode("\r\n", trim($ehloResponse));
170 array_shift($lines);
171 foreach ($lines as $line) {
172 if (preg_match('/^[0-9]{3}[ -]([A-Z0-9-]+)((?:[ =].*)?)$/Di', $line, $matches)) {
173 $value = strtoupper(ltrim($matches[2], ' ='));
174 $capabilities[strtoupper($matches[1])] = $value ? explode(' ', $value) : [];
175 }
176 }
177
178 return $capabilities;

181 private function handleAuth(array $modes): void
182 {
183 if (!$this->username) {
184 return;
185 }
186
187 $code = null;
188 $authNames = [];
189 $errors = [];
190 $modes = array_map('strtolower', $modes);
191 foreach ($this->authenticators as $authenticator) {
192 if (!\in_array(strtolower($authenticator->getAuthKeyword()), $modes, true)) {
193 continue;
194 }
195
196 $code = null;
197 $authNames[] = $authenticator->getAuthKeyword();
198 try {
199 $authenticator->authenticate($this);
200
201 return;
202 } catch (TransportExceptionInterface $e) {
203 $code = $e->getCode();
204
205 try {
206 $this->executeCommand("RSET\r\n", [250]);
207 } catch (TransportExceptionInterface) {
208 // ignore this exception as it probably means that the server error was final
209 }
210
211 // keep the error message, but tries the other authenticators
212 $errors[$authenticator->getAuthKeyword()] = $e->getMessage();
213 }
214 }
215
216 if (!$authNames) {
217 throw new TransportException(sprintf('Failed to find an authenticator supported by the SMTP server, which currently supports: "%s".', implode('", "', $modes)), $code ?: 504);
218 }
219
220 $message = sprintf('Failed to authenticate on SMTP server with username "%s" using the following authenticators: "%s".', $this->username, implode('", "', $authNames));
221 foreach ($errors as $name => $error) {
222 $message .= sprintf(' Authenticator "%s" returned "%s".', $name, $error);
223 }
224
225 throw new TransportException($message, $code ?: 535);
226 }
