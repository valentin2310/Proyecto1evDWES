123 {
124 $file = $this->getFilename($profile->getToken());
125
126 $profileIndexed = is_file($file);
127 if (!$profileIndexed) {
128 // Create directory
129 $dir = \dirname($file);
130 if (!is_dir($dir) && false === @mkdir($dir, 0777, true) && !is_dir($dir)) {
131 throw new \RuntimeException(sprintf('Unable to create the storage directory (%s).', $dir));
132 }

138 $parentToken = $profile->getParentToken() !== $profileToken ? $profile->getParentToken() : null;
139 $childrenToken = array_filter(array_map(fn (Profile $p) => $profileToken !== $p->getToken() ? $p->getToken() : null, $profile->getChildren()));
140
141 // Store profile
142 $data = [
143 'token' => $profileToken,
144 'parent' => $parentToken,
145 'children' => $childrenToken,
146 'data' => $profile->getCollectors(),
147 'ip' => $profile->getIp(),
148 'method' => $profile->getMethod(),

151 'status_code' => $profile->getStatusCode(),

216 protected function readLineFromFile($file): mixed
217 {
218 $line = '';
219 $position = ftell($file);
220
221 if (0 === $position) {
222 return null;
223 }
224
225 while (true) {
226 $chunkSize = min($position, 1024);
227 $position -= $chunkSize;
228 fseek($file, $position);
229
230 if (0 === $chunkSize) {
231 // bof reached
232 break;
233 }
234
235 $buffer = fread($file, $chunkSize);
236
237 if (false === ($upTo = strrpos($buffer, "\n"))) {
238 $line = $buffer.$line;
239 continue;
240 }
241
242 $position += $upTo;
243 $line = substr($buffer, $upTo + 1).$line;
244 fseek($file, max(0, $position), \SEEK_SET);
245
246 if ('' !== $line) {
247 break;
248 }
249 }
250
251 return '' === $line ? null : $line;
252 }

284 private function doRead($token, Profile $profile = null): ?Profile
285 {
286 if (!$token || !file_exists($file = $this->getFilename($token))) {
287 return null;
288 }
289
290 $h = fopen($file, 'r');
291 flock($h, \LOCK_SH);

293 flock($h, \LOCK_UN);
294 fclose($h);
295

308 {

310 $file = $this->getIndexFilename();

313 if ($offset = is_file($file.'.offset') ? (int) file_get_contents($file.'.offset') : 0) {
