208 private function findLinesIgnoredByLineBasedAnnotations(string $filename, string $source, bool $useAnnotationsForIgnoringCode): void
209 {
210 if (!$useAnnotationsForIgnoringCode) {
211 return;
212 }
213
214 $start = false;
215
216 foreach (token_get_all($source) as $token) {
217 if (!is_array($token) ||
218 !(T_COMMENT === $token[0] || T_DOC_COMMENT === $token[0])) {
219 continue;
220 }
221
222 $comment = trim($token[1]);
223
224 if ($comment === '// @codeCoverageIgnore' ||
225 $comment === '//@codeCoverageIgnore') {
226 $this->ignoredLines[$filename][] = $token[2];
227
228 continue;
229 }
230
231 if ($comment === '// @codeCoverageIgnoreStart' ||
232 $comment === '//@codeCoverageIgnoreStart') {
233 $start = $token[2];
234
235 continue;
236 }
237

239 $comment === '//@codeCoverageIgnoreEnd') {
240 if (false === $start) {
241 $start = $token[2];
242 }
243
244 $this->ignoredLines[$filename] = array_merge(
