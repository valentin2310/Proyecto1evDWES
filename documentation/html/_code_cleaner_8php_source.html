188 private function addImplicitDebugContext(array $passes)
189 {
190 $file = $this->getDebugFile();
191 if ($file === null) {
192 return;
193 }
194
195 try {
196 $code = @\file_get_contents($file);
197 if (!$code) {
198 return;
199 }
200
201 $stmts = $this->parse($code, true);
202 if ($stmts === false) {
203 return;
204 }
205
206 // Set up a clean traverser for just these code cleaner passes
207 $traverser = new NodeTraverser();
208 foreach ($passes as $pass) {
209 $traverser->addVisitor($pass);
210 }
211
212 $traverser->traverse($stmts);
213 } catch (\Throwable $e) {
214 // Don't care.
215 }

223 private static function getDebugFile()
224 {
225 $trace = \debug_backtrace(\DEBUG_BACKTRACE_IGNORE_ARGS);
226
227 foreach (\array_reverse($trace) as $stackFrame) {
228 if (!self::isDebugCall($stackFrame)) {

232 if (\preg_match('/eval\‍(/', $stackFrame['file'])) {

266 public function clean(array $codeLines, bool $requireSemicolons = false)
267 {
268 $stmts = $this->parse('<?php '.\implode(\PHP_EOL, $codeLines).\PHP_EOL, $requireSemicolons);
269 if ($stmts === false) {
270 return false;
271 }
272
273 // Catch fatal errors before they happen
274 $stmts = $this->traverser->traverse($stmts);
275
276 // Work around https://github.com/nikic/PHP-Parser/issues/399
277 $oldLocale = \setlocale(\LC_NUMERIC, 0);
278 \setlocale(\LC_NUMERIC, 'C');
279
280 $code = $this->printer->prettyPrint($stmts);
281
282 // Now put the locale back
283 \setlocale(\LC_NUMERIC, $oldLocale);
284
285 return $code;
286 }

321 protected function parse(string $code, bool $requireSemicolons = false)
322 {
323 try {
324 return $this->parser->parse($code);
325 } catch (\PhpParser\Error $e) {
326 if ($this->parseErrorIsUnclosedString($e, $code)) {
327 return false;
328 }
329
330 if ($this->parseErrorIsUnterminatedComment($e, $code)) {
331 return false;
332 }
333
334 if ($this->parseErrorIsTrailingComma($e, $code)) {
335 return false;
336 }
337
338 if (!$this->parseErrorIsEOF($e)) {
339 throw ParseErrorException::fromParseError($e);
340 }
341
342 if ($requireSemicolons) {
343 return false;
344 }
345
346 try {
347 // Unexpected EOF, try again with an implicit semicolon
348 return $this->parser->parse($code.';');
349 } catch (\PhpParser\Error $e) {
350 return false;
351 }
352 }
