174 public function trans(?string $id, array $parameters = [], string $domain = null, string $locale = null): string
175 {
176 if (null === $id || '' === $id) {
177 return '';
178 }
179
180 $domain ??= 'messages';
181
182 $catalogue = $this->getCatalogue($locale);
183 $locale = $catalogue->getLocale();
184 while (!$catalogue->defines($id, $domain)) {
185 if ($cat = $catalogue->getFallbackCatalogue()) {
186 $catalogue = $cat;
187 $locale = $catalogue->getLocale();
188 } else {
189 break;
190 }
191 }
192
193 $parameters = array_map(fn ($parameter) => $parameter instanceof TranslatableInterface ? $parameter->trans($this, $locale) : $parameter, $parameters);
194
195 $len = \strlen(MessageCatalogue::INTL_DOMAIN_SUFFIX);
196 if ($this->hasIntlFormatter
197 && ($catalogue->defines($id, $domain.MessageCatalogue::INTL_DOMAIN_SUFFIX)
198 || (\strlen($domain) > $len && 0 === substr_compare($domain, MessageCatalogue::INTL_DOMAIN_SUFFIX, -$len, $len)))
199 ) {
200 return $this->formatter->formatIntl($catalogue->get($id, $domain), $locale, $parameters);
201 }
202
203 return $this->formatter->format($catalogue->get($id, $domain), $locale, $parameters);
204 }

313 private function getFallbackContent(MessageCatalogue $catalogue): string
314 {
315 $fallbackContent = '';
316 $current = '';

321 $fallbackSuffix = ucfirst(preg_replace($replacementPattern, '_', $fallback));

334 $fallbackSuffix

369 private function loadFallbackCatalogues(string $locale): void
370 {
371 $current = $this->catalogues[$locale];
372
373 foreach ($this->computeFallbackLocales($locale) as $fallback) {
374 if (!isset($this->catalogues[$fallback])) {
375 $this->initializeCatalogue($fallback);
376 }
377
378 $fallbackCatalogue = new MessageCatalogue($fallback, $this->getAllMessages($this->catalogues[$fallback]));
379 foreach ($this->catalogues[$fallback]->getResources() as $resource) {
380 $fallbackCatalogue->addResource($resource);
381 }
382 $current->addFallbackCatalogue($fallbackCatalogue);
383 $current = $fallbackCatalogue;
384 }

390 protected function computeFallbackLocales(string $locale)
391 {
392 $this->parentLocales ??= json_decode(file_get_contents(__DIR__.'/Resources/data/parents.json'), true);
393
394 $originLocale = $locale;
395 $locales = [];
396
397 while ($locale) {
398 $parent = $this->parentLocales[$locale] ?? null;
399
400 if ($parent) {
401 $locale = 'root' !== $parent ? $parent : null;
402 } elseif (\function_exists('locale_parse')) {
403 $localeSubTags = locale_parse($locale);
404 $locale = null;
405 if (1 < \count($localeSubTags)) {
406 array_pop($localeSubTags);
407 $locale = locale_compose($localeSubTags) ?: null;
408 }
409 } elseif ($i = strrpos($locale, '_') ?: strrpos($locale, '-')) {
410 $locale = substr($locale, 0, $i);
411 } else {
412 $locale = null;
413 }
414
415 if (null !== $locale) {
416 $locales[] = $locale;
417 }
418 }
419
420 foreach ($this->fallbackLocales as $fallback) {
421 if ($fallback === $originLocale) {
422 continue;
423 }
424
425 $locales[] = $fallback;
426 }
427
428 return array_unique($locales);
429 }

456 private function getAllMessages(MessageCatalogueInterface $catalogue): array
457 {
458 $allMessages = [];

461 if ($intlMessages = $catalogue->all($domain.MessageCatalogue::INTL_DOMAIN_SUFFIX)) {
462 $allMessages[$domain.MessageCatalogue::INTL_DOMAIN_SUFFIX] = $intlMessages;
463 $messages = array_diff_key($messages, $intlMessages);
