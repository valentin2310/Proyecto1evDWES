542 {
543 // ditch the anchors
544 $regex = preg_replace('/^\/?\^?/', '', $regex);
545 $regex = preg_replace('/\$?\/?$/', '', $regex);
546 // All {2} become {2,2}
547 $regex = preg_replace('/\{(\d+)\}/', '{\1,\1}', $regex);
548 // Single-letter quantifiers (?, *, +) become bracket quantifiers ({0,1}, {0,rand}, {1, rand})
549 $regex = preg_replace('/(?<!\\\‍)\?/', '{0,1}', $regex);
550 $regex = preg_replace('/(?<!\\\‍)\*/', '{0,' . static::randomDigitNotNull() . '}', $regex);
551 $regex = preg_replace('/(?<!\\\‍)\+/', '{1,' . static::randomDigitNotNull() . '}', $regex);
552 // [12]{1,2} becomes [12] or [12][12]
553 $regex = preg_replace_callback('/(\[[^\]]+\])\{(\d+),(\d+)\}/', static function ($matches) {
554 return str_repeat($matches[1], Base::randomElement(range($matches[2], $matches[3])));
555 }, $regex);
556 // (12|34){1,2} becomes (12|34) or (12|34)(12|34)
557 $regex = preg_replace_callback('/(\‍([^\‍)]+\‍))\{(\d+),(\d+)\}/', static function ($matches) {
558 return str_repeat($matches[1], Base::randomElement(range($matches[2], $matches[3])));
559 }, $regex);
560 // A{1,2} becomes A or AA or \d{3} becomes \d\d\d
561 $regex = preg_replace_callback('/(\\\?.)\{(\d+),(\d+)\}/', static function ($matches) {
562 return str_repeat($matches[1], Base::randomElement(range($matches[2], $matches[3])));
563 }, $regex);
564 // (this|that) becomes 'this' or 'that'
565 $regex = preg_replace_callback('/\‍((.*?)\‍)/', static function ($matches) {
566 return Base::randomElement(explode('|', str_replace(['(', ')'], '', $matches[1])));
567 }, $regex);
568 // All A-F inside of [] become ABCDEF
569 $regex = preg_replace_callback('/\[([^\]]+)\]/', static function ($matches) {
570 return '[' . preg_replace_callback('/(\w|\d)\-(\w|\d)/', static function ($range) {
571 return implode('', range($range[1], $range[2]));

573 }, $regex);
574 // All [ABC] become B (or A or C)
575 $regex = preg_replace_callback('/\[([^\]]+)\]/', static function ($matches) {

577 $match = preg_replace('/\\\‍(?!\\\‍)/', '', $matches[1]);
