559 private static function formatArgument($argument, $depth = 0)
560 {
561 if ($argument instanceof MatcherAbstract) {
562 return (string) $argument;
563 }
564
565 if (is_object($argument)) {
566 return 'object(' . get_class($argument) . ')';
567 }
568
569 if (is_int($argument) || is_float($argument)) {
570 return $argument;
571 }
572
573 if (is_array($argument)) {
574 if ($depth === 1) {
575 $argument = '[...]';
576 } else {
577 $sample = array();
578 foreach ($argument as $key => $value) {
579 $key = is_int($key) ? $key : "'$key'";
580 $value = self::formatArgument($value, $depth + 1);
581 $sample[] = "$key => $value";
582 }
583
584 $argument = "[" . implode(", ", $sample) . "]";
585 }
586
587 return ((strlen($argument) > 1000) ? substr($argument, 0, 1000) . '...]' : $argument);
588 }
589
590 if (is_bool($argument)) {
591 return $argument ? 'true' : 'false';
592 }
593
594 if (is_resource($argument)) {
595 return 'resource(...)';
596 }
597
598 if (is_null($argument)) {
599 return 'NULL';
600 }
601
602 return "'" . (string) $argument . "'";
