124 public function writeHistory(): bool
125 {
126 // We have to write history first, since it is used
127 // by Libedit to list history
128 if ($this->historyFile !== false) {
129 $res = \readline_write_history($this->historyFile);
130 } else {
131 $res = true;
132 }
133
134 if (!$res || !$this->eraseDups && !$this->historySize > 0) {
135 return $res;
136 }
137
138 $hist = $this->listHistory();
139 if (!$hist) {
140 return true;
141 }
142
143 if ($this->eraseDups) {
144 // flip-flip technique: removes duplicates, latest entries win.
145 $hist = \array_flip(\array_flip($hist));
146 // sort on keys to get the order back
147 \ksort($hist);
148 }
149
150 if ($this->historySize > 0) {
151 $histsize = \count($hist);
152 if ($histsize > $this->historySize) {
153 $hist = \array_slice($hist, $histsize - $this->historySize);
154 }
155 }
156
157 \readline_clear_history();
158 foreach ($hist as $line) {
159 \readline_add_history($line);
160 }
161
162 if ($this->historyFile !== false) {
163 return \readline_write_history($this->historyFile);
164 }
