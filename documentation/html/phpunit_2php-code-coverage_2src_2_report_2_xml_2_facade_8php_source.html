169 }

173 );

175 $this->saveDocument($fileReport->asDom(), $file->id());

178 private function processUnit(array $unit, Report $report): void
179 {
180 if (isset($unit['className'])) {

182 } else {
183 $unitObject = $report->traitObject($unit['traitName']);
184 }
185
186 $unitObject->setLines(

188 $unit['executableLines'],
189 $unit['executedLines']
190 );
191
192 $unitObject->setCrap((float) $unit['crap']);
193 $unitObject->setNamespace($unit['namespace']);
194
195 foreach ($unit['methods'] as $method) {
196 $methodObject = $unitObject->addMethod($method['methodName']);
197 $methodObject->setSignature($method['signature']);
198 $methodObject->setLines((string) $method['startLine'], (string) $method['endLine']);
199 $methodObject->setCrap($method['crap']);
200 $methodObject->setTotals(
201 (string) $method['executableLines'],
202 (string) $method['executedLines'],

204 );
205 }
