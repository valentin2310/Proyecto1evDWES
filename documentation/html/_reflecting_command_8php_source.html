82 protected function getTarget(string $valueName): array
83 {
84 $valueName = \trim($valueName);
85 $matches = [];
86 switch (true) {
87 case \preg_match(self::CLASS_OR_FUNC, $valueName, $matches):
88 return [$this->resolveName($matches[0], true), null, 0];
89
90 case \preg_match(self::CLASS_MEMBER, $valueName, $matches):
91 return [$this->resolveName($matches[1]), $matches[2], Mirror::CONSTANT | Mirror::METHOD];
92
93 case \preg_match(self::CLASS_STATIC, $valueName, $matches):
94 return [$this->resolveName($matches[1]), $matches[2], Mirror::STATIC_PROPERTY | Mirror::PROPERTY];
95
96 case \preg_match(self::INSTANCE_MEMBER, $valueName, $matches):
97 if ($matches[2] === '->') {
98 $kind = Mirror::METHOD | Mirror::PROPERTY;
99 } else {
100 $kind = Mirror::CONSTANT | Mirror::METHOD;
101 }
102
103 return [$this->resolveObject($matches[1]), $matches[3], $kind];
104
105 default:
106 return [$this->resolveObject($valueName), null, 0];
107 }

118 protected function resolveName(string $name, bool $includeFunctions = false): string
119 {
120 $shell = $this->getApplication();
121
122 // While not *technically* 100% accurate, let's treat `self` and `static` as equivalent.
123 if (\in_array(\strtolower($name), ['self', 'static'])) {
124 if ($boundClass = $shell->getBoundClass()) {
125 return $boundClass;
126 }
127
128 if ($boundObject = $shell->getBoundObject()) {
129 return \get_class($boundObject);
130 }
131
132 $msg = \sprintf('Cannot use "%s" when no class scope is active', \strtolower($name));
133 throw new ErrorException($msg, 0, \E_USER_ERROR, "eval()'d code", 1);
134 }
135
136 if (\substr($name, 0, 1) === '\\') {
137 return $name;
138 }
139
140 // Check $name against the current namespace and use statements.
141 if (self::couldBeClassName($name)) {
142 try {
143 $name = $this->resolveCode($name.'::class');
144 } catch (RuntimeException $e) {
145 // /shrug
146 }
147 }
148
149 if ($namespace = $shell->getNamespace()) {
150 $fullName = $namespace.'\\'.$name;
151
152 if (\class_exists($fullName) || \interface_exists($fullName) || ($includeFunctions && \function_exists($fullName))) {
153 return $fullName;
154 }
155 }
156
157 return $name;
158 }

192 protected function resolveCode(string $code)
193 {
194 try {
195 // Add an implicit `sudo` to target resolution.
196 $nodes = $this->traverser->traverse($this->parser->parse($code));
197 $sudoCode = $this->printer->prettyPrint($nodes);
198 $value = $this->getApplication()->execute($sudoCode, true);
199 } catch (\Throwable $e) {
200 // Swallow all exceptions?
201 }
202
203 if (!isset($value) || $value instanceof NoReturnValue) {
204 throw new RuntimeException('Unknown target: '.$code);
205 }
206
207 return $value;
208 }

273 protected function setCommandScopeVariables(\Reflector $reflector)
274 {
275 $vars = [];
276
277 switch (\get_class($reflector)) {
278 case \ReflectionClass::class:
279 case \ReflectionObject::class:
280 $vars['__class'] = $reflector->name;
281 if ($reflector->inNamespace()) {
282 $vars['__namespace'] = $reflector->getNamespaceName();
283 }
284 break;
285
286 case \ReflectionMethod::class:
287 $vars['__method'] = \sprintf('%s::%s', $reflector->class, $reflector->name);
288 $vars['__class'] = $reflector->class;
289 $classReflector = $reflector->getDeclaringClass();
290 if ($classReflector->inNamespace()) {
291 $vars['__namespace'] = $classReflector->getNamespaceName();
292 }
293 break;
294
295 case \ReflectionFunction::class:
296 $vars['__function'] = $reflector->name;
297 if ($reflector->inNamespace()) {
298 $vars['__namespace'] = $reflector->getNamespaceName();
299 }
300 break;
301
302 case \ReflectionGenerator::class:
303 $funcReflector = $reflector->getFunction();
304 $vars['__function'] = $funcReflector->name;
305 if ($funcReflector->inNamespace()) {
306 $vars['__namespace'] = $funcReflector->getNamespaceName();
307 }
308 if ($fileName = $reflector->getExecutingFile()) {
309 $vars['__file'] = $fileName;
310 $vars['__line'] = $reflector->getExecutingLine();
311 $vars['__dir'] = \dirname($fileName);
312 }
313 break;
314
315 case \ReflectionProperty::class:
316 case \ReflectionClassConstant::class:
317 case ReflectionClassConstant::class:
318 $classReflector = $reflector->getDeclaringClass();
319 $vars['__class'] = $classReflector->name;
320 if ($classReflector->inNamespace()) {
321 $vars['__namespace'] = $classReflector->getNamespaceName();
322 }
323 // no line for these, but this'll do
324 if ($fileName = $reflector->getDeclaringClass()->getFileName()) {
325 $vars['__file'] = $fileName;
326 $vars['__dir'] = \dirname($fileName);
327 }
328 break;
329
330 case ReflectionConstant_::class:
331 if ($reflector->inNamespace()) {
332 $vars['__namespace'] = $reflector->getNamespaceName();
333 }
334 break;
335 }
336
337 if ($reflector instanceof \ReflectionClass || $reflector instanceof \ReflectionFunctionAbstract) {
338 if ($fileName = $reflector->getFileName()) {
339 $vars['__file'] = $fileName;
340 $vars['__line'] = $reflector->getStartLine();
341 $vars['__dir'] = \dirname($fileName);
342 }
343 }
344
345 $this->context->setCommandScopeVariables($vars);
346 }
