112 public function getSlackData(LogRecord $record): array
113 {
114 $dataArray = [];
115
116 if ($this->username !== null) {
117 $dataArray['username'] = $this->username;
118 }
119
120 if ($this->channel !== null) {
121 $dataArray['channel'] = $this->channel;
122 }
123
124 if ($this->formatter !== null && !$this->useAttachment) {
125 $message = $this->formatter->format($record);
126 } else {
127 $message = $record->message;
128 }
129
130 $recordData = $this->removeExcludedFields($record);
131
132 if ($this->useAttachment) {
133 $attachment = [
134 'fallback' => $message,
135 'text' => $message,
136 'color' => $this->getAttachmentColor($record->level),
137 'fields' => [],
138 'mrkdwn_in' => ['fields'],
139 'ts' => $recordData['datetime']->getTimestamp(),
140 'footer' => $this->username,
141 'footer_icon' => $this->userIcon,
142 ];
143
144 if ($this->useShortAttachment) {
145 $attachment['title'] = $recordData['level_name'];
146 } else {
147 $attachment['title'] = 'Message';
148 $attachment['fields'][] = $this->generateAttachmentField('Level', $recordData['level_name']);
149 }
150
151 if ($this->includeContextAndExtra) {
152 foreach (['extra', 'context'] as $key) {
153 if (!isset($recordData[$key]) || \count($recordData[$key]) === 0) {
154 continue;
155 }
156
157 if ($this->useShortAttachment) {
158 $attachment['fields'][] = $this->generateAttachmentField(
159 $key,
160 $recordData[$key]
161 );
162 } else {
163 // Add all extra fields as individual fields in attachment
164 $attachment['fields'] = array_merge(
165 $attachment['fields'],
166 $this->generateAttachmentFields($recordData[$key])
167 );
168 }
169 }
170 }
171
172 $dataArray['attachments'] = [$attachment];
173 } else {
174 $dataArray['text'] = $message;
175 }
176
177 if ($this->userIcon !== null) {
178 if (false !== ($iconUrl = filter_var($this->userIcon, FILTER_VALIDATE_URL))) {
179 $dataArray['icon_url'] = $iconUrl;
180 } else {
181 $dataArray['icon_emoji'] = ":{$this->userIcon}:";
182 }
183 }
184
185 return $dataArray;
186 }

207 public function stringify(array $fields): string
208 {
210 $normalized = $this->normalizerFormatter->normalizeValue($fields);
211
212 $hasSecondDimension = \count(array_filter($normalized, 'is_array')) > 0;
213 $hasOnlyNonNumericKeys = \count(array_filter(array_keys($normalized), 'is_numeric')) === 0;

360 private function removeExcludedFields(LogRecord $record): array
361 {
362 $recordData = $record->toArray();
363 foreach ($this->excludeFields as $field) {
364 $keys = explode('.', $field);
365 $node = &$recordData;
366 $lastKey = end($keys);
367 foreach ($keys as $key) {
368 if (!isset($node[$key])) {
369 break;
370 }
371 if ($lastKey === $key) {
372 unset($node[$key]);
373 break;
374 }
375 $node = &$node[$key];
376 }
377 }
378
379 return $recordData;
380 }
