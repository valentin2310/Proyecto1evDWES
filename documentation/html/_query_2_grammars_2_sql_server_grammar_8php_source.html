398 public function compileUpsert(Builder $query, array $values, array $uniqueBy, array $update)
399 {
400 $columns = $this->columnize(array_keys(reset($values)));
401
402 $sql = 'merge '.$this->wrapTable($query->from).' ';
403
404 $parameters = collect($values)->map(function ($record) {
405 return '('.$this->parameterize($record).')';
406 })->implode(', ');
407
408 $sql .= 'using (values '.$parameters.') '.$this->wrapTable('laravel_source').' ('.$columns.') ';
409
410 $on = collect($uniqueBy)->map(function ($column) use ($query) {
411 return $this->wrap('laravel_source.'.$column).' = '.$this->wrap($query->from.'.'.$column);
412 })->implode(' and ');
413
414 $sql .= 'on '.$on.' ';
415
416 if ($update) {
417 $update = collect($update)->map(function ($value, $key) {
418 return is_numeric($key)
419 ? $this->wrap($value).' = '.$this->wrap('laravel_source.'.$value)
420 : $this->wrap($key).' = '.$this->parameter($value);
421 })->implode(', ');
422
423 $sql .= 'when matched then update set '.$update.' ';
424 }
425
426 $sql .= 'when not matched then insert ('.$columns.') values ('.$columns.');';
427
428 return $sql;
