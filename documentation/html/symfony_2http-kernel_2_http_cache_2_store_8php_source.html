77 public function lock(Request $request): bool|string
78 {
79 $key = $this->getCacheKey($request);
80
81 if (!isset($this->locks[$key])) {
82 $path = $this->getPath($key);
83 if (!is_dir(\dirname($path)) && false === @mkdir(\dirname($path), 0777, true) && !is_dir(\dirname($path))) {
84 return $path;
85 }

142 public function lookup(Request $request): ?Response
143 {
144 $key = $this->getCacheKey($request);
145
146 if (!$entries = $this->getMetadata($key)) {
147 return null;
148 }
149
150 // find a cached entry that matches the request.
151 $match = null;
152 foreach ($entries as $entry) {
153 if ($this->requestsMatch(isset($entry[1]['vary'][0]) ? implode(', ', $entry[1]['vary']) : '', $request->headers->all(), $entry[0])) {
154 $match = $entry;
155
156 break;
157 }
158 }
159
160 if (null === $match) {
161 return null;
162 }
163
164 $headers = $match[1];
165 if (file_exists($path = $this->getPath($headers['x-content-digest'][0]))) {
166 return $this->restoreResponse($headers, $path);
167 }
168
169 // TODO the metaStore referenced an entity that doesn't exist in
170 // the entityStore. We definitely want to return nil but we should
171 // also purge the entry from the meta-store when this is detected.
172 return null;
173 }

183 public function write(Request $request, Response $response): string
184 {
185 $key = $this->getCacheKey($request);
186 $storedEnv = $this->persistRequest($request);
187
188 if ($response->headers->has('X-Body-File')) {
189 // Assume the response came from disk, but at least perform some safeguard checks
190 if (!$response->headers->has('X-Content-Digest')) {
191 throw new \RuntimeException('A restored response must have the X-Content-Digest header.');
192 }
193
194 $digest = $response->headers->get('X-Content-Digest');
195 if ($this->getPath($digest) !== $response->headers->get('X-Body-File')) {
196 throw new \RuntimeException('X-Body-File and X-Content-Digest do not match.');
197 }
198 // Everything seems ok, omit writing content to disk
199 } else {
200 $digest = $this->generateContentDigest($response);
201 $response->headers->set('X-Content-Digest', $digest);
202
203 if (!$this->save($digest, $response->getContent(), false)) {
204 throw new \RuntimeException('Unable to store the entity.');
205 }
206
207 if (!$response->headers->has('Transfer-Encoding')) {
208 $response->headers->set('Content-Length', \strlen($response->getContent()));
209 }
210 }
211
212 // read existing cache entries, remove non-varying, and add this one to the list
213 $entries = [];
214 $vary = $response->headers->get('vary');
215 foreach ($this->getMetadata($key) as $entry) {
216 if (!isset($entry[1]['vary'][0])) {
217 $entry[1]['vary'] = [''];
218 }
219
220 if ($entry[1]['vary'][0] != $vary || !$this->requestsMatch($vary ?? '', $entry[0], $storedEnv)) {
221 $entries[] = $entry;
222 }
223 }
224
225 $headers = $this->persistResponse($response);
226 unset($headers['age']);
227
228 foreach ($this->options['private_headers'] as $h) {
229 unset($headers[strtolower($h)]);
230 }
231
232 array_unshift($entries, [$storedEnv, $headers]);
233
234 if (!$this->save($key, serialize($entries))) {
235 throw new \RuntimeException('Unable to store the metadata.');
236 }
237
238 return $key;
239 }

256 public function invalidate(Request $request)
257 {
258 $modified = false;
259 $key = $this->getCacheKey($request);
260
261 $entries = [];
262 foreach ($this->getMetadata($key) as $entry) {
263 $response = $this->restoreResponse($entry[1]);
264 if ($response->isFresh()) {
265 $response->expire();
266 $modified = true;
267 $entries[] = [$entry[0], $this->persistResponse($response)];
268 } else {
269 $entries[] = $entry;
270 }
271 }
272
273 if ($modified && !$this->save($key, serialize($entries))) {
274 throw new \RuntimeException('Unable to store the metadata.');
275 }

286 private function requestsMatch(?string $vary, array $env1, array $env2): bool
287 {
288 if (empty($vary)) {
289 return true;
290 }
291
292 foreach (preg_split('/[\s,]+/', $vary) as $header) {
293 $key = str_replace('_', '-', strtolower($header));
294 $v1 = $env1[$key] ?? null;
295 $v2 = $env2[$key] ?? null;
296 if ($v1 !== $v2) {
297 return false;
298 }
299 }
300
301 return true;
302 }

370 private function save(string $key, string $data, bool $overwrite = true): bool
371 {
372 $path = $this->getPath($key);
373
374 if (!$overwrite && file_exists($path)) {
375 return true;
376 }
377
378 if (isset($this->locks[$key])) {
379 $fp = $this->locks[$key];
380 @ftruncate($fp, 0);
381 @fseek($fp, 0);
382 $len = @fwrite($fp, $data);
383 if (\strlen($data) !== $len) {
384 @ftruncate($fp, 0);
385
386 return false;
387 }
388 } else {
389 if (!is_dir(\dirname($path)) && false === @mkdir(\dirname($path), 0777, true) && !is_dir(\dirname($path))) {
390 return false;
391 }
392
393 $tmpFile = tempnam(\dirname($path), basename($path));
394 if (false === $fp = @fopen($tmpFile, 'w')) {
395 @unlink($tmpFile);
396
397 return false;
398 }
399 @fwrite($fp, $data);
400 @fclose($fp);
401
402 if ($data != file_get_contents($tmpFile)) {
403 @unlink($tmpFile);
404
405 return false;
406 }
407
408 if (false === @rename($tmpFile, $path)) {
409 @unlink($tmpFile);
410
411 return false;
412 }
413 }
414
415 @chmod($path, 0666 & ~umask());
416
417 return true;
418 }

477 private function restoreResponse(array $headers, string $path = null): ?Response
478 {
479 $status = $headers['X-Status'][0];
