232 protected function write(LogRecord $record): void
233 {
234 if ($record->level->isLowerThan(Level::Notice)) {
235 $this->handleDebugRecord($record);
236 } elseif (isset($record->context['exception']) && $record->context['exception'] instanceof \Throwable) {
237 $this->handleExceptionRecord($record);

243 private function handleDebugRecord(LogRecord $record): void
244 {
245 [$tags, $filteredContext] = $this->getRecordTags($record);
246 $message = $record->message;
247 if (\count($filteredContext) > 0) {
248 $message .= ' ' . Utils::jsonEncode($this->connector->getDumper()->dump(array_filter($filteredContext)), null, true);
249 }
250 $this->connector->getDebugDispatcher()->dispatchDebug($message, $tags, $this->options['classesPartialsTraceIgnore']);
251 }

274 private function getRecordTags(LogRecord $record): array
275 {
276 $tags = null;
277 $filteredContext = [];
278 if ($record->context !== []) {
279 $filteredContext = $record->context;
280 foreach ($this->options['debugTagsKeysInContext'] as $key) {
281 if (isset($filteredContext[$key])) {
282 $tags = $filteredContext[$key];
283 if ($key === 0) {
284 array_shift($filteredContext);
285 } else {
286 unset($filteredContext[$key]);
287 }
288 break;
289 }
290 }
291 }
292
293 return [$tags ?? $record->level->toPsrLogLevel(), $filteredContext];
