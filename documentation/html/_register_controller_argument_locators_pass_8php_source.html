136 } elseif ('' === $target = (string) substr($target, 1)) {

143
144 [$bindingValue, $bindingId, , $bindingType, $bindingFile] = $binding->getValues();
145 $binding->setValues([$bindingValue, $bindingId, true, $bindingType, $bindingFile]);

150 } elseif (!$autowire || (!($autowireAttributes ??= $p->getAttributes(Autowire::class, \ReflectionAttribute::IS_INSTANCEOF)) && (!$type || '\\' !== $target[0]))) {

175 $container->register((string) $args[$p->name], 'mixed')

183 if ($type && !$p->isOptional() && !$p->allowsNull() && !class_exists($type) && !interface_exists($type, false)) {

185
186 // see if the type-hint lives in the same namespace as the controller

189 }
190
191 $container->register($erroredId = '.errored.'.$container->hash($message), $type)
192 ->addError($message);
193
194 $args[$p->name] = new Reference($erroredId, ContainerInterface::RUNTIME_EXCEPTION_ON_INVALID_REFERENCE);
195 } else {
196 $target = preg_replace('/(^|[(|&])\\\\/', '\1', $target);
197 $args[$p->name] = $type ? new TypedReference($target, $type, $invalidBehavior, Target::parseName($p)) : new Reference($target, $invalidBehavior);
198 }
199 }
200 // register the maps as a per-method service-locators
201 if ($args) {
202 $controllers[$id.'::'.$r->name] = ServiceLocatorTagPass::register($container, $args);
203
204 foreach ($publicAliases[$id] ?? [] as $alias) {
205 $controllers[$alias.'::'.$r->name] = clone $controllers[$id.'::'.$r->name];
206 }
207 }
208 }
209 }
210
211 $controllerLocatorRef = ServiceLocatorTagPass::register($container, $controllers);
212
213 if ($container->hasDefinition('argument_resolver.service')) {
214 $container->getDefinition('argument_resolver.service')
215 ->replaceArgument(0, $controllerLocatorRef);
216 }
217
218 if ($container->hasDefinition('argument_resolver.not_tagged_controller')) {
219 $container->getDefinition('argument_resolver.not_tagged_controller')
220 ->replaceArgument(0, $controllerLocatorRef);
221 }
222
223 $container->setAlias('argument_resolver.controller_locator', (string) $controllerLocatorRef);
224 }
