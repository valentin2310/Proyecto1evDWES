1048 public function getRelativeUriForPath(string $path): string
1049 {
1050 // be sure that we are dealing with an absolute path
1051 if (!isset($path[0]) || '/' !== $path[0]) {
1052 return $path;
1053 }
1054
1055 if ($path === $basePath = $this->getPathInfo()) {
1056 return '';
1057 }
1058
1059 $sourceDirs = explode('/', isset($basePath[0]) && '/' === $basePath[0] ? substr($basePath, 1) : $basePath);
1060 $targetDirs = explode('/', substr($path, 1));
1061 array_pop($sourceDirs);
1062 $targetFile = array_pop($targetDirs);
1063
1064 foreach ($sourceDirs as $i => $dir) {
1065 if (isset($targetDirs[$i]) && $dir === $targetDirs[$i]) {
1066 unset($sourceDirs[$i], $targetDirs[$i]);
1067 } else {
1068 break;
1069 }
1070 }
1071
1072 $targetDirs[] = $targetFile;
1073 $path = str_repeat('../', \count($sourceDirs)).implode('/', $targetDirs);
1074
1075 // A reference to the same base directory or an empty subdirectory must be prefixed with "./".
1076 // This also applies to a segment with a colon character (e.g., "file:colon") that cannot be used
1077 // as the first segment of a relative-path reference, as it would be mistaken for a scheme name
1078 // (see https://tools.ietf.org/html/rfc3986#section-4.2).
1079 return !isset($path[0]) || '/' === $path[0]
1080 || false !== ($colonPos = strpos($path, ':')) && ($colonPos < ($slashPos = strpos($path, '/')) || false === $slashPos)
1081 ? "./$path" : $path;

1127 {

1129 $host = $host[0];
1130 } elseif (!$host = $this->headers->get('HOST')) {
1131 if (!$host = $this->server->get('SERVER_NAME')) {
1132 $host = $this->server->get('SERVER_ADDR', '');
1133 }
1134 }
1135
1136 // trim and remove port number from host
1137 // host is lowercase as per RFC 952/2181
1138 $host = strtolower(preg_replace('/:\d+$/', '', trim($host)));
1139
1140 // as the host can come from the user (HTTP_HOST and depending on the configuration, SERVER_NAME too can come from the user)
1141 // check that it does not contain forbidden characters (see RFC 952 and RFC 2181)
1142 // use preg_replace() instead of preg_match() to prevent DoS attacks with long host names
1143 if ($host && '' !== preg_replace('/(?:^\[)?[a-zA-Z0-9-:\]_]+\.?/', '', $host)) {
1144 if (!$this->isHostValid) {
1145 return '';
1146 }
1147 $this->isHostValid = false;
1148
1149 throw new SuspiciousOperationException(sprintf('Invalid Host "%s".', $host));
1150 }
1151

1153 // to avoid host header injection attacks, you should provide a list of trusted host patterns
1154
1155 if (\in_array($host, self::$trustedHosts)) {
1156 return $host;
1157 }
1158
1159 foreach (self::$trustedHostPatterns as $pattern) {
1160 if (preg_match($pattern, $host)) {
1161 self::$trustedHosts[] = $host;
1162
1163 return $host;
1164 }
1165 }
1166
1167 if (!$this->isHostValid) {
1168 return '';
1169 }
1170 $this->isHostValid = false;
1171
1172 throw new SuspiciousOperationException(sprintf('Untrusted Host "%s".', $host));
1173 }

1808 protected function prepareBaseUrl(): string
1809 {
1810 $filename = basename($this->server->get('SCRIPT_FILENAME', ''));
1811
1812 if (basename($this->server->get('SCRIPT_NAME', '')) === $filename) {
1813 $baseUrl = $this->server->get('SCRIPT_NAME');
1814 } elseif (basename($this->server->get('PHP_SELF', '')) === $filename) {
1815 $baseUrl = $this->server->get('PHP_SELF');
1816 } elseif (basename($this->server->get('ORIG_SCRIPT_NAME', '')) === $filename) {
1817 $baseUrl = $this->server->get('ORIG_SCRIPT_NAME'); // 1and1 shared hosting compatibility
1818 } else {
1819 // Backtrack up the script_filename to find the portion matching
1820 // php_self
1821 $path = $this->server->get('PHP_SELF', '');
1822 $file = $this->server->get('SCRIPT_FILENAME', '');
1823 $segs = explode('/', trim($file, '/'));
1824 $segs = array_reverse($segs);
1825 $index = 0;
1826 $last = \count($segs);
1827 $baseUrl = '';
1828 do {
1829 $seg = $segs[$index];
1830 $baseUrl = '/'.$seg.$baseUrl;
1831 ++$index;
1832 } while ($last > $index && (false !== $pos = strpos($path, $baseUrl)) && 0 != $pos);
1833 }
1834
1835 // Does the baseUrl have anything in common with the request_uri?
1836 $requestUri = $this->getRequestUri();
1837 if ('' !== $requestUri && '/' !== $requestUri[0]) {
1838 $requestUri = '/'.$requestUri;
1839 }
1840
1841 if ($baseUrl && null !== $prefix = $this->getUrlencodedPrefix($requestUri, $baseUrl)) {
1842 // full $baseUrl matches
1843 return $prefix;
1844 }
1845
1846 if ($baseUrl && null !== $prefix = $this->getUrlencodedPrefix($requestUri, rtrim(\dirname($baseUrl), '/'.\DIRECTORY_SEPARATOR).'/')) {
1847 // directory portion of $baseUrl matches
1848 return rtrim($prefix, '/'.\DIRECTORY_SEPARATOR);
1849 }
1850
1851 $truncatedRequestUri = $requestUri;
1852 if (false !== $pos = strpos($requestUri, '?')) {
1853 $truncatedRequestUri = substr($requestUri, 0, $pos);
1854 }
1855
1856 $basename = basename($baseUrl ?? '');
1857 if (empty($basename) || !strpos(rawurldecode($truncatedRequestUri), $basename)) {
1858 // no match whatsoever; set it blank
1859 return '';
1860 }
1861
1862 // If using mod_rewrite or ISAPI_Rewrite strip the script filename
1863 // out of baseUrl. $pos !== 0 makes sure it is not matching a value
1864 // from PATH_INFO or QUERY_STRING
1865 if (\strlen($requestUri) >= \strlen($baseUrl) && (false !== $pos = strpos($requestUri, $baseUrl)) && 0 !== $pos) {
1866 $baseUrl = substr($requestUri, 0, $pos + \strlen($baseUrl));
1867 }
1868
1869 return rtrim($baseUrl, '/'.\DIRECTORY_SEPARATOR);
1870 }

2063 private function normalizeAndFilterClientIps(array $clientIps, string $ip): array
2064 {

2067 }
2068 $clientIps[] = $ip; // Complete the IP chain with the IP the request actually came from
2069 $firstTrustedIp = null;
2070
2071 foreach ($clientIps as $key => $clientIp) {
2072 if (strpos($clientIp, '.')) {
2073 // Strip :port from IPv4 addresses. This is allowed in Forwarded
2074 // and may occur in X-Forwarded-For.
2075 $i = strpos($clientIp, ':');
2076 if ($i) {
2077 $clientIps[$key] = $clientIp = substr($clientIp, 0, $i);
2078 }
2079 } elseif (str_starts_with($clientIp, '[')) {
2080 // Strip brackets and :port from IPv6 addresses.
2081 $i = strpos($clientIp, ']', 1);
2082 $clientIps[$key] = $clientIp = substr($clientIp, 1, $i - 1);
2083 }
