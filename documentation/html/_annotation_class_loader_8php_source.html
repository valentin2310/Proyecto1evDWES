183 foreach ($path as $locale => $localePath) {
184 $paths[$locale] = $prefix.$localePath;

186 } elseif ($missing = array_diff_key($prefix, $path)) {
187 throw new \LogicException(sprintf('Route to "%s" is missing paths for locale(s) "%s".', $class->name.'::'.$method->name, implode('", "', array_keys($missing))));

212 $defaults[$param->name] = $defaultValue;

218 }
219 }

221 foreach ($paths as $locale => $path) {
