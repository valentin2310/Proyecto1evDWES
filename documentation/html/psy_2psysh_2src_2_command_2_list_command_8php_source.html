117 protected function execute(InputInterface $input, OutputInterface $output)
118 {
119 $this->validateInput($input);
120 $this->initEnumerators();
121
122 $method = $input->getOption('long') ? 'writeLong' : 'write';
123
124 if ($target = $input->getArgument('target')) {
125 list($target, $reflector) = $this->getTargetAndReflector($target);
126 } else {
127 $reflector = null;
128 }
129
130 // @todo something cleaner than this :-/
131 if ($output instanceof ShellOutput && $input->getOption('long')) {
132 $output->startPaging();
133 }
134
135 foreach ($this->enumerators as $enumerator) {
136 $this->$method($output, $enumerator->enumerate($input, $reflector, $target));
137 }
138
139 if ($output instanceof ShellOutput && $input->getOption('long')) {
140 $output->stopPaging();
141 }
142
143 // Set some magic local variables
144 if ($reflector !== null) {
145 $this->setCommandScopeVariables($reflector);
146 }
147
148 return 0;
149 }

178 protected function write(OutputInterface $output, array $result)
179 {
180 if (\count($result) === 0) {
181 return;
182 }
183
184 foreach ($result as $label => $items) {
185 $names = \array_map([$this, 'formatItemName'], $items);
186 $output->writeln(\sprintf('<strong>%s</strong>: %s', $label, \implode(', ', $names)));
187 }

198 protected function writeLong(OutputInterface $output, array $result)
199 {
200 if (\count($result) === 0) {
201 return;
202 }
203
204 $table = $this->getTable($output);
205
206 foreach ($result as $label => $items) {
207 $output->writeln('');
208 $output->writeln(\sprintf('<strong>%s:</strong>', $label));
209
210 $table->setRows([]);
211 foreach ($items as $item) {
212 $table->addRow([$this->formatItemName($item), $item['value']]);
213 }
214
215 $table->render();
216 }

236 private function validateInput(InputInterface $input)
237 {
238 if (!$input->getArgument('target')) {
239 // if no target is passed, there can be no properties or methods
240 foreach (['properties', 'methods', 'no-inherit'] as $option) {
241 if ($input->getOption($option)) {
242 throw new RuntimeException('--'.$option.' does not make sense without a specified target');
243 }
244 }
245
246 foreach (['globals', 'vars', 'constants', 'functions', 'classes', 'interfaces', 'traits'] as $option) {
247 if ($input->getOption($option)) {
248 return;
249 }
250 }
251
252 // default to --vars if no other options are passed
253 $input->setOption('vars', true);
254 } else {
255 // if a target is passed, classes, functions, etc don't make sense
256 foreach (['vars', 'globals'] as $option) {
257 if ($input->getOption($option)) {
258 throw new RuntimeException('--'.$option.' does not make sense with a specified target');
259 }
260 }
261
262 // @todo ensure that 'functions', 'classes', 'interfaces', 'traits' only accept namespace target?
263 foreach (['constants', 'properties', 'methods', 'functions', 'classes', 'interfaces', 'traits'] as $option) {
264 if ($input->getOption($option)) {
265 return;
266 }
267 }
268
269 // default to --constants --properties --methods if no other options are passed
270 $input->setOption('constants', true);
271 $input->setOption('properties', true);
272 $input->setOption('methods', true);
273 }
274 }
