251 public function testSuiteFinished(TestSuiteFinished $event): void
252 {
253 if ($this->currentTestSuiteForTestClassFailed) {
254 return;
255 }
256
257 $testSuite = $event->testSuite();
258
259 if ($testSuite->isWithName()) {
260 return;
261 }
262
263 if ($testSuite->isForTestMethodWithDataProvider()) {
264 assert($testSuite instanceof TestSuiteForTestMethodWithDataProvider);
265
266 $test = $testSuite->tests()->asArray()[0];
267
268 assert($test instanceof TestMethod);
269
270 PassedTests::instance()->testMethodPassed($test, null);
271
272 return;
273 }
274
275 assert($testSuite instanceof TestSuiteForTestClass);
276
277 PassedTests::instance()->testClassPassed($testSuite->className());
278 }
