277 private function coverageColor(int $numberOfCoveredElements, int $totalNumberOfElements): string
278 {
279 $coverage = Percentage::fromFractionAndTotal(
280 $numberOfCoveredElements,
281 $totalNumberOfElements
282 );
283
284 if ($coverage->asFloat() >= $this->thresholds->highLowerBound()) {
285 return self::COLOR_GREEN;
286 }
287
288 if ($coverage->asFloat() > $this->thresholds->lowUpperBound()) {
289 return self::COLOR_YELLOW;
290 }
291
292 return self::COLOR_RED;
293 }

295 private function printCoverageCounts(int $numberOfCoveredElements, int $totalNumberOfElements, int $precision): string
296 {
297 $format = '%' . $precision . 's';
298
299 return Percentage::fromFractionAndTotal(

301 $totalNumberOfElements
302 )->asFixedWidthString() .

304 sprintf($format, $totalNumberOfElements) . ')';
305 }
