413 public static function calculateSize(
414 int $srcWidth,
415 int $srcHeight,
416 $newWidth,
417 $newHeight,
418 int $mode = self::OrSmaller,
419 ): array
420 {
421 if ($newWidth === null) {
422 } elseif (self::isPercent($newWidth)) {
423 $newWidth = (int) round($srcWidth / 100 * abs($newWidth));
424 $percents = true;
425 } else {
426 $newWidth = abs($newWidth);
427 }
428
429 if ($newHeight === null) {
430 } elseif (self::isPercent($newHeight)) {
431 $newHeight = (int) round($srcHeight / 100 * abs($newHeight));
432 $mode |= empty($percents) ? 0 : self::Stretch;
433 } else {
434 $newHeight = abs($newHeight);
435 }
436
437 if ($mode & self::Stretch) { // non-proportional
438 if (!$newWidth || !$newHeight) {
439 throw new Nette\InvalidArgumentException('For stretching must be both width and height specified.');
440 }
441
442 if ($mode & self::ShrinkOnly) {
443 $newWidth = min($srcWidth, $newWidth);
444 $newHeight = min($srcHeight, $newHeight);
445 }
446 } else { // proportional
447 if (!$newWidth && !$newHeight) {
448 throw new Nette\InvalidArgumentException('At least width or height must be specified.');
449 }
450
451 $scale = [];
452 if ($newWidth > 0) { // fit width
453 $scale[] = $newWidth / $srcWidth;
454 }
455
456 if ($newHeight > 0) { // fit height
457 $scale[] = $newHeight / $srcHeight;
458 }
459
460 if ($mode & self::OrBigger) {
461 $scale = [max($scale)];
462 }
463
464 if ($mode & self::ShrinkOnly) {
465 $scale[] = 1;
466 }
467
468 $scale = min($scale);
469 $newWidth = (int) round($srcWidth * $scale);
470 $newHeight = (int) round($srcHeight * $scale);
471 }
472
473 return [max($newWidth, 1), max($newHeight, 1)];
474 }

711 {
712 switch ($type) {

715 $success = @imagejpeg($this->image, $file, $quality); // @ is escalated to exception
716 break;
717

719 $quality = $quality === null ? 9 : max(0, min(9, $quality));
720 $success = @imagepng($this->image, $file, $quality); // @ is escalated to exception
721 break;
722
723 case ImageType::GIF:
724 $success = @imagegif($this->image, $file); // @ is escalated to exception
725 break;
726
727 case ImageType::WEBP:
728 $quality = $quality === null ? 80 : max(0, min(100, $quality));
729 $success = @imagewebp($this->image, $file, $quality); // @ is escalated to exception
730 break;
731

733 $quality = $quality === null ? 30 : max(0, min(100, $quality));
734 $success = @imageavif($this->image, $file, $quality); // @ is escalated to exception

739 break;

741 default:
