60 private function expandClasses(array $patterns, array $classes): array
61 {
62 $expanded = [];
63
64 // Explicit classes declared in the patterns are returned directly
65 foreach ($patterns as $key => $pattern) {
66 if (!str_ends_with($pattern, '\\') && !str_contains($pattern, '*')) {
67 unset($patterns[$key]);
68 $expanded[] = ltrim($pattern, '\\');
69 }
70 }
71
72 // Match patterns with the classes list
73 $regexps = $this->patternsToRegexps($patterns);
74
75 foreach ($classes as $class) {
76 $class = ltrim($class, '\\');
77
78 if ($this->matchAnyRegexps($class, $regexps)) {
79 $expanded[] = $class;
80 }
81 }
82
83 return array_unique($expanded);
84 }

86 private function getClassesInComposerClassMaps(): array
87 {
88 $classes = [];

107 private function patternsToRegexps(array $patterns): array
108 {
109 $regexps = [];
110
111 foreach ($patterns as $pattern) {
112 // Escape user input
113 $regex = preg_quote(ltrim($pattern, '\\'));
114
115 // Wildcards * and **
116 $regex = strtr($regex, ['\\*\\*' => '.*?', '\\*' => '[^\\\\]*?']);
117
118 // If this class does not end by a slash, anchor the end
119 if (!str_ends_with($regex, '\\')) {
120 $regex .= '$';
121 }
122
123 $regexps[] = '{^\\\\'.$regex.'}';
124 }
125
126 return $regexps;
127 }

129 private function matchAnyRegexps(string $class, array $regexps): bool

133 foreach ($regexps as $regex) {

135 continue;
136 }

138 if (preg_match($regex, '\\'.$class)) {
139 return true;
140 }
141 }
142
143 return false;
