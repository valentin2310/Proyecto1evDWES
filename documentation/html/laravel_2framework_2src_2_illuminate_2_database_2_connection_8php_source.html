468 public function cursor($query, $bindings = [], $useReadPdo = true)
469 {
470 $statement = $this->run($query, $bindings, function ($query, $bindings) use ($useReadPdo) {
471 if ($this->pretending()) {
472 return [];
473 }
474
475 // First we will create a statement for the query. Then, we will set the fetch
476 // mode and prepare the bindings for the query. Once that's done we will be
477 // ready to execute the query against the database and return the cursor.
478 $statement = $this->prepared($this->getPdoForSelect($useReadPdo)

481 $this->bindValues(
482 $statement, $this->prepareBindings($bindings)

486 // so we can return the cursor. The cursor will use a PHP generator to give
487 // back one row at a time without using a bunch of memory to render them.
488 $statement->execute();

490 return $statement;
491 });

762 protected function run($query, $bindings, Closure $callback)
763 {
764 foreach ($this->beforeExecutingCallbacks as $beforeExecutingCallback) {
765 $beforeExecutingCallback($query, $bindings, $this);
766 }
767
768 $this->reconnectIfMissingConnection();
769
770 $start = microtime(true);
771
772 // Here we will run this query. If an exception occurs we'll determine if it was
773 // caused by a connection that has been lost. If that is the cause, we'll try
774 // to re-establish connection and re-run the query with a fresh connection.
775 try {
776 $result = $this->runQueryCallback($query, $bindings, $callback);
777 } catch (QueryException $e) {
778 $result = $this->handleQueryException(
779 $e, $query, $bindings, $callback
780 );
781 }
782
783 // Once we have run the query we will calculate the time that it took to run and
784 // then log the query, bindings, and execution time so we will report them on
785 // the event that the developer needs them. We'll log time in milliseconds.
786 $this->logQuery(
787 $query, $bindings, $this->getElapsedTime($start)
788 );
789
790 return $result;
791 }

803 protected function runQueryCallback($query, $bindings, Closure $callback)
804 {
805 // To execute the statement, we'll simply call the callback, which will actually
806 // run the SQL against the PDO connection. Then we can calculate the time it
807 // took to execute and log the query SQL, bindings and time in our memory.
808 try {
809 return $callback($query, $bindings);
810 }
811
812 // If an exception occurs when attempting to run a query, we'll format the error
813 // message to include the bindings with SQL, which will make this exception a
814 // lot more helpful to the developer instead of just the database's errors.
815 catch (Exception $e) {
816 if ($this->isUniqueConstraintError($e)) {
817 throw new UniqueConstraintViolationException(
818 $this->getName(), $query, $this->prepareBindings($bindings), $e
819 );
820 }
821
822 throw new QueryException(
823 $this->getName(), $query, $this->prepareBindings($bindings), $e
824 );
825 }
826 }

1094 public function escape($value, $binary = false)
1095 {
1096 if ($value === null) {
1097 return 'null';
1098 } elseif ($binary) {
1099 return $this->escapeBinary($value);
1100 } elseif (is_int($value) || is_float($value)) {

1102 } elseif (is_bool($value)) {
