138 public function addHeader(string $name, mixed $argument, array $more = []): static
139 {
140 $headerClass = self::HEADER_CLASS_MAP[strtolower($name)] ?? UnstructuredHeader::class;
141 if (\is_array($headerClass)) {
142 $headerClass = $headerClass[0];
143 }
144 $parts = explode('\\', $headerClass);
145 $method = 'add'.ucfirst(array_pop($parts));
146 if ('addUnstructuredHeader' === $method) {
147 $method = 'addTextHeader';
148 } elseif ('addIdentificationHeader' === $method) {
149 $method = 'addIdHeader';
150 }

163 public function add(HeaderInterface $header): static
164 {
165 self::checkHeaderClass($header);
166
167 $header->setMaxLineLength($this->lineLength);
168 $name = strtolower($header->getName());
169
170 if (\in_array($name, self::UNIQUE_HEADERS, true) && isset($this->headers[$name]) && \count($this->headers[$name]) > 0) {
171 throw new LogicException(sprintf('Impossible to set header "%s" as it\'s already defined and must be unique.', $header->getName()));
172 }
173
174 $this->headers[$name][] = $header;

176 return $this;

224 public static function checkHeaderClass(HeaderInterface $header): void
225 {
226 $name = strtolower($header->getName());
227 $headerClasses = self::HEADER_CLASS_MAP[$name] ?? [];
228 if (!\is_array($headerClasses)) {
229 $headerClasses = [$headerClasses];
230 }
231
232 if (!$headerClasses) {
233 return;
234 }
235
236 foreach ($headerClasses as $c) {
237 if ($header instanceof $c) {
238 return;
239 }
240 }
241
242 throw new LogicException(sprintf('The "%s" header must be an instance of "%s" (got "%s").', $header->getName(), implode('" or "', $headerClasses), get_debug_type($header)));
243 }
