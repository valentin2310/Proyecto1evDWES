120 public function generate(string $name, array $parameters = [], int $referenceType = self::ABSOLUTE_PATH): string
121 {
122 $route = null;
123 $locale = $parameters['_locale'] ?? $this->context->getParameter('_locale') ?: $this->defaultLocale;
124
125 if (null !== $locale) {
126 do {
127 if (null !== ($route = $this->routes->get($name.'.'.$locale)) && $route->getDefault('_canonical_route') === $name) {
128 break;
129 }
130 } while (false !== $locale = strstr($locale, '_', true));
131 }
132
133 if (null === $route ??= $this->routes->get($name)) {
134 throw new RouteNotFoundException(sprintf('Unable to generate a URL for the named route "%s" as such route does not exist.', $name));
135 }
136
137 // the Route has a cache of its own and is not recompiled as long as it does not get modified
138 $compiledRoute = $route->compile();
139
140 $defaults = $route->getDefaults();
141 $variables = $compiledRoute->getVariables();
142
143 if (isset($defaults['_canonical_route']) && isset($defaults['_locale'])) {
144 if (!\in_array('_locale', $variables, true)) {
145 unset($parameters['_locale']);
146 } elseif (!isset($parameters['_locale'])) {
147 $parameters['_locale'] = $defaults['_locale'];
148 }
149 }
150
151 return $this->doGenerate($variables, $defaults, $route->getRequirements(), $compiledRoute->getTokens(), $parameters, $name, $referenceType, $compiledRoute->getHostTokens(), $route->getSchemes());
152 }

159 protected function doGenerate(array $variables, array $defaults, array $requirements, array $tokens, array $parameters, string $name, int $referenceType, array $hostTokens, array $requiredSchemes = []): string
160 {
161 $variables = array_flip($variables);
162 $mergedParams = array_replace($defaults, $this->context->getParameters(), $parameters);
163
164 // all params must be given
165 if ($diff = array_diff_key($variables, $mergedParams)) {
166 throw new MissingMandatoryParametersException($name, array_keys($diff));
167 }
168
169 $url = '';
170 $optional = true;
171 $message = 'Parameter "{parameter}" for route "{route}" must match "{expected}" ("{given}" given) to generate a corresponding URL.';
172 foreach ($tokens as $token) {
173 if ('variable' === $token[0]) {
174 $varName = $token[3];
175 // variable is not important by default
176 $important = $token[5] ?? false;
177
178 if (!$optional || $important || !\array_key_exists($varName, $defaults) || (null !== $mergedParams[$varName] && (string) $mergedParams[$varName] !== (string) $defaults[$varName])) {
179 // check requirement (while ignoring look-around patterns)
180 if (null !== $this->strictRequirements && !preg_match('#^'.preg_replace('/\‍(\?(?:=|<=|!|<!)((?:[^()\\\\]+|\\\\.|\‍((?1)\‍))*)\‍)/', '', $token[2]).'$#i'.(empty($token[4]) ? '' : 'u'), $mergedParams[$token[3]] ?? '')) {
181 if ($this->strictRequirements) {
182 throw new InvalidParameterException(strtr($message, ['{parameter}' => $varName, '{route}' => $name, '{expected}' => $token[2], '{given}' => $mergedParams[$varName]]));
183 }
184
185 $this->logger?->error($message, ['parameter' => $varName, 'route' => $name, 'expected' => $token[2], 'given' => $mergedParams[$varName]]);
186
187 return '';
188 }
189
190 $url = $token[1].$mergedParams[$varName].$url;
191 $optional = false;
192 }
193 } else {
194 // static text
195 $url = $token[1].$url;
196 $optional = false;
197 }
198 }
199
200 if ('' === $url) {
201 $url = '/';
202 }
203
204 // the contexts base URL is already encoded (see Symfony\Component\HttpFoundation\Request)
205 $url = strtr(rawurlencode($url), $this->decodedChars);
206
207 // the path segments "." and ".." are interpreted as relative reference when resolving a URI; see http://tools.ietf.org/html/rfc3986#section-3.3
208 // so we need to encode them as they are not used for this purpose here
209 // otherwise we would generate a URI that, when followed by a user agent (e.g. browser), does not match this route
210 $url = strtr($url, ['/../' => '/%2E%2E/', '/./' => '/%2E/']);
211 if (str_ends_with($url, '/..')) {
212 $url = substr($url, 0, -2).'%2E%2E';
213 } elseif (str_ends_with($url, '/.')) {
214 $url = substr($url, 0, -1).'%2E';
215 }
216
217 $schemeAuthority = '';
218 $host = $this->context->getHost();
219 $scheme = $this->context->getScheme();
220
221 if ($requiredSchemes) {
222 if (!\in_array($scheme, $requiredSchemes, true)) {
223 $referenceType = self::ABSOLUTE_URL;
224 $scheme = current($requiredSchemes);
225 }
226 }
227
228 if ($hostTokens) {
229 $routeHost = '';
230 foreach ($hostTokens as $token) {
231 if ('variable' === $token[0]) {
232 // check requirement (while ignoring look-around patterns)
233 if (null !== $this->strictRequirements && !preg_match('#^'.preg_replace('/\‍(\?(?:=|<=|!|<!)((?:[^()\\\\]+|\\\\.|\‍((?1)\‍))*)\‍)/', '', $token[2]).'$#i'.(empty($token[4]) ? '' : 'u'), $mergedParams[$token[3]])) {
234 if ($this->strictRequirements) {
235 throw new InvalidParameterException(strtr($message, ['{parameter}' => $token[3], '{route}' => $name, '{expected}' => $token[2], '{given}' => $mergedParams[$token[3]]]));
236 }
237
238 $this->logger?->error($message, ['parameter' => $token[3], 'route' => $name, 'expected' => $token[2], 'given' => $mergedParams[$token[3]]]);
239
240 return '';
241 }
242
243 $routeHost = $token[1].$mergedParams[$token[3]].$routeHost;
244 } else {
245 $routeHost = $token[1].$routeHost;
246 }
247 }
248
249 if ($routeHost !== $host) {
250 $host = $routeHost;
251 if (self::ABSOLUTE_URL !== $referenceType) {
252 $referenceType = self::NETWORK_PATH;
253 }
254 }
255 }
256
257 if (self::ABSOLUTE_URL === $referenceType || self::NETWORK_PATH === $referenceType) {
258 if ('' !== $host || ('' !== $scheme && 'http' !== $scheme && 'https' !== $scheme)) {
259 $port = '';
260 if ('http' === $scheme && 80 !== $this->context->getHttpPort()) {
261 $port = ':'.$this->context->getHttpPort();
262 } elseif ('https' === $scheme && 443 !== $this->context->getHttpsPort()) {
263 $port = ':'.$this->context->getHttpsPort();
264 }
265
266 $schemeAuthority = self::NETWORK_PATH === $referenceType || '' === $scheme ? '//' : "$scheme://";
267 $schemeAuthority .= $host.$port;
268 }
269 }
270
271 if (self::RELATIVE_PATH === $referenceType) {
272 $url = self::getRelativePath($this->context->getPathInfo(), $url);
273 } else {
274 $url = $schemeAuthority.$this->context->getBaseUrl().$url;
275 }
276
277 // add a query string if needed
278 $extra = array_udiff_assoc(array_diff_key($parameters, $variables), $defaults, fn ($a, $b) => $a == $b ? 0 : 1);
279
280 array_walk_recursive($extra, $caster = static function (&$v) use (&$caster) {
281 if (\is_object($v)) {
282 if ($vars = get_object_vars($v)) {
283 array_walk_recursive($vars, $caster);
284 $v = $vars;
285 } elseif (method_exists($v, '__toString')) {
286 $v = (string) $v;
287 }
288 }
289 });
290
291 // extract fragment
292 $fragment = $defaults['_fragment'] ?? '';
293
294 if (isset($extra['_fragment'])) {
295 $fragment = $extra['_fragment'];
296 unset($extra['_fragment']);
297 }
298
299 if ($extra && $query = http_build_query($extra, '', '&', \PHP_QUERY_RFC3986)) {
300 $url .= '?'.strtr($query, self::QUERY_FRAGMENT_DECODED);
301 }
302
303 if ('' !== $fragment) {
304 $url .= '#'.strtr(rawurlencode($fragment), self::QUERY_FRAGMENT_DECODED);
305 }
306
307 return $url;
308 }

328 public static function getRelativePath(string $basePath, string $targetPath): string
329 {
330 if ($basePath === $targetPath) {
331 return '';
332 }
333
334 $sourceDirs = explode('/', isset($basePath[0]) && '/' === $basePath[0] ? substr($basePath, 1) : $basePath);
335 $targetDirs = explode('/', isset($targetPath[0]) && '/' === $targetPath[0] ? substr($targetPath, 1) : $targetPath);
336 array_pop($sourceDirs);
337 $targetFile = array_pop($targetDirs);
338
339 foreach ($sourceDirs as $i => $dir) {
340 if (isset($targetDirs[$i]) && $dir === $targetDirs[$i]) {
341 unset($sourceDirs[$i], $targetDirs[$i]);
342 } else {
343 break;
344 }
345 }
346
347 $targetDirs[] = $targetFile;
348 $path = str_repeat('../', \count($sourceDirs)).implode('/', $targetDirs);
349
350 // A reference to the same base directory or an empty subdirectory must be prefixed with "./".
351 // This also applies to a segment with a colon character (e.g., "file:colon") that cannot be used
352 // as the first segment of a relative-path reference, as it would be mistaken for a scheme name
353 // (see http://tools.ietf.org/html/rfc3986#section-4.2).
354 return '' === $path || '/' === $path[0]
355 || false !== ($colonPos = strpos($path, ':')) && ($colonPos < ($slashPos = strpos($path, '/')) || false === $slashPos)
356 ? "./$path" : $path;
357 }
