91 protected function addForeignKeys(Blueprint $blueprint)
92 {
93 $foreigns = $this->getCommandsByName($blueprint, 'foreign');
94
95 return collect($foreigns)->reduce(function ($sql, $foreign) {
96 // Once we have all the foreign key commands for the table creation statement
97 // we'll loop through each of them and add them to the create table SQL we
98 // are building, since SQLite needs foreign keys on the tables creation.
99 $sql .= $this->getForeignKey($foreign);
100
101 if (! is_null($foreign->onDelete)) {
102 $sql .= " on delete {$foreign->onDelete}";
103 }
104
105 // If this foreign key specifies the action to be taken on update we will add
106 // that to the statement here. We'll append it to this SQL and then return
107 // the SQL so we can keep adding any other foreign constraints onto this.
108 if (! is_null($foreign->onUpdate)) {
109 $sql .= " on update {$foreign->onUpdate}";
110 }
111
112 return $sql;
113 }, '');
114 }

414 public function compileRenameIndex(Blueprint $blueprint, Fluent $command, Connection $connection)
415 {
416 $schemaManager = $connection->getDoctrineSchemaManager();
417
418 $indexes = $schemaManager->listTableIndexes($this->getTablePrefix().$blueprint->getTable());
419
420 $index = Arr::get($indexes, $command->from);
421
422 if (! $index) {
423 throw new RuntimeException("Index [{$command->from}] does not exist.");
424 }
425
426 $newIndex = new Index(
427 $command->to, $index->getColumns(), $index->isUnique(),
428 $index->isPrimary(), $index->getFlags(), $index->getOptions()
429 );
430
431 $platform = $connection->getDoctrineConnection()->getDatabasePlatform();
432
433 return [
434 $platform->getDropIndexSQL($command->from, $this->getTablePrefix().$blueprint->getTable()),
435 $platform->getCreateIndexSQL($newIndex, $this->getTablePrefix().$blueprint->getTable()),
