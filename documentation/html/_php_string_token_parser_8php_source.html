57 protected static $replacements = [
58 '\\' => '\\',
59 '$' => '$',
60 'n' => "\n",
61 'r' => "\r",
62 't' => "\t",
63 'f' => "\f",
64 'v' => "\v",
65 'e' => "\x1B",
66 ];

73 public static function parse(string $str): string
74 {
75 $bLength = 0;
76 if ('b' === $str[0]) {
77 $bLength = 1;
78 }
79
80 if ('\'' === $str[$bLength]) {
81 return str_replace(
82 ['\\\\', '\\\''],
83 ['\\', '\''],
84 substr($str, $bLength + 1, -1)
85 );
86 } else {
87 return self::parseEscapeSequences(substr($str, $bLength + 1, -1), '"');
88 }
89 }
90
97 public static function parseEscapeSequences(string $str, string $quote = null): string
98 {
99 if (null !== $quote) {
100 $str = str_replace('\\'.$quote, $quote, $str);
101 }
102
103 return preg_replace_callback(
104 '~\\\\([\\\\$nrtfve]|[xX][0-9a-fA-F]{1,2}|[0-7]{1,3})~',
105 [__CLASS__, 'parseCallback'],
106 $str
107 );
108 }
109
110 private static function parseCallback(array $matches): string
111 {
112 $str = $matches[1];
113
114 if (isset(self::$replacements[$str])) {
115 return self::$replacements[$str];
116 } elseif ('x' === $str[0] || 'X' === $str[0]) {
117 return \chr(hexdec($str));
118 } else {
119 return \chr(octdec($str));
120 }
121 }
122
129 public static function parseDocString(string $startToken, string $str): string
130 {
131 // strip last newline (thanks tokenizer for sticking it into the string!)
132 $str = preg_replace('~(\r\n|\n|\r)$~', '', $str);
133
134 // nowdoc string
135 if (str_contains($startToken, '\'')) {
136 return $str;
137 }
138
139 return self::parseEscapeSequences($str, null);
140 }
