154 public function merge(self $newData): void
155 {
156 foreach ($newData->lineCoverage as $file => $lines) {
157 if (!isset($this->lineCoverage[$file])) {
158 $this->lineCoverage[$file] = $lines;
159
160 continue;
161 }
162
163 // we should compare the lines if any of two contains data
164 $compareLineNumbers = array_unique(
165 array_merge(
166 array_keys($this->lineCoverage[$file]),
167 array_keys($newData->lineCoverage[$file])
168 )
169 );
170
171 foreach ($compareLineNumbers as $line) {
172 $thatPriority = $this->priorityForLine($newData->lineCoverage[$file], $line);
173 $thisPriority = $this->priorityForLine($this->lineCoverage[$file], $line);
174
175 if ($thatPriority > $thisPriority) {
176 $this->lineCoverage[$file][$line] = $newData->lineCoverage[$file][$line];
177 } elseif ($thatPriority === $thisPriority && is_array($this->lineCoverage[$file][$line])) {
178 $this->lineCoverage[$file][$line] = array_unique(
179 array_merge($this->lineCoverage[$file][$line], $newData->lineCoverage[$file][$line])
180 );
181 }
182 }
183 }
184
185 foreach ($newData->functionCoverage as $file => $functions) {
186 if (!isset($this->functionCoverage[$file])) {
187 $this->functionCoverage[$file] = $functions;
188
189 continue;
190 }
191
192 foreach ($functions as $functionName => $functionData) {
193 if (isset($this->functionCoverage[$file][$functionName])) {
194 $this->initPreviouslySeenFunction($file, $functionName, $functionData);
195 } else {
196 $this->initPreviouslyUnseenFunction($file, $functionName, $functionData);
197 }
198
199 foreach ($functionData['branches'] as $branchId => $branchData) {
200 $this->functionCoverage[$file][$functionName]['branches'][$branchId]['hit'] = array_unique(array_merge($this->functionCoverage[$file][$functionName]['branches'][$branchId]['hit'], $branchData['hit']));
201 }
202
203 foreach ($functionData['paths'] as $pathId => $pathData) {

220 private function priorityForLine(array $data, int $line): int
221 {
222 if (!array_key_exists($line, $data)) {
223 return 1;
224 }
225
226 if (is_array($data[$line]) && count($data[$line]) === 0) {
227 return 2;
228 }
229
230 if ($data[$line] === null) {
231 return 3;
232 }
233
234 return 4;
