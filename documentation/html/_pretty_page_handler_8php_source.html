561 protected function getEditor($filePath, $line)
562 {
563 if (!$this->editor || (!is_string($this->editor) && !is_callable($this->editor))) {
564 return [];
565 }
566
567 if (is_string($this->editor) && isset($this->editors[$this->editor]) && !is_callable($this->editors[$this->editor])) {
568 return [
569 'ajax' => false,
570 'url' => $this->editors[$this->editor],
571 ];
572 }
573
574 if (is_callable($this->editor) || (isset($this->editors[$this->editor]) && is_callable($this->editors[$this->editor]))) {
575 if (is_callable($this->editor)) {
576 $callback = call_user_func($this->editor, $filePath, $line);
577 } else {
578 $callback = call_user_func($this->editors[$this->editor], $filePath, $line);
579 }
580
581 if (empty($callback)) {
582 return [];
583 }
584
585 if (is_string($callback)) {
586 return [
587 'ajax' => false,
588 'url' => $callback,
589 ];
590 }
591
592 return [
593 'ajax' => isset($callback['ajax']) ? $callback['ajax'] : false,
594 'url' => isset($callback['url']) ? $callback['url'] : $callback,
595 ];
596 }

692 protected function getResource($resource)
693 {
694 // If the resource was found before, we can speed things up
695 // by caching its absolute, resolved path:
696 if (isset($this->resourceCache[$resource])) {
697 return $this->resourceCache[$resource];
698 }
699
700 // Search through available search paths, until we find the
701 // resource we're after:
702 foreach ($this->searchPaths as $path) {
703 $fullPath = $path . "/$resource";
704
705 if (is_file($fullPath)) {
706 // Cache the result:
707 $this->resourceCache[$resource] = $fullPath;
708 return $fullPath;
709 }
710 }
711
712 // If we got this far, nothing was found.
713 throw new RuntimeException(
714 "Could not find resource '$resource' in any resource paths."
715 . "(searched: " . join(", ", $this->searchPaths). ")"
716 );
717 }
