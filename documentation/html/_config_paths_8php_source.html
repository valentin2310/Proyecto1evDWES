326 private function allDirNames(array $baseDirs): array
327 {
328 $dirs = \array_map(function ($dir) {
329 return \strtr($dir, '\\', '/').'/psysh';
330 }, $baseDirs);
331
332 // Add ~/.psysh
333 if ($home = $this->getEnv('HOME')) {
334 $dirs[] = \strtr($home, '\\', '/').'/.psysh';
335 }
336
337 // Add some Windows specific ones :)
338 if (\defined('PHP_WINDOWS_VERSION_MAJOR')) {
339 if ($appData = $this->getEnv('APPDATA')) {
340 // AppData gets preference
341 \array_unshift($dirs, \strtr($appData, '\\', '/').'/PsySH');
342 }
343
344 if ($windowsHomeDir = $this->windowsHomeDir()) {
345 $dir = \strtr($windowsHomeDir, '\\', '/').'/.psysh';
346 if (!\in_array($dir, $dirs)) {
347 $dirs[] = $dir;
348 }
349 }
350 }
351
352 return $dirs;
353 }
