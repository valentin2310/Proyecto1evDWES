210 public function to($path, $extra = [], $secure = null)
211 {
212 // First we will check if the URL is already a valid URL. If it is we will not
213 // try to generate a new one but will simply return the URL as is, which is
214 // convenient since developers do not always have to check if it's valid.
215 if ($this->isValidUrl($path)) {
216 return $path;
217 }
218
219 $tail = implode('/', array_map(
220 'rawurlencode', (array) $this->formatParameters($extra))
221 );
222
223 // Once we have the scheme we will compile the "tail" by collapsing the values
224 // into a single string delimited by slashes. This just makes it convenient
225 // for passing the array of parameters to this URL as a list of segments.
226 $root = $this->formatRoot($this->formatScheme($secure));
227
228 [$path, $query] = $this->extractQueryString($path);
229
230 return $this->format(
231 $root, '/'.trim($path.'/'.$tail, '/')
232 ).$query;
233 }

428 public function hasCorrectSignature(Request $request, $absolute = true, array $ignoreQuery = [])
429 {
430 $ignoreQuery[] = 'signature';
431
432 $url = $absolute ? $request->url() : '/'.$request->path();
433
434 $queryString = collect(explode('&', (string) $request->server->get('QUERY_STRING')))
435 ->reject(fn ($parameter) => in_array(Str::before($parameter, '='), $ignoreQuery))
436 ->join('&');
437
438 $original = rtrim($url.'?'.$queryString, '?');
439
440 $signature = hash_hmac('sha256', $original, call_user_func($this->keyResolver));
441
442 return hash_equals($signature, (string) $request->query('signature', ''));
443 }
