424 public function compileUpdateFrom(Builder $query, $values)
425 {
426 $table = $this->wrapTable($query->from);
427
428 // Each one of the columns in the update statements needs to be wrapped in the
429 // keyword identifiers, also a place-holder needs to be created for each of
430 // the values in the list of bindings so we can make the sets statements.
431 $columns = $this->compileUpdateColumns($query, $values);
432
433 $from = '';
434
435 if (isset($query->joins)) {
436 // When using Postgres, updates with joins list the joined tables in the from
437 // clause, which is different than other systems like MySQL. Here, we will
438 // compile out the tables that are joined and add them to a from clause.
439 $froms = collect($query->joins)->map(function ($join) {
440 return $this->wrapTable($join->table);
441 })->all();
442
443 if (count($froms) > 0) {
444 $from = ' from '.implode(', ', $froms);
445 }
446 }
447
448 $where = $this->compileUpdateWheres($query);
449
450 return trim("update {$table} set {$columns}{$from} {$where}");
451 }

459 protected function compileUpdateWheres(Builder $query)
460 {
461 $baseWheres = $this->compileWheres($query);
462
463 if (! isset($query->joins)) {
464 return $baseWheres;
465 }
466
467 // Once we compile the join constraints, we will either use them as the where
468 // clause or append them to the existing base where clauses. If we need to
469 // strip the leading boolean we will do so when using as the only where.
470 $joinWheres = $this->compileUpdateJoinWheres($query);
471
472 if (trim($baseWheres) == '') {
473 return 'where '.$this->removeLeadingBoolean($joinWheres);
474 }
475
476 return $baseWheres.' '.$joinWheres;

485 protected function compileUpdateJoinWheres(Builder $query)
486 {
487 $joinWheres = [];
488
489 // Here we will just loop through all of the join constraints and compile them
490 // all out then implode them. This should give us "where" like syntax after
491 // everything has been built and then we will join it to the real wheres.
492 foreach ($query->joins as $join) {
493 foreach ($join->wheres as $where) {
494 $method = "where{$where['type']}";

497 }
498 }
