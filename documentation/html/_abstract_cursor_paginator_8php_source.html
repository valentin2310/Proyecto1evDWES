102 public function url($cursor)
103 {
104 // If we have any extra query string key / value pairs that need to be added
105 // onto the URL, we will put them in query string form and then attach it
106 // to the URL. This allows for extra information like sortings storage.
107 $parameters = is_null($cursor) ? [] : [$this->cursorName => $cursor->encode()];
108
109 if (count($this->query) > 0) {
110 $parameters = array_merge($this->query, $parameters);
111 }
112
113 return $this->path()
114 .(str_contains($this->path(), '?') ? '&' : '?')
115 .Arr::query($parameters)
116 .$this->buildFragment();

205 public function getParametersForItem($item)
206 {
207 return collect($this->parameters)
208 ->flip()
209 ->map(function ($_, $parameterName) use ($item) {
210 if ($item instanceof JsonResource) {
211 $item = $item->resource;
212 }
213
214 if ($item instanceof Model &&
215 ! is_null($parameter = $this->getPivotParameterForItem($item, $parameterName))) {
216 return $parameter;
217 } elseif ($item instanceof ArrayAccess || is_array($item)) {
218 return $this->ensureParameterIsPrimitive(
219 $item[$parameterName] ?? $item[Str::afterLast($parameterName, '.')]
220 );
221 } elseif (is_object($item)) {
222 return $this->ensureParameterIsPrimitive(
223 $item->{$parameterName} ?? $item->{Str::afterLast($parameterName, '.')}
224 );
225 }
226
227 throw new Exception('Only arrays and objects are supported when cursor paginating items.');
228 })->toArray();
229 }
