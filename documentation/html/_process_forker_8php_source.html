107 public function beforeRun(Shell $shell)
108 {
109 list($up, $down) = \stream_socket_pair(\STREAM_PF_UNIX, \STREAM_SOCK_STREAM, \STREAM_IPPROTO_IP);
110
111 if (!$up) {
112 throw new \RuntimeException('Unable to create socket pair');
113 }
114
115 $pid = \pcntl_fork();
116 if ($pid < 0) {
117 throw new \RuntimeException('Unable to start execution loop');
118 } elseif ($pid > 0) {
119 // This is the main thread. We'll just wait for a while.
120
121 // We won't be needing this one.
122 \fclose($up);
123
124 // Wait for a return value from the loop process.
125 $read = [$down];
126 $write = null;
127 $except = null;
128
129 do {
130 $n = @\stream_select($read, $write, $except, null);
131
132 if ($n === 0) {
133 throw new \RuntimeException('Process timed out waiting for execution loop');
134 }
135
136 if ($n === false) {
137 $err = \error_get_last();
138 if (!isset($err['message']) || \stripos($err['message'], 'interrupted system call') === false) {
139 $msg = $err['message'] ?
140 \sprintf('Error waiting for execution loop: %s', $err['message']) :
141 'Error waiting for execution loop';
142 throw new \RuntimeException($msg);
143 }
144 }
145 } while ($n < 1);
146
147 $content = \stream_get_contents($down);
148 \fclose($down);
149
150 if ($content) {
151 $shell->setScopeVariables(@\unserialize($content));
152 }
153
154 throw new BreakException('Exiting main thread');
155 }
156
157 // This is the child process. It's going to do all the work.
158 if (!@\cli_set_process_title('psysh (loop)')) {
159 // Fall back to `setproctitle` if that wasn't succesful.
160 if (\function_exists('setproctitle')) {
161 @\setproctitle('psysh (loop)');
162 }
163 }
164
165 // We won't be needing this one.
166 \fclose($down);
167
168 // Save this; we'll need to close it in `afterRun`
169 $this->up = $up;
170 }

220 private function createSavegame()
221 {
222 // the current process will become the savegame
223 $this->savegame = \posix_getpid();
224
225 $pid = \pcntl_fork();
226 if ($pid < 0) {
227 throw new \RuntimeException('Unable to create savegame fork');
228 } elseif ($pid > 0) {
229 // we're the savegame now... let's wait and see what happens
230 \pcntl_waitpid($pid, $status);
231
232 // worker exited cleanly, let's bail
233 if (!\pcntl_wexitstatus($status)) {
234 \posix_kill(\posix_getpid(), \SIGKILL);
235 }
236
237 // worker didn't exit cleanly, we'll need to have another go
238 $this->createSavegame();
239 }

252 private function serializeReturn(array $return): string
253 {
254 $serializable = [];
255
256 foreach ($return as $key => $value) {
257 // No need to return magic variables
258 if (Context::isSpecialVariableName($key)) {
259 continue;
260 }
261
262 // Resources and Closures don't error, but they don't serialize well either.
263 if (\is_resource($value) || $value instanceof \Closure) {
264 continue;
265 }
266
267 if (\version_compare(\PHP_VERSION, '8.1', '>=') && $value instanceof \UnitEnum) {
268 // Enums defined in the REPL session can't be unserialized.
269 $ref = new \ReflectionObject($value);
270 if (\strpos($ref->getFileName(), ": eval()'d code") !== false) {
271 continue;
272 }
273 }
274
275 try {
276 @\serialize($value);
277 $serializable[$key] = $value;
278 } catch (\Throwable $e) {
279 // we'll just ignore this one...
280 }
281 }
282
283 return @\serialize($serializable);
284 }
