601 // an array. Then we determine if the given rule accepts other field names as parameters.
602 // If so, we will replace any asterisks found in the parameters with the correct keys.
603 if ($this->dependsOnOtherFields($rule)) {

865 protected function shouldStopValidating($attribute)
866 {
867 $cleanedAttribute = $this->replacePlaceholderInString($attribute);
868
869 if ($this->hasRule($attribute, ['Bail'])) {
870 return $this->messages->has($cleanedAttribute);
871 }

874 array_key_exists('uploaded', $this->failedRules[$cleanedAttribute])) {
875 return true;
