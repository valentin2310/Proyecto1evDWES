384 public function handleError(int $type, string $message, string $file, int $line): bool
385 {
386 if (\E_WARNING === $type && '"' === $message[0] && str_contains($message, '" targeting switch is equivalent to "break')) {
387 $type = \E_DEPRECATED;
388 }
389
390 // Level is the current error reporting level to manage silent error.
391 $level = error_reporting();
392 $silenced = 0 === ($level & $type);
393 // Strong errors are not authorized to be silenced.
394 $level |= \E_RECOVERABLE_ERROR | \E_USER_ERROR | \E_DEPRECATED | \E_USER_DEPRECATED;
395 $log = $this->loggedErrors & $type;
396 $throw = $this->thrownErrors & $type & $level;
397 $type &= $level | $this->screamedErrors;
398
399 // Never throw on warnings triggered by assert()
400 if (\E_WARNING === $type && 'a' === $message[0] && 0 === strncmp($message, 'assert(): ', 10)) {
401 $throw = 0;
402 }
403
404 if (!$type || (!$log && !$throw)) {
405 return false;
406 }
407
408 $logMessage = $this->levels[$type].': '.$message;
409
410 if (!$throw && !($type & $level)) {
411 if (!isset(self::$silencedErrorCache[$id = $file.':'.$line])) {
412 $lightTrace = $this->tracedErrors & $type ? $this->cleanTrace(debug_backtrace(\DEBUG_BACKTRACE_IGNORE_ARGS, 5), $type, $file, $line, false) : [];
413 $errorAsException = new SilencedErrorContext($type, $file, $line, isset($lightTrace[1]) ? [$lightTrace[0]] : $lightTrace);
414 } elseif (isset(self::$silencedErrorCache[$id][$message])) {
415 $lightTrace = null;
416 $errorAsException = self::$silencedErrorCache[$id][$message];
417 ++$errorAsException->count;
418 } else {
419 $lightTrace = [];
420 $errorAsException = null;
421 }
422
423 if (100 < ++self::$silencedErrorCount) {
424 self::$silencedErrorCache = $lightTrace = [];
425 self::$silencedErrorCount = 1;
426 }
427 if ($errorAsException) {
428 self::$silencedErrorCache[$id][$message] = $errorAsException;
429 }
430 if (null === $lightTrace) {
431 return true;
432 }
433 } else {
434 if (str_contains($message, '@anonymous')) {
435 $backtrace = debug_backtrace(false, 5);
436
437 for ($i = 1; isset($backtrace[$i]); ++$i) {
438 if (isset($backtrace[$i]['function'], $backtrace[$i]['args'][0])
439 && ('trigger_error' === $backtrace[$i]['function'] || 'user_error' === $backtrace[$i]['function'])
440 ) {
441 if ($backtrace[$i]['args'][0] !== $message) {
442 $message = $this->parseAnonymousClass($backtrace[$i]['args'][0]);
443 $logMessage = $this->levels[$type].': '.$message;
444 }
445
446 break;
447 }
448 }
449 }
450
451 $errorAsException = new \ErrorException($logMessage, 0, $type, $file, $line);
452
453 if ($throw || $this->tracedErrors & $type) {
454 $backtrace = $errorAsException->getTrace();
455 $backtrace = $this->cleanTrace($backtrace, $type, $file, $line, $throw);
456 ($this->configureException)($errorAsException, $backtrace, $file, $line);
457 } else {
458 ($this->configureException)($errorAsException, []);
459 }
460 }
461
462 if ($throw) {
463 throw $errorAsException;
464 }
465
466 if ($this->isRecursive) {
467 $log = 0;
468 } else {
469 try {
470 $this->isRecursive = true;
471 $level = ($type & $level) ? $this->loggers[$type][1] : LogLevel::DEBUG;
472 $this->loggers[$type][0]->log($level, $logMessage, $errorAsException ? ['exception' => $errorAsException] : []);
473 } finally {
474 $this->isRecursive = false;
475 }
476 }
477
478 return !$silenced && $type && $log;
479 }

486 public function handleException(\Throwable $exception): void
487 {
488 $handlerException = null;
489
490 if (!$exception instanceof FatalError) {
491 self::$exitCode = 255;
492
493 $type = ThrowableUtils::getSeverity($exception);
494 } else {
495 $type = $exception->getError()['type'];
496 }
497
498 if ($this->loggedErrors & $type) {
499 if (str_contains($message = $exception->getMessage(), "@anonymous\0")) {
500 $message = $this->parseAnonymousClass($message);
501 }
502
503 if ($exception instanceof FatalError) {
504 $message = 'Fatal '.$message;
505 } elseif ($exception instanceof \Error) {
506 $message = 'Uncaught Error: '.$message;
507 } elseif ($exception instanceof \ErrorException) {
508 $message = 'Uncaught '.$message;
509 } else {
510 $message = 'Uncaught Exception: '.$message;
511 }
512
513 try {
514 $this->loggers[$type][0]->log($this->loggers[$type][1], $message, ['exception' => $exception]);
515 } catch (\Throwable $handlerException) {
516 }
517 }
518
519 if (!$exception instanceof OutOfMemoryError) {
520 foreach ($this->getErrorEnhancers() as $errorEnhancer) {
521 if ($e = $errorEnhancer->enhance($exception)) {
522 $exception = $e;
523 break;
524 }
525 }
526 }
527
528 $exceptionHandler = $this->exceptionHandler;
529 $this->exceptionHandler = [$this, 'renderException'];
530
531 if (null === $exceptionHandler || $exceptionHandler === $this->exceptionHandler) {
532 $this->exceptionHandler = null;
533 }
534
535 try {
536 if (null !== $exceptionHandler) {
537 $exceptionHandler($exception);
538
539 return;
540 }
541 $handlerException ??= $exception;
542 } catch (\Throwable $handlerException) {
543 }
544 if ($exception === $handlerException && null === $this->exceptionHandler) {
545 self::$reservedMemory = null; // Disable the fatal error handler
546 throw $exception; // Give back $exception to the native handler
547 }
548
549 $loggedErrors = $this->loggedErrors;
550 if ($exception === $handlerException) {
551 $this->loggedErrors &= ~$type;
552 }
553
554 try {
555 $this->handleException($handlerException);
556 } finally {
557 $this->loggedErrors = $loggedErrors;
558 }

568 public static function handleFatalError(array $error = null): void
569 {
570 if (null === self::$reservedMemory) {
571 return;
572 }
573
574 $handler = self::$reservedMemory = null;
575 $handlers = [];
576 $previousHandler = null;
577 $sameHandlerLimit = 10;
578
579 while (!\is_array($handler) || !$handler[0] instanceof self) {
580 $handler = set_exception_handler('is_int');
581 restore_exception_handler();
582
583 if (!$handler) {
584 break;
585 }
586 restore_exception_handler();
587
588 if ($handler !== $previousHandler) {
589 array_unshift($handlers, $handler);
590 $previousHandler = $handler;
591 } elseif (0 === --$sameHandlerLimit) {
592 $handler = null;
593 break;
594 }
595 }
596 foreach ($handlers as $h) {
597 set_exception_handler($h);
598 }
599 if (!$handler) {
600 return;
601 }
602 if ($handler !== $h) {
603 $handler[0]->setExceptionHandler($h);
604 }
605 $handler = $handler[0];
606 $handlers = [];
607
608 if ($exit = null === $error) {
609 $error = error_get_last();
610 }
611
612 if ($error && $error['type'] &= \E_PARSE | \E_ERROR | \E_CORE_ERROR | \E_COMPILE_ERROR) {
613 // Let's not throw anymore but keep logging
614 $handler->throwAt(0, true);
615 $trace = $error['backtrace'] ?? null;
616
617 if (str_starts_with($error['message'], 'Allowed memory') || str_starts_with($error['message'], 'Out of memory')) {
618 $fatalError = new OutOfMemoryError($handler->levels[$error['type']].': '.$error['message'], 0, $error, 2, false, $trace);
619 } else {
620 $fatalError = new FatalError($handler->levels[$error['type']].': '.$error['message'], 0, $error, 2, true, $trace);
621 }
622 } else {
623 $fatalError = null;
624 }
625
626 try {
627 if (null !== $fatalError) {
628 self::$exitCode = 255;
629 $handler->handleException($fatalError);
630 }
631 } catch (FatalError) {
632 // Ignore this re-throw
633 }
634
635 if ($exit && self::$exitCode) {
636 $exitCode = self::$exitCode;
637 register_shutdown_function('register_shutdown_function', function () use ($exitCode) { exit($exitCode); });
638 }
639 }

681 private function cleanTrace(array $backtrace, int $type, string &$file, int &$line, bool $throw): array
682 {
683 $lightTrace = $backtrace;
684
685 for ($i = 0; isset($backtrace[$i]); ++$i) {
686 if (isset($backtrace[$i]['file'], $backtrace[$i]['line']) && $backtrace[$i]['line'] === $line && $backtrace[$i]['file'] === $file) {
687 $lightTrace = \array_slice($lightTrace, 1 + $i);
688 break;
689 }
690 }
691 if (\E_USER_DEPRECATED === $type) {
692 for ($i = 0; isset($lightTrace[$i]); ++$i) {
693 if (!isset($lightTrace[$i]['file'], $lightTrace[$i]['line'], $lightTrace[$i]['function'])) {
694 continue;
695 }
696 if (!isset($lightTrace[$i]['class']) && 'trigger_deprecation' === $lightTrace[$i]['function']) {
697 $file = $lightTrace[$i]['file'];
698 $line = $lightTrace[$i]['line'];
699 $lightTrace = \array_slice($lightTrace, 1 + $i);
700 break;
701 }
702 }
703 }
704 if (class_exists(DebugClassLoader::class, false)) {
705 for ($i = \count($lightTrace) - 2; 0 < $i; --$i) {
706 if (DebugClassLoader::class === ($lightTrace[$i]['class'] ?? null)) {
707 array_splice($lightTrace, --$i, 2);
708 }
709 }
710 }
711 if (!($throw || $this->scopedErrors & $type)) {
712 for ($i = 0; isset($lightTrace[$i]); ++$i) {
713 unset($lightTrace[$i]['args'], $lightTrace[$i]['object']);
714 }
715 }
716
717 return $lightTrace;
