245 private function determineSymfonyState(): string
246 {
247 $now = new \DateTimeImmutable();
248 $eom = \DateTimeImmutable::createFromFormat('d/m/Y', '01/'.Kernel::END_OF_MAINTENANCE)->modify('last day of this month');

251 if ($now > $eol) {

254 $versionState = 'eom';
255 } elseif ('' !== Kernel::EXTRA_VERSION) {
