362 public function setProgress(int $step): void
363 {
364 if ($this->max && $step > $this->max) {
365 $this->max = $step;
366 } elseif ($step < 0) {
367 $step = 0;
368 }
369
370 $redrawFreq = $this->redrawFreq ?? (($this->max ?: 10) / 10);
371 $prevPeriod = (int) ($this->step / $redrawFreq);
372 $currPeriod = (int) ($step / $redrawFreq);
373 $this->step = $step;
374 $this->percent = $this->max ? (float) $this->step / $this->max : 0;
375 $timeInterval = microtime(true) - $this->lastWriteTime;
376
377 // Draw regardless of other limits
378 if ($this->max === $step) {
379 $this->display();
380
381 return;
382 }
383
384 // Throttling
385 if ($timeInterval < $this->minSecondsBetweenRedraws) {
386 return;
387 }
388
389 // Draw each step period, but not too late
390 if ($prevPeriod !== $currPeriod || $timeInterval >= $this->maxSecondsBetweenRedraws) {
391 $this->display();
392 }
393 }

470 private function overwrite(string $message): void
471 {
472 if ($this->previousMessage === $message) {
473 return;
474 }
475
476 $originalMessage = $message;
477
478 if ($this->overwrite) {
479 if (null !== $this->previousMessage) {
480 if ($this->output instanceof ConsoleSectionOutput) {
481 $messageLines = explode("\n", $this->previousMessage);
482 $lineCount = \count($messageLines);
483 foreach ($messageLines as $messageLine) {
484 $messageLineLength = Helper::width(Helper::removeDecoration($this->output->getFormatter(), $messageLine));
485 if ($messageLineLength > $this->terminal->getWidth()) {
486 $lineCount += floor($messageLineLength / $this->terminal->getWidth());
487 }
488 }
489 $this->output->clear($lineCount);
490 } else {
491 $lineCount = substr_count($this->previousMessage, "\n");
492 for ($i = 0; $i < $lineCount; ++$i) {
493 $this->cursor->moveToColumn(1);
494 $this->cursor->clearLine();
495 $this->cursor->moveUp();
496 }
497
498 $this->cursor->moveToColumn(1);
499 $this->cursor->clearLine();
500 }
501 }
502 } elseif ($this->step > 0) {
503 $message = \PHP_EOL.$message;
504 }
505
506 $this->previousMessage = $originalMessage;
507 $this->lastWriteTime = microtime(true);
508
509 $this->output->write($message);
510 ++$this->writeCount;
511 }

513 private function determineBestFormat(): string
514 {
515 return match ($this->output->getVerbosity()) {
516 // OutputInterface::VERBOSITY_QUIET: display is disabled anyway
517 OutputInterface::VERBOSITY_VERBOSE => $this->max ? self::FORMAT_VERBOSE : self::FORMAT_VERBOSE_NOMAX,
518 OutputInterface::VERBOSITY_VERY_VERBOSE => $this->max ? self::FORMAT_VERY_VERBOSE : self::FORMAT_VERY_VERBOSE_NOMAX,
519 OutputInterface::VERBOSITY_DEBUG => $this->max ? self::FORMAT_DEBUG : self::FORMAT_DEBUG_NOMAX,
520 default => $this->max ? self::FORMAT_NORMAL : self::FORMAT_NORMAL_NOMAX,
521 };

524 private static function initPlaceholderFormatters(): array
525 {
526 return [
527 'bar' => function (self $bar, OutputInterface $output) {
528 $completeBars = $bar->getBarOffset();
529 $display = str_repeat($bar->getBarCharacter(), $completeBars);
530 if ($completeBars < $bar->getBarWidth()) {
531 $emptyBars = $bar->getBarWidth() - $completeBars - Helper::length(Helper::removeDecoration($output->getFormatter(), $bar->getProgressCharacter()));
532 $display .= $bar->getProgressCharacter().str_repeat($bar->getEmptyBarCharacter(), $emptyBars);
533 }
534
535 return $display;
536 },
537 'elapsed' => fn (self $bar) => Helper::formatTime(time() - $bar->getStartTime()),
538 'remaining' => function (self $bar) {
539 if (!$bar->getMaxSteps()) {
540 throw new LogicException('Unable to display the remaining time if the maximum number of steps is not set.');
541 }
542
543 return Helper::formatTime($bar->getRemaining());
544 },
545 'estimated' => function (self $bar) {
546 if (!$bar->getMaxSteps()) {
547 throw new LogicException('Unable to display the estimated time if the maximum number of steps is not set.');
548 }
549
550 return Helper::formatTime($bar->getEstimated());
551 },
552 'memory' => fn (self $bar) => Helper::formatMemory(memory_get_usage(true)),
553 'current' => fn (self $bar) => str_pad($bar->getProgress(), $bar->getStepWidth(), ' ', \STR_PAD_LEFT),
554 'max' => fn (self $bar) => $bar->getMaxSteps(),
555 'percent' => fn (self $bar) => floor($bar->getProgressPercent() * 100),
556 ];

576 private function buildLine(): string
577 {
578 \assert(null !== $this->format);
579
580 $regex = "{%([a-z\-_]+)(?:\:([^%]+))?%}i";
581 $callback = function ($matches) {
582 if ($formatter = $this->getPlaceholderFormatter($matches[1])) {
583 $text = $formatter($this, $this->output);
584 } elseif (isset($this->messages[$matches[1]])) {
585 $text = $this->messages[$matches[1]];
586 } else {
587 return $matches[0];
588 }
589
590 if (isset($matches[2])) {
591 $text = sprintf('%'.$matches[2], $text);
592 }
593
594 return $text;
595 };
596 $line = preg_replace_callback($regex, $callback, $this->format);
597
598 // gets string length for each sub line with multiline format
599 $linesLength = array_map(fn ($subLine) => Helper::width(Helper::removeDecoration($this->output->getFormatter(), rtrim($subLine, "\r"))), explode("\n", $line));
600
601 $linesWidth = max($linesLength);
602
603 $terminalWidth = $this->terminal->getWidth();
604 if ($linesWidth <= $terminalWidth) {
605 return $line;
606 }
607
608 $this->setBarWidth($this->barWidth - $linesWidth + $terminalWidth);
609
610 return preg_replace_callback($regex, $callback, $this->format);
611 }
