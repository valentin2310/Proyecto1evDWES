119 protected function rotate(): void
120 {
121 // update filename
122 $this->url = $this->getTimedFilename();
123 $this->nextRotation = $this->getNextRotation();
124
125 // skip GC of old logs if files are unlimited
126 if (0 === $this->maxFiles) {
127 return;
128 }
129
130 $logFiles = glob($this->getGlobPattern());
131 if (false === $logFiles) {
132 // failed to glob
133 return;
134 }
135
136 if ($this->maxFiles >= count($logFiles)) {
137 // no files to remove
138 return;
139 }
140
141 // Sorting the files by name to remove the older ones
142 usort($logFiles, function ($a, $b) {
143 return strcmp($b, $a);
144 });
145
146 foreach (array_slice($logFiles, $this->maxFiles) as $file) {
147 if (is_writable($file)) {
148 // suppress errors here as unlink() might fail if two processes
149 // are cleaning up/rotating at the same time
150 set_error_handler(function (int $errno, string $errstr, string $errfile, int $errline): bool {
151 return false;
152 });
153 unlink($file);
154 restore_error_handler();
155 }
156 }
157
158 $this->mustRotate = false;
159 }
