85 protected function execute(InputInterface $input, OutputInterface $output)
86 {
87 $code = $input->getArgument('code');
88 $num = (int) ($input->getOption('num') ?: 1);
89 $shell = $this->getApplication();
90
91 $instrumentedCode = $this->instrumentCode($code);
92
93 self::$times = [];
94
95 do {
96 $_ = $shell->execute($instrumentedCode);
97 $this->ensureEndMarked();
98 } while (\count(self::$times) < $num);
99
100 $shell->writeReturnValue($_);
101
102 $times = self::$times;
103 self::$times = [];
104
105 if ($num === 1) {
106 $output->writeln(\sprintf(self::RESULT_MSG, $times[0] / 1e+9));
107 } else {
108 $total = \array_sum($times);
109 \rsort($times);
110 $median = $times[\round($num / 2)];
111
112 $output->writeln(\sprintf(self::AVG_RESULT_MSG, ($total / $num) / 1e+9, $median / 1e+9, $total / 1e+9));
113 }
114
115 return 0;
116 }
