231 public function addOption(InputOption $option)
232 {
233 if (isset($this->options[$option->getName()]) && !$option->equals($this->options[$option->getName()])) {
234 throw new LogicException(sprintf('An option named "%s" already exists.', $option->getName()));
235 }
236 if (isset($this->negations[$option->getName()])) {
237 throw new LogicException(sprintf('An option named "%s" already exists.', $option->getName()));
238 }
239
240 if ($option->getShortcut()) {
241 foreach (explode('|', $option->getShortcut()) as $shortcut) {
242 if (isset($this->shortcuts[$shortcut]) && !$option->equals($this->options[$this->shortcuts[$shortcut]])) {
243 throw new LogicException(sprintf('An option with shortcut "%s" already exists.', $shortcut));
244 }
245 }
246 }
247
248 $this->options[$option->getName()] = $option;
249 if ($option->getShortcut()) {
250 foreach (explode('|', $option->getShortcut()) as $shortcut) {
251 $this->shortcuts[$shortcut] = $option->getName();
252 }
253 }
254
255 if ($option->isNegatable()) {
256 $negatedName = 'no-'.$option->getName();
257 if (isset($this->options[$negatedName])) {
258 throw new LogicException(sprintf('An option named "%s" already exists.', $negatedName));
259 }
260 $this->negations[$negatedName] = $option->getName();
261 }

392 }
393 }

396 $elements[] = '[--]';
