102 public function read(int $length): ?string
103 {
104 if ($this->currentPos >= $this->charCount) {
105 return null;
106 }
107 $length = ($this->currentPos + $length > $this->charCount) ? $this->charCount - $this->currentPos : $length;
108 if ($this->fixedWidth > 0) {
109 $len = $length * $this->fixedWidth;
110 $ret = substr($this->data, $this->currentPos * $this->fixedWidth, $len);
111 $this->currentPos += $length;
112 } else {
113 $end = $this->currentPos + $length;
114 $end = $end > $this->charCount ? $this->charCount : $end;
115 $ret = '';
116 $start = 0;
117 if ($this->currentPos > 0) {
118 $start = $this->map['p'][$this->currentPos - 1];
119 }
120 $to = $start;
121 for (; $this->currentPos < $end; ++$this->currentPos) {
122 if (isset($this->map['i'][$this->currentPos])) {
123 $ret .= substr($this->data, $start, $to - $start).'?';
124 $start = $this->map['p'][$this->currentPos];
125 } else {
126 $to = $this->map['p'][$this->currentPos];
127 }
128 }
129 $ret .= substr($this->data, $start, $to - $start);
130 }

167 private function getUtf8CharPositions(string $string, int $startOffset, string &$ignoredChars): int
168 {
169 $strlen = \strlen($string);
170 $charPos = \count($this->map['p']);
171 $foundChars = 0;
172 $invalid = false;
173 for ($i = 0; $i < $strlen; ++$i) {
174 $char = $string[$i];
175 $size = self::UTF8_LENGTH_MAP[$char];
176 if (0 == $size) {
177 /* char is invalid, we must wait for a resync */
178 $invalid = true;
179 continue;
180 }
181
182 if ($invalid) {
183 /* We mark the chars as invalid and start a new char */
184 $this->map['p'][$charPos + $foundChars] = $startOffset + $i;
185 $this->map['i'][$charPos + $foundChars] = true;
186 ++$foundChars;
187 $invalid = false;
188 }
189 if (($i + $size) > $strlen) {
190 $ignoredChars = substr($string, $i);
191 break;
192 }
193 for ($j = 1; $j < $size; ++$j) {
194 $char = $string[$i + $j];
195 if ($char > "\x7F" && $char < "\xC0") {
196 // Valid - continue parsing
197 } else {
198 /* char is invalid, we must wait for a resync */
199 $invalid = true;
200 continue 2;
201 }
202 }
203 /* Ok we got a complete char here */
204 $this->map['p'][$charPos + $foundChars] = $startOffset + $i + $size;
205 $i += $j - 1;
206 ++$foundChars;
207 }
