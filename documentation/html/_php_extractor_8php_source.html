181 private function skipMethodArgument(\Iterator $tokenIterator): void
182 {
183 $openBraces = 0;
184
185 for (; $tokenIterator->valid(); $tokenIterator->next()) {
186 $t = $tokenIterator->current();
187
188 if ('[' === $t[0] || '(' === $t[0]) {
189 ++$openBraces;
190 }
191
192 if (']' === $t[0] || ')' === $t[0]) {
193 --$openBraces;
194 }
195
196 if ((0 === $openBraces && ',' === $t[0]) || (-1 === $openBraces && ')' === $t[0])) {
197 break;
198 }

206 private function getValue(\Iterator $tokenIterator): string
207 {
208 $message = '';
209 $docToken = '';
210 $docPart = '';
211
212 for (; $tokenIterator->valid(); $tokenIterator->next()) {
213 $t = $tokenIterator->current();
214 if ('.' === $t) {
215 // Concatenate with next token
216 continue;
217 }
218 if (!isset($t[1])) {
219 break;
220 }
221
222 switch ($t[0]) {
223 case \T_START_HEREDOC:
224 $docToken = $t[1];
225 break;
226 case \T_ENCAPSED_AND_WHITESPACE:
227 case \T_CONSTANT_ENCAPSED_STRING:
228 if ('' === $docToken) {
229 $message .= PhpStringTokenParser::parse($t[1]);
230 } else {
231 $docPart = $t[1];
232 }
233 break;
234 case \T_END_HEREDOC:
235 if ($indentation = strspn($t[1], ' ')) {
236 $docPartWithLineBreaks = $docPart;
237 $docPart = '';
238
239 foreach (preg_split('~(\r\n|\n|\r)~', $docPartWithLineBreaks, -1, \PREG_SPLIT_DELIM_CAPTURE) as $str) {
240 if (\in_array($str, ["\r\n", "\n", "\r"], true)) {
241 $docPart .= $str;
242 } else {
243 $docPart .= substr($str, $indentation);
244 }
245 }
246 }
247
248 $message .= PhpStringTokenParser::parseDocString($docToken, $docPart);
249 $docToken = '';
250 $docPart = '';
251 break;
252 case \T_WHITESPACE:
253 break;
254 default:
255 break 2;
256 }
257 }
258
259 return $message;
260 }

267 protected function parseTokens(array $tokens, MessageCatalogue $catalog, string $filename)
268 {
269 $tokenIterator = new \ArrayIterator($tokens);
270
271 for ($key = 0; $key < $tokenIterator->count(); ++$key) {
272 foreach ($this->sequences as $sequence) {
273 $message = '';
274 $domain = 'messages';
275 $tokenIterator->seek($key);
276
277 foreach ($sequence as $sequenceKey => $item) {
278 $this->seekToNextRelevantToken($tokenIterator);
279
280 if ($this->normalizeToken($tokenIterator->current()) === $item) {
281 $tokenIterator->next();
282 continue;
283 } elseif (self::MESSAGE_TOKEN === $item) {
284 $message = $this->getValue($tokenIterator);
285
286 if (\count($sequence) === ($sequenceKey + 1)) {
287 break;
288 }
289 } elseif (self::METHOD_ARGUMENTS_TOKEN === $item) {
290 $this->skipMethodArgument($tokenIterator);
291 } elseif (self::DOMAIN_TOKEN === $item) {
292 $domainToken = $this->getValue($tokenIterator);
293 if ('' !== $domainToken) {
294 $domain = $domainToken;
295 }
296
297 break;
298 } else {
299 break;
300 }
301 }
302
303 if ($message) {
304 $catalog->set($message, $this->prefix.$message, $domain);
305 $metadata = $catalog->getMetadata($message, $domain) ?? [];
306 $normalizedFilename = preg_replace('{[\\\\/]+}', '/', $filename);
307 $metadata['sources'][] = $normalizedFilename.':'.$tokens[$key][2];
308 $catalog->setMetadata($message, $metadata, $domain);
309 break;
310 }
311 }
312 }
