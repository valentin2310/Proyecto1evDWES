113 protected function execute(InputInterface $input, OutputInterface $output): int
114 {
115 $provider = $this->providers->get($input->getArgument('provider'));
116
117 if (!$this->enabledLocales) {
118 throw new InvalidArgumentException(sprintf('You must define "framework.enabled_locales" or "framework.translator.providers.%s.locales" config key in order to work with translation providers.', parse_url($provider, \PHP_URL_SCHEME)));
119 }
120
121 $io = new SymfonyStyle($input, $output);
122 $domains = $input->getOption('domains');
123 $locales = $input->getOption('locales');
124 $force = $input->getOption('force');
125 $deleteMissing = $input->getOption('delete-missing');
126
127 if (!$domains && $provider instanceof FilteringProvider) {
128 $domains = $provider->getDomains();
129 }
130
131 // Reading local translations must be done after retrieving the domains from the provider
132 // in order to manage only translations from configured domains
133 $localTranslations = $this->readLocalTranslations($locales, $domains, $this->transPaths);
134
135 if (!$domains) {
136 $domains = $this->getDomainsFromTranslatorBag($localTranslations);
137 }
138
139 if (!$deleteMissing && $force) {
140 $provider->write($localTranslations);
141
142 $io->success(sprintf('All local translations has been sent to "%s" (for "%s" locale(s), and "%s" domain(s)).', parse_url($provider, \PHP_URL_SCHEME), implode(', ', $locales), implode(', ', $domains)));
143
144 return 0;
145 }
146
147 $providerTranslations = $provider->read($domains, $locales);
148
149 if ($deleteMissing) {
150 $provider->delete($providerTranslations->diff($localTranslations));
151
152 $io->success(sprintf('Missing translations on "%s" has been deleted (for "%s" locale(s), and "%s" domain(s)).', parse_url($provider, \PHP_URL_SCHEME), implode(', ', $locales), implode(', ', $domains)));
153
154 // Read provider translations again, after missing translations deletion,
155 // to avoid push freshly deleted translations.
156 $providerTranslations = $provider->read($domains, $locales);
157 }
158
159 $translationsToWrite = $localTranslations->diff($providerTranslations);
160
161 if ($force) {
162 $translationsToWrite->addBag($localTranslations->intersect($providerTranslations));
163 }
164
165 $provider->write($translationsToWrite);
166
167 $io->success(sprintf('%s local translations has been sent to "%s" (for "%s" locale(s), and "%s" domain(s)).', $force ? 'All' : 'New', parse_url($provider, \PHP_URL_SCHEME), implode(', ', $locales), implode(', ', $domains)));
168
169 return 0;
170 }
