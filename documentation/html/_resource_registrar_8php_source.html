81 {

97 // place-holder on the route parameters, which should be the base resources.

226 protected function getResourcePrefix($name)
227 {
228 $segments = explode('/', $name);
229
230 // To get the prefix, we will take all of the name segments and implode them on
231 // a slash. This will generate a proper URI prefix for us. Then we take this
232 // last segment, which will be considered the final resources name we use.
233 $prefix = implode('/', array_slice($segments, 0, -1));
234
235 return [end($segments), $prefix];
236 }

552 public function getResourceUri($resource)
553 {
554 if (! str_contains($resource, '.')) {
555 return $resource;
556 }
557
558 // Once we have built the base URI, we'll remove the parameter holder for this
559 // base resource name so that the individual route adders can suffix these
560 // paths however they need to, as some do not have any parameters at all.
561 $segments = explode('.', $resource);
562
563 $uri = $this->getNestedResourceUri($segments);
564
565 return str_replace('/{'.$this->getResourceWildcard(end($segments)).'}', '', $uri);
566 }

574 protected function getNestedResourceUri(array $segments)
575 {
576 // We will spin through the segments and create a place-holder for each of the
577 // resource segments, as well as the resource itself. Then we should get an
578 // entire string for the resource URI that contains all nested resources.
579 return implode('/', array_map(function ($s) {
580 return $s.'/{'.$this->getResourceWildcard($s).'}';
581 }, $segments));
582 }

645 protected function getResourceRouteName($resource, $method, $options)
646 {
647 $name = $resource;
648
649 // If the names array has been provided to us we will check for an entry in the
650 // array first. We will also check for the specific method within this array
651 // so the names may be specified on a more "granular" level using methods.
652 if (isset($options['names'])) {
653 if (is_string($options['names'])) {
654 $name = $options['names'];
655 } elseif (isset($options['names'][$method])) {
656 return $options['names'][$method];
657 }
658 }
659
660 // If a global prefix has been assigned to all names for this resource, we will
661 // grab that so we can prepend it onto the name when we create this name for
662 // the resource action. Otherwise we'll just use an empty string for here.
663 $prefix = isset($options['as']) ? $options['as'].'.' : '';
664
665 return trim(sprintf('%s%s.%s', $prefix, $name, $method), '.');
666 }
