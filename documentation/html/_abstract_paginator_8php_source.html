168 public function url($page)
169 {
170 if ($page <= 0) {
171 $page = 1;
172 }
173
174 // If we have any extra query string key / value pairs that need to be added
175 // onto the URL, we will put them in query string form and then attach it
176 // to the URL. This allows for extra information like sortings storage.
177 $parameters = [$this->pageName => $page];
178
179 if (count($this->query) > 0) {
180 $parameters = array_merge($this->query, $parameters);
181 }
182
183 return $this->path()
184 .(str_contains($this->path(), '?') ? '&' : '?')
185 .Arr::query($parameters)
186 .$this->buildFragment();
