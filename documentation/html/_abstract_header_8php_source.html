96 protected function createPhrase(HeaderInterface $header, string $string, string $charset, bool $shorten = false): string
97 {
98 // Treat token as exactly what was given
99 $phraseStr = $string;
100
101 // If it's not valid
102 if (!preg_match('/^'.self::PHRASE_PATTERN.'$/D', $phraseStr)) {
103 // .. but it is just ascii text, try escaping some characters
104 // and make it a quoted-string
105 if (preg_match('/^[\x00-\x08\x0B\x0C\x0E-\x7F]*$/D', $phraseStr)) {
106 foreach (['\\', '"'] as $char) {
107 $phraseStr = str_replace($char, '\\'.$char, $phraseStr);
108 }
109 $phraseStr = '"'.$phraseStr.'"';
110 } else {
111 // ... otherwise it needs encoding
112 // Determine space remaining on line if first line
113 if ($shorten) {
114 $usedLength = \strlen($header->getName().': ');
115 } else {
116 $usedLength = 0;
117 }
118 $phraseStr = $this->encodeWords($header, $string, $usedLength);
119 }
120 } elseif (str_contains($phraseStr, '(')) {
121 foreach (['\\', '"'] as $char) {
122 $phraseStr = str_replace($char, '\\'.$char, $phraseStr);
123 }
124 $phraseStr = '"'.$phraseStr.'"';
125 }
126
127 return $phraseStr;
128 }

133 protected function encodeWords(HeaderInterface $header, string $input, int $usedLength = -1): string
134 {
135 $value = '';
136 $tokens = $this->getEncodableWordTokens($input);
137 foreach ($tokens as $token) {
138 // See RFC 2822, Sect 2.2 (really 2.2 ??)
139 if ($this->tokenNeedsEncoding($token)) {
140 // Don't encode starting WSP
141 $firstChar = substr($token, 0, 1);
142 switch ($firstChar) {
143 case ' ':
144 case "\t":
145 $value .= $firstChar;
146 $token = substr($token, 1);
147 }
148
149 if (-1 == $usedLength) {
150 $usedLength = \strlen($header->getName().': ') + \strlen($value);
151 }
152 $value .= $this->getTokenAsEncodedWord($token, $usedLength);
153 } else {
154 $value .= $token;
155 }
156 }
157
158 return $value;
159 }

172 {

182 $encodedToken = '';

262 private function tokensToString(array $tokens): string
263 {
264 $lineCount = 0;
265 $headerLines = [];
266 $headerLines[] = $this->name.': ';
267 $currentLine = &$headerLines[$lineCount++];
268
269 // Build all tokens back into compliant header
270 foreach ($tokens as $i => $token) {
271 // Line longer than specified maximum or token was just a new line
272 if (("\r\n" === $token)
273 || ($i > 0 && \strlen($currentLine.$token) > $this->lineLength)
274 && '' !== $currentLine) {
275 $headerLines[] = '';
276 $currentLine = &$headerLines[$lineCount++];
277 }
278
279 // Append token to the line
280 if ("\r\n" !== $token) {
281 $currentLine .= $token;
282 }
283 }
284
285 // Implode with FWS (RFC 2822, 2.2.3)
286 return implode("\r\n", $headerLines);
287 }
