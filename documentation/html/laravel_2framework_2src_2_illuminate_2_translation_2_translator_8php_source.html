102 public function has($key, $locale = null, $fallback = true)
103 {
104 $locale = $locale ?: $this->locale;
105

109 // Otherwise, we must assume we are handling typical translation file
110 // and check if the returned line is not the same as the given key.
111 if (! is_null($this->loaded['*']['*'][$locale][$key] ?? null)) {
112 return true;
113 }

127 public function get($key, array $replace = [], $locale = null, $fallback = true)
128 {
129 $locale = $locale ?: $this->locale;
130
131 // For JSON translations, there is only one file per locale, so we will simply load
132 // that file and then we will be ready to check the array for the key. These are
133 // only one level deep so we do not need to do any fancy searching through it.
134 $this->load('*', '*', $locale);
135
136 $line = $this->loaded['*']['*'][$locale][$key] ?? null;
137
138 // If we can't find a translation for the JSON key, we will attempt to translate it
139 // using the typical translation file. This way developers can always just use a
140 // helper such as __ instead of having to pick between trans or __ with views.
141 if (! isset($line)) {
142 [$namespace, $group, $item] = $this->parseKey($key);
143
144 // Here we will get the locale that should be used for the language line. If one
145 // was not passed, we will use the default locales which was given to us when
146 // the translator was instantiated. Then, we can load the lines and return.
147 $locales = $fallback ? $this->localeArray($locale) : [$locale];
148
149 foreach ($locales as $locale) {
150 if (! is_null($line = $this->getLine(
151 $namespace, $group, $locale, $item, $replace
152 ))) {
153 return $line;
154 }
155 }
156 }
157
158 // If the line doesn't exist, we will return back the key which was requested as
159 // that will be quick to spot in the UI if language keys are wrong or missing
160 // from the application's language files. Otherwise we can return the line.
161 return $this->makeReplacements($line ?: $key, $replace);
162 }

173 public function choice($key, $number, array $replace = [], $locale = null)
174 {
175 $line = $this->get(
176 $key, $replace, $locale = $this->localeForChoice($locale)
177 );
178

214 protected function getLine($namespace, $group, $locale, $item, array $replace)
215 {
216 $this->load($namespace, $group, $locale);
217
218 $line = Arr::get($this->loaded[$namespace][$group][$locale], $item);
219
220 if (is_string($line)) {
221 return $this->makeReplacements($line, $replace);
222 } elseif (is_array($line) && count($line) > 0) {
223 array_walk_recursive($line, function (&$value, $key) use ($replace) {
224 $value = $this->makeReplacements($value, $replace);
225 });

238 protected function makeReplacements($line, array $replace)
239 {
240 if (empty($replace)) {
241 return $line;
242 }
243
244 $shouldReplace = [];
245
246 foreach ($replace as $key => $value) {
247 if (is_object($value) && isset($this->stringableHandlers[get_class($value)])) {
248 $value = call_user_func($this->stringableHandlers[get_class($value)], $value);
249 }
250
251 $shouldReplace[':'.Str::ucfirst($key ?? '')] = Str::ucfirst($value ?? '');
252 $shouldReplace[':'.Str::upper($key ?? '')] = Str::upper($value ?? '');
253 $shouldReplace[':'.$key] = $value;
254 }
255
256 return strtr($line, $shouldReplace);
257 }
