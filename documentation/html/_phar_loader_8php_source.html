41 $loadedExtensions = [];
42
43 foreach ((new FileIteratorFacade)->getFilesAsArray($directory, '.phar') as $file) {
44 if (!$pharExtensionLoaded) {
45 Event\Facade::emitter()->testRunnerTriggeredWarning(
46 sprintf(
47 'Cannot load extension from %s because the PHAR extension is not available',
48 $file,
49 ),
50 );
51
52 continue;
53 }
54
55 if (!is_file('phar://' . $file . '/manifest.xml')) {
56 Event\Facade::emitter()->testRunnerTriggeredWarning(
57 sprintf(
58 '%s is not an extension for PHPUnit',
59 $file,
60 ),
61 );
62
63 continue;
64 }

67 $applicationName = new ApplicationName('phpunit/phpunit');

69 $manifest = ManifestLoader::fromFile('phar://' . $file . '/manifest.xml');
70
71 if (!$manifest->isExtensionFor($applicationName)) {

75 $file,
76 ),

100 );
