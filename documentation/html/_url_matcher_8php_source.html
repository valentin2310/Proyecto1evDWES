121 protected function matchCollection(string $pathinfo, RouteCollection $routes): array
122 {
123 // HEAD and GET are equivalent as per RFC
124 if ('HEAD' === $method = $this->context->getMethod()) {
125 $method = 'GET';
126 }
127 $supportsTrailingSlash = 'GET' === $method && $this instanceof RedirectableUrlMatcherInterface;
128 $trimmedPathinfo = rtrim($pathinfo, '/') ?: '/';
129
130 foreach ($routes as $name => $route) {
131 $compiledRoute = $route->compile();
132 $staticPrefix = rtrim($compiledRoute->getStaticPrefix(), '/');
133 $requiredMethods = $route->getMethods();
134
135 // check the static prefix of the URL first. Only use the more expensive preg_match when it matches
136 if ('' !== $staticPrefix && !str_starts_with($trimmedPathinfo, $staticPrefix)) {
137 continue;
138 }
139 $regex = $compiledRoute->getRegex();
140
141 $pos = strrpos($regex, '$');
142 $hasTrailingSlash = '/' === $regex[$pos - 1];
143 $regex = substr_replace($regex, '/?$', $pos - $hasTrailingSlash, 1 + $hasTrailingSlash);
144
145 if (!preg_match($regex, $pathinfo, $matches)) {
146 continue;
147 }
148
149 $hasTrailingVar = $trimmedPathinfo !== $pathinfo && preg_match('#\{[\w\x80-\xFF]+\}/?$#', $route->getPath());
150
151 if ($hasTrailingVar && ($hasTrailingSlash || (null === $m = $matches[\count($compiledRoute->getPathVariables())] ?? null) || '/' !== ($m[-1] ?? '/')) && preg_match($regex, $trimmedPathinfo, $m)) {
152 if ($hasTrailingSlash) {
153 $matches = $m;
154 } else {
155 $hasTrailingVar = false;
156 }
157 }
158
159 $hostMatches = [];
160 if ($compiledRoute->getHostRegex() && !preg_match($compiledRoute->getHostRegex(), $this->context->getHost(), $hostMatches)) {
161 continue;
162 }
163
164 $attributes = $this->getAttributes($route, $name, array_replace($matches, $hostMatches));
165
166 $status = $this->handleRouteRequirements($pathinfo, $name, $route, $attributes);
167
168 if (self::REQUIREMENT_MISMATCH === $status[0]) {
169 continue;
170 }
171
172 if ('/' !== $pathinfo && !$hasTrailingVar && $hasTrailingSlash === ($trimmedPathinfo === $pathinfo)) {
173 if ($supportsTrailingSlash && (!$requiredMethods || \in_array('GET', $requiredMethods))) {
174 return $this->allow = $this->allowSchemes = [];
175 }
176 continue;
177 }
178
179 if ($route->getSchemes() && !$route->hasScheme($this->context->getScheme())) {
180 $this->allowSchemes = array_merge($this->allowSchemes, $route->getSchemes());
181 continue;
182 }
183
184 if ($requiredMethods && !\in_array($method, $requiredMethods)) {
185 $this->allow = array_merge($this->allow, $requiredMethods);
186 continue;
187 }
188
189 return array_replace($attributes, $status[1] ?? []);
190 }
191
192 return [];
193 }

219 protected function handleRouteRequirements(string $pathinfo, string $name, Route $route/* , array $routeParameters */): array
220 {
221 if (\func_num_args() < 4) {
222 trigger_deprecation('symfony/routing', '6.1', 'The "%s()" method will have a new "array $routeParameters" argument in version 7.0, not defining it is deprecated.', __METHOD__);
223 $routeParameters = [];
224 } else {
225 $routeParameters = func_get_arg(3);
226
227 if (!\is_array($routeParameters)) {
228 throw new \TypeError(sprintf('"%s": Argument $routeParameters is expected to be an array, got "%s".', __METHOD__, get_debug_type($routeParameters)));
229 }
230 }
231
232 // expression condition
233 if ($route->getCondition() && !$this->getExpressionLanguage()->evaluate($route->getCondition(), [
234 'context' => $this->context,
235 'request' => $this->request ?: $this->createRequest($pathinfo),
236 'params' => $routeParameters,
237 ])) {
238 return [self::REQUIREMENT_MISMATCH, null];
239 }
240
241 return [self::REQUIREMENT_MATCH, null];
