147 {

152 $size = TestSize::unknown();
153
154 if ($test->size()->isSmall()) {
155 $size = TestSize::small();

164 $this->codeCoverage->start(

172 public function stop(bool $append = true, array|false $linesToBeCovered = [], array $linesToBeUsed = []): void
173 {
174 if (!$this->collecting) {
175 return;
176 }
177
178 $status = TestStatus::unknown();
179
180 if ($this->test !== null) {
181 if ($this->test->status()->isSuccess()) {
182 $status = TestStatus::success();
183 } else {
184 $status = TestStatus::failure();
185 }
186 }
187
188 /* @noinspection UnusedFunctionResultInspection */
189 $this->codeCoverage->stop($append, $status, $linesToBeCovered, $linesToBeUsed, $this->linesToBeIgnored);
190
191 $this->test = null;
192 $this->collecting = false;
193 }
